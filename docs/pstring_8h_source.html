271 bool ensure_capacity( std::uint32_t required ) noexcept
272 {
273 if ( required <= _capacity )
274 return true;
275
276 // Новая ёмкость: удвоение текущей или required, что больше.
277 // Минимум 16 символов для избежания частых реаллокаций.
278 std::uint32_t new_cap = _capacity * 2;
279 if ( new_cap < required )
280 new_cap = required;
281 if ( new_cap < 16 )
282 new_cap = 16;
283
284 // Выделяем новый блок данных: new_cap + 1 байт для null-terminator.
285 std::size_t alloc_size = static_cast<std::size_t>( new_cap ) + 1;
286 void* new_raw = ManagerT::allocate( alloc_size );
287 if ( new_raw == nullptr )
288 return false;
289
290 // Создаём новый индекс.
291 std::uint8_t* base = ManagerT::backend().base_ptr();
292 std::size_t byte_off = static_cast<std::uint8_t*>( new_raw ) - base;
293 index_type new_dat_idx = static_cast<index_type>( byte_off / ManagerT::address_traits::granule_size );
294
295 // Копируем старые данные.
296 if ( _length > 0 && _data_idx != static_cast<index_type>( 0 ) )
297 {
298 char* old_data = resolve_data();
299 if ( old_data != nullptr )
300 std::memcpy( new_raw, old_data, static_cast<std::size_t>( _length ) + 1 );
301 }
302 else
303 {
304 // Инициализируем пустую строку.
305 static_cast<char*>( new_raw )[0] = '\0';
306 }
307
308 // Освобождаем старый блок.
309 if ( _data_idx != static_cast<index_type>( 0 ) )
310 {
311 void* old_raw = base + static_cast<std::size_t>( _data_idx ) * ManagerT::address_traits::granule_size;
312 ManagerT::deallocate( old_raw );
313 }
314
315 _data_idx = new_dat_idx;
316 _capacity = new_cap;
317 return true;
318 }
