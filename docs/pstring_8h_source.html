270 bool ensure_capacity( std::uint32_t required ) noexcept
271 {
272 if ( required <= _capacity )
273 return true;
274
275 // Новая ёмкость: удвоение текущей или required, что больше.
276 // Минимум 16 символов для избежания частых реаллокаций.
277 std::uint32_t new_cap = _capacity * 2;
278 if ( new_cap < required )
279 new_cap = required;
280 if ( new_cap < 16 )
281 new_cap = 16;
282
283 // Выделяем новый блок данных: new_cap + 1 байт для null-terminator.
284 std::size_t alloc_size = static_cast<std::size_t>( new_cap ) + 1;
285 void* new_raw = ManagerT::allocate( alloc_size );
286 if ( new_raw == nullptr )
287 return false;
288
289 // Создаём новый индекс.
290 std::uint8_t* base = ManagerT::backend().base_ptr();
291 index_type new_dat_idx = detail::ptr_to_granule_idx<typename ManagerT::address_traits>( base, new_raw );
292
293 // Копируем старые данные.
294 if ( _length > 0 && _data_idx != static_cast<index_type>( 0 ) )
295 {
296 char* old_data = resolve_data();
297 if ( old_data != nullptr )
298 std::memcpy( new_raw, old_data, static_cast<std::size_t>( _length ) + 1 );
299 }
300 else
301 {
302 // Инициализируем пустую строку.
303 static_cast<char*>( new_raw )[0] = '\0';
304 }
305
306 // Освобождаем старый блок.
307 if ( _data_idx != static_cast<index_type>( 0 ) )
308 ManagerT::deallocate( detail::resolve_granule_ptr<typename ManagerT::address_traits>( base, _data_idx ) );
