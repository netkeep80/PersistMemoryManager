84 static constexpr index_type bytes_to_granules( std::size_t bytes ) noexcept
85 {
86 if ( bytes == 0 )
87 return static_cast<index_type>( 0 );
88 // Overflow-safe ceiling division: (bytes + granule_size - 1) / granule_size
89 if ( bytes > std::numeric_limits<std::size_t>::max() - ( granule_size - 1 ) )
90 return static_cast<index_type>( 0 ); // overflow
91 std::size_t granules = ( bytes + granule_size - 1 ) / granule_size;
92 if ( granules > static_cast<std::size_t>( std::numeric_limits<IndexT>::max() ) )
93 return static_cast<index_type>( 0 ); // overflow for IndexT
94 return static_cast<index_type>( granules );
95 }
