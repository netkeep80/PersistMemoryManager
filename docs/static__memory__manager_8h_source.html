117 static bool create( std::size_t initial_size ) noexcept
118 {
119 typename thread_policy::unique_lock_type lock( _mutex );
120 if ( initial_size < detail::kMinMemorySize )
121 return false;
122 std::size_t aligned = ( ( initial_size + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
123 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
124 {
125 // Либо буфера нет, либо он меньше требуемого — расширяем
126 std::size_t additional =
127 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
128 if ( !_backend.expand( additional ) )
129 return false;
130 }
131 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
132 return false;
133 return init_layout( _backend.base_ptr(), _backend.total_size() );
134 }

154 static bool load() noexcept
155 {
156 typename thread_policy::unique_lock_type lock( _mutex );
157 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
158 return false;
159 std::uint8_t* base = _backend.base_ptr();
160 detail::ManagerHeader* hdr = get_header( base );
161 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )
162 return false;
163 if ( hdr->granule_size != static_cast<std::uint16_t>( kGranuleSize ) )
164 return false;
165 hdr->owns_memory = hdr->prev_owns_memory = false;
166 hdr->prev_total_size = 0;
167 hdr->prev_base_ptr = nullptr;
168 allocator::repair_linked_list( base, hdr );
169 allocator::recompute_counters( base, hdr );
170 allocator::rebuild_free_tree( base, hdr );
171 _initialized = true;
172 return true;
173 }

202 static void* allocate( std::size_t user_size ) noexcept
203 {
204 typename thread_policy::unique_lock_type lock( _mutex );
205 if ( !_initialized || user_size == 0 )
206 return nullptr;
207
208 std::uint8_t* base = _backend.base_ptr();
209 detail::ManagerHeader* hdr = get_header( base );
210 std::uint32_t needed = detail::required_block_granules( user_size );
211 std::uint32_t idx = free_block_tree::find_best_fit( base, hdr, needed );
212
213 if ( idx != detail::kNoBlock )
214 return allocator::allocate_from_block( base, hdr, idx, user_size );
215
216 // Попытка расширить (если бэкенд поддерживает)
217 if ( !do_expand( user_size ) )
218 return nullptr;
219
220 base = _backend.base_ptr();
221 hdr = get_header( base );
222 idx = free_block_tree::find_best_fit( base, hdr, needed );
223 if ( idx != detail::kNoBlock )
224 return allocator::allocate_from_block( base, hdr, idx, user_size );
225 return nullptr;
226 }

231 static void deallocate( void* ptr ) noexcept
232 {
233 typename thread_policy::unique_lock_type lock( _mutex );
234 if ( !_initialized || ptr == nullptr )
235 return;
236 std::uint8_t* base = _backend.base_ptr();
237 detail::ManagerHeader* hdr = get_header( base );
238 pmm::Block<address_traits>* blk =
239 detail::header_from_ptr( base, ptr, static_cast<std::size_t>( hdr->total_size ) );
240 if ( blk == nullptr || blk->weight == 0 )
241 return;
242
243 std::uint32_t freed = blk->weight;
244 std::uint32_t blk_idx = detail::block_idx( base, blk );
245
246 AllocatedBlock<address_traits>* alloc = AllocatedBlock<address_traits>::cast_from_raw( blk );
247 alloc->mark_as_free();
248
249 hdr->alloc_count--;
250 hdr->free_count++;
251 if ( hdr->used_size >= freed )
252 hdr->used_size -= freed;
253 allocator::coalesce( base, hdr, blk_idx );
254 }

406 {
407 static constexpr std::uint32_t kHdrBlkIdx = 0;
408 static constexpr std::uint32_t kFreeBlkIdx = detail::kBlockHeaderGranules + detail::kManagerHeaderGranules;
409

411 return false;
412
413 Block<address_traits>* hdr_blk = reinterpret_cast<Block<address_traits>*>( base );
414 std::memset( hdr_blk, 0, sizeof( Block<address_traits> ) );
415 hdr_blk->weight = detail::kManagerHeaderGranules;
416 hdr_blk->prev_offset = detail::kNoBlock;
417 hdr_blk->next_offset = kFreeBlkIdx;
418 hdr_blk->left_offset = detail::kNoBlock;
419 hdr_blk->right_offset = detail::kNoBlock;
420 hdr_blk->parent_offset = detail::kNoBlock;
421 hdr_blk->avl_height = 0;
422 hdr_blk->root_offset = kHdrBlkIdx;

426 hdr->magic = kMagic;

431 hdr->granule_size = static_cast<std::uint16_t>( kGranuleSize );
432
433 Block<address_traits>* blk =

436 blk->prev_offset = kHdrBlkIdx;

439 blk->right_offset = detail::kNoBlock;

442 blk->weight = 0;
443 blk->root_offset = 0;

447 hdr->block_count = 2;
448 hdr->free_count = 1;

456 static bool do_expand( std::size_t user_size ) noexcept
457 {
458 if ( !_initialized )
459 return false;
460 std::uint8_t* base = _backend.base_ptr();
461 detail::ManagerHeader* hdr = get_header( base );
462 std::size_t old_size = hdr->total_size;
463
464 std::size_t min_need =
465 detail::granules_to_bytes( detail::required_block_granules( user_size ) + detail::kBlockHeaderGranules );
466 std::size_t growth = old_size / 4;
467 if ( growth < min_need )
468 growth = min_need;
469
470 if ( !_backend.expand( growth ) )
471 return false;
472
473 std::uint8_t* new_base = _backend.base_ptr();
474 std::size_t new_size = _backend.total_size();
475 if ( new_base == nullptr || new_size <= old_size )
476 return false;
477
478 hdr = get_header( new_base );
479
480 std::uint32_t extra_idx = detail::byte_off_to_idx( old_size );
481 std::size_t extra_size = new_size - old_size;
482
483 Block<address_traits>* last_blk = ( hdr->last_block_offset != detail::kNoBlock )
484 ? reinterpret_cast<Block<address_traits>*>(
485 new_base + detail::idx_to_byte_off( hdr->last_block_offset ) )
486 : nullptr;
487
488 if ( last_blk != nullptr && last_blk->weight == 0 )
489 {
490 std::uint32_t loff = detail::block_idx( new_base, last_blk );
491 free_block_tree::remove( new_base, hdr, loff );
492 hdr->total_size = new_size;
493 free_block_tree::insert( new_base, hdr, loff );
494 }
495 else
496 {
497 if ( extra_size < sizeof( Block<address_traits> ) + detail::kMinBlockSize )
498 return false;
499 Block<address_traits>* nb_blk =
500 reinterpret_cast<Block<address_traits>*>( new_base + detail::idx_to_byte_off( extra_idx ) );
501 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
502 nb_blk->left_offset = detail::kNoBlock;
503 nb_blk->right_offset = detail::kNoBlock;
504 nb_blk->parent_offset = detail::kNoBlock;
505 nb_blk->avl_height = 1;
506 nb_blk->weight = 0;
507 nb_blk->root_offset = 0;
508 if ( last_blk != nullptr )
509 {
510 std::uint32_t loff = detail::block_idx( new_base, last_blk );
511 nb_blk->prev_offset = loff;
512 nb_blk->next_offset = detail::kNoBlock;
513 last_blk->next_offset = extra_idx;
514 }
515 else
516 {
517 nb_blk->prev_offset = detail::kNoBlock;
518 nb_blk->next_offset = detail::kNoBlock;
519 hdr->first_block_offset = extra_idx;
520 }
521 hdr->last_block_offset = extra_idx;
522 hdr->block_count++;
523 hdr->free_count++;
524 hdr->total_size = new_size;
525 free_block_tree::insert( new_base, hdr, extra_idx );
526 }
527 return true;
528 }
