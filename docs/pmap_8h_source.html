178 node_pptr insert( const _K& key, const _V& val ) noexcept
179 {
180 // Ищем существующий узел.
181 node_pptr existing = _avl_find( key );
182 if ( !existing.is_null() )
183 {
184 // Ключ найден — обновляем значение.
185 node_type* obj = ManagerT::template resolve<node_type>( existing );
186 if ( obj != nullptr )
187 obj->value = val;
188 return existing;
189 }
190
191 // Создаём новый узел в ПАП.
192 node_pptr new_node = ManagerT::template allocate_typed<node_type>();
193 if ( new_node.is_null() )
194 return node_pptr();
195
196 node_type* obj = ManagerT::template resolve<node_type>( new_node );
197 if ( obj == nullptr )
198 return node_pptr();
199
200 obj->key = key;
201 obj->value = val;
202
203 // Инициализируем AVL-поля нового узла (no_block = нет связи).
204 auto& tn = new_node.tree_node();
205 tn.set_left( no_block );
206 tn.set_right( no_block );
207 tn.set_parent( no_block );
208 tn.set_height( static_cast<std::int16_t>( 1 ) );
209
210 // Вставляем в AVL-дерево.
211 _avl_insert( new_node );
212
213 return new_node;
214 }

301 iterator& operator++() noexcept
302 {
303 if ( _current_idx == static_cast<index_type>( 0 ) || _current_idx == no_block )
304 return *this;

306 node_pptr cur( _current_idx );
307
308 // Если есть правый потомок — идём в его крайний левый узел.
309 auto right_idx = cur.tree_node().get_right();
310 if ( right_idx != no_block )
311 {
312 node_pptr right( right_idx );
313 while ( true )
314 {
315 auto left_idx = right.tree_node().get_left();
316 if ( left_idx == no_block )
317 break;
318 right = node_pptr( left_idx );
319 }
320 _current_idx = right.offset();
321 return *this;
322 }
323
324 // Иначе — идём вверх, пока не окажемся левым потомком.
325 while ( true )
326 {
327 auto parent_idx = cur.tree_node().get_parent();
328 if ( parent_idx == no_block )
329 {
330 // Достигли корня снизу справа — конец обхода.
331 _current_idx = static_cast<index_type>( 0 );
332 return *this;
333 }
334 node_pptr parent( parent_idx );
335 auto parent_left = parent.tree_node().get_left();
336 if ( parent_left == cur.offset() )
337 {
338 // cur — левый потомок: parent — следующий.
339 _current_idx = parent_idx;
