186template <typename AddressTraitsT> inline std::uint32_t bytes_to_granules_t( std::size_t bytes )
187{
188 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
189 if ( bytes > std::numeric_limits<std::size_t>::max() - ( kGranSz - 1 ) )
190 return 0;
191 std::size_t granules = ( bytes + kGranSz - 1 ) / kGranSz;
192 if ( granules > std::numeric_limits<std::uint32_t>::max() )
193 return 0;
194 return static_cast<std::uint32_t>( granules );
195}

198template <typename AddressTraitsT> inline typename AddressTraitsT::index_type bytes_to_idx_t( std::size_t bytes )
199{
200 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
201 using IndexT = typename AddressTraitsT::index_type;
202 if ( bytes == 0 )
203 return static_cast<IndexT>( 0 );
204 if ( bytes > std::numeric_limits<std::size_t>::max() - ( kGranSz - 1 ) )
205 return AddressTraitsT::no_block; // overflow
206 std::size_t granules = ( bytes + kGranSz - 1 ) / kGranSz;
207 if ( granules > static_cast<std::size_t>( std::numeric_limits<IndexT>::max() ) )
208 return AddressTraitsT::no_block; // overflow for IndexT
209 return static_cast<IndexT>( granules );
210}

328template <typename AddressTraitsT>
329inline std::uint32_t block_total_granules( const std::uint8_t* base, const ManagerHeader* hdr,
330 const pmm::Block<AddressTraitsT>* blk )
331{
332 using BlockState = pmm::BlockStateBase<AddressTraitsT>;
333 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
334 using IndexT = typename AddressTraitsT::index_type;
335 static constexpr IndexT kNoBlk = AddressTraitsT::no_block;
336
337 std::size_t byte_off = reinterpret_cast<const std::uint8_t*>( blk ) - base;
338 IndexT this_idx = static_cast<IndexT>( byte_off / kGranSz );
339 IndexT next_off = BlockState::get_next_offset( blk );
340 IndexT total_gran = static_cast<IndexT>( hdr->total_size / kGranSz );
341 if ( next_off != kNoBlk )
342 return static_cast<std::uint32_t>( next_off - this_idx );
343 return static_cast<std::uint32_t>( total_gran - this_idx );
344}

348inline bool is_valid_block( const std::uint8_t* base, const ManagerHeader* hdr, std::uint32_t idx )
349{
350 using BlockState = pmm::BlockStateBase<pmm::DefaultAddressTraits>;
351 if ( idx == kNoBlock )
352 return false;
353 if ( idx_to_byte_off( idx ) + sizeof( pmm::Block<pmm::DefaultAddressTraits> ) > hdr->total_size )
354 return false;
355
356 const void* blk = base + idx_to_byte_off( idx );
357 auto next_off = BlockState::get_next_offset( blk );
358 std::uint32_t total_gran = ( next_off != kNoBlock )
359 ? ( next_off - idx )
360 : ( byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - idx );
361 if ( BlockState::get_weight( blk ) >= total_gran )
362 return false;
363 auto prev_off = BlockState::get_prev_offset( blk );
364 if ( prev_off != kNoBlock && prev_off >= idx )
365 return false;
366 if ( next_off != kNoBlock && next_off <= idx )
367 return false;
368 if ( BlockState::get_avl_height( blk ) >= 32 )
369 return false;
370 auto left_off = BlockState::get_left_offset( blk );
371 auto right_off = BlockState::get_right_offset( blk );
372 auto parent_off = BlockState::get_parent_offset( blk );
373 const bool l = ( left_off != kNoBlock );
374 const bool r = ( right_off != kNoBlock );
375 const bool p = ( parent_off != kNoBlock );
376 if ( ( l || r || p ) && ( ( l && r && left_off == right_off ) || ( l && p && left_off == parent_off ) ||
377 ( r && p && right_off == parent_off ) ) )
378 return false;
379 return true;
380}

395inline pmm::Block<pmm::DefaultAddressTraits>* header_from_ptr( std::uint8_t* base, void* ptr, std::size_t total_size )
396{
397 using BlockState = pmm::BlockStateBase<pmm::DefaultAddressTraits>;
398 if ( ptr == nullptr )
399 return nullptr;
400 std::uint8_t* raw_ptr = reinterpret_cast<std::uint8_t*>( ptr );
401 // First user data starts after Block_0 + ManagerHeader + Block_1 (Issue #75, #112)
402 static constexpr std::size_t kBlockSize = sizeof( pmm::Block<pmm::DefaultAddressTraits> );
403 std::uint8_t* min_addr = base + kBlockSize + sizeof( ManagerHeader ) + kBlockSize;
404 if ( raw_ptr < min_addr )
405 return nullptr;
406 if ( raw_ptr > base + total_size )
407 return nullptr;
408 std::uint8_t* cand_addr = raw_ptr - kBlockSize;
409 if ( ( reinterpret_cast<std::size_t>( cand_addr ) - reinterpret_cast<std::size_t>( base ) ) % kGranuleSize != 0 )
410 return nullptr;
411 std::uint32_t cand_idx = static_cast<std::uint32_t>( ( cand_addr - base ) / kGranuleSize );
412 // Issue #83: ManagerHeader is at base + kBlockSize.
413 const ManagerHeader* hdr_const = reinterpret_cast<const ManagerHeader*>( base + kBlockSize );
414 if ( !is_valid_block( base, hdr_const, cand_idx ) )
415 return nullptr;
416 if ( BlockState::get_weight( cand_addr ) == 0 )
417 return nullptr;
418 return reinterpret_cast<pmm::Block<pmm::DefaultAddressTraits>*>( cand_addr );
419}

423template <typename AddressTraitsT>
424inline pmm::Block<AddressTraitsT>* header_from_ptr_t( std::uint8_t* base, void* ptr, std::size_t total_size )
425{
426 using BlockState = pmm::BlockStateBase<AddressTraitsT>;
427 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
428 static constexpr std::size_t kBlockSize = sizeof( pmm::Block<AddressTraitsT> );
429 using IndexT = typename AddressTraitsT::index_type;
430 static constexpr IndexT kNoBlk = AddressTraitsT::no_block;
431
432 if ( ptr == nullptr )
433 return nullptr;
434 std::uint8_t* raw_ptr = reinterpret_cast<std::uint8_t*>( ptr );
435 // First user data starts after Block_0 + ManagerHeader + Block_1
436 std::uint8_t* min_addr = base + kBlockSize + sizeof( ManagerHeader ) + kBlockSize;
437 if ( raw_ptr < min_addr )
438 return nullptr;
439 if ( raw_ptr > base + total_size )
440 return nullptr;
441 std::uint8_t* cand_addr = raw_ptr - kBlockSize;
442 if ( ( reinterpret_cast<std::size_t>( cand_addr ) - reinterpret_cast<std::size_t>( base ) ) % kGranSz != 0 )
443 return nullptr;
444 // Validate via BlockState (uses AddressTraitsT::no_block for sentinel checks)
445 if ( BlockState::get_weight( cand_addr ) == 0 )
446 return nullptr;
447 // Basic sanity: candidate address is within bounds
448 if ( cand_addr < base || cand_addr + kBlockSize > base + total_size )
449 return nullptr;
450 return reinterpret_cast<pmm::Block<AddressTraitsT>*>( cand_addr );
451}
