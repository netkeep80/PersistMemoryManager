170inline std::uint32_t bytes_to_granules( std::size_t bytes )
171{
172 if ( bytes > std::numeric_limits<std::size_t>::max() - ( kGranuleSize - 1 ) )
173 return 0;
174 std::size_t granules = ( bytes + kGranuleSize - 1 ) / kGranuleSize;
175 if ( granules > std::numeric_limits<std::uint32_t>::max() )
176 return 0;
177 return static_cast<std::uint32_t>( granules );
178}

230inline std::uint32_t block_total_granules( const std::uint8_t* base, const ManagerHeader* hdr,
231 const pmm::Block<pmm::DefaultAddressTraits>* blk )
232{
233 using BlockState = pmm::BlockStateBase<pmm::DefaultAddressTraits>;
234 std::uint32_t this_idx = block_idx( base, blk );
235 auto next_off = BlockState::get_next_offset( blk );
236 if ( next_off != kNoBlock )
237 return next_off - this_idx;

243inline bool is_valid_block( const std::uint8_t* base, const ManagerHeader* hdr, std::uint32_t idx )
244{
245 using BlockState = pmm::BlockStateBase<pmm::DefaultAddressTraits>;
246 if ( idx == kNoBlock )
247 return false;
248 if ( idx_to_byte_off( idx ) + sizeof( pmm::Block<pmm::DefaultAddressTraits> ) > hdr->total_size )
249 return false;
250
251 const void* blk = base + idx_to_byte_off( idx );
252 auto next_off = BlockState::get_next_offset( blk );
253 std::uint32_t total_gran = ( next_off != kNoBlock )
254 ? ( next_off - idx )
255 : ( byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - idx );
256 if ( BlockState::get_weight( blk ) >= total_gran )
257 return false;
258 auto prev_off = BlockState::get_prev_offset( blk );
259 if ( prev_off != kNoBlock && prev_off >= idx )
260 return false;
261 if ( next_off != kNoBlock && next_off <= idx )
262 return false;
263 if ( BlockState::get_avl_height( blk ) >= 32 )
264 return false;
265 auto left_off = BlockState::get_left_offset( blk );
266 auto right_off = BlockState::get_right_offset( blk );
267 auto parent_off = BlockState::get_parent_offset( blk );
268 const bool l = ( left_off != kNoBlock );
269 const bool r = ( right_off != kNoBlock );
270 const bool p = ( parent_off != kNoBlock );
271 if ( ( l || r || p ) && ( ( l && r && left_off == right_off ) || ( l && p && left_off == parent_off ) ||
272 ( r && p && right_off == parent_off ) ) )
273 return false;
274 return true;
275}

285inline pmm::Block<pmm::DefaultAddressTraits>* header_from_ptr( std::uint8_t* base, void* ptr, std::size_t total_size )
286{
287 using BlockState = pmm::BlockStateBase<pmm::DefaultAddressTraits>;
288 if ( ptr == nullptr )
289 return nullptr;
290 std::uint8_t* raw_ptr = reinterpret_cast<std::uint8_t*>( ptr );
291 // First user data starts after Block_0 + ManagerHeader + Block_1 (Issue #75, #112)
292 static constexpr std::size_t kBlockSize = sizeof( pmm::Block<pmm::DefaultAddressTraits> );
293 std::uint8_t* min_addr = base + kBlockSize + sizeof( ManagerHeader ) + kBlockSize;
294 if ( raw_ptr < min_addr )
295 return nullptr;
296 if ( raw_ptr > base + total_size )
297 return nullptr;
298 std::uint8_t* cand_addr = raw_ptr - kBlockSize;
299 if ( ( reinterpret_cast<std::size_t>( cand_addr ) - reinterpret_cast<std::size_t>( base ) ) % kGranuleSize != 0 )
300 return nullptr;
301 std::uint32_t cand_idx = static_cast<std::uint32_t>( ( cand_addr - base ) / kGranuleSize );
302 // Issue #83: ManagerHeader is at base + kBlockSize.
303 const ManagerHeader* hdr_const = reinterpret_cast<const ManagerHeader*>( base + kBlockSize );
304 if ( !is_valid_block( base, hdr_const, cand_idx ) )
305 return nullptr;
306 if ( BlockState::get_weight( cand_addr ) == 0 )
307 return nullptr;
308 return reinterpret_cast<pmm::Block<pmm::DefaultAddressTraits>*>( cand_addr );
309}
