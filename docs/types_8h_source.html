200inline std::uint32_t bytes_to_granules( std::size_t bytes )
201{
202 if ( bytes > std::numeric_limits<std::size_t>::max() - ( kGranuleSize - 1 ) )
203 return 0;
204 std::size_t granules = ( bytes + kGranuleSize - 1 ) / kGranuleSize;
205 if ( granules > std::numeric_limits<std::uint32_t>::max() )
206 return 0;
207 return static_cast<std::uint32_t>( granules );
208}

303inline bool is_valid_block( const std::uint8_t* base, const ManagerHeader* hdr, std::uint32_t idx )
304{
305 if ( idx == kNoBlock )
306 return false;
307 if ( idx_to_byte_off( idx ) + sizeof( BlockHeader ) > hdr->total_size )
308 return false;
309
310 const auto* blk = reinterpret_cast<const pmm::Block<pmm::DefaultAddressTraits>*>( base + idx_to_byte_off( idx ) );
311 std::uint32_t total_gran = ( blk->next_offset != kNoBlock )
312 ? ( blk->next_offset - idx )
313 : ( byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - idx );
314 if ( blk->weight >= total_gran )
315 return false;
316 if ( blk->prev_offset != kNoBlock && blk->prev_offset >= idx )
317 return false;
318 if ( blk->next_offset != kNoBlock && blk->next_offset <= idx )
319 return false;
320 if ( blk->avl_height >= 32 )
321 return false;
322 const bool l = ( blk->left_offset != kNoBlock );
323 const bool r = ( blk->right_offset != kNoBlock );
324 const bool p = ( blk->parent_offset != kNoBlock );
325 if ( ( l || r || p ) && ( ( l && r && blk->left_offset == blk->right_offset ) ||
326 ( l && p && blk->left_offset == blk->parent_offset ) ||
327 ( r && p && blk->right_offset == blk->parent_offset ) ) )
328 return false;
329 return true;
330}

346inline pmm::Block<pmm::DefaultAddressTraits>* header_from_ptr( std::uint8_t* base, void* ptr, std::size_t total_size )
347{
348 if ( ptr == nullptr )
349 return nullptr;
350 std::uint8_t* raw_ptr = reinterpret_cast<std::uint8_t*>( ptr );
351 // First user data starts after BlockHeader_0 + ManagerHeader + BlockHeader_1 (Issue #75)
352 std::uint8_t* min_addr = base + sizeof( BlockHeader ) + sizeof( ManagerHeader ) + sizeof( BlockHeader );
353 if ( raw_ptr < min_addr )
354 return nullptr;
355 if ( raw_ptr > base + total_size )
356 return nullptr;
357 std::uint8_t* cand_addr = raw_ptr - sizeof( BlockHeader );
358 if ( ( reinterpret_cast<std::size_t>( cand_addr ) - reinterpret_cast<std::size_t>( base ) ) % kGranuleSize != 0 )
359 return nullptr;
360 std::uint32_t cand_idx = static_cast<std::uint32_t>( ( cand_addr - base ) / kGranuleSize );
361 // Issue #83: ManagerHeader is at base + sizeof(BlockHeader), not base.
362 const ManagerHeader* hdr_const = reinterpret_cast<const ManagerHeader*>( base + sizeof( BlockHeader ) );
363 if ( !is_valid_block( base, hdr_const, cand_idx ) )
364 return nullptr;
365 auto* cand = reinterpret_cast<pmm::Block<pmm::DefaultAddressTraits>*>( cand_addr );
366 if ( cand->weight == 0 )
367 return nullptr;
368 return cand;
369}
