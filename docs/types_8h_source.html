124//
125// Software CRC32 (ISO 3309 / ITU-T V.42 polynomial 0xEDB88320).
126// Header-only, no external dependencies. Used by io.h save/load functions.

130inline std::uint32_t crc32_accumulate_byte( std::uint32_t crc, std::uint8_t byte ) noexcept
131{
132 crc ^= byte;
133 for ( int bit = 0; bit < 8; ++bit )
134 crc = ( crc >> 1 ) ^ ( 0xEDB88320U & ( ~( crc & 1U ) + 1U ) );
135 return crc;
136}

140inline std::uint32_t compute_crc32( const std::uint8_t* data, std::size_t length ) noexcept
141{
142 std::uint32_t crc = 0xFFFFFFFFU;
143 for ( std::size_t i = 0; i < length; ++i )
144 crc = crc32_accumulate_byte( crc, data[i] );
145 return crc ^ 0xFFFFFFFFU;
146}

224template <typename AddressTraitsT>
225inline std::uint32_t compute_image_crc32( const std::uint8_t* data, std::size_t length ) noexcept
226{
227 // Offset of the crc32 field within ManagerHeader, which itself is located
228 // after Block_0 (sizeof(Block<AT>) bytes from base).
229 constexpr std::size_t kHdrOffset = sizeof( pmm::Block<AddressTraitsT> );
230 constexpr std::size_t kCrcOffset = kHdrOffset + offsetof( ManagerHeader<AddressTraitsT>, crc32 );
231 constexpr std::size_t kCrcSize = sizeof( std::uint32_t );
232 constexpr std::size_t kAfterCrc = kCrcOffset + kCrcSize;
233
234 // CRC everything before the crc32 field
235 std::uint32_t crc = 0xFFFFFFFFU;
236 for ( std::size_t i = 0; i < kCrcOffset && i < length; ++i )
237 crc = crc32_accumulate_byte( crc, data[i] );
238 // Skip the crc32 field (treat as 4 zero bytes)
239 for ( std::size_t i = 0; i < kCrcSize; ++i )
240 crc = crc32_accumulate_byte( crc, 0x00U );
241 // CRC everything after the crc32 field
242 for ( std::size_t i = kAfterCrc; i < length; ++i )
243 crc = crc32_accumulate_byte( crc, data[i] );
244 return crc ^ 0xFFFFFFFFU;
245}

266template <typename AddressTraitsT> inline typename AddressTraitsT::index_type bytes_to_granules_t( std::size_t bytes )
267{
268 using IndexT = typename AddressTraitsT::index_type;
269 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
270 if ( bytes > std::numeric_limits<std::size_t>::max() - ( kGranSz - 1 ) )
271 return static_cast<IndexT>( 0 );
272 std::size_t granules = ( bytes + kGranSz - 1 ) / kGranSz;
273 if ( granules > static_cast<std::size_t>( std::numeric_limits<IndexT>::max() ) )
274 return static_cast<IndexT>( 0 );
275 return static_cast<IndexT>( granules );
276}

279template <typename AddressTraitsT> inline typename AddressTraitsT::index_type bytes_to_idx_t( std::size_t bytes )
280{
281 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
282 using IndexT = typename AddressTraitsT::index_type;
283 if ( bytes == 0 )
284 return static_cast<IndexT>( 0 );
285 if ( bytes > std::numeric_limits<std::size_t>::max() - ( kGranSz - 1 ) )
286 return AddressTraitsT::no_block; // overflow
287 std::size_t granules = ( bytes + kGranSz - 1 ) / kGranSz;
288 if ( granules > static_cast<std::size_t>( std::numeric_limits<IndexT>::max() ) )
289 return AddressTraitsT::no_block; // overflow for IndexT
290 return static_cast<IndexT>( granules );
291}

370template <typename AddressTraitsT>
371inline typename AddressTraitsT::index_type block_total_granules( const std::uint8_t* base,
372 const ManagerHeader<AddressTraitsT>* hdr,
373 const pmm::Block<AddressTraitsT>* blk )
374{
375 using BlockState = pmm::BlockStateBase<AddressTraitsT>;
376 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
377 using IndexT = typename AddressTraitsT::index_type;
378 static constexpr IndexT kNoBlk = AddressTraitsT::no_block;
379
380 std::size_t byte_off = reinterpret_cast<const std::uint8_t*>( blk ) - base;
381 IndexT this_idx = static_cast<IndexT>( byte_off / kGranSz );
382 IndexT next_off = BlockState::get_next_offset( blk );
383 IndexT total_gran = static_cast<IndexT>( hdr->total_size / kGranSz );
384 if ( next_off != kNoBlk )
385 return static_cast<IndexT>( next_off - this_idx );
386 return static_cast<IndexT>( total_gran - this_idx );
387}

421template <typename AddressTraitsT>
422inline pmm::Block<AddressTraitsT>* header_from_ptr_t( std::uint8_t* base, void* ptr, std::size_t total_size )
423{
424 using BlockState = pmm::BlockStateBase<AddressTraitsT>;
425 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
426 static constexpr std::size_t kBlockSize = sizeof( pmm::Block<AddressTraitsT> );
427
428 if ( ptr == nullptr )
429 return nullptr;
430 std::uint8_t* raw_ptr = reinterpret_cast<std::uint8_t*>( ptr );
431 // First user data starts after Block_0 + ManagerHeader<AddressTraitsT> + Block_1
432 std::uint8_t* min_addr = base + kBlockSize + sizeof( ManagerHeader<AddressTraitsT> ) + kBlockSize;
433 if ( raw_ptr < min_addr )
434 return nullptr;
435 if ( raw_ptr > base + total_size )
436 return nullptr;
437 std::uint8_t* cand_addr = raw_ptr - kBlockSize;
438 if ( ( reinterpret_cast<std::size_t>( cand_addr ) - reinterpret_cast<std::size_t>( base ) ) % kGranSz != 0 )
439 return nullptr;
440 // Validate via BlockState (uses AddressTraitsT::no_block for sentinel checks)
441 if ( BlockState::get_weight( cand_addr ) == 0 )
442 return nullptr;
443 // Basic sanity: candidate address is within bounds
444 if ( cand_addr < base || cand_addr + kBlockSize > base + total_size )
445 return nullptr;
446 return reinterpret_cast<pmm::Block<AddressTraitsT>*>( cand_addr );
447}
