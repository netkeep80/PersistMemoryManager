134// Software CRC32 (ISO 3309 / ITU-T V.42 polynomial 0xEDB88320).

139inline std::uint32_t crc32_accumulate_byte( std::uint32_t crc, std::uint8_t byte ) noexcept
140{
141 crc ^= byte;
142 for ( int bit = 0; bit < 8; ++bit )
143 crc = ( crc >> 1 ) ^ ( 0xEDB88320U & ( ~( crc & 1U ) + 1U ) );
144 return crc;
145}

149inline std::uint32_t compute_crc32( const std::uint8_t* data, std::size_t length ) noexcept
150{
151 std::uint32_t crc = 0xFFFFFFFFU;
152 for ( std::size_t i = 0; i < length; ++i )
153 crc = crc32_accumulate_byte( crc, data[i] );
154 return crc ^ 0xFFFFFFFFU;
155}

230template <typename AddressTraitsT>
231inline std::uint32_t compute_image_crc32( const std::uint8_t* data, std::size_t length ) noexcept
232{
233 // Offset of the crc32 field within ManagerHeader, which itself is located
234 // after Block_0 (sizeof(Block<AT>) bytes from base).
235 constexpr std::size_t kHdrOffset = sizeof( pmm::Block<AddressTraitsT> );
236 constexpr std::size_t kCrcOffset = kHdrOffset + offsetof( ManagerHeader<AddressTraitsT>, crc32 );
237 constexpr std::size_t kCrcSize = sizeof( std::uint32_t );
238 constexpr std::size_t kAfterCrc = kCrcOffset + kCrcSize;
239
240 // CRC everything before the crc32 field
241 std::uint32_t crc = 0xFFFFFFFFU;
242 for ( std::size_t i = 0; i < kCrcOffset && i < length; ++i )
243 crc = crc32_accumulate_byte( crc, data[i] );
244 // Skip the crc32 field (treat as 4 zero bytes)
245 for ( std::size_t i = 0; i < kCrcSize; ++i )
246 crc = crc32_accumulate_byte( crc, 0x00U );
247 // CRC everything after the crc32 field
248 for ( std::size_t i = kAfterCrc; i < length; ++i )
249 crc = crc32_accumulate_byte( crc, data[i] );
250 return crc ^ 0xFFFFFFFFU;
251}

281template <typename AddressTraitsT> inline typename AddressTraitsT::index_type bytes_to_granules_t( std::size_t bytes )
282{
283 using IndexT = typename AddressTraitsT::index_type;
284 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
285 if ( bytes > std::numeric_limits<std::size_t>::max() - ( kGranSz - 1 ) )
286 return static_cast<IndexT>( 0 );
287 std::size_t granules = ( bytes + kGranSz - 1 ) / kGranSz;
288 if ( granules > static_cast<std::size_t>( std::numeric_limits<IndexT>::max() ) )
289 return static_cast<IndexT>( 0 );
290 return static_cast<IndexT>( granules );
291}

297 using IndexT = typename AddressTraitsT::index_type;

443template <typename AddressTraitsT>
444inline typename AddressTraitsT::index_type block_total_granules( const std::uint8_t* base,
445 const ManagerHeader<AddressTraitsT>* hdr,
446 const pmm::Block<AddressTraitsT>* blk )
447{
448 using BlockState = pmm::BlockStateBase<AddressTraitsT>;
449 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
450 using IndexT = typename AddressTraitsT::index_type;
451 static constexpr IndexT kNoBlk = AddressTraitsT::no_block;

453 std::size_t byte_off = reinterpret_cast<const std::uint8_t*>( blk ) - base;
454 IndexT this_idx = static_cast<IndexT>( byte_off / kGranSz );

457 if ( next_off != kNoBlk )
458 return static_cast<IndexT>( next_off - this_idx );
459 return static_cast<IndexT>( total_gran - this_idx );

465inline bool is_valid_block( const std::uint8_t* base, const ManagerHeader<pmm::DefaultAddressTraits>* hdr,
466 std::uint32_t idx )
467{
468 using BlockState = pmm::BlockStateBase<pmm::DefaultAddressTraits>;
469 if ( idx == kNoBlock )
470 return false;
471 if ( idx_to_byte_off_t<pmm::DefaultAddressTraits>( idx ) + sizeof( pmm::Block<pmm::DefaultAddressTraits> ) >
472 hdr->total_size )
473 return false;
474
475 const void* blk = base + idx_to_byte_off_t<pmm::DefaultAddressTraits>( idx );
476 auto next_off = BlockState::get_next_offset( blk );
477 std::uint32_t total_gran =
478 ( next_off != kNoBlock )
479 ? ( next_off - idx )
480 : ( byte_off_to_idx_t<pmm::DefaultAddressTraits>( static_cast<std::size_t>( hdr->total_size ) ) - idx );
481 if ( BlockState::get_weight( blk ) >= total_gran )
482 return false;
483 auto prev_off = BlockState::get_prev_offset( blk );
484 if ( prev_off != kNoBlock && prev_off >= idx )
485 return false;
486 if ( next_off != kNoBlock && next_off <= idx )
487 return false;
488 if ( BlockState::get_avl_height( blk ) >= 32 )
489 return false;
490 auto left_off = BlockState::get_left_offset( blk );
491 auto right_off = BlockState::get_right_offset( blk );
492 auto parent_off = BlockState::get_parent_offset( blk );
493 const bool l = ( left_off != kNoBlock );
494 const bool r = ( right_off != kNoBlock );
495 const bool p = ( parent_off != kNoBlock );
496 if ( ( l || r || p ) && ( ( l && r && left_off == right_off ) || ( l && p && left_off == parent_off ) ||
497 ( r && p && right_off == parent_off ) ) )
498 return false;
499 return true;
500}

534template <typename AddressTraitsT>
535inline pmm::Block<AddressTraitsT>* header_from_ptr_t( std::uint8_t* base, void* ptr, std::size_t total_size )
536{
537 using BlockState = pmm::BlockStateBase<AddressTraitsT>;
538 static constexpr std::size_t kGranSz = AddressTraitsT::granule_size;
539 static constexpr std::size_t kBlockSize = sizeof( pmm::Block<AddressTraitsT> );
540
541 if ( ptr == nullptr )
542 return nullptr;
543 std::uint8_t* raw_ptr = reinterpret_cast<std::uint8_t*>( ptr );
544 // First user data starts after Block_0 + ManagerHeader<AddressTraitsT> + Block_1
545 std::uint8_t* min_addr = base + kBlockSize + sizeof( ManagerHeader<AddressTraitsT> ) + kBlockSize;
546 if ( raw_ptr < min_addr )
547 return nullptr;
548 if ( raw_ptr > base + total_size )
549 return nullptr;
550 std::uint8_t* cand_addr = raw_ptr - kBlockSize;
551 if ( ( reinterpret_cast<std::size_t>( cand_addr ) - reinterpret_cast<std::size_t>( base ) ) % kGranSz != 0 )
552 return nullptr;
553 // Validate via BlockState (uses AddressTraitsT::no_block for sentinel checks)
554 if ( BlockState::get_weight( cand_addr ) == 0 )
555 return nullptr;
556 // Basic sanity: candidate address is within bounds
557 if ( cand_addr < base || cand_addr + kBlockSize > base + total_size )
558 return nullptr;
559 return reinterpret_cast<pmm::Block<AddressTraitsT>*>( cand_addr );
560}
