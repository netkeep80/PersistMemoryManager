141 bool create( std::size_t initial_size ) noexcept
142 {
143 typename ThreadPolicyT::unique_lock_type lock( _mutex );
144 if ( initial_size < detail::kMinMemorySize )
145 return false;
146 // Align to granule_size
147 std::size_t aligned = ( ( initial_size + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;

166 bool load() noexcept
167 {
168 typename ThreadPolicyT::unique_lock_type lock( _mutex );
169 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
170 return false;
171 std::uint8_t* base = _backend.base_ptr();
172 detail::ManagerHeader* hdr = get_header( base );
173 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )
174 return false;
175 if ( hdr->granule_size != static_cast<std::uint16_t>( kGranuleSize ) )
176 return false;
177 hdr->owns_memory = hdr->prev_owns_memory = false;
178 hdr->prev_total_size = 0;
179 hdr->prev_base_ptr = nullptr;
180 allocator::repair_linked_list( base, hdr );
181 allocator::recompute_counters( base, hdr );
182 allocator::rebuild_free_tree( base, hdr );
183 _initialized = true;
184 return true;
185 }

209 void* allocate( std::size_t user_size ) noexcept
210 {
211 typename ThreadPolicyT::unique_lock_type lock( _mutex );
212 if ( !_initialized || user_size == 0 )
213 return nullptr;
214
215 std::uint8_t* base = _backend.base_ptr();
216 detail::ManagerHeader* hdr = get_header( base );
217 std::uint32_t needed = detail::required_block_granules( user_size );
218 std::uint32_t idx = FreeBlockTreeT::find_best_fit( base, hdr, needed );
219
220 if ( idx != detail::kNoBlock )
221 return allocator::allocate_from_block( base, hdr, idx, user_size );
222
223 // Попытка расширить (если бэкенд поддерживает)
224 if ( !do_expand( user_size ) )
225 return nullptr;
226
227 base = _backend.base_ptr();
228 hdr = get_header( base );
229 idx = FreeBlockTreeT::find_best_fit( base, hdr, needed );
230 if ( idx != detail::kNoBlock )
231 return allocator::allocate_from_block( base, hdr, idx, user_size );
232 return nullptr;
233 }

238 void deallocate( void* ptr ) noexcept
239 {
240 typename ThreadPolicyT::unique_lock_type lock( _mutex );
241 if ( !_initialized || ptr == nullptr )
242 return;
243 std::uint8_t* base = _backend.base_ptr();
244 detail::ManagerHeader* hdr = get_header( base );
245 pmm::Block<AddressTraitsT>* blk =
246 detail::header_from_ptr( base, ptr, static_cast<std::size_t>( hdr->total_size ) );
247 if ( blk == nullptr || blk->weight == 0 )
248 return;
249
250 std::uint32_t freed = blk->weight;
251 std::uint32_t blk_idx = detail::block_idx( base, blk );
252
253 // State: AllocatedBlock → FreeBlockNotInAVL [mark_as_free]
254 AllocatedBlock<AddressTraitsT>* alloc = AllocatedBlock<AddressTraitsT>::cast_from_raw( blk );
255 alloc->mark_as_free();
256
257 hdr->alloc_count--;
258 hdr->free_count++;
259 if ( hdr->used_size >= freed )
260 hdr->used_size -= freed;
261 allocator::coalesce( base, hdr, blk_idx );
262 }

410 bool init_layout( std::uint8_t* base, std::size_t size ) noexcept
411 {
412 static constexpr std::uint32_t kHdrBlkIdx = 0;
413 static constexpr std::uint32_t kFreeBlkIdx = detail::kBlockHeaderGranules + detail::kManagerHeaderGranules;
414
415 if ( detail::idx_to_byte_off( kFreeBlkIdx ) + sizeof( Block<AddressTraitsT> ) + detail::kMinBlockSize > size )
416 return false;
417
418 // BlockHeader_0: allocated block containing ManagerHeader (Block<A> layout)
419 Block<AddressTraitsT>* hdr_blk = reinterpret_cast<Block<AddressTraitsT>*>( base );
420 std::memset( hdr_blk, 0, sizeof( Block<AddressTraitsT> ) );
421 hdr_blk->weight = detail::kManagerHeaderGranules; // data size = kManagerHeaderGranules
422 hdr_blk->prev_offset = detail::kNoBlock;
423 hdr_blk->next_offset = kFreeBlkIdx;
424 hdr_blk->left_offset = detail::kNoBlock;
425 hdr_blk->right_offset = detail::kNoBlock;
426 hdr_blk->parent_offset = detail::kNoBlock;
427 hdr_blk->avl_height = 0;
428 hdr_blk->root_offset = kHdrBlkIdx; // allocated: root_offset == own_idx
429
430 detail::ManagerHeader* hdr = get_header( base );
431 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
432 hdr->magic = kMagic;
433 hdr->total_size = size;
434 hdr->first_block_offset = kHdrBlkIdx;
435 hdr->last_block_offset = detail::kNoBlock;
436 hdr->free_tree_root = detail::kNoBlock;
437 hdr->granule_size = static_cast<std::uint16_t>( kGranuleSize );
438
439 // Initial free block (Block<A> layout)
440 Block<AddressTraitsT>* blk =
441 reinterpret_cast<Block<AddressTraitsT>*>( base + detail::idx_to_byte_off( kFreeBlkIdx ) );
442 std::memset( blk, 0, sizeof( Block<AddressTraitsT> ) );
443 blk->prev_offset = kHdrBlkIdx;
444 blk->next_offset = detail::kNoBlock;
445 blk->left_offset = detail::kNoBlock;
446 blk->right_offset = detail::kNoBlock;
447 blk->parent_offset = detail::kNoBlock;
448 blk->avl_height = 1; // ready for AVL insertion
449 blk->weight = 0; // free block
450 blk->root_offset = 0; // free block: root_offset == 0
451
452 hdr->last_block_offset = kFreeBlkIdx;
453 hdr->free_tree_root = kFreeBlkIdx;
454 hdr->block_count = 2;
455 hdr->free_count = 1;
456 hdr->alloc_count = 1;
457 hdr->used_size = kFreeBlkIdx + detail::kBlockHeaderGranules;
458
459 _initialized = true;
460 return true;
461 }

463 bool do_expand( std::size_t user_size ) noexcept
464 {
465 if ( !_initialized )
466 return false;
467 std::uint8_t* base = _backend.base_ptr();
468 detail::ManagerHeader* hdr = get_header( base );
469 std::size_t old_size = hdr->total_size;
470
471 // Вычисляем минимально необходимый прирост
472 std::size_t min_need =
473 detail::granules_to_bytes( detail::required_block_granules( user_size ) + detail::kBlockHeaderGranules );
474 std::size_t growth = old_size / 4; // 25% рост
475 if ( growth < min_need )
476 growth = min_need;
477
478 if ( !_backend.expand( growth ) )
479 return false;
480
481 std::uint8_t* new_base = _backend.base_ptr();
482 std::size_t new_size = _backend.total_size();
483 if ( new_base == nullptr || new_size <= old_size )
484 return false;
485
486 // Обновляем total_size в заголовке и добавляем новый свободный блок
487 hdr = get_header( new_base );
488
489 std::uint32_t extra_idx = detail::byte_off_to_idx( old_size );
490 std::size_t extra_size = new_size - old_size;
491
492 Block<AddressTraitsT>* last_blk = ( hdr->last_block_offset != detail::kNoBlock )
493 ? reinterpret_cast<Block<AddressTraitsT>*>(
494 new_base + detail::idx_to_byte_off( hdr->last_block_offset ) )
495 : nullptr;
496
497 if ( last_blk != nullptr && last_blk->weight == 0 ) // free block: weight == 0
498 {
499 // Расширяем последний свободный блок
500 std::uint32_t loff = detail::block_idx( new_base, last_blk );
501 FreeBlockTreeT::remove( new_base, hdr, loff );
502 hdr->total_size = new_size;
503 FreeBlockTreeT::insert( new_base, hdr, loff );
504 }
505 else
506 {
507 if ( extra_size < sizeof( Block<AddressTraitsT> ) + detail::kMinBlockSize )
508 return false;
509 Block<AddressTraitsT>* nb_blk =
510 reinterpret_cast<Block<AddressTraitsT>*>( new_base + detail::idx_to_byte_off( extra_idx ) );
511 std::memset( nb_blk, 0, sizeof( Block<AddressTraitsT> ) );
512 nb_blk->left_offset = detail::kNoBlock;
513 nb_blk->right_offset = detail::kNoBlock;
514 nb_blk->parent_offset = detail::kNoBlock;
515 nb_blk->avl_height = 1;
516 nb_blk->weight = 0; // free block
517 nb_blk->root_offset = 0; // free block
518 if ( last_blk != nullptr )
519 {
520 std::uint32_t loff = detail::block_idx( new_base, last_blk );
521 nb_blk->prev_offset = loff;
522 nb_blk->next_offset = detail::kNoBlock;
523 last_blk->next_offset = extra_idx;
524 }
525 else
526 {
527 nb_blk->prev_offset = detail::kNoBlock;
528 nb_blk->next_offset = detail::kNoBlock;
529 hdr->first_block_offset = extra_idx;
530 }
531 hdr->last_block_offset = extra_idx;
532 hdr->block_count++;
533 hdr->free_count++;
534 hdr->total_size = new_size;
535 FreeBlockTreeT::insert( new_base, hdr, extra_idx );
536 }
537 return true;
538 }
