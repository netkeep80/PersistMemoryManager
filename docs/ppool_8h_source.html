179 // --- Allocation / Deallocation ----------------------------------------------
180
189 T* allocate() noexcept
190 {
191 // If free-list is empty, allocate a new chunk.
192 if ( _free_head_idx == static_cast<index_type>( 0 ) )
193 {
194 if ( !allocate_chunk() )
195 return nullptr;
196 }
197
198 // Pop from free-list.
199 std::uint8_t* base = ManagerT::backend().base_ptr();
200 std::uint8_t* slot_raw = base + static_cast<std::size_t>( _free_head_idx ) * granule_size;
201
202 // Read the next free index from the slot.
203 index_type next_free;
204 std::memcpy( &next_free, slot_raw, sizeof( index_type ) );
205
206 _free_head_idx = next_free;
207 ++_total_allocated;
208
209 // Zero the slot before returning.
210 std::memset( slot_raw, 0, slot_bytes );
211
212 return reinterpret_cast<T*>( slot_raw );
213 }

223 void deallocate( T* ptr ) noexcept
224 {
225 if ( ptr == nullptr )
226 return;
227
228 std::uint8_t* base = ManagerT::backend().base_ptr();
229 std::uint8_t* slot_raw = reinterpret_cast<std::uint8_t*>( ptr );
230
231 // Compute the granule index of this slot.
232 std::size_t byte_off = static_cast<std::size_t>( slot_raw - base );
233 index_type slot_idx = static_cast<index_type>( byte_off / granule_size );
234
235 // Push onto free-list: store current free_head in the slot.
236 std::memcpy( slot_raw, &_free_head_idx, sizeof( index_type ) );
237
238 _free_head_idx = slot_idx;
239 --_total_allocated;
240 }

248 void free_all() noexcept
249 {
250 std::uint8_t* base = ManagerT::backend().base_ptr();
251
252 // Walk the chunk list and deallocate each chunk.
253 index_type chunk_idx = _chunk_head_idx;
254 while ( chunk_idx != static_cast<index_type>( 0 ) )
255 {
256 std::uint8_t* chunk_raw = base + static_cast<std::size_t>( chunk_idx ) * granule_size;
257
258 // Read the next chunk index from the chunk header.
259 index_type next_chunk;
260 std::memcpy( &next_chunk, chunk_raw, sizeof( index_type ) );
261
262 // Deallocate this chunk.
263 ManagerT::deallocate( chunk_raw );
264
265 chunk_idx = next_chunk;
266 }
267
268 _free_head_idx = static_cast<index_type>( 0 );
269 _chunk_head_idx = static_cast<index_type>( 0 );
270 _total_allocated = 0;
271 _total_capacity = 0;
272 }

275 // --- Internal helpers -------------------------------------------------------
276
287 bool allocate_chunk() noexcept
288 {
289 std::size_t n_objects = static_cast<std::size_t>( _objects_per_chunk );
290
291 // Chunk = 1 header granule + n_objects * granules_per_slot granules.
292 std::size_t total_granules = 1 + n_objects * granules_per_slot;
293 std::size_t alloc_size = total_granules * granule_size;
294
295 // Overflow check.
296 if ( n_objects > 0 && ( alloc_size / granule_size - 1 ) / granules_per_slot != n_objects )
297 return false;
298
299 void* raw = ManagerT::allocate( alloc_size );
300 if ( raw == nullptr )
301 return false;
302
303 std::uint8_t* chunk_raw = static_cast<std::uint8_t*>( raw );
304 std::uint8_t* base = ManagerT::backend().base_ptr();
305
306 // Compute chunk granule index.
307 std::size_t chunk_byte_off = static_cast<std::size_t>( chunk_raw - base );
308 index_type chunk_idx = static_cast<index_type>( chunk_byte_off / granule_size );
309
310 // Write chunk header: link to previous chunk head.
311 // Zero the full header granule first, then write the index.
312 std::memset( chunk_raw, 0, granule_size );
313 std::memcpy( chunk_raw, &_chunk_head_idx, sizeof( index_type ) );
314 _chunk_head_idx = chunk_idx;
315
316 // Initialize all slots and link them into the free-list.
317 // Slots start 1 granule after the chunk start.
318 std::uint8_t* slots_start = chunk_raw + granule_size;
319
320 for ( std::size_t i = 0; i < n_objects; ++i )
321 {
322 std::uint8_t* slot = slots_start + i * slot_bytes;
323 std::size_t slot_byte_off = static_cast<std::size_t>( slot - base );
324 index_type slot_idx = static_cast<index_type>( slot_byte_off / granule_size );
325
326 // Store current free_head in the slot (push onto free-list).
327 std::memset( slot, 0, slot_bytes );
328 std::memcpy( slot, &_free_head_idx, sizeof( index_type ) );
329 _free_head_idx = slot_idx;
330 }
331
332 _total_capacity += _objects_per_chunk;
333 return true;
334 }
