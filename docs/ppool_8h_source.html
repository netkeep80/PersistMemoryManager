192 T* allocate() noexcept
193 {
194 // If free-list is empty, allocate a new chunk.
195 if ( _free_head_idx == detail::kNullIdx_v<typename ManagerT::address_traits> )
196 {
197 if ( !allocate_chunk() )
198 return nullptr;
199 }
200
201 // Pop from free-list.
202 std::uint8_t* slot_raw =
203 reinterpret_cast<std::uint8_t*>( detail::resolve_granule_ptr<typename ManagerT::address_traits>(
204 ManagerT::backend().base_ptr(), _free_head_idx ) );
205
206 // Read the next free index from the slot.
207 index_type next_free;
208 std::memcpy( &next_free, slot_raw, sizeof( index_type ) );
209
210 _free_head_idx = next_free;
211 ++_total_allocated;
212
213 // Zero the slot before returning.
214 std::memset( slot_raw, 0, slot_bytes );
215
216 return reinterpret_cast<T*>( slot_raw );
217 }

227 void deallocate( T* ptr ) noexcept
228 {
229 if ( ptr == nullptr )
230 return;
231
232 std::uint8_t* slot_raw = reinterpret_cast<std::uint8_t*>( ptr );
233
234 // Compute the granule index of this slot.
235 index_type slot_idx =
236 detail::ptr_to_granule_idx<typename ManagerT::address_traits>( ManagerT::backend().base_ptr(), slot_raw );
237
238 // Push onto free-list: store current free_head in the slot.
239 std::memcpy( slot_raw, &_free_head_idx, sizeof( index_type ) );

241 _free_head_idx = slot_idx;

251 void free_all() noexcept
252 {
253 // Walk the chunk list and deallocate each chunk.
254 std::uint8_t* base = ManagerT::backend().base_ptr();
255 index_type chunk_idx = _chunk_head_idx;
256 while ( chunk_idx != detail::kNullIdx_v<typename ManagerT::address_traits> )
257 {
258 std::uint8_t* chunk_raw = reinterpret_cast<std::uint8_t*>(
259 detail::resolve_granule_ptr<typename ManagerT::address_traits>( base, chunk_idx ) );
260
261 // Read the next chunk index from the chunk header.
262 index_type next_chunk;
263 std::memcpy( &next_chunk, chunk_raw, sizeof( index_type ) );
264
265 // Deallocate this chunk.
266 ManagerT::deallocate( chunk_raw );
267
268 chunk_idx = next_chunk;
269 }

273 _total_allocated = 0;

290 bool allocate_chunk() noexcept
291 {
292 std::size_t n_objects = static_cast<std::size_t>( _objects_per_chunk );
293
294 // Chunk = 1 header granule + n_objects * granules_per_slot granules.
295 std::size_t total_granules = 1 + n_objects * granules_per_slot;
296 std::size_t alloc_size = total_granules * granule_size;
297
298 // Overflow check.
299 if ( n_objects > 0 && ( alloc_size / granule_size - 1 ) / granules_per_slot != n_objects )
300 return false;
301
302 void* raw = ManagerT::allocate( alloc_size );
303 if ( raw == nullptr )
304 return false;
305
306 std::uint8_t* chunk_raw = static_cast<std::uint8_t*>( raw );
307 std::uint8_t* base = ManagerT::backend().base_ptr();
308
309 // Compute chunk granule index.
310 index_type chunk_idx = detail::ptr_to_granule_idx<typename ManagerT::address_traits>( base, chunk_raw );
311
312 // Write chunk header: link to previous chunk head.
313 // Zero the full header granule first, then write the index.
314 std::memset( chunk_raw, 0, granule_size );
315 std::memcpy( chunk_raw, &_chunk_head_idx, sizeof( index_type ) );
316 _chunk_head_idx = chunk_idx;
317
318 // Initialize all slots and link them into the free-list.
319 // Slots start 1 granule after the chunk start.
320 std::uint8_t* slots_start = chunk_raw + granule_size;
321
322 for ( std::size_t i = 0; i < n_objects; ++i )
323 {
324 std::uint8_t* slot = slots_start + i * slot_bytes;
325 index_type slot_idx = detail::ptr_to_granule_idx<typename ManagerT::address_traits>( base, slot );
326
327 // Store current free_head in the slot (push onto free-list).
328 std::memset( slot, 0, slot_bytes );
329 std::memcpy( slot, &_free_head_idx, sizeof( index_type ) );
330 _free_head_idx = slot_idx;
331 }
332
333 _total_capacity += _objects_per_chunk;
334 return true;
335 }
