120 bool expand( std::size_t additional_bytes ) noexcept
121 {
122 if ( additional_bytes == 0 )
123 return _size > 0;
124 // Grow by 25% or by additional_bytes, whichever is larger.
125 // Issue #235: for initial allocation from zero, use a minimum of 4096 bytes
126 // (or additional_bytes if larger) to avoid inefficient tiny allocations.
127 static constexpr std::size_t kMinInitialSize = 4096;
128 std::size_t growth =
129 ( _size > 0 ) ? ( _size / 4 + additional_bytes ) : std::max( additional_bytes, kMinInitialSize );
130 std::size_t new_size = _size + growth;
131 // Align to granule_size
132 new_size = ( ( new_size + AddressTraitsT::granule_size - 1 ) / AddressTraitsT::granule_size ) *
133 AddressTraitsT::granule_size;
134 if ( new_size <= _size )
135 return false;

137 void* new_buf = std::malloc( new_size );
138 if ( new_buf == nullptr )
139 return false;
140 if ( _buffer != nullptr )
141 std::memcpy( new_buf, _buffer, _size );
142 if ( _owns_memory && _buffer != nullptr )
143 std::free( _buffer );
144 _buffer = static_cast<std::uint8_t*>( new_buf );
145 _size = new_size;
146 _owns_memory = true;
