46template <typename AT> inline bool validate_block_index( std::size_t total_size, typename AT::index_type idx ) noexcept
47{
48 if ( idx == AT::no_block )
49 return false;
50 std::size_t byte_off = static_cast<std::size_t>( idx ) * AT::granule_size;
51 // Overflow check: if idx * granule_size wrapped around, byte_off < idx for large idx.
52 if ( idx != 0 && byte_off / AT::granule_size != static_cast<std::size_t>( idx ) )
53 return false;
54 if ( byte_off + sizeof( pmm::Block<AT> ) > total_size )
55 return false;
56 return true;
57}

74template <typename AT>
75inline bool validate_user_ptr( const std::uint8_t* base, std::size_t total_size, const void* ptr,
76 std::size_t min_user_offset ) noexcept
77{
78 if ( ptr == nullptr || base == nullptr )
79 return false;
80 const auto* raw_ptr = static_cast<const std::uint8_t*>( ptr );
81 // Must be within managed area.
82 if ( raw_ptr < base || raw_ptr >= base + total_size )
83 return false;
84 // Must be past minimum user-data address.
85 if ( static_cast<std::size_t>( raw_ptr - base ) < min_user_offset )
86 return false;
87 // The block header candidate must be granule-aligned relative to base.
88 static constexpr std::size_t kBlockSize = sizeof( pmm::Block<AT> );
89 std::size_t cand_off = static_cast<std::size_t>( raw_ptr - base ) - kBlockSize;
90 if ( cand_off % AT::granule_size != 0 )
91 return false;
92 return true;
93}

132template <typename AT>
133inline void validate_block_header_full( const std::uint8_t* base, std::size_t total_size, typename AT::index_type idx,
134 VerifyResult& result ) noexcept
135{
136 using BlockState = pmm::BlockStateBase<AT>;
137 using index_type = typename AT::index_type;
138
139 if ( !validate_block_index<AT>( total_size, idx ) )
140 {
141 result.add( ViolationType::BlockStateInconsistent, DiagnosticAction::NoAction,
142 static_cast<std::uint64_t>( idx ), 0, 0 );
143 return;
144 }
145
146 const void* blk_raw = base + static_cast<std::size_t>( idx ) * AT::granule_size;
147
148 // 1. Weight / root_offset consistency (delegates to BlockState::verify_state).
149 BlockState::verify_state( blk_raw, idx, result );
150
151 // 2. next_offset within bounds.
152 index_type next = BlockState::get_next_offset( blk_raw );
153 if ( next != AT::no_block && !validate_block_index<AT>( total_size, next ) )
154 {
155 result.add( ViolationType::BlockStateInconsistent, DiagnosticAction::NoAction,
156 static_cast<std::uint64_t>( idx ), static_cast<std::uint64_t>( AT::no_block ),
157 static_cast<std::uint64_t>( next ) );
158 }
159
160 // 3. prev_offset within bounds.
161 index_type prev = BlockState::get_prev_offset( blk_raw );
162 if ( prev != AT::no_block && !validate_block_index<AT>( total_size, prev ) )
163 {
164 result.add( ViolationType::BlockStateInconsistent, DiagnosticAction::NoAction,
165 static_cast<std::uint64_t>( idx ), static_cast<std::uint64_t>( AT::no_block ),
166 static_cast<std::uint64_t>( prev ) );
167 }
168
169 // 4. node_type is a known value.
170 std::uint16_t nt = BlockState::get_node_type( blk_raw );
171 if ( nt != pmm::kNodeReadWrite && nt != pmm::kNodeReadOnly )
172 {
173 result.add( ViolationType::BlockStateInconsistent, DiagnosticAction::NoAction,
174 static_cast<std::uint64_t>( idx ), 0, static_cast<std::uint64_t>( nt ) );
175 }
176
177 // 5. If allocated, verify block data range fits in image.
178 index_type w = BlockState::get_weight( blk_raw );
179 if ( w > 0 )
180 {
181 static constexpr std::size_t kBlkHdrBytes = sizeof( pmm::Block<AT> );
182 std::size_t blk_byte_off = static_cast<std::size_t>( idx ) * AT::granule_size;
183 std::size_t data_bytes = static_cast<std::size_t>( w ) * AT::granule_size;
184 if ( blk_byte_off + kBlkHdrBytes + data_bytes > total_size )
185 {
186 result.add( ViolationType::BlockStateInconsistent, DiagnosticAction::NoAction,
187 static_cast<std::uint64_t>( idx ), total_size,
188 static_cast<std::uint64_t>( blk_byte_off + kBlkHdrBytes + data_bytes ) );
189 }
190 }
