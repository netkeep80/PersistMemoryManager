126 static bool create( std::size_t initial_size ) noexcept
127 {
128 typename thread_policy::unique_lock_type lock( _mutex );
129 if ( initial_size < detail::kMinMemorySize )
130 return false;
131 std::size_t aligned = ( ( initial_size + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
132 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
133 {
134 // Либо буфера нет, либо он меньше требуемого — расширяем
135 std::size_t additional =
136 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
137 if ( !_backend.expand( additional ) )
138 return false;
139 }
140 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
141 return false;
142 return init_layout( _backend.base_ptr(), _backend.total_size() );
143 }

163 static bool load() noexcept
164 {
165 typename thread_policy::unique_lock_type lock( _mutex );
166 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
167 return false;
168 std::uint8_t* base = _backend.base_ptr();
169 detail::ManagerHeader* hdr = get_header( base );
170 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )
171 return false;
172 if ( hdr->granule_size != static_cast<std::uint16_t>( kGranuleSize ) )
173 return false;
174 hdr->owns_memory = hdr->prev_owns_memory = false;
175 hdr->prev_total_size = 0;
176 hdr->prev_base_ptr = nullptr;
177 allocator::repair_linked_list( base, hdr );
178 allocator::recompute_counters( base, hdr );
179 allocator::rebuild_free_tree( base, hdr );
180 _initialized = true;
181 return true;
182 }

211 static void* allocate( std::size_t user_size ) noexcept
212 {
213 typename thread_policy::unique_lock_type lock( _mutex );
214 if ( !_initialized || user_size == 0 )
215 return nullptr;
216
217 std::uint8_t* base = _backend.base_ptr();
218 detail::ManagerHeader* hdr = get_header( base );
219 std::uint32_t needed = detail::required_block_granules( user_size );
220 std::uint32_t idx = free_block_tree::find_best_fit( base, hdr, needed );
221
222 if ( idx != detail::kNoBlock )
223 return allocator::allocate_from_block( base, hdr, idx, user_size );
224
225 // Попытка расширить (если бэкенд поддерживает)
226 if ( !do_expand( user_size ) )
227 return nullptr;
228
229 base = _backend.base_ptr();
230 hdr = get_header( base );
231 idx = free_block_tree::find_best_fit( base, hdr, needed );
232 if ( idx != detail::kNoBlock )
233 return allocator::allocate_from_block( base, hdr, idx, user_size );
234 return nullptr;

242 static void deallocate( void* ptr ) noexcept
243 {
244 typename thread_policy::unique_lock_type lock( _mutex );
245 if ( !_initialized || ptr == nullptr )
246 return;
247 std::uint8_t* base = _backend.base_ptr();
248 detail::ManagerHeader* hdr = get_header( base );
249 pmm::Block<address_traits>* blk =
250 detail::header_from_ptr( base, ptr, static_cast<std::size_t>( hdr->total_size ) );
251 if ( blk == nullptr )
252 return;
253 std::uint32_t freed = BlockStateBase<address_traits>::get_weight( blk );
254 if ( freed == 0 )
255 return;
256
257 // Issue #126: Permanently locked blocks cannot be freed.
258 if ( BlockStateBase<address_traits>::get_node_type( blk ) == pmm::kNodeReadOnly )
259 return;
260
261 std::uint32_t blk_idx = detail::block_idx( base, blk );
262
263 AllocatedBlock<address_traits>* alloc = AllocatedBlock<address_traits>::cast_from_raw( blk );
264 alloc->mark_as_free();
265
266 hdr->alloc_count--;
267 hdr->free_count++;
268 if ( hdr->used_size >= freed )
269 hdr->used_size -= freed;
270 allocator::coalesce( base, hdr, blk_idx );
271 }

654 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
655 {
656 typename thread_policy::shared_lock_type lock( _mutex );
657 if ( !_initialized )
658 return false;
659 const std::uint8_t* base = _backend.base_ptr();
660 using BlockState = BlockStateBase<address_traits>;
661 const detail::ManagerHeader* hdr = get_header_c( base );
662 std::uint32_t idx = hdr->first_block_offset;
663 while ( idx != detail::kNoBlock )
664 {
665 if ( detail::idx_to_byte_off( idx ) + sizeof( Block<address_traits> ) > hdr->total_size )
666 break;
667 const void* blk_raw = base + detail::idx_to_byte_off( idx );
668 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
669 std::uint32_t total_gran = detail::block_total_granules( base, hdr, blk );
670 auto w = BlockState::get_weight( blk_raw );
671 bool is_used = ( w > 0 );
672 std::size_t hdr_bytes = sizeof( Block<address_traits> );
673 std::size_t data_bytes = is_used ? detail::granules_to_bytes( w ) : 0;
674
675 BlockView view;
676 view.index = idx;
677 view.offset = static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( idx ) );
678 view.total_size = detail::granules_to_bytes( total_gran );
679 view.header_size = hdr_bytes;
680 view.user_size = data_bytes;
681 view.alignment = kGranuleSize;
682 view.used = is_used;
683 callback( view );
684 idx = BlockState::get_next_offset( blk_raw );
685 }
686 return true;
687 }

732 template <typename Callback>
733 static void for_each_free_block_inorder( const std::uint8_t* base, const detail::ManagerHeader* hdr,
734 std::uint32_t node_idx, int depth, Callback&& callback ) noexcept
735 {
736 using BlockState = BlockStateBase<address_traits>;
737 if ( node_idx == detail::kNoBlock )
738 return;
739 if ( detail::idx_to_byte_off( node_idx ) + sizeof( Block<address_traits> ) > hdr->total_size )
740 return;
741 const void* blk_raw = base + detail::idx_to_byte_off( node_idx );
742 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
743
744 auto left_off = BlockState::get_left_offset( blk_raw );
745 auto right_off = BlockState::get_right_offset( blk_raw );
746 auto parent_off = BlockState::get_parent_offset( blk_raw );
747
748 // Visit left subtree first (smaller blocks)
749 for_each_free_block_inorder( base, hdr, left_off, depth + 1, callback );
750
751 // Visit current node
752 std::uint32_t total_gran = detail::block_total_granules( base, hdr, blk );
753 FreeBlockView view;
754 view.offset = static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( node_idx ) );
755 view.total_size = detail::granules_to_bytes( total_gran );
756 view.free_size = detail::granules_to_bytes( total_gran - detail::kBlockHeaderGranules );
757 view.left_offset =
758 ( left_off != detail::kNoBlock ) ? static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( left_off ) ) : -1;
759 view.right_offset = ( right_off != detail::kNoBlock )
760 ? static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( right_off ) )
761 : -1;
762 view.parent_offset = ( parent_off != detail::kNoBlock )
763 ? static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( parent_off ) )
764 : -1;
765 view.avl_height = BlockState::get_avl_height( blk_raw );
766 view.avl_depth = depth;
767 callback( view );
768
769 // Visit right subtree (larger blocks)
770 for_each_free_block_inorder( base, hdr, right_off, depth + 1, callback );
771 }

783 static bool init_layout( std::uint8_t* base, std::size_t size ) noexcept
784 {
785 using BlockState = BlockStateBase<address_traits>;
786 static constexpr std::uint32_t kHdrBlkIdx = 0;
787 static constexpr std::uint32_t kFreeBlkIdx = detail::kBlockHeaderGranules + detail::kManagerHeaderGranules;
788
789 if ( detail::idx_to_byte_off( kFreeBlkIdx ) + sizeof( Block<address_traits> ) + detail::kMinBlockSize > size )
790 return false;
791
792 // Инициализация блока-заголовка (Block_0) через state machine утилиты
793 void* hdr_blk = base;
794 std::memset( hdr_blk, 0, sizeof( Block<address_traits> ) );
795 BlockState::init_fields( hdr_blk,
796 /*prev*/ detail::kNoBlock,
797 /*next*/ kFreeBlkIdx,
798 /*avl_height*/ 0,
799 /*weight*/ detail::kManagerHeaderGranules,
800 /*root_offset*/ kHdrBlkIdx );
801
802 detail::ManagerHeader* hdr = get_header( base );
803 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
804 hdr->magic = kMagic;
805 hdr->total_size = size;
806 hdr->first_block_offset = kHdrBlkIdx;
807 hdr->last_block_offset = detail::kNoBlock;
808 hdr->free_tree_root = detail::kNoBlock;
809 hdr->granule_size = static_cast<std::uint16_t>( kGranuleSize );
810
811 // Инициализация первого свободного блока через state machine утилиты
812 void* blk = base + detail::idx_to_byte_off( kFreeBlkIdx );
813 std::memset( blk, 0, sizeof( Block<address_traits> ) );
814 BlockState::init_fields( blk,
815 /*prev*/ kHdrBlkIdx,
816 /*next*/ detail::kNoBlock,
817 /*avl_height*/ 1,
818 /*weight*/ 0,
819 /*root_offset*/ 0 );
820
821 hdr->last_block_offset = kFreeBlkIdx;
822 hdr->free_tree_root = kFreeBlkIdx;
823 hdr->block_count = 2;
824 hdr->free_count = 1;
825 hdr->alloc_count = 1;
826 hdr->used_size = kFreeBlkIdx + detail::kBlockHeaderGranules;
827
828 _initialized = true;
829 return true;
830 }

832 static bool do_expand( std::size_t user_size ) noexcept
833 {
834 using BlockState = BlockStateBase<address_traits>;
835 if ( !_initialized )
836 return false;
837 std::uint8_t* base = _backend.base_ptr();
838 detail::ManagerHeader* hdr = get_header( base );
839 std::size_t old_size = hdr->total_size;
840
841 std::size_t min_need =
842 detail::granules_to_bytes( detail::required_block_granules( user_size ) + detail::kBlockHeaderGranules );
843 std::size_t growth = old_size / 4;
844 if ( growth < min_need )
845 growth = min_need;
846
847 if ( !_backend.expand( growth ) )
848 return false;
849
850 std::uint8_t* new_base = _backend.base_ptr();
851 std::size_t new_size = _backend.total_size();
852 if ( new_base == nullptr || new_size <= old_size )
853 return false;
854
855 hdr = get_header( new_base );
856
857 std::uint32_t extra_idx = detail::byte_off_to_idx( old_size );
858 std::size_t extra_size = new_size - old_size;
859
860 void* last_blk_raw = ( hdr->last_block_offset != detail::kNoBlock )
861 ? static_cast<void*>( new_base + detail::idx_to_byte_off( hdr->last_block_offset ) )
862 : nullptr;
863
864 if ( last_blk_raw != nullptr && BlockState::get_weight( last_blk_raw ) == 0 )
865 {
866 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
867 std::uint32_t loff = detail::block_idx( new_base, last_blk );
868 free_block_tree::remove( new_base, hdr, loff );
869 hdr->total_size = new_size;
870 free_block_tree::insert( new_base, hdr, loff );
871 }
872 else
873 {
874 if ( extra_size < sizeof( Block<address_traits> ) + detail::kMinBlockSize )
875 return false;
876 void* nb_blk = new_base + detail::idx_to_byte_off( extra_idx );
877 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
878 if ( last_blk_raw != nullptr )
879 {
880 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
881 std::uint32_t loff = detail::block_idx( new_base, last_blk );
882 BlockState::init_fields( nb_blk,
883 /*prev*/ loff,
884 /*next*/ detail::kNoBlock,
885 /*avl_height*/ 1,
886 /*weight*/ 0,
887 /*root_offset*/ 0 );
888 BlockState::set_next_offset_of( last_blk_raw, extra_idx );
889 }
890 else
891 {
892 BlockState::init_fields( nb_blk,
893 /*prev*/ detail::kNoBlock,
894 /*next*/ detail::kNoBlock,
895 /*avl_height*/ 1,
896 /*weight*/ 0,
897 /*root_offset*/ 0 );
898 hdr->first_block_offset = extra_idx;
899 }
900 hdr->last_block_offset = extra_idx;
901 hdr->block_count++;
902 hdr->free_count++;
903 hdr->total_size = new_size;
904 free_block_tree::insert( new_base, hdr, extra_idx );
905 }
906 return true;
907 }
908};
