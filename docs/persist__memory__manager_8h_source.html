231 {

239 // Issue #172: guard against overflow when initial_size is close to size_t max.
240 static constexpr std::size_t kGranSzCreate = address_traits::granule_size;
241 if ( initial_size > std::numeric_limits<std::size_t>::max() - ( kGranSzCreate - 1 ) )
242 {
243 _last_error = PmmError::Overflow;

369 {
370 typename thread_policy::unique_lock_type lock( _mutex );
371 if ( !_initialized )
372 {
373 _last_error = PmmError::NotInitialized;
374 logging_policy::on_allocation_failure( user_size, PmmError::NotInitialized );
375 return nullptr;
376 }
377 if ( user_size == 0 )
378 {
379 _last_error = PmmError::InvalidSize;
380 logging_policy::on_allocation_failure( user_size, PmmError::InvalidSize );
381 return nullptr;
382 }
383
384 std::uint8_t* base = _backend.base_ptr();
385 detail::ManagerHeader<address_traits>* hdr = get_header( base );
386 // Issue #146: use AddressTraits-specific granule size for required granule computation.
387 index_type data_gran = detail::bytes_to_granules_t<address_traits>( user_size );
388 if ( data_gran == 0 )
389 data_gran = 1;
390 // Issue #43 Phase 1.3: Overflow protection — check before adding header granules.
391 if ( data_gran > std::numeric_limits<index_type>::max() - kBlockHdrGranules )
392 {
393 _last_error = PmmError::Overflow;
394 logging_policy::on_allocation_failure( user_size, PmmError::Overflow );
395 return nullptr;
396 }
397 index_type needed = kBlockHdrGranules + data_gran;
398 index_type idx = free_block_tree::find_best_fit( base, hdr, needed );
399
400 if ( idx != address_traits::no_block )
401 {
402 _last_error = PmmError::Ok;
403 return allocator::allocate_from_block( base, hdr, idx, user_size );
404 }
405
406 // Попытка расширить (если бэкенд поддерживает)
407 if ( !do_expand( user_size ) )
408 {
409 _last_error = PmmError::OutOfMemory;
410 logging_policy::on_allocation_failure( user_size, PmmError::OutOfMemory );
411 return nullptr;
412 }
413
414 base = _backend.base_ptr();
415 hdr = get_header( base );
416 idx = free_block_tree::find_best_fit( base, hdr, needed );
417 if ( idx != address_traits::no_block )
418 {
419 _last_error = PmmError::Ok;
420 return allocator::allocate_from_block( base, hdr, idx, user_size );
421 }
422 _last_error = PmmError::OutOfMemory;

432 static void deallocate( void* ptr ) noexcept
433 {
434 typename thread_policy::unique_lock_type lock( _mutex );
435 if ( !_initialized || ptr == nullptr )
436 return;
437 pmm::Block<address_traits>* blk = find_block_from_user_ptr( ptr );
438 if ( blk == nullptr )
439 return;
440 index_type freed = BlockStateBase<address_traits>::get_weight( blk );
441 if ( freed == 0 )
442 return;
443
444 // Issue #126: Permanently locked blocks cannot be freed.
445 if ( BlockStateBase<address_traits>::get_node_type( blk ) == pmm::kNodeReadOnly )
446 return;
447
448 std::uint8_t* base = _backend.base_ptr();
449 detail::ManagerHeader<address_traits>* hdr = get_header( base );
450 index_type blk_idx = detail::block_idx_t<address_traits>( base, blk );
451
452 AllocatedBlock<address_traits>* alloc = AllocatedBlock<address_traits>::cast_from_raw( blk );
453 alloc->mark_as_free();
454
455 hdr->alloc_count--;
456 hdr->free_count++;
457 if ( hdr->used_size >= freed )
458 hdr->used_size -= freed;
459 allocator::coalesce( base, hdr, blk_idx );
460 }

562 template <typename T>
563 static pptr<T> reallocate_typed( pptr<T> p, std::size_t old_count, std::size_t new_count ) noexcept
564 {
565 static_assert( std::is_trivially_copyable_v<T>,
566 "reallocate_typed<T>: T must be trivially copyable for safe memcpy reallocation." );
567 if ( new_count == 0 )
568 {
569 _last_error = PmmError::InvalidSize;
570 return pptr<T>();
571 }
572 if ( p.is_null() )
573 return allocate_typed<T>( new_count );
574 if ( sizeof( T ) > 0 && new_count > ( std::numeric_limits<std::size_t>::max )() / sizeof( T ) )
575 {
576 _last_error = PmmError::Overflow;
577 return pptr<T>();
578 }
579 std::size_t new_user_size = sizeof( T ) * new_count;
580 typename thread_policy::unique_lock_type lock( _mutex );
581 if ( !_initialized )
582 {
583 _last_error = PmmError::NotInitialized;
584 return pptr<T>();
585 }
586 std::uint8_t* base = _backend.base_ptr();
587 detail::ManagerHeader<address_traits>* hdr = get_header( base );
588 // blk_idx = pptr.offset - floor(sizeof(Block) / granule)
589 static constexpr index_type kBlkHdrFloorGran =
590 static_cast<index_type>( sizeof( Block<address_traits> ) / address_traits::granule_size );
591 index_type blk_idx = static_cast<index_type>( p.offset() - kBlkHdrFloorGran );
592 void* blk_raw = detail::block_at<address_traits>( base, blk_idx );
593 index_type old_data_gran = BlockStateBase<address_traits>::get_weight( blk_raw );
594 index_type new_data_gran = detail::bytes_to_granules_t<address_traits>( new_user_size );
595 if ( new_data_gran == 0 )
596 new_data_gran = 1;
597 if ( new_data_gran == old_data_gran )
598 {
599 _last_error = PmmError::Ok;
600 return p;
601 }
602 // Skip in-place paths when resolve() overlaps block header (SmallAT).
603 static constexpr bool kBlockAligned = ( sizeof( Block<address_traits> ) % address_traits::granule_size == 0 );
604
605 if constexpr ( kBlockAligned )
606 {
607 if ( new_data_gran < old_data_gran )
608 {
609 allocator::realloc_shrink( base, hdr, blk_idx, blk_raw, old_data_gran, new_data_gran );
610 _last_error = PmmError::Ok;
611 return p;
612 }
613 if ( new_data_gran > old_data_gran )
614 {
615 if ( allocator::realloc_grow( base, hdr, blk_idx, blk_raw, old_data_gran, new_data_gran ) )
616 {
617 _last_error = PmmError::Ok;
618 return p;
619 }
620 }
621 }
622 // Fallback: allocate new + memmove + free old (under same lock).
623 static constexpr index_type kBlkHdrFloorGranFb =
624 static_cast<index_type>( sizeof( Block<address_traits> ) / address_traits::granule_size );
625 index_type new_data_gran_alloc = detail::bytes_to_granules_t<address_traits>( new_user_size );
626 if ( new_data_gran_alloc == 0 )
627 new_data_gran_alloc = 1;
628 if ( new_data_gran_alloc > std::numeric_limits<index_type>::max() - kBlockHdrGranules )
629 {
630 _last_error = PmmError::Overflow;
631 return pptr<T>();
632 }
633 index_type needed = kBlockHdrGranules + new_data_gran_alloc;
634 index_type new_idx = free_block_tree::find_best_fit( base, hdr, needed );
635 if ( new_idx == address_traits::no_block )
636 {
637 if ( !do_expand( new_user_size ) )
638 {
639 _last_error = PmmError::OutOfMemory;
640 logging_policy::on_allocation_failure( new_user_size, PmmError::OutOfMemory );
641 return pptr<T>();
642 }
643 base = _backend.base_ptr();
644 hdr = get_header( base );
645 new_idx = free_block_tree::find_best_fit( base, hdr, needed );
646 if ( new_idx == address_traits::no_block )
647 {
648 _last_error = PmmError::OutOfMemory;
649 logging_policy::on_allocation_failure( new_user_size, PmmError::OutOfMemory );
650 return pptr<T>();
651 }
652 }
653 void* new_raw = allocator::allocate_from_block( base, hdr, new_idx, new_user_size );
654 if ( new_raw == nullptr )
655 {
656 _last_error = PmmError::OutOfMemory;
657 return pptr<T>();
658 }
659 pptr<T> new_p = make_pptr_from_raw<T>( new_raw );
660 void* new_dst = base + static_cast<std::size_t>( new_p.offset() ) * address_traits::granule_size;
661 void* old_src = base + static_cast<std::size_t>( p.offset() ) * address_traits::granule_size;
662 std::size_t copy_sz = ( new_count < old_count ? new_count : old_count ) * sizeof( T );
663 std::memmove( new_dst, old_src, copy_sz );
664 // Free old block
665 index_type old_blk_idx = static_cast<index_type>( p.offset() - kBlkHdrFloorGranFb );
666 void* old_blk_raw = detail::block_at<address_traits>( base, old_blk_idx );
667 index_type freed_w = BlockStateBase<address_traits>::get_weight( old_blk_raw );
668 if ( BlockStateBase<address_traits>::get_node_type( old_blk_raw ) != pmm::kNodeReadOnly )
669 {
670 auto* old_alloc = AllocatedBlock<address_traits>::cast_from_raw( old_blk_raw );
671 if ( old_alloc != nullptr )
672 {
673 old_alloc->mark_as_free();
674 hdr->alloc_count--;
675 hdr->free_count++;
676 if ( hdr->used_size >= freed_w )
677 hdr->used_size -= freed_w;
678 allocator::coalesce( base, hdr, old_blk_idx );
679 }
680 }
681 _last_error = PmmError::Ok;
682 return new_p;
683 }

1165 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
1166 view.header_size = hdr_bytes;
1167 view.user_size = data_bytes;

1281 template <typename Callback>
1282 static void for_each_free_block_inorder( const std::uint8_t* base, const detail::ManagerHeader<address_traits>* hdr,
1283 index_type node_idx, int depth, Callback&& callback ) noexcept
1284 {
1285 using BlockState = BlockStateBase<address_traits>;
1286 // Issue #146: use address_traits::granule_size for correct byte offset computations.
1287 static constexpr std::size_t kGranSz = address_traits::granule_size;
1288 if ( node_idx == address_traits::no_block )
1289 return;
1290 if ( static_cast<std::size_t>( node_idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
1291 return;
1292 const void* blk_raw = base + static_cast<std::size_t>( node_idx ) * kGranSz;
1293 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
1294
1295 index_type left_off = BlockState::get_left_offset( blk_raw );
1296 index_type right_off = BlockState::get_right_offset( blk_raw );
1297 index_type parent_off = BlockState::get_parent_offset( blk_raw );
1298
1299 // Visit left subtree first (smaller blocks)
1300 for_each_free_block_inorder( base, hdr, left_off, depth + 1, callback );
1301
1302 // Visit current node
1303 index_type total_gran = detail::block_total_granules( base, hdr, blk );
1304 FreeBlockView view;
1305 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( node_idx ) * kGranSz );
1306 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
1307 view.free_size = static_cast<std::size_t>( total_gran - kBlockHdrGranules ) * kGranSz;
1308 view.left_offset = ( left_off != address_traits::no_block )
1309 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( left_off ) * kGranSz )
1310 : -1;
1311 view.right_offset = ( right_off != address_traits::no_block )
1312 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( right_off ) * kGranSz )
1313 : -1;
1314 view.parent_offset = ( parent_off != address_traits::no_block )
1315 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( parent_off ) * kGranSz )
1316 : -1;
1317 view.avl_height = BlockState::get_avl_height( blk_raw );
1318 view.avl_depth = depth;
1319 callback( view );
1320
1321 // Visit right subtree (larger blocks)
1322 for_each_free_block_inorder( base, hdr, right_off, depth + 1, callback );
1323 }

1411 static bool do_expand( std::size_t user_size ) noexcept
1412 {
1413 using BlockState = BlockStateBase<address_traits>;
1414 if ( !_initialized )
1415 return false;
1416 std::uint8_t* base = _backend.base_ptr();
1417 detail::ManagerHeader<address_traits>* hdr = get_header( base );
1418 std::size_t old_size = hdr->total_size;
1419
1420 // Issue #146: use AddressTraitsT-specific granule size for all computations.
1421 static constexpr std::size_t kGranSz = address_traits::granule_size;
1422 index_type data_gran_need = detail::bytes_to_granules_t<address_traits>( user_size );
1423 if ( data_gran_need == 0 )
1424 data_gran_need = 1;
1425 // min_need = block_header + data + another block_header (for the new expand block)
1426 std::size_t min_need =
1427 static_cast<std::size_t>( kBlockHdrGranules + data_gran_need + kBlockHdrGranules ) * kGranSz;
1428 std::size_t growth = old_size / 4;
1429 if ( growth < min_need )
1430 growth = min_need;
1431
1432 if ( !_backend.expand( growth ) )
1433 return false;
1434
1435 std::uint8_t* new_base = _backend.base_ptr();
1436 std::size_t new_size = _backend.total_size();
1437 if ( new_base == nullptr || new_size <= old_size )
1438 return false;
1439
1440 logging_policy::on_expand( old_size, new_size );
1441 hdr = get_header( new_base );
1442
1443 // Issue #146: compute extra_idx using address_traits::granule_size.
1444 index_type extra_idx = detail::byte_off_to_idx_t<address_traits>( old_size );
1445 std::size_t extra_size = new_size - old_size;
1446
1447 void* last_blk_raw =
1448 ( hdr->last_block_offset != address_traits::no_block )
1449 ? static_cast<void*>( new_base + static_cast<std::size_t>( hdr->last_block_offset ) * kGranSz )
1450 : nullptr;
1451
1452 if ( last_blk_raw != nullptr && BlockState::get_weight( last_blk_raw ) == 0 )
1453 {
1454 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1455 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1456 free_block_tree::remove( new_base, hdr, loff );
1457 hdr->total_size = new_size;
1458 free_block_tree::insert( new_base, hdr, loff );
1459 }
1460 else
1461 {
1462 // Issue #146: minimum extra size check uses address_traits granule size.
1463 if ( extra_size < sizeof( Block<address_traits> ) + kGranSz )
1464 return false;
1465 void* nb_blk = new_base + static_cast<std::size_t>( extra_idx ) * kGranSz;
1466 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
1467 if ( last_blk_raw != nullptr )
1468 {
1469 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1470 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1471 BlockState::init_fields( nb_blk,
1472 /*prev*/ loff,
1473 /*next*/ address_traits::no_block,
1474 /*avl_height*/ 1,
1475 /*weight*/ 0,
1476 /*root_offset*/ 0 );
1477 BlockState::set_next_offset_of( last_blk_raw, static_cast<index_type>( extra_idx ) );
1478 }
1479 else
1480 {
1481 BlockState::init_fields( nb_blk,
1482 /*prev*/ address_traits::no_block,
1483 /*next*/ address_traits::no_block,
1484 /*avl_height*/ 1,
1485 /*weight*/ 0,
1486 /*root_offset*/ 0 );
1487 hdr->first_block_offset = extra_idx;
1488 }
1489 hdr->last_block_offset = extra_idx;
1490 hdr->block_count++;
1491 hdr->free_count++;
1492 hdr->total_size = new_size;
1493 free_block_tree::insert( new_base, hdr, extra_idx );
1494 }
1495 return true;
