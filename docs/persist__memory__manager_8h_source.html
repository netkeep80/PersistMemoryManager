185 static bool create( std::size_t initial_size ) noexcept
186 {
187 typename thread_policy::unique_lock_type lock( _mutex );
188 if ( initial_size < detail::kMinMemorySize )
189 return false;
190 // Issue #146: use address_traits::granule_size instead of hardcoded kGranuleSize.
191 // Issue #172: guard against overflow when initial_size is close to size_t max.
192 static constexpr std::size_t kGranSzCreate = address_traits::granule_size;
193 if ( initial_size > std::numeric_limits<std::size_t>::max() - ( kGranSzCreate - 1 ) )
194 return false; // overflow in alignment computation
195 std::size_t aligned = ( ( initial_size + kGranSzCreate - 1 ) / kGranSzCreate ) * kGranSzCreate;
196 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
197 {
198 // Либо буфера нет, либо он меньше требуемого — расширяем
199 std::size_t additional =
200 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
201 if ( !_backend.expand( additional ) )
202 return false;
203 }
204 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
205 return false;
206 return init_layout( _backend.base_ptr(), _backend.total_size() );
207 }

227 static bool load() noexcept
228 {
229 typename thread_policy::unique_lock_type lock( _mutex );
230 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
231 return false;
232 std::uint8_t* base = _backend.base_ptr();
233 detail::ManagerHeader<address_traits>* hdr = get_header( base );
234 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )
235 return false;
236 // Issue #146: compare stored granule size against address_traits::granule_size.
237 if ( hdr->granule_size != static_cast<std::uint16_t>( address_traits::granule_size ) )
238 return false;
239 hdr->owns_memory = false;
240 hdr->prev_total_size = 0;
241 allocator::repair_linked_list( base, hdr );
242 allocator::recompute_counters( base, hdr );
243 allocator::rebuild_free_tree( base, hdr );
244 _initialized = true;
245 return true;
246 }

276 static void* allocate( std::size_t user_size ) noexcept
277 {
278 typename thread_policy::unique_lock_type lock( _mutex );
279 if ( !_initialized || user_size == 0 )
280 return nullptr;
281
282 std::uint8_t* base = _backend.base_ptr();
283 detail::ManagerHeader<address_traits>* hdr = get_header( base );
284 // Issue #146: use AddressTraits-specific granule size for required granule computation.
285 index_type data_gran = detail::bytes_to_granules_t<address_traits>( user_size );
286 if ( data_gran == 0 )
287 data_gran = 1;
288 // Issue #43 Phase 1.3: Overflow protection — check before adding header granules.
289 if ( data_gran > std::numeric_limits<index_type>::max() - kBlockHdrGranules )
290 return nullptr;
291 index_type needed = kBlockHdrGranules + data_gran;
292 index_type idx = free_block_tree::find_best_fit( base, hdr, needed );
293
294 if ( idx != address_traits::no_block )
295 return allocator::allocate_from_block( base, hdr, idx, user_size );
296
297 // Попытка расширить (если бэкенд поддерживает)
298 if ( !do_expand( user_size ) )
299 return nullptr;
300
301 base = _backend.base_ptr();
302 hdr = get_header( base );
303 idx = free_block_tree::find_best_fit( base, hdr, needed );
304 if ( idx != address_traits::no_block )
305 return allocator::allocate_from_block( base, hdr, idx, user_size );
306 return nullptr;
307 }

314 static void deallocate( void* ptr ) noexcept
315 {
316 typename thread_policy::unique_lock_type lock( _mutex );
317 if ( !_initialized || ptr == nullptr )
318 return;
319 pmm::Block<address_traits>* blk = find_block_from_user_ptr( ptr );
320 if ( blk == nullptr )
321 return;
322 index_type freed = BlockStateBase<address_traits>::get_weight( blk );
323 if ( freed == 0 )
324 return;
325
326 // Issue #126: Permanently locked blocks cannot be freed.
327 if ( BlockStateBase<address_traits>::get_node_type( blk ) == pmm::kNodeReadOnly )
328 return;
329
330 std::uint8_t* base = _backend.base_ptr();
331 detail::ManagerHeader<address_traits>* hdr = get_header( base );
332 index_type blk_idx = detail::block_idx_t<address_traits>( base, blk );
333
334 AllocatedBlock<address_traits>* alloc = AllocatedBlock<address_traits>::cast_from_raw( blk );
335 alloc->mark_as_free();
336
337 hdr->alloc_count--;
338 hdr->free_count++;
339 if ( hdr->used_size >= freed )
340 hdr->used_size -= freed;
341 allocator::coalesce( base, hdr, blk_idx );
342 }

850 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
851 {
852 typename thread_policy::shared_lock_type lock( _mutex );
853 if ( !_initialized )
854 return false;
855 const std::uint8_t* base = _backend.base_ptr();
856 using BlockState = BlockStateBase<address_traits>;
857 const detail::ManagerHeader<address_traits>* hdr = get_header_c( base );
858 index_type idx = hdr->first_block_offset;
859 // Issue #146: use address_traits::granule_size for correct byte offset computations.
860 static constexpr std::size_t kGranSz = address_traits::granule_size;
861 while ( idx != address_traits::no_block )
862 {
863 if ( static_cast<std::size_t>( idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
864 break;
865 const void* blk_raw = base + static_cast<std::size_t>( idx ) * kGranSz;
866 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
867 index_type total_gran = detail::block_total_granules( base, hdr, blk );
868 auto w = BlockState::get_weight( blk_raw );
869 bool is_used = ( w > 0 );
870 std::size_t hdr_bytes = sizeof( Block<address_traits> );
871 std::size_t data_bytes = is_used ? static_cast<std::size_t>( w ) * kGranSz : 0;
872
873 BlockView view;
874 view.index = idx;
875 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( idx ) * kGranSz );
876 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
877 view.header_size = hdr_bytes;
878 view.user_size = data_bytes;
879 view.alignment = kGranSz;
880 view.used = is_used;
881 callback( view );
882 idx = BlockState::get_next_offset( blk_raw );
883 }
884 return true;
885 }

989 template <typename Callback>
990 static void for_each_free_block_inorder( const std::uint8_t* base, const detail::ManagerHeader<address_traits>* hdr,
991 index_type node_idx, int depth, Callback&& callback ) noexcept
992 {
993 using BlockState = BlockStateBase<address_traits>;
994 // Issue #146: use address_traits::granule_size for correct byte offset computations.
995 static constexpr std::size_t kGranSz = address_traits::granule_size;
996 if ( node_idx == address_traits::no_block )
997 return;
998 if ( static_cast<std::size_t>( node_idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
999 return;
1000 const void* blk_raw = base + static_cast<std::size_t>( node_idx ) * kGranSz;
1001 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
1002
1003 index_type left_off = BlockState::get_left_offset( blk_raw );
1004 index_type right_off = BlockState::get_right_offset( blk_raw );
1005 index_type parent_off = BlockState::get_parent_offset( blk_raw );
1006
1007 // Visit left subtree first (smaller blocks)
1008 for_each_free_block_inorder( base, hdr, left_off, depth + 1, callback );
1009
1010 // Visit current node
1011 index_type total_gran = detail::block_total_granules( base, hdr, blk );
1012 FreeBlockView view;
1013 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( node_idx ) * kGranSz );
1014 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
1015 view.free_size = static_cast<std::size_t>( total_gran - kBlockHdrGranules ) * kGranSz;
1016 view.left_offset = ( left_off != address_traits::no_block )
1017 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( left_off ) * kGranSz )
1018 : -1;
1019 view.right_offset = ( right_off != address_traits::no_block )
1020 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( right_off ) * kGranSz )
1021 : -1;
1022 view.parent_offset = ( parent_off != address_traits::no_block )
1023 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( parent_off ) * kGranSz )
1024 : -1;
1025 view.avl_height = BlockState::get_avl_height( blk_raw );
1026 view.avl_depth = depth;
1027 callback( view );
1028
1029 // Visit right subtree (larger blocks)
1030 for_each_free_block_inorder( base, hdr, right_off, depth + 1, callback );
1031 }

1033 // ─── Address-traits-specific layout constants (Issue #146) ──────────────────
1034 // These compute the correct granule indices based on the actual address_traits
1035 // granule size, rather than using the hardcoded DefaultAddressTraits constants.
1036
1039 static constexpr std::size_t kBlockHdrByteSize =
1040 ( ( sizeof( Block<address_traits> ) + address_traits::granule_size - 1 ) / address_traits::granule_size ) *
1041 address_traits::granule_size;
1042
1044 static constexpr index_type kBlockHdrGranules =
1045 static_cast<index_type>( kBlockHdrByteSize / address_traits::granule_size );
1046
1049 static constexpr index_type kMgrHdrGranules = detail::kManagerHeaderGranules_t<address_traits>;
1050
1052 static constexpr index_type kFreeBlkIdxLayout = kBlockHdrGranules + kMgrHdrGranules;
1053
1054 static detail::ManagerHeader<address_traits>* get_header( std::uint8_t* base ) noexcept
1055 {
1056 // Place ManagerHeader at a granule-aligned offset after Block_0 (Issue #146).
1057 return reinterpret_cast<detail::ManagerHeader<address_traits>*>( base + kBlockHdrByteSize );
1058 }
1059
1060 static const detail::ManagerHeader<address_traits>* get_header_c( const std::uint8_t* base ) noexcept
1061 {
1062 return reinterpret_cast<const detail::ManagerHeader<address_traits>*>( base + kBlockHdrByteSize );
1063 }
1064
1065 static bool init_layout( std::uint8_t* base, std::size_t size ) noexcept
1066 {
1067 using BlockState = BlockStateBase<address_traits>;
1068 static constexpr index_type kHdrBlkIdx = 0;
1069 static constexpr index_type kFreeBlkIdx = kFreeBlkIdxLayout;
1070 static constexpr std::size_t kGranSz = address_traits::granule_size;
1071
1072 // Minimum size check: Block_0 + ManagerHeader + Block_1 + at least 1 data granule
1073 static constexpr std::size_t kMinBlockDataSize = kGranSz; // 1 data granule
1074 if ( static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz + sizeof( Block<address_traits> ) + kMinBlockDataSize >
1075 size )
1076 return false;
1077
1078 // Инициализация блока-заголовка (Block_0) через state machine утилиты
1079 void* hdr_blk = base;
1080 std::memset( hdr_blk, 0, kBlockHdrByteSize ); // zero entire aligned region (including padding)
1081 BlockState::init_fields( hdr_blk,
1082 /*prev*/ address_traits::no_block,
1083 /*next*/ kFreeBlkIdx,
1084 /*avl_height*/ 0,
1085 /*weight*/ kMgrHdrGranules,
1086 /*root_offset*/ kHdrBlkIdx );
1087
1088 detail::ManagerHeader<address_traits>* hdr = get_header( base );
1089 std::memset( hdr, 0, sizeof( detail::ManagerHeader<address_traits> ) );
1090 hdr->magic = kMagic;
1091 hdr->total_size = size;
1092 hdr->first_block_offset = kHdrBlkIdx;
1093 hdr->last_block_offset = address_traits::no_block;
1094 hdr->free_tree_root = address_traits::no_block;
1095 hdr->granule_size = static_cast<std::uint16_t>( kGranSz );
1096
1097 // Инициализация первого свободного блока через state machine утилиты
1098 void* blk = base + static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz;
1099 std::memset( blk, 0, sizeof( Block<address_traits> ) );
1100 BlockState::init_fields( blk,
1101 /*prev*/ kHdrBlkIdx,
1102 /*next*/ address_traits::no_block,
1103 /*avl_height*/ 1,
1104 /*weight*/ 0,
1105 /*root_offset*/ 0 );
1106
1107 hdr->last_block_offset = kFreeBlkIdx;
1108 hdr->free_tree_root = kFreeBlkIdx;
1109 hdr->block_count = 2;
1110 hdr->free_count = 1;
1111 hdr->alloc_count = 1;
1112 hdr->used_size = kFreeBlkIdx + kBlockHdrGranules;
1113
1114 _initialized = true;
1115 return true;
1116 }
1117
1118 static bool do_expand( std::size_t user_size ) noexcept
1119 {
1120 using BlockState = BlockStateBase<address_traits>;
1121 if ( !_initialized )
1122 return false;
1123 std::uint8_t* base = _backend.base_ptr();
1124 detail::ManagerHeader<address_traits>* hdr = get_header( base );
1125 std::size_t old_size = hdr->total_size;
1126
1127 // Issue #146: use AddressTraitsT-specific granule size for all computations.
1128 static constexpr std::size_t kGranSz = address_traits::granule_size;
1129 index_type data_gran_need = detail::bytes_to_granules_t<address_traits>( user_size );
1130 if ( data_gran_need == 0 )
1131 data_gran_need = 1;
1132 // min_need = block_header + data + another block_header (for the new expand block)
1133 std::size_t min_need =
1134 static_cast<std::size_t>( kBlockHdrGranules + data_gran_need + kBlockHdrGranules ) * kGranSz;
1135 std::size_t growth = old_size / 4;
1136 if ( growth < min_need )
1137 growth = min_need;
1138
1139 if ( !_backend.expand( growth ) )
1140 return false;
1141
1142 std::uint8_t* new_base = _backend.base_ptr();
1143 std::size_t new_size = _backend.total_size();
1144 if ( new_base == nullptr || new_size <= old_size )
1145 return false;
1146
1147 hdr = get_header( new_base );
1148
1149 // Issue #146: compute extra_idx using address_traits::granule_size.
1150 index_type extra_idx = detail::byte_off_to_idx_t<address_traits>( old_size );
1151 std::size_t extra_size = new_size - old_size;
1152
1153 void* last_blk_raw =
1154 ( hdr->last_block_offset != address_traits::no_block )
1155 ? static_cast<void*>( new_base + static_cast<std::size_t>( hdr->last_block_offset ) * kGranSz )
1156 : nullptr;
1157
1158 if ( last_blk_raw != nullptr && BlockState::get_weight( last_blk_raw ) == 0 )
1159 {
1160 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1161 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1162 free_block_tree::remove( new_base, hdr, loff );
1163 hdr->total_size = new_size;
1164 free_block_tree::insert( new_base, hdr, loff );
1165 }
1166 else
1167 {
1168 // Issue #146: minimum extra size check uses address_traits granule size.
1169 if ( extra_size < sizeof( Block<address_traits> ) + kGranSz )
1170 return false;
1171 void* nb_blk = new_base + static_cast<std::size_t>( extra_idx ) * kGranSz;
1172 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
1173 if ( last_blk_raw != nullptr )
1174 {
1175 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1176 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1177 BlockState::init_fields( nb_blk,
1178 /*prev*/ loff,
1179 /*next*/ address_traits::no_block,
1180 /*avl_height*/ 1,
1181 /*weight*/ 0,
1182 /*root_offset*/ 0 );
1183 BlockState::set_next_offset_of( last_blk_raw, static_cast<index_type>( extra_idx ) );
1184 }
1185 else
1186 {
1187 BlockState::init_fields( nb_blk,
1188 /*prev*/ address_traits::no_block,
1189 /*next*/ address_traits::no_block,
1190 /*avl_height*/ 1,
1191 /*weight*/ 0,
1192 /*root_offset*/ 0 );
1193 hdr->first_block_offset = extra_idx;
1194 }
1195 hdr->last_block_offset = extra_idx;
1196 hdr->block_count++;
1197 hdr->free_count++;
1198 hdr->total_size = new_size;
1199 free_block_tree::insert( new_base, hdr, extra_idx );
1200 }
1201 return true;
1202 }
