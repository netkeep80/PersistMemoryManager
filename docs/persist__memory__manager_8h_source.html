220 static bool create( std::size_t initial_size ) noexcept
221 {
222 typename thread_policy::unique_lock_type lock( _mutex );
223 if ( initial_size < detail::kMinMemorySize )
224 {
225 _last_error = PmmError::InvalidSize;
226 return false;

228 // Issue #146: use address_traits::granule_size instead of hardcoded kGranuleSize.
229 // Issue #172: guard against overflow when initial_size is close to size_t max.
230 static constexpr std::size_t kGranSzCreate = address_traits::granule_size;
231 if ( initial_size > std::numeric_limits<std::size_t>::max() - ( kGranSzCreate - 1 ) )
232 {
233 _last_error = PmmError::Overflow;
234 return false;
235 }
236 std::size_t aligned = ( ( initial_size + kGranSzCreate - 1 ) / kGranSzCreate ) * kGranSzCreate;
237 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
238 {
239 // Либо буфера нет, либо он меньше требуемого — расширяем
240 std::size_t additional =
241 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
242 if ( !_backend.expand( additional ) )
243 {
244 _last_error = PmmError::ExpandFailed;
245 return false;
246 }
247 }
248 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
249 {
250 _last_error = PmmError::BackendError;
251 return false;
252 }

255 ok = bootstrap_forest_registry_unlocked();

294 {
295 typename thread_policy::unique_lock_type lock( _mutex );
296 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
297 {
298 _last_error = ( _backend.base_ptr() == nullptr ) ? PmmError::BackendError : PmmError::InvalidSize;
299 return false;

302 detail::ManagerHeader<address_traits>* hdr = get_header( base );
303 if ( hdr->magic != kMagic )
304 {
305 _last_error = PmmError::InvalidMagic;
306 logging_policy::on_corruption_detected( PmmError::InvalidMagic );
307 return false;
308 }

310 {
311 _last_error = PmmError::SizeMismatch;
312 logging_policy::on_corruption_detected( PmmError::SizeMismatch );

316 if ( hdr->granule_size != static_cast<std::uint16_t>( address_traits::granule_size ) )
317 {
318 _last_error = PmmError::GranuleMismatch;
319 logging_policy::on_corruption_detected( PmmError::GranuleMismatch );
320 return false;
321 }
322 hdr->owns_memory = false;
323 hdr->prev_total_size = 0;
324 allocator::repair_linked_list( base, hdr );
325 allocator::recompute_counters( base, hdr );
326 allocator::rebuild_free_tree( base, hdr );
327 _initialized = true;
328 if ( !validate_or_bootstrap_forest_registry_unlocked() )
329 {
330 _initialized = false;
331 return false;
332 }
333 _last_error = PmmError::Ok;

476 static pptr<T> reallocate_typed( pptr<T> p, std::size_t old_count, std::size_t new_count ) noexcept
477 {
478 static_assert( std::is_trivially_copyable_v<T>,
479 "reallocate_typed<T>: T must be trivially copyable for safe memcpy reallocation." );
480 if ( new_count == 0 )
481 {
482 _last_error = PmmError::InvalidSize;
483 return pptr<T>();
484 }
485 if ( p.is_null() )
486 return allocate_typed<T>( new_count );
487 if ( sizeof( T ) > 0 && new_count > ( std::numeric_limits<std::size_t>::max )() / sizeof( T ) )
488 {
489 _last_error = PmmError::Overflow;
490 return pptr<T>();
491 }
492 std::size_t new_user_size = sizeof( T ) * new_count;
493 typename thread_policy::unique_lock_type lock( _mutex );
494 if ( !_initialized )
495 {
496 _last_error = PmmError::NotInitialized;
497 return pptr<T>();
498 }
499 std::uint8_t* base = _backend.base_ptr();
500 detail::ManagerHeader<address_traits>* hdr = get_header( base );
501 // blk_idx = pptr.offset - floor(sizeof(Block) / granule)
502 static constexpr index_type kBlkHdrFloorGran =
503 static_cast<index_type>( sizeof( Block<address_traits> ) / address_traits::granule_size );
504 index_type blk_idx = static_cast<index_type>( p.offset() - kBlkHdrFloorGran );
505 void* blk_raw = detail::block_at<address_traits>( base, blk_idx );
506 index_type old_data_gran = BlockStateBase<address_traits>::get_weight( blk_raw );
507 index_type new_data_gran = detail::bytes_to_granules_t<address_traits>( new_user_size );
508 if ( new_data_gran == 0 )
509 new_data_gran = 1;
510 if ( new_data_gran == old_data_gran )
511 {
512 _last_error = PmmError::Ok;
513 return p;
514 }
515 // Skip in-place paths when resolve() overlaps block header (SmallAT).
516 static constexpr bool kBlockAligned = ( sizeof( Block<address_traits> ) % address_traits::granule_size == 0 );
517
518 if constexpr ( kBlockAligned )
519 {
520 if ( new_data_gran < old_data_gran )
521 {
522 allocator::realloc_shrink( base, hdr, blk_idx, blk_raw, old_data_gran, new_data_gran );
523 _last_error = PmmError::Ok;
524 return p;
525 }
526 if ( new_data_gran > old_data_gran )
527 {
528 if ( allocator::realloc_grow( base, hdr, blk_idx, blk_raw, old_data_gran, new_data_gran ) )
529 {
530 _last_error = PmmError::Ok;
531 return p;
532 }
533 }
534 }
535 // Fallback: allocate new + memmove + free old (under same lock).
536 static constexpr index_type kBlkHdrFloorGranFb =
537 static_cast<index_type>( sizeof( Block<address_traits> ) / address_traits::granule_size );
538 index_type new_data_gran_alloc = detail::bytes_to_granules_t<address_traits>( new_user_size );
539 if ( new_data_gran_alloc == 0 )
540 new_data_gran_alloc = 1;
541 if ( new_data_gran_alloc > std::numeric_limits<index_type>::max() - kBlockHdrGranules )
542 {
543 _last_error = PmmError::Overflow;
544 return pptr<T>();
545 }
546 index_type needed = kBlockHdrGranules + new_data_gran_alloc;
547 index_type new_idx = free_block_tree::find_best_fit( base, hdr, needed );
548 if ( new_idx == address_traits::no_block )
549 {
550 if ( !do_expand( new_user_size ) )
551 {
552 _last_error = PmmError::OutOfMemory;
553 logging_policy::on_allocation_failure( new_user_size, PmmError::OutOfMemory );
554 return pptr<T>();
555 }
556 base = _backend.base_ptr();
557 hdr = get_header( base );
558 new_idx = free_block_tree::find_best_fit( base, hdr, needed );
559 if ( new_idx == address_traits::no_block )
560 {
561 _last_error = PmmError::OutOfMemory;
562 logging_policy::on_allocation_failure( new_user_size, PmmError::OutOfMemory );
563 return pptr<T>();
564 }
565 }
566 void* new_raw = allocator::allocate_from_block( base, hdr, new_idx, new_user_size );
567 if ( new_raw == nullptr )
568 {
569 _last_error = PmmError::OutOfMemory;
570 return pptr<T>();
571 }
572 pptr<T> new_p = make_pptr_from_raw<T>( new_raw );
573 void* new_dst = base + static_cast<std::size_t>( new_p.offset() ) * address_traits::granule_size;
574 void* old_src = base + static_cast<std::size_t>( p.offset() ) * address_traits::granule_size;
575 std::size_t copy_sz = ( new_count < old_count ? new_count : old_count ) * sizeof( T );
576 std::memmove( new_dst, old_src, copy_sz );
577 // Free old block
578 index_type old_blk_idx = static_cast<index_type>( p.offset() - kBlkHdrFloorGranFb );
579 void* old_blk_raw = detail::block_at<address_traits>( base, old_blk_idx );
580 index_type freed_w = BlockStateBase<address_traits>::get_weight( old_blk_raw );
581 if ( BlockStateBase<address_traits>::get_node_type( old_blk_raw ) != pmm::kNodeReadOnly )
582 {
583 auto* old_alloc = AllocatedBlock<address_traits>::cast_from_raw( old_blk_raw );
584 if ( old_alloc != nullptr )
585 {
586 old_alloc->mark_as_free();
587 hdr->alloc_count--;
588 hdr->free_count++;
589 if ( hdr->used_size >= freed_w )
590 hdr->used_size -= freed_w;
591 allocator::coalesce( base, hdr, old_blk_idx );
592 }
593 }
594 _last_error = PmmError::Ok;
595 return new_p;
596 }

1041 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
1042 {
1043 typename thread_policy::shared_lock_type lock( _mutex );
1044 if ( !_initialized )
1045 return false;
1046 const std::uint8_t* base = _backend.base_ptr();
1047 using BlockState = BlockStateBase<address_traits>;
1048 const detail::ManagerHeader<address_traits>* hdr = get_header_c( base );
1049 index_type idx = hdr->first_block_offset;
1050 // Issue #146: use address_traits::granule_size for correct byte offset computations.
1051 static constexpr std::size_t kGranSz = address_traits::granule_size;
1052 while ( idx != address_traits::no_block )
1053 {
1054 if ( static_cast<std::size_t>( idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
1055 break;
1056 const void* blk_raw = base + static_cast<std::size_t>( idx ) * kGranSz;
1057 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
1058 index_type total_gran = detail::block_total_granules( base, hdr, blk );
1059 auto w = BlockState::get_weight( blk_raw );
1060 bool is_used = ( w > 0 );
1061 std::size_t hdr_bytes = sizeof( Block<address_traits> );
1062 std::size_t data_bytes = is_used ? static_cast<std::size_t>( w ) * kGranSz : 0;
1063
1064 BlockView view;
1065 view.index = idx;
1066 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( idx ) * kGranSz );
1067 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
1068 view.header_size = hdr_bytes;
1069 view.user_size = data_bytes;
1070 view.alignment = kGranSz;
1071 view.used = is_used;
1072 callback( view );
1073 idx = BlockState::get_next_offset( blk_raw );
1074 }
1075 return true;
1076 }

1556 template <typename Callback>
1557 static void for_each_free_block_inorder( const std::uint8_t* base, const detail::ManagerHeader<address_traits>* hdr,
1558 index_type node_idx, int depth, Callback&& callback ) noexcept
1559 {
1560 using BlockState = BlockStateBase<address_traits>;
1561 // Issue #146: use address_traits::granule_size for correct byte offset computations.
1562 static constexpr std::size_t kGranSz = address_traits::granule_size;
1563 if ( node_idx == address_traits::no_block )
1564 return;
1565 if ( static_cast<std::size_t>( node_idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
1566 return;
1567 const void* blk_raw = base + static_cast<std::size_t>( node_idx ) * kGranSz;
1568 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
1569
1570 index_type left_off = BlockState::get_left_offset( blk_raw );
1571 index_type right_off = BlockState::get_right_offset( blk_raw );
1572 index_type parent_off = BlockState::get_parent_offset( blk_raw );
1573
1574 // Visit left subtree first (smaller blocks)
1575 for_each_free_block_inorder( base, hdr, left_off, depth + 1, callback );
1576
1577 // Visit current node
1578 index_type total_gran = detail::block_total_granules( base, hdr, blk );
1579 FreeBlockView view;
1580 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( node_idx ) * kGranSz );
1581 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
1582 view.free_size = static_cast<std::size_t>( total_gran - kBlockHdrGranules ) * kGranSz;
1583 view.left_offset = ( left_off != address_traits::no_block )
1584 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( left_off ) * kGranSz )
1585 : -1;
1586 view.right_offset = ( right_off != address_traits::no_block )
1587 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( right_off ) * kGranSz )
1588 : -1;
1589 view.parent_offset = ( parent_off != address_traits::no_block )
1590 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( parent_off ) * kGranSz )
1591 : -1;
1592 view.avl_height = BlockState::get_avl_height( blk_raw );
1593 view.avl_depth = depth;
1594 callback( view );
1595
1596 // Visit right subtree (larger blocks)
1597 for_each_free_block_inorder( base, hdr, right_off, depth + 1, callback );
1598 }

1686 static bool do_expand( std::size_t user_size ) noexcept
1687 {
1688 using BlockState = BlockStateBase<address_traits>;
1689 if ( !_initialized )
1690 return false;
1691 std::uint8_t* base = _backend.base_ptr();
1692 detail::ManagerHeader<address_traits>* hdr = get_header( base );
1693 std::size_t old_size = hdr->total_size;
1694
1695 // Issue #146: use AddressTraitsT-specific granule size for all computations.
1696 static constexpr std::size_t kGranSz = address_traits::granule_size;
1697 index_type data_gran_need = detail::bytes_to_granules_t<address_traits>( user_size );
1698 if ( data_gran_need == 0 )
1699 data_gran_need = 1;
1700 // min_need = block_header + data + another block_header (for the new expand block)
1701 std::size_t min_need =
1702 static_cast<std::size_t>( kBlockHdrGranules + data_gran_need + kBlockHdrGranules ) * kGranSz;
1703 std::size_t growth = old_size / 4;
1704 if ( growth < min_need )
1705 growth = min_need;
1706
1707 if ( !_backend.expand( growth ) )
1708 return false;
1709
1710 std::uint8_t* new_base = _backend.base_ptr();
1711 std::size_t new_size = _backend.total_size();
1712 if ( new_base == nullptr || new_size <= old_size )
1713 return false;
1714
1715 logging_policy::on_expand( old_size, new_size );
1716 hdr = get_header( new_base );
1717
1718 // Issue #146: compute extra_idx using address_traits::granule_size.
1719 index_type extra_idx = detail::byte_off_to_idx_t<address_traits>( old_size );
1720 std::size_t extra_size = new_size - old_size;
1721
1722 void* last_blk_raw =
1723 ( hdr->last_block_offset != address_traits::no_block )
1724 ? static_cast<void*>( new_base + static_cast<std::size_t>( hdr->last_block_offset ) * kGranSz )
1725 : nullptr;
1726
1727 if ( last_blk_raw != nullptr && BlockState::get_weight( last_blk_raw ) == 0 )
1728 {
1729 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1730 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1731 free_block_tree::remove( new_base, hdr, loff );
1732 hdr->total_size = new_size;
1733 free_block_tree::insert( new_base, hdr, loff );
1734 }
1735 else
1736 {
1737 // Issue #146: minimum extra size check uses address_traits granule size.
1738 if ( extra_size < sizeof( Block<address_traits> ) + kGranSz )
1739 return false;
1740 void* nb_blk = new_base + static_cast<std::size_t>( extra_idx ) * kGranSz;
1741 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
1742 if ( last_blk_raw != nullptr )
1743 {
1744 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1745 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1746 BlockState::init_fields( nb_blk,
1747 /*prev*/ loff,
1748 /*next*/ address_traits::no_block,
1749 /*avl_height*/ 1,
1750 /*weight*/ 0,
1751 /*root_offset*/ 0 );
1752 BlockState::set_next_offset_of( last_blk_raw, static_cast<index_type>( extra_idx ) );
1753 }
1754 else
1755 {
1756 BlockState::init_fields( nb_blk,
1757 /*prev*/ address_traits::no_block,
1758 /*next*/ address_traits::no_block,
1759 /*avl_height*/ 1,
1760 /*weight*/ 0,
1761 /*root_offset*/ 0 );
1762 hdr->first_block_offset = extra_idx;
1763 }
1764 hdr->last_block_offset = extra_idx;
1765 hdr->block_count++;
1766 hdr->free_count++;
1767 hdr->total_size = new_size;
1768 free_block_tree::insert( new_base, hdr, extra_idx );
1769 }
1770 return true;
1771 }
