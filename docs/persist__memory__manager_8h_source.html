220 static bool create( std::size_t initial_size ) noexcept
221 {
222 typename thread_policy::unique_lock_type lock( _mutex );
223 if ( initial_size < detail::kMinMemorySize )
224 {
225 _last_error = PmmError::InvalidSize;
226 return false;
227 }
228 // Issue #146: use address_traits::granule_size instead of hardcoded kGranuleSize.
229 // Issue #172: guard against overflow when initial_size is close to size_t max.
230 static constexpr std::size_t kGranSzCreate = address_traits::granule_size;
231 if ( initial_size > std::numeric_limits<std::size_t>::max() - ( kGranSzCreate - 1 ) )
232 {
233 _last_error = PmmError::Overflow;
234 return false;
235 }
236 std::size_t aligned = ( ( initial_size + kGranSzCreate - 1 ) / kGranSzCreate ) * kGranSzCreate;
237 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
238 {
239 // Либо буфера нет, либо он меньше требуемого — расширяем
240 std::size_t additional =
241 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
242 if ( !_backend.expand( additional ) )
243 {
244 _last_error = PmmError::ExpandFailed;
245 return false;
246 }
247 }
248 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
249 {
250 _last_error = PmmError::BackendError;
251 return false;
252 }
253 bool ok = init_layout( _backend.base_ptr(), _backend.total_size() );
254 if ( ok )
255 ok = bootstrap_forest_registry_unlocked();
256 if ( ok )
257 ok = validate_bootstrap_invariants_unlocked(); // Issue #241
258 if ( ok )

260 _last_error = PmmError::Ok;

297 static bool load() noexcept
298 {
299 typename thread_policy::unique_lock_type lock( _mutex );
300 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
301 {
302 _last_error = ( _backend.base_ptr() == nullptr ) ? PmmError::BackendError : PmmError::InvalidSize;
303 return false;
304 }
305 std::uint8_t* base = _backend.base_ptr();
306 detail::ManagerHeader<address_traits>* hdr = get_header( base );
307 if ( hdr->magic != kMagic )
308 {
309 _last_error = PmmError::InvalidMagic;
310 logging_policy::on_corruption_detected( PmmError::InvalidMagic );
311 return false;
312 }
313 if ( hdr->total_size != _backend.total_size() )
314 {
315 _last_error = PmmError::SizeMismatch;
316 logging_policy::on_corruption_detected( PmmError::SizeMismatch );
317 return false;
318 }
319 // Issue #146: compare stored granule size against address_traits::granule_size.
320 if ( hdr->granule_size != static_cast<std::uint16_t>( address_traits::granule_size ) )
321 {
322 _last_error = PmmError::GranuleMismatch;
323 logging_policy::on_corruption_detected( PmmError::GranuleMismatch );
324 return false;
325 }
326 hdr->owns_memory = false;
327 hdr->prev_total_size = 0;
328 allocator::repair_linked_list( base, hdr );
329 allocator::recompute_counters( base, hdr );
330 allocator::rebuild_free_tree( base, hdr );
331 _initialized = true;
332 if ( !validate_or_bootstrap_forest_registry_unlocked() )
333 {
334 _initialized = false;
335 return false;
336 }
337 if ( !validate_bootstrap_invariants_unlocked() ) // Issue #241
338 {
339 _initialized = false;
340 return false;
341 }
342 _last_error = PmmError::Ok;
343 logging_policy::on_load();
344 return true;
345 }

484 template <typename T>
485 static pptr<T> reallocate_typed( pptr<T> p, std::size_t old_count, std::size_t new_count ) noexcept
486 {
487 static_assert( std::is_trivially_copyable_v<T>,
488 "reallocate_typed<T>: T must be trivially copyable for safe memcpy reallocation." );
489 if ( new_count == 0 )
490 {
491 _last_error = PmmError::InvalidSize;
492 return pptr<T>();
493 }
494 if ( p.is_null() )
495 return allocate_typed<T>( new_count );
496 if ( sizeof( T ) > 0 && new_count > ( std::numeric_limits<std::size_t>::max )() / sizeof( T ) )
497 {
498 _last_error = PmmError::Overflow;
499 return pptr<T>();
500 }
501 std::size_t new_user_size = sizeof( T ) * new_count;
502 typename thread_policy::unique_lock_type lock( _mutex );
503 if ( !_initialized )
504 {
505 _last_error = PmmError::NotInitialized;
506 return pptr<T>();
507 }
508 std::uint8_t* base = _backend.base_ptr();
509 detail::ManagerHeader<address_traits>* hdr = get_header( base );
510 // blk_idx = pptr.offset - floor(sizeof(Block) / granule)
511 static constexpr index_type kBlkHdrFloorGran =
512 static_cast<index_type>( sizeof( Block<address_traits> ) / address_traits::granule_size );
513 index_type blk_idx = static_cast<index_type>( p.offset() - kBlkHdrFloorGran );
514 void* blk_raw = detail::block_at<address_traits>( base, blk_idx );
515 index_type old_data_gran = BlockStateBase<address_traits>::get_weight( blk_raw );
516 index_type new_data_gran = detail::bytes_to_granules_t<address_traits>( new_user_size );
517 if ( new_data_gran == 0 )
518 new_data_gran = 1;
519 if ( new_data_gran == old_data_gran )
520 {
521 _last_error = PmmError::Ok;
522 return p;
523 }
524 // Skip in-place paths when resolve() overlaps block header (SmallAT).
525 static constexpr bool kBlockAligned = ( sizeof( Block<address_traits> ) % address_traits::granule_size == 0 );
526
527 if constexpr ( kBlockAligned )
528 {
529 if ( new_data_gran < old_data_gran )
530 {
531 allocator::realloc_shrink( base, hdr, blk_idx, blk_raw, old_data_gran, new_data_gran );
532 _last_error = PmmError::Ok;
533 return p;
534 }
535 if ( new_data_gran > old_data_gran )
536 {
537 if ( allocator::realloc_grow( base, hdr, blk_idx, blk_raw, old_data_gran, new_data_gran ) )
538 {
539 _last_error = PmmError::Ok;
540 return p;
541 }
542 }
543 }
544 // Fallback: allocate new + memmove + free old (under same lock).
545 static constexpr index_type kBlkHdrFloorGranFb =
546 static_cast<index_type>( sizeof( Block<address_traits> ) / address_traits::granule_size );
547 index_type new_data_gran_alloc = detail::bytes_to_granules_t<address_traits>( new_user_size );
548 if ( new_data_gran_alloc == 0 )
549 new_data_gran_alloc = 1;
550 if ( new_data_gran_alloc > std::numeric_limits<index_type>::max() - kBlockHdrGranules )
551 {
552 _last_error = PmmError::Overflow;
553 return pptr<T>();
554 }
555 index_type needed = kBlockHdrGranules + new_data_gran_alloc;
556 index_type new_idx = free_block_tree::find_best_fit( base, hdr, needed );
557 if ( new_idx == address_traits::no_block )
558 {
559 if ( !do_expand( new_user_size ) )
560 {
561 _last_error = PmmError::OutOfMemory;
562 logging_policy::on_allocation_failure( new_user_size, PmmError::OutOfMemory );
563 return pptr<T>();
564 }
565 base = _backend.base_ptr();
566 hdr = get_header( base );
567 new_idx = free_block_tree::find_best_fit( base, hdr, needed );
568 if ( new_idx == address_traits::no_block )
569 {
570 _last_error = PmmError::OutOfMemory;
571 logging_policy::on_allocation_failure( new_user_size, PmmError::OutOfMemory );
572 return pptr<T>();
573 }
574 }
575 void* new_raw = allocator::allocate_from_block( base, hdr, new_idx, new_user_size );
576 if ( new_raw == nullptr )
577 {
578 _last_error = PmmError::OutOfMemory;
579 return pptr<T>();
580 }
581 pptr<T> new_p = make_pptr_from_raw<T>( new_raw );
582 void* new_dst = base + static_cast<std::size_t>( new_p.offset() ) * address_traits::granule_size;
583 void* old_src = base + static_cast<std::size_t>( p.offset() ) * address_traits::granule_size;
584 std::size_t copy_sz = ( new_count < old_count ? new_count : old_count ) * sizeof( T );
585 std::memmove( new_dst, old_src, copy_sz );
586 // Free old block
587 index_type old_blk_idx = static_cast<index_type>( p.offset() - kBlkHdrFloorGranFb );
588 void* old_blk_raw = detail::block_at<address_traits>( base, old_blk_idx );
589 index_type freed_w = BlockStateBase<address_traits>::get_weight( old_blk_raw );
590 if ( BlockStateBase<address_traits>::get_node_type( old_blk_raw ) != pmm::kNodeReadOnly )
591 {
592 auto* old_alloc = AllocatedBlock<address_traits>::cast_from_raw( old_blk_raw );
593 if ( old_alloc != nullptr )
594 {
595 old_alloc->mark_as_free();
596 hdr->alloc_count--;
597 hdr->free_count++;
598 if ( hdr->used_size >= freed_w )
599 hdr->used_size -= freed_w;
600 allocator::coalesce( base, hdr, old_blk_idx );
601 }
602 }
603 _last_error = PmmError::Ok;
604 return new_p;
605 }

1058 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
1059 {
1060 typename thread_policy::shared_lock_type lock( _mutex );
1061 if ( !_initialized )
1062 return false;
1063 const std::uint8_t* base = _backend.base_ptr();
1064 using BlockState = BlockStateBase<address_traits>;
1065 const detail::ManagerHeader<address_traits>* hdr = get_header_c( base );
1066 index_type idx = hdr->first_block_offset;
1067 // Issue #146: use address_traits::granule_size for correct byte offset computations.
1068 static constexpr std::size_t kGranSz = address_traits::granule_size;
1069 while ( idx != address_traits::no_block )
1070 {
1071 if ( static_cast<std::size_t>( idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
1072 break;
1073 const void* blk_raw = base + static_cast<std::size_t>( idx ) * kGranSz;
1074 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
1075 index_type total_gran = detail::block_total_granules( base, hdr, blk );
1076 auto w = BlockState::get_weight( blk_raw );
1077 bool is_used = ( w > 0 );
1078 std::size_t hdr_bytes = sizeof( Block<address_traits> );
1079 std::size_t data_bytes = is_used ? static_cast<std::size_t>( w ) * kGranSz : 0;
1080
1081 BlockView view;
1082 view.index = idx;
1083 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( idx ) * kGranSz );
1084 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
1085 view.header_size = hdr_bytes;
1086 view.user_size = data_bytes;
1087 view.alignment = kGranSz;
1088 view.used = is_used;
1089 callback( view );
1090 idx = BlockState::get_next_offset( blk_raw );
1091 }
1092 return true;
1093 }

1399 static bool do_expand( std::size_t user_size ) noexcept
1400 {
1401 using BlockState = BlockStateBase<address_traits>;
1402 if ( !_initialized )
1403 return false;
1404 std::uint8_t* base = _backend.base_ptr();
1405 detail::ManagerHeader<address_traits>* hdr = get_header( base );
1406 std::size_t old_size = hdr->total_size;
1407
1408 // Issue #146: use AddressTraitsT-specific granule size for all computations.
1409 static constexpr std::size_t kGranSz = address_traits::granule_size;
1410 index_type data_gran_need = detail::bytes_to_granules_t<address_traits>( user_size );
1411 if ( data_gran_need == 0 )
1412 data_gran_need = 1;
1413 // min_need = block_header + data + another block_header (for the new expand block)
1414 std::size_t min_need =
1415 static_cast<std::size_t>( kBlockHdrGranules + data_gran_need + kBlockHdrGranules ) * kGranSz;
1416 std::size_t growth = old_size / 4;
1417 if ( growth < min_need )
1418 growth = min_need;
1419
1420 if ( !_backend.expand( growth ) )
1421 return false;
1422
1423 std::uint8_t* new_base = _backend.base_ptr();
1424 std::size_t new_size = _backend.total_size();
1425 if ( new_base == nullptr || new_size <= old_size )
1426 return false;
1427
1428 logging_policy::on_expand( old_size, new_size );
1429 hdr = get_header( new_base );
1430
1431 // Issue #146: compute extra_idx using address_traits::granule_size.
1432 index_type extra_idx = detail::byte_off_to_idx_t<address_traits>( old_size );
1433 std::size_t extra_size = new_size - old_size;
1434
1435 void* last_blk_raw =
1436 ( hdr->last_block_offset != address_traits::no_block )
1437 ? static_cast<void*>( new_base + static_cast<std::size_t>( hdr->last_block_offset ) * kGranSz )
1438 : nullptr;
1439
1440 if ( last_blk_raw != nullptr && BlockState::get_weight( last_blk_raw ) == 0 )
1441 {
