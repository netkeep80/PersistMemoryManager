185 static bool create( std::size_t initial_size ) noexcept
186 {
187 typename thread_policy::unique_lock_type lock( _mutex );
188 if ( initial_size < detail::kMinMemorySize )
189 return false;
190 // Issue #146: use address_traits::granule_size instead of hardcoded kGranuleSize.
191 // Issue #172: guard against overflow when initial_size is close to size_t max.
192 static constexpr std::size_t kGranSzCreate = address_traits::granule_size;
193 if ( initial_size > std::numeric_limits<std::size_t>::max() - ( kGranSzCreate - 1 ) )
194 return false; // overflow in alignment computation
195 std::size_t aligned = ( ( initial_size + kGranSzCreate - 1 ) / kGranSzCreate ) * kGranSzCreate;
196 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
197 {
198 // Либо буфера нет, либо он меньше требуемого — расширяем
199 std::size_t additional =
200 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
201 if ( !_backend.expand( additional ) )
202 return false;
203 }
204 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
205 return false;
206 return init_layout( _backend.base_ptr(), _backend.total_size() );
207 }

227 static bool load() noexcept
228 {
229 typename thread_policy::unique_lock_type lock( _mutex );
230 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
231 return false;
232 std::uint8_t* base = _backend.base_ptr();
233 detail::ManagerHeader<address_traits>* hdr = get_header( base );
234 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )
235 return false;
236 // Issue #146: compare stored granule size against address_traits::granule_size.
237 if ( hdr->granule_size != static_cast<std::uint16_t>( address_traits::granule_size ) )
238 return false;
239 hdr->owns_memory = false;
240 hdr->prev_total_size = 0;
241 allocator::repair_linked_list( base, hdr );
242 allocator::recompute_counters( base, hdr );
243 allocator::rebuild_free_tree( base, hdr );
244 _initialized = true;
245 return true;

276 static void* allocate( std::size_t user_size ) noexcept
277 {
278 typename thread_policy::unique_lock_type lock( _mutex );
279 if ( !_initialized || user_size == 0 )
280 return nullptr;
281
282 std::uint8_t* base = _backend.base_ptr();
283 detail::ManagerHeader<address_traits>* hdr = get_header( base );
284 // Issue #146: use AddressTraits-specific granule size for required granule computation.
285 index_type data_gran = detail::bytes_to_granules_t<address_traits>( user_size );
286 if ( data_gran == 0 )
287 data_gran = 1;
288 index_type needed = kBlockHdrGranules + data_gran;
289 index_type idx = free_block_tree::find_best_fit( base, hdr, needed );
290
291 if ( idx != address_traits::no_block )
292 return allocator::allocate_from_block( base, hdr, idx, user_size );
293
294 // Попытка расширить (если бэкенд поддерживает)
295 if ( !do_expand( user_size ) )
296 return nullptr;
297
298 base = _backend.base_ptr();
299 hdr = get_header( base );
300 idx = free_block_tree::find_best_fit( base, hdr, needed );
301 if ( idx != address_traits::no_block )
302 return allocator::allocate_from_block( base, hdr, idx, user_size );
303 return nullptr;
304 }

311 static void deallocate( void* ptr ) noexcept
312 {
313 typename thread_policy::unique_lock_type lock( _mutex );
314 if ( !_initialized || ptr == nullptr )
315 return;
316 pmm::Block<address_traits>* blk = find_block_from_user_ptr( ptr );
317 if ( blk == nullptr )
318 return;
319 index_type freed = BlockStateBase<address_traits>::get_weight( blk );
320 if ( freed == 0 )
321 return;
322
323 // Issue #126: Permanently locked blocks cannot be freed.
324 if ( BlockStateBase<address_traits>::get_node_type( blk ) == pmm::kNodeReadOnly )
325 return;
326
327 std::uint8_t* base = _backend.base_ptr();
328 detail::ManagerHeader<address_traits>* hdr = get_header( base );
329 index_type blk_idx = detail::block_idx_t<address_traits>( base, blk );
330
331 AllocatedBlock<address_traits>* alloc = AllocatedBlock<address_traits>::cast_from_raw( blk );
332 alloc->mark_as_free();
333
334 hdr->alloc_count--;
335 hdr->free_count++;
336 if ( hdr->used_size >= freed )
337 hdr->used_size -= freed;
338 allocator::coalesce( base, hdr, blk_idx );
339 }

820 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
821 {
822 typename thread_policy::shared_lock_type lock( _mutex );
823 if ( !_initialized )
824 return false;
825 const std::uint8_t* base = _backend.base_ptr();
826 using BlockState = BlockStateBase<address_traits>;
827 const detail::ManagerHeader<address_traits>* hdr = get_header_c( base );
828 index_type idx = hdr->first_block_offset;
829 // Issue #146: use address_traits::granule_size for correct byte offset computations.
830 static constexpr std::size_t kGranSz = address_traits::granule_size;
831 while ( idx != address_traits::no_block )
832 {
833 if ( static_cast<std::size_t>( idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
834 break;
835 const void* blk_raw = base + static_cast<std::size_t>( idx ) * kGranSz;
836 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
837 index_type total_gran = detail::block_total_granules( base, hdr, blk );
838 auto w = BlockState::get_weight( blk_raw );
839 bool is_used = ( w > 0 );
840 std::size_t hdr_bytes = sizeof( Block<address_traits> );
841 std::size_t data_bytes = is_used ? static_cast<std::size_t>( w ) * kGranSz : 0;
842
843 BlockView view;
844 view.index = idx;
845 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( idx ) * kGranSz );
846 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
847 view.header_size = hdr_bytes;
848 view.user_size = data_bytes;
849 view.alignment = kGranSz;
850 view.used = is_used;
851 callback( view );
852 idx = BlockState::get_next_offset( blk_raw );
853 }
854 return true;
855 }

936 // ─── Issue #179: blk_raw helpers ──────────────────────────────────────────
937 // Compute the raw block-header pointer from a pptr<T>.
938 // Formula: base + offset * granule_size - sizeof(Block<AT>) points to the

959 template <typename Callback>
960 static void for_each_free_block_inorder( const std::uint8_t* base, const detail::ManagerHeader<address_traits>* hdr,
961 index_type node_idx, int depth, Callback&& callback ) noexcept
962 {
963 using BlockState = BlockStateBase<address_traits>;
964 // Issue #146: use address_traits::granule_size for correct byte offset computations.
965 static constexpr std::size_t kGranSz = address_traits::granule_size;
966 if ( node_idx == address_traits::no_block )
967 return;
968 if ( static_cast<std::size_t>( node_idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
969 return;
970 const void* blk_raw = base + static_cast<std::size_t>( node_idx ) * kGranSz;
971 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
972
973 index_type left_off = BlockState::get_left_offset( blk_raw );
974 index_type right_off = BlockState::get_right_offset( blk_raw );
975 index_type parent_off = BlockState::get_parent_offset( blk_raw );
976
977 // Visit left subtree first (smaller blocks)
978 for_each_free_block_inorder( base, hdr, left_off, depth + 1, callback );
979
980 // Visit current node
981 index_type total_gran = detail::block_total_granules( base, hdr, blk );
982 FreeBlockView view;
983 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( node_idx ) * kGranSz );
984 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
985 view.free_size = static_cast<std::size_t>( total_gran - kBlockHdrGranules ) * kGranSz;
986 view.left_offset = ( left_off != address_traits::no_block )
987 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( left_off ) * kGranSz )
988 : -1;
989 view.right_offset = ( right_off != address_traits::no_block )
990 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( right_off ) * kGranSz )
991 : -1;
992 view.parent_offset = ( parent_off != address_traits::no_block )
993 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( parent_off ) * kGranSz )
994 : -1;
995 view.avl_height = BlockState::get_avl_height( blk_raw );
996 view.avl_depth = depth;
997 callback( view );
998
999 // Visit right subtree (larger blocks)
1000 for_each_free_block_inorder( base, hdr, right_off, depth + 1, callback );
1001 }

1003 // ─── Address-traits-specific layout constants (Issue #146) ──────────────────
1004 // These compute the correct granule indices based on the actual address_traits
1005 // granule size, rather than using the hardcoded DefaultAddressTraits constants.
1006
1009 static constexpr std::size_t kBlockHdrByteSize =
1010 ( ( sizeof( Block<address_traits> ) + address_traits::granule_size - 1 ) / address_traits::granule_size ) *
1011 address_traits::granule_size;
1012
1014 static constexpr index_type kBlockHdrGranules =
1015 static_cast<index_type>( kBlockHdrByteSize / address_traits::granule_size );
1016
1019 static constexpr index_type kMgrHdrGranules = detail::kManagerHeaderGranules_t<address_traits>;
1020

1035 static bool init_layout( std::uint8_t* base, std::size_t size ) noexcept
1036 {
1037 using BlockState = BlockStateBase<address_traits>;
1038 static constexpr index_type kHdrBlkIdx = 0;
1039 static constexpr index_type kFreeBlkIdx = kFreeBlkIdxLayout;
1040 static constexpr std::size_t kGranSz = address_traits::granule_size;
1041
1042 // Minimum size check: Block_0 + ManagerHeader + Block_1 + at least 1 data granule
1043 static constexpr std::size_t kMinBlockDataSize = kGranSz; // 1 data granule
1044 if ( static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz + sizeof( Block<address_traits> ) + kMinBlockDataSize >
1045 size )
1046 return false;
1047
1048 // Инициализация блока-заголовка (Block_0) через state machine утилиты
1049 void* hdr_blk = base;
1050 std::memset( hdr_blk, 0, kBlockHdrByteSize ); // zero entire aligned region (including padding)
1051 BlockState::init_fields( hdr_blk,
1052 /*prev*/ address_traits::no_block,
1053 /*next*/ kFreeBlkIdx,
1054 /*avl_height*/ 0,
1055 /*weight*/ kMgrHdrGranules,
1056 /*root_offset*/ kHdrBlkIdx );
1057
1058 detail::ManagerHeader<address_traits>* hdr = get_header( base );
1059 std::memset( hdr, 0, sizeof( detail::ManagerHeader<address_traits> ) );
1060 hdr->magic = kMagic;
1061 hdr->total_size = size;
1062 hdr->first_block_offset = kHdrBlkIdx;
1063 hdr->last_block_offset = address_traits::no_block;
1064 hdr->free_tree_root = address_traits::no_block;
1065 hdr->granule_size = static_cast<std::uint16_t>( kGranSz );
1066
1067 // Инициализация первого свободного блока через state machine утилиты
1068 void* blk = base + static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz;
1069 std::memset( blk, 0, sizeof( Block<address_traits> ) );
1070 BlockState::init_fields( blk,
1071 /*prev*/ kHdrBlkIdx,
1072 /*next*/ address_traits::no_block,
1073 /*avl_height*/ 1,
1074 /*weight*/ 0,
1075 /*root_offset*/ 0 );
1076
1077 hdr->last_block_offset = kFreeBlkIdx;
1078 hdr->free_tree_root = kFreeBlkIdx;
1079 hdr->block_count = 2;
1080 hdr->free_count = 1;
1081 hdr->alloc_count = 1;
1082 hdr->used_size = kFreeBlkIdx + kBlockHdrGranules;
1083
1084 _initialized = true;
1085 return true;
1086 }

1088 static bool do_expand( std::size_t user_size ) noexcept
1089 {
1090 using BlockState = BlockStateBase<address_traits>;
1091 if ( !_initialized )
1092 return false;
1093 std::uint8_t* base = _backend.base_ptr();
1094 detail::ManagerHeader<address_traits>* hdr = get_header( base );
1095 std::size_t old_size = hdr->total_size;
1096
1097 // Issue #146: use AddressTraitsT-specific granule size for all computations.
1098 static constexpr std::size_t kGranSz = address_traits::granule_size;
1099 index_type data_gran_need = detail::bytes_to_granules_t<address_traits>( user_size );
1100 if ( data_gran_need == 0 )
1101 data_gran_need = 1;
1102 // min_need = block_header + data + another block_header (for the new expand block)
1103 std::size_t min_need =
1104 static_cast<std::size_t>( kBlockHdrGranules + data_gran_need + kBlockHdrGranules ) * kGranSz;
1105 std::size_t growth = old_size / 4;
1106 if ( growth < min_need )
1107 growth = min_need;
1108
1109 if ( !_backend.expand( growth ) )
1110 return false;
1111
1112 std::uint8_t* new_base = _backend.base_ptr();
1113 std::size_t new_size = _backend.total_size();
1114 if ( new_base == nullptr || new_size <= old_size )
1115 return false;
1116
1117 hdr = get_header( new_base );
1118
1119 // Issue #146: compute extra_idx using address_traits::granule_size.
1120 index_type extra_idx = detail::byte_off_to_idx_t<address_traits>( old_size );
1121 std::size_t extra_size = new_size - old_size;
1122
1123 void* last_blk_raw =
1124 ( hdr->last_block_offset != address_traits::no_block )
1125 ? static_cast<void*>( new_base + static_cast<std::size_t>( hdr->last_block_offset ) * kGranSz )
1126 : nullptr;
1127
1128 if ( last_blk_raw != nullptr && BlockState::get_weight( last_blk_raw ) == 0 )
1129 {
1130 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1131 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1132 free_block_tree::remove( new_base, hdr, loff );
1133 hdr->total_size = new_size;
1134 free_block_tree::insert( new_base, hdr, loff );
1135 }
1136 else
1137 {
1138 // Issue #146: minimum extra size check uses address_traits granule size.
1139 if ( extra_size < sizeof( Block<address_traits> ) + kGranSz )
1140 return false;
1141 void* nb_blk = new_base + static_cast<std::size_t>( extra_idx ) * kGranSz;
1142 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
1143 if ( last_blk_raw != nullptr )
1144 {
1145 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1146 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1147 BlockState::init_fields( nb_blk,
1148 /*prev*/ loff,
1149 /*next*/ address_traits::no_block,
1150 /*avl_height*/ 1,
1151 /*weight*/ 0,
1152 /*root_offset*/ 0 );
1153 BlockState::set_next_offset_of( last_blk_raw, static_cast<index_type>( extra_idx ) );
1154 }
1155 else
1156 {
1157 BlockState::init_fields( nb_blk,
1158 /*prev*/ address_traits::no_block,
1159 /*next*/ address_traits::no_block,
1160 /*avl_height*/ 1,
1161 /*weight*/ 0,
1162 /*root_offset*/ 0 );
1163 hdr->first_block_offset = extra_idx;
1164 }
1165 hdr->last_block_offset = extra_idx;
1166 hdr->block_count++;
1167 hdr->free_count++;
1168 hdr->total_size = new_size;
1169 free_block_tree::insert( new_base, hdr, extra_idx );
1170 }
1171 return true;
1172 }
