155 static bool create( std::size_t initial_size ) noexcept
156 {
157 typename thread_policy::unique_lock_type lock( _mutex );
158 if ( initial_size < detail::kMinMemorySize )
159 return false;
160 // Issue #146: use address_traits::granule_size instead of hardcoded kGranuleSize.
161 static constexpr std::size_t kGranSzCreate = address_traits::granule_size;
162 std::size_t aligned = ( ( initial_size + kGranSzCreate - 1 ) / kGranSzCreate ) * kGranSzCreate;
163 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
164 {
165 // Либо буфера нет, либо он меньше требуемого — расширяем
166 std::size_t additional =
167 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
168 if ( !_backend.expand( additional ) )
169 return false;
170 }
171 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
172 return false;
173 return init_layout( _backend.base_ptr(), _backend.total_size() );
174 }

194 static bool load() noexcept
195 {
196 typename thread_policy::unique_lock_type lock( _mutex );
197 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
198 return false;
199 std::uint8_t* base = _backend.base_ptr();
200 detail::ManagerHeader* hdr = get_header( base );
201 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )
202 return false;
203 // Issue #146: compare stored granule size against address_traits::granule_size.
204 if ( hdr->granule_size != static_cast<std::uint16_t>( address_traits::granule_size ) )
205 return false;
206 hdr->owns_memory = hdr->prev_owns_memory = false;
207 hdr->prev_total_size = 0;
208 hdr->prev_base_ptr = nullptr;
209 allocator::repair_linked_list( base, hdr );
210 allocator::recompute_counters( base, hdr );
211 allocator::rebuild_free_tree( base, hdr );
212 _initialized = true;
213 return true;

243 static void* allocate( std::size_t user_size ) noexcept
244 {
245 typename thread_policy::unique_lock_type lock( _mutex );
246 if ( !_initialized || user_size == 0 )
247 return nullptr;

250 detail::ManagerHeader* hdr = get_header( base );
251 // Issue #146: use AddressTraits-specific granule size for required granule computation.
252 std::uint32_t data_gran = detail::bytes_to_granules_t<address_traits>( user_size );
253 if ( data_gran == 0 )

278 static void deallocate( void* ptr ) noexcept
279 {
280 typename thread_policy::unique_lock_type lock( _mutex );
281 if ( !_initialized || ptr == nullptr )
282 return;
283 std::uint8_t* base = _backend.base_ptr();
284 detail::ManagerHeader* hdr = get_header( base );
285 pmm::Block<address_traits>* blk =
286 detail::header_from_ptr_t<address_traits>( base, ptr, static_cast<std::size_t>( hdr->total_size ) );
287 if ( blk == nullptr )
288 return;
289 std::uint32_t freed = BlockStateBase<address_traits>::get_weight( blk );
290 if ( freed == 0 )
291 return;
292
293 // Issue #126: Permanently locked blocks cannot be freed.
294 if ( BlockStateBase<address_traits>::get_node_type( blk ) == pmm::kNodeReadOnly )
295 return;
296
297 index_type blk_idx = detail::block_idx_t<address_traits>( base, blk );
298
299 AllocatedBlock<address_traits>* alloc = AllocatedBlock<address_traits>::cast_from_raw( blk );
300 alloc->mark_as_free();
301
302 hdr->alloc_count--;
303 hdr->free_count++;
304 if ( hdr->used_size >= freed )
305 hdr->used_size -= freed;
306 allocator::coalesce( base, hdr, blk_idx );
307 }

738 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
739 {
740 typename thread_policy::shared_lock_type lock( _mutex );
741 if ( !_initialized )
742 return false;
743 const std::uint8_t* base = _backend.base_ptr();
744 using BlockState = BlockStateBase<address_traits>;
745 const detail::ManagerHeader* hdr = get_header_c( base );
746 std::uint32_t idx = hdr->first_block_offset;
747 // Issue #146: use address_traits::granule_size for correct byte offset computations.
748 static constexpr std::size_t kGranSz = address_traits::granule_size;
749 while ( idx != detail::kNoBlock )
750 {
751 if ( static_cast<std::size_t>( idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
752 break;
753 const void* blk_raw = base + static_cast<std::size_t>( idx ) * kGranSz;
754 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
755 std::uint32_t total_gran = detail::block_total_granules( base, hdr, blk );
756 auto w = BlockState::get_weight( blk_raw );
757 bool is_used = ( w > 0 );
758 std::size_t hdr_bytes = sizeof( Block<address_traits> );
759 std::size_t data_bytes = is_used ? static_cast<std::size_t>( w ) * kGranSz : 0;
760
761 BlockView view;
762 view.index = idx;
763 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( idx ) * kGranSz );
764 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
765 view.header_size = hdr_bytes;
766 view.user_size = data_bytes;
767 view.alignment = kGranSz;
768 view.used = is_used;
769 callback( view );
770 idx = detail::to_u32_idx<address_traits>( BlockState::get_next_offset( blk_raw ) );
771 }
772 return true;
773 }

818 template <typename Callback>
819 static void for_each_free_block_inorder( const std::uint8_t* base, const detail::ManagerHeader* hdr,
820 std::uint32_t node_idx, int depth, Callback&& callback ) noexcept
821 {
822 using BlockState = BlockStateBase<address_traits>;
823 // Issue #146: use address_traits::granule_size for correct byte offset computations.
824 static constexpr std::size_t kGranSz = address_traits::granule_size;
825 if ( node_idx == detail::kNoBlock )
826 return;
827 if ( static_cast<std::size_t>( node_idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
828 return;
829 const void* blk_raw = base + static_cast<std::size_t>( node_idx ) * kGranSz;
830 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
831
832 auto left_off_idx = BlockState::get_left_offset( blk_raw );
833 auto right_off_idx = BlockState::get_right_offset( blk_raw );
834 auto parent_off_idx = BlockState::get_parent_offset( blk_raw );
835 std::uint32_t left_off = detail::to_u32_idx<address_traits>( left_off_idx );
836 std::uint32_t right_off = detail::to_u32_idx<address_traits>( right_off_idx );
837 std::uint32_t parent_off = detail::to_u32_idx<address_traits>( parent_off_idx );
838
839 // Visit left subtree first (smaller blocks)
840 for_each_free_block_inorder( base, hdr, left_off, depth + 1, callback );
841
842 // Visit current node
843 std::uint32_t total_gran = detail::block_total_granules( base, hdr, blk );
844 FreeBlockView view;
845 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( node_idx ) * kGranSz );
846 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
847 view.free_size = static_cast<std::size_t>( total_gran - kBlockHdrGranules ) * kGranSz;
848 view.left_offset = ( left_off != detail::kNoBlock )
849 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( left_off ) * kGranSz )
850 : -1;
851 view.right_offset = ( right_off != detail::kNoBlock )
852 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( right_off ) * kGranSz )
853 : -1;
854 view.parent_offset = ( parent_off != detail::kNoBlock )
855 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( parent_off ) * kGranSz )
856 : -1;
857 view.avl_height = BlockState::get_avl_height( blk_raw );
858 view.avl_depth = depth;
859 callback( view );
860
861 // Visit right subtree (larger blocks)
862 for_each_free_block_inorder( base, hdr, right_off, depth + 1, callback );
863 }

865 // ─── Address-traits-specific layout constants (Issue #146) ──────────────────
866 // These compute the correct granule indices based on the actual address_traits
867 // granule size, rather than using the hardcoded DefaultAddressTraits constants.
868
871 static constexpr std::size_t kBlockHdrByteSize =
872 ( ( sizeof( Block<address_traits> ) + address_traits::granule_size - 1 ) / address_traits::granule_size ) *
873 address_traits::granule_size;
874
876 static constexpr std::uint32_t kBlockHdrGranules =
877 static_cast<std::uint32_t>( kBlockHdrByteSize / address_traits::granule_size );
878
880 static constexpr std::uint32_t kMgrHdrGranules =
881 static_cast<std::uint32_t>( sizeof( detail::ManagerHeader ) / address_traits::granule_size );
882
884 static constexpr std::uint32_t kFreeBlkIdxLayout = kBlockHdrGranules + kMgrHdrGranules;
885
886 static detail::ManagerHeader* get_header( std::uint8_t* base ) noexcept
887 {
888 // Place ManagerHeader at a granule-aligned offset after Block_0 (Issue #146).
889 return reinterpret_cast<detail::ManagerHeader*>( base + kBlockHdrByteSize );
890 }
891
892 static const detail::ManagerHeader* get_header_c( const std::uint8_t* base ) noexcept
893 {
894 return reinterpret_cast<const detail::ManagerHeader*>( base + kBlockHdrByteSize );
895 }
896
897 static bool init_layout( std::uint8_t* base, std::size_t size ) noexcept
898 {
899 using BlockState = BlockStateBase<address_traits>;
900 static constexpr std::uint32_t kHdrBlkIdx = 0;
901 static constexpr std::uint32_t kFreeBlkIdx = kFreeBlkIdxLayout;
902 static constexpr std::size_t kGranSz = address_traits::granule_size;
903
904 // Minimum size check: Block_0 + ManagerHeader + Block_1 + at least 1 data granule
905 static constexpr std::size_t kMinBlockDataSize = kGranSz; // 1 data granule
906 if ( static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz + sizeof( Block<address_traits> ) + kMinBlockDataSize >
907 size )
908 return false;
909
910 // Инициализация блока-заголовка (Block_0) через state machine утилиты
911 void* hdr_blk = base;
912 std::memset( hdr_blk, 0, kBlockHdrByteSize ); // zero entire aligned region (including padding)
913 BlockState::init_fields( hdr_blk,
914 /*prev*/ address_traits::no_block,
915 /*next*/ static_cast<index_type>( kFreeBlkIdx ),
916 /*avl_height*/ 0,
917 /*weight*/ static_cast<index_type>( kMgrHdrGranules ),
918 /*root_offset*/ static_cast<index_type>( kHdrBlkIdx ) );
919
920 detail::ManagerHeader* hdr = get_header( base );
921 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
922 hdr->magic = kMagic;
923 hdr->total_size = size;
924 hdr->first_block_offset = kHdrBlkIdx;
925 hdr->last_block_offset = detail::kNoBlock;
926 hdr->free_tree_root = detail::kNoBlock;
927 hdr->granule_size = static_cast<std::uint16_t>( kGranSz );
928
929 // Инициализация первого свободного блока через state machine утилиты
930 void* blk = base + static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz;
931 std::memset( blk, 0, sizeof( Block<address_traits> ) );
932 BlockState::init_fields( blk,
933 /*prev*/ static_cast<index_type>( kHdrBlkIdx ),
934 /*next*/ address_traits::no_block,
935 /*avl_height*/ 1,
936 /*weight*/ 0,
937 /*root_offset*/ 0 );
938
939 hdr->last_block_offset = kFreeBlkIdx;
940 hdr->free_tree_root = kFreeBlkIdx;
941 hdr->block_count = 2;
942 hdr->free_count = 1;
943 hdr->alloc_count = 1;
944 hdr->used_size = kFreeBlkIdx + kBlockHdrGranules;
945
946 _initialized = true;
947 return true;
948 }

950 static bool do_expand( std::size_t user_size ) noexcept
951 {
952 using BlockState = BlockStateBase<address_traits>;
953 if ( !_initialized )
954 return false;
955 std::uint8_t* base = _backend.base_ptr();
956 detail::ManagerHeader* hdr = get_header( base );
957 std::size_t old_size = hdr->total_size;
958
959 // Issue #146: use AddressTraitsT-specific granule size for all computations.
960 static constexpr std::size_t kGranSz = address_traits::granule_size;
961 std::uint32_t data_gran_need = detail::bytes_to_granules_t<address_traits>( user_size );
962 if ( data_gran_need == 0 )
963 data_gran_need = 1;
964 // min_need = block_header + data + another block_header (for the new expand block)
965 std::size_t min_need =
966 static_cast<std::size_t>( kBlockHdrGranules + data_gran_need + kBlockHdrGranules ) * kGranSz;
967 std::size_t growth = old_size / 4;
968 if ( growth < min_need )
969 growth = min_need;
970
971 if ( !_backend.expand( growth ) )
972 return false;
973
974 std::uint8_t* new_base = _backend.base_ptr();
975 std::size_t new_size = _backend.total_size();
976 if ( new_base == nullptr || new_size <= old_size )
977 return false;
978
979 hdr = get_header( new_base );
980
981 // Issue #146: compute extra_idx using address_traits::granule_size.
982 std::uint32_t extra_idx = detail::byte_off_to_idx_t<address_traits>( old_size );
983 std::size_t extra_size = new_size - old_size;
984
985 void* last_blk_raw =
986 ( hdr->last_block_offset != detail::kNoBlock )
987 ? static_cast<void*>( new_base + static_cast<std::size_t>( hdr->last_block_offset ) * kGranSz )
988 : nullptr;
989
990 if ( last_blk_raw != nullptr && BlockState::get_weight( last_blk_raw ) == 0 )
991 {
992 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
993 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
994 free_block_tree::remove( new_base, hdr, loff );
995 hdr->total_size = new_size;
996 free_block_tree::insert( new_base, hdr, loff );
997 }
998 else
999 {
1000 // Issue #146: minimum extra size check uses address_traits granule size.
1001 if ( extra_size < sizeof( Block<address_traits> ) + kGranSz )
1002 return false;
1003 void* nb_blk = new_base + static_cast<std::size_t>( extra_idx ) * kGranSz;
1004 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
1005 if ( last_blk_raw != nullptr )
1006 {
1007 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1008 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1009 BlockState::init_fields( nb_blk,
1010 /*prev*/ loff,
1011 /*next*/ address_traits::no_block,
1012 /*avl_height*/ 1,
1013 /*weight*/ 0,
1014 /*root_offset*/ 0 );
1015 BlockState::set_next_offset_of( last_blk_raw, static_cast<index_type>( extra_idx ) );
1016 }
1017 else
1018 {
1019 BlockState::init_fields( nb_blk,
1020 /*prev*/ address_traits::no_block,
1021 /*next*/ address_traits::no_block,
1022 /*avl_height*/ 1,
1023 /*weight*/ 0,
1024 /*root_offset*/ 0 );
1025 hdr->first_block_offset = extra_idx;
1026 }
1027 hdr->last_block_offset = extra_idx;
1028 hdr->block_count++;
1029 hdr->free_count++;
1030 hdr->total_size = new_size;
1031 free_block_tree::insert( new_base, hdr, extra_idx );
1032 }
1033 return true;
1034 }
