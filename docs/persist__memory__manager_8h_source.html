169 static bool create( std::size_t initial_size ) noexcept
170 {
171 typename thread_policy::unique_lock_type lock( _mutex );
172 if ( initial_size < detail::kMinMemorySize )
173 return false;
174 // Issue #146: use address_traits::granule_size instead of hardcoded kGranuleSize.
175 // Issue #172: guard against overflow when initial_size is close to size_t max.
176 static constexpr std::size_t kGranSzCreate = address_traits::granule_size;
177 if ( initial_size > std::numeric_limits<std::size_t>::max() - ( kGranSzCreate - 1 ) )
178 return false; // overflow in alignment computation
179 std::size_t aligned = ( ( initial_size + kGranSzCreate - 1 ) / kGranSzCreate ) * kGranSzCreate;
180 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
181 {
182 // Либо буфера нет, либо он меньше требуемого — расширяем
183 std::size_t additional =
184 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
185 if ( !_backend.expand( additional ) )
186 return false;
187 }
188 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
189 return false;
190 return init_layout( _backend.base_ptr(), _backend.total_size() );
191 }

212 {
213 typename thread_policy::unique_lock_type lock( _mutex );
214 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
215 return false;
216 std::uint8_t* base = _backend.base_ptr();

218 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )

260 static void* allocate( std::size_t user_size ) noexcept
261 {
262 typename thread_policy::unique_lock_type lock( _mutex );
263 if ( !_initialized || user_size == 0 )
264 return nullptr;
265
266 std::uint8_t* base = _backend.base_ptr();
267 detail::ManagerHeader* hdr = get_header( base );
268 // Issue #146: use AddressTraits-specific granule size for required granule computation.
269 std::uint32_t data_gran = detail::bytes_to_granules_t<address_traits>( user_size );
270 if ( data_gran == 0 )
271 data_gran = 1;
272 std::uint32_t needed = kBlockHdrGranules + data_gran;
273 std::uint32_t idx = free_block_tree::find_best_fit( base, hdr, needed );
274
275 if ( idx != detail::kNoBlock )
276 return allocator::allocate_from_block( base, hdr, idx, user_size );
277
278 // Попытка расширить (если бэкенд поддерживает)
279 if ( !do_expand( user_size ) )
280 return nullptr;
281
282 base = _backend.base_ptr();
283 hdr = get_header( base );
284 idx = free_block_tree::find_best_fit( base, hdr, needed );
285 if ( idx != detail::kNoBlock )
286 return allocator::allocate_from_block( base, hdr, idx, user_size );
287 return nullptr;

295 static void deallocate( void* ptr ) noexcept
296 {
297 typename thread_policy::unique_lock_type lock( _mutex );
298 if ( !_initialized || ptr == nullptr )
299 return;
300 std::uint8_t* base = _backend.base_ptr();
301 detail::ManagerHeader* hdr = get_header( base );
302 pmm::Block<address_traits>* blk =
303 detail::header_from_ptr_t<address_traits>( base, ptr, static_cast<std::size_t>( hdr->total_size ) );
304 if ( blk == nullptr )
305 return;
306 std::uint32_t freed = BlockStateBase<address_traits>::get_weight( blk );
307 if ( freed == 0 )
308 return;
309
310 // Issue #126: Permanently locked blocks cannot be freed.
311 if ( BlockStateBase<address_traits>::get_node_type( blk ) == pmm::kNodeReadOnly )
312 return;
313
314 index_type blk_idx = detail::block_idx_t<address_traits>( base, blk );
315
316 AllocatedBlock<address_traits>* alloc = AllocatedBlock<address_traits>::cast_from_raw( blk );
317 alloc->mark_as_free();
318
319 hdr->alloc_count--;
320 hdr->free_count++;
321 if ( hdr->used_size >= freed )
322 hdr->used_size -= freed;
323 allocator::coalesce( base, hdr, blk_idx );
324 }

840 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
841 {
842 typename thread_policy::shared_lock_type lock( _mutex );
843 if ( !_initialized )
844 return false;
845 const std::uint8_t* base = _backend.base_ptr();
846 using BlockState = BlockStateBase<address_traits>;
847 const detail::ManagerHeader* hdr = get_header_c( base );
848 std::uint32_t idx = hdr->first_block_offset;
849 // Issue #146: use address_traits::granule_size for correct byte offset computations.
850 static constexpr std::size_t kGranSz = address_traits::granule_size;
851 while ( idx != detail::kNoBlock )
852 {
853 if ( static_cast<std::size_t>( idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
854 break;
855 const void* blk_raw = base + static_cast<std::size_t>( idx ) * kGranSz;
856 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
857 std::uint32_t total_gran = detail::block_total_granules( base, hdr, blk );
858 auto w = BlockState::get_weight( blk_raw );
859 bool is_used = ( w > 0 );
860 std::size_t hdr_bytes = sizeof( Block<address_traits> );
861 std::size_t data_bytes = is_used ? static_cast<std::size_t>( w ) * kGranSz : 0;
862
863 BlockView view;
864 view.index = idx;
865 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( idx ) * kGranSz );
866 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
867 view.header_size = hdr_bytes;
868 view.user_size = data_bytes;
869 view.alignment = kGranSz;
870 view.used = is_used;
871 callback( view );
872 idx = detail::to_u32_idx<address_traits>( BlockState::get_next_offset( blk_raw ) );
873 }
874 return true;
875 }

922 template <typename Callback>
923 static void for_each_free_block_inorder( const std::uint8_t* base, const detail::ManagerHeader* hdr,
924 std::uint32_t node_idx, int depth, Callback&& callback ) noexcept
925 {
926 using BlockState = BlockStateBase<address_traits>;
927 // Issue #146: use address_traits::granule_size for correct byte offset computations.
928 static constexpr std::size_t kGranSz = address_traits::granule_size;
929 if ( node_idx == detail::kNoBlock )
930 return;
931 if ( static_cast<std::size_t>( node_idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
932 return;
933 const void* blk_raw = base + static_cast<std::size_t>( node_idx ) * kGranSz;
934 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
935
936 auto left_off_idx = BlockState::get_left_offset( blk_raw );
937 auto right_off_idx = BlockState::get_right_offset( blk_raw );
938 auto parent_off_idx = BlockState::get_parent_offset( blk_raw );
939 std::uint32_t left_off = detail::to_u32_idx<address_traits>( left_off_idx );
940 std::uint32_t right_off = detail::to_u32_idx<address_traits>( right_off_idx );
941 std::uint32_t parent_off = detail::to_u32_idx<address_traits>( parent_off_idx );
942
943 // Visit left subtree first (smaller blocks)
944 for_each_free_block_inorder( base, hdr, left_off, depth + 1, callback );
945
946 // Visit current node
947 std::uint32_t total_gran = detail::block_total_granules( base, hdr, blk );
948 FreeBlockView view;
949 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( node_idx ) * kGranSz );
950 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
951 view.free_size = static_cast<std::size_t>( total_gran - kBlockHdrGranules ) * kGranSz;
952 view.left_offset = ( left_off != detail::kNoBlock )
953 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( left_off ) * kGranSz )
954 : -1;
955 view.right_offset = ( right_off != detail::kNoBlock )
956 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( right_off ) * kGranSz )
957 : -1;
958 view.parent_offset = ( parent_off != detail::kNoBlock )
959 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( parent_off ) * kGranSz )
960 : -1;
961 view.avl_height = BlockState::get_avl_height( blk_raw );
962 view.avl_depth = depth;
963 callback( view );
964
965 // Visit right subtree (larger blocks)
966 for_each_free_block_inorder( base, hdr, right_off, depth + 1, callback );
967 }

969 // ─── Address-traits-specific layout constants (Issue #146) ──────────────────
970 // These compute the correct granule indices based on the actual address_traits
971 // granule size, rather than using the hardcoded DefaultAddressTraits constants.
972
975 static constexpr std::size_t kBlockHdrByteSize =
976 ( ( sizeof( Block<address_traits> ) + address_traits::granule_size - 1 ) / address_traits::granule_size ) *
977 address_traits::granule_size;
978
980 static constexpr std::uint32_t kBlockHdrGranules =
981 static_cast<std::uint32_t>( kBlockHdrByteSize / address_traits::granule_size );
982
984 static constexpr std::uint32_t kMgrHdrGranules =
985 static_cast<std::uint32_t>( sizeof( detail::ManagerHeader ) / address_traits::granule_size );
986
988 static constexpr std::uint32_t kFreeBlkIdxLayout = kBlockHdrGranules + kMgrHdrGranules;
989
990 static detail::ManagerHeader* get_header( std::uint8_t* base ) noexcept
991 {
992 // Place ManagerHeader at a granule-aligned offset after Block_0 (Issue #146).
993 return reinterpret_cast<detail::ManagerHeader*>( base + kBlockHdrByteSize );
994 }
995
996 static const detail::ManagerHeader* get_header_c( const std::uint8_t* base ) noexcept
997 {
998 return reinterpret_cast<const detail::ManagerHeader*>( base + kBlockHdrByteSize );
999 }
1000
1001 static bool init_layout( std::uint8_t* base, std::size_t size ) noexcept
1002 {
1003 using BlockState = BlockStateBase<address_traits>;
1004 static constexpr std::uint32_t kHdrBlkIdx = 0;
1005 static constexpr std::uint32_t kFreeBlkIdx = kFreeBlkIdxLayout;
1006 static constexpr std::size_t kGranSz = address_traits::granule_size;
1007
1008 // Minimum size check: Block_0 + ManagerHeader + Block_1 + at least 1 data granule
1009 static constexpr std::size_t kMinBlockDataSize = kGranSz; // 1 data granule
1010 if ( static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz + sizeof( Block<address_traits> ) + kMinBlockDataSize >
1011 size )
1012 return false;
1013
1014 // Инициализация блока-заголовка (Block_0) через state machine утилиты
1015 void* hdr_blk = base;
1016 std::memset( hdr_blk, 0, kBlockHdrByteSize ); // zero entire aligned region (including padding)
1017 BlockState::init_fields( hdr_blk,
1018 /*prev*/ address_traits::no_block,
1019 /*next*/ static_cast<index_type>( kFreeBlkIdx ),
1020 /*avl_height*/ 0,
1021 /*weight*/ static_cast<index_type>( kMgrHdrGranules ),
1022 /*root_offset*/ static_cast<index_type>( kHdrBlkIdx ) );
1023
1024 detail::ManagerHeader* hdr = get_header( base );
1025 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
1026 hdr->magic = kMagic;
1027 hdr->total_size = size;
1028 hdr->first_block_offset = kHdrBlkIdx;
1029 hdr->last_block_offset = detail::kNoBlock;
1030 hdr->free_tree_root = detail::kNoBlock;
1031 hdr->granule_size = static_cast<std::uint16_t>( kGranSz );
1032
1033 // Инициализация первого свободного блока через state machine утилиты
1034 void* blk = base + static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz;
1035 std::memset( blk, 0, sizeof( Block<address_traits> ) );
1036 BlockState::init_fields( blk,
1037 /*prev*/ static_cast<index_type>( kHdrBlkIdx ),
1038 /*next*/ address_traits::no_block,
1039 /*avl_height*/ 1,
1040 /*weight*/ 0,
1041 /*root_offset*/ 0 );
1042
1043 hdr->last_block_offset = kFreeBlkIdx;
1044 hdr->free_tree_root = kFreeBlkIdx;
1045 hdr->block_count = 2;
1046 hdr->free_count = 1;
1047 hdr->alloc_count = 1;
1048 hdr->used_size = kFreeBlkIdx + kBlockHdrGranules;
1049
1050 _initialized = true;
1051 return true;
1052 }
1053
1054 static bool do_expand( std::size_t user_size ) noexcept
1055 {
1056 using BlockState = BlockStateBase<address_traits>;
1057 if ( !_initialized )
1058 return false;
1059 std::uint8_t* base = _backend.base_ptr();
1060 detail::ManagerHeader* hdr = get_header( base );
1061 std::size_t old_size = hdr->total_size;
1062
1063 // Issue #146: use AddressTraitsT-specific granule size for all computations.
1064 static constexpr std::size_t kGranSz = address_traits::granule_size;
1065 std::uint32_t data_gran_need = detail::bytes_to_granules_t<address_traits>( user_size );
1066 if ( data_gran_need == 0 )
1067 data_gran_need = 1;
1068 // min_need = block_header + data + another block_header (for the new expand block)
1069 std::size_t min_need =
1070 static_cast<std::size_t>( kBlockHdrGranules + data_gran_need + kBlockHdrGranules ) * kGranSz;
1071 std::size_t growth = old_size / 4;
1072 if ( growth < min_need )
1073 growth = min_need;
1074
1075 if ( !_backend.expand( growth ) )
1076 return false;
1077
1078 std::uint8_t* new_base = _backend.base_ptr();
1079 std::size_t new_size = _backend.total_size();
1080 if ( new_base == nullptr || new_size <= old_size )
1081 return false;
1082
1083 hdr = get_header( new_base );
1084
1085 // Issue #146: compute extra_idx using address_traits::granule_size.
1086 std::uint32_t extra_idx = detail::byte_off_to_idx_t<address_traits>( old_size );
1087 std::size_t extra_size = new_size - old_size;
1088
1089 void* last_blk_raw =
1090 ( hdr->last_block_offset != detail::kNoBlock )
1091 ? static_cast<void*>( new_base + static_cast<std::size_t>( hdr->last_block_offset ) * kGranSz )
1092 : nullptr;
1093
1094 if ( last_blk_raw != nullptr && BlockState::get_weight( last_blk_raw ) == 0 )
1095 {
1096 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1097 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1098 free_block_tree::remove( new_base, hdr, loff );
1099 hdr->total_size = new_size;
1100 free_block_tree::insert( new_base, hdr, loff );
1101 }
1102 else
1103 {
1104 // Issue #146: minimum extra size check uses address_traits granule size.
1105 if ( extra_size < sizeof( Block<address_traits> ) + kGranSz )
1106 return false;
1107 void* nb_blk = new_base + static_cast<std::size_t>( extra_idx ) * kGranSz;
1108 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
1109 if ( last_blk_raw != nullptr )
1110 {
1111 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1112 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1113 BlockState::init_fields( nb_blk,
1114 /*prev*/ loff,
1115 /*next*/ address_traits::no_block,
1116 /*avl_height*/ 1,
1117 /*weight*/ 0,
1118 /*root_offset*/ 0 );
1119 BlockState::set_next_offset_of( last_blk_raw, static_cast<index_type>( extra_idx ) );
1120 }
1121 else
1122 {
1123 BlockState::init_fields( nb_blk,
1124 /*prev*/ address_traits::no_block,
1125 /*next*/ address_traits::no_block,
1126 /*avl_height*/ 1,
1127 /*weight*/ 0,
1128 /*root_offset*/ 0 );
1129 hdr->first_block_offset = extra_idx;
1130 }
1131 hdr->last_block_offset = extra_idx;
1132 hdr->block_count++;
1133 hdr->free_count++;
1134 hdr->total_size = new_size;
1135 free_block_tree::insert( new_base, hdr, extra_idx );
1136 }
1137 return true;
1138 }
