221 static bool create( std::size_t initial_size ) noexcept
222 {
223 typename thread_policy::unique_lock_type lock( _mutex );
224 if ( initial_size < detail::kMinMemorySize )
225 {
226 _last_error = PmmError::InvalidSize;

229 // Use address_traits::granule_size instead of hardcoded kGranuleSize.
230 // Guard against overflow when initial_size is close to size_t max.
231 static constexpr std::size_t kGranSzCreate = address_traits::granule_size;
232 if ( initial_size > std::numeric_limits<std::size_t>::max() - ( kGranSzCreate - 1 ) )
233 {
234 _last_error = PmmError::Overflow;
235 return false;
236 }
237 std::size_t aligned = ( ( initial_size + kGranSzCreate - 1 ) / kGranSzCreate ) * kGranSzCreate;
238 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )

240 // Либо буфера нет, либо он меньше требуемого — расширяем

242 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
243 if ( !_backend.expand( additional ) )
244 {
245 _last_error = PmmError::ExpandFailed;

247 }
248 }
249 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
250 {
251 _last_error = PmmError::BackendError;
252 return false;
253 }

256 ok = bootstrap_forest_registry_unlocked();
257 if ( ok )
258 ok = validate_bootstrap_invariants_unlocked();

315 static bool load( VerifyResult& result ) noexcept
316 {
317 result.mode = RecoveryMode::Repair;
318 result.ok = true;
319 typename thread_policy::unique_lock_type lock( _mutex );
320 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
321 {
322 _last_error = ( _backend.base_ptr() == nullptr ) ? PmmError::BackendError : PmmError::InvalidSize;
323 result.add( ViolationType::HeaderCorruption, DiagnosticAction::Aborted );
324 return false;
325 }
326 std::uint8_t* base = _backend.base_ptr();
327 detail::ManagerHeader<address_traits>* hdr = get_header( base );
328 if ( hdr->magic != kMagic )
329 {
330 _last_error = PmmError::InvalidMagic;
331 logging_policy::on_corruption_detected( PmmError::InvalidMagic );
332 result.add( ViolationType::HeaderCorruption, DiagnosticAction::Aborted, 0,
333 static_cast<std::uint64_t>( kMagic ), static_cast<std::uint64_t>( hdr->magic ) );
334 return false;
335 }
336 if ( hdr->total_size != _backend.total_size() )
337 {
338 _last_error = PmmError::SizeMismatch;
339 logging_policy::on_corruption_detected( PmmError::SizeMismatch );
340 result.add( ViolationType::HeaderCorruption, DiagnosticAction::Aborted, 0, _backend.total_size(),
341 static_cast<std::uint64_t>( hdr->total_size ) );
342 return false;
343 }
344 if ( hdr->granule_size != static_cast<std::uint16_t>( address_traits::granule_size ) )
345 {
346 _last_error = PmmError::GranuleMismatch;
347 logging_policy::on_corruption_detected( PmmError::GranuleMismatch );
348 result.add( ViolationType::HeaderCorruption, DiagnosticAction::Aborted, 0, address_traits::granule_size,
349 static_cast<std::uint64_t>( hdr->granule_size ) );
350 return false;
351 }
352 // Verify before repair — detect violations in the raw image.
353 // Verify before repair — detect violations, then mark with repair actions.
354 auto mark_entries = []( VerifyResult& r, std::size_t from, DiagnosticAction act )
355 {
356 for ( std::size_t i = from; i < r.entry_count; ++i )
357 r.entries[i].action = act;
358 };
359 std::size_t pre = result.entry_count;
360 allocator::verify_block_states( base, hdr, result ); // Phase 1: Repaired
361 mark_entries( result, pre, DiagnosticAction::Repaired );
362 pre = result.entry_count;
363 allocator::verify_linked_list( base, hdr, result ); // Phase 2: Repaired
364 mark_entries( result, pre, DiagnosticAction::Repaired );
365 pre = result.entry_count;
366 allocator::verify_counters( base, hdr, result ); // Phase 3: Rebuilt
367 mark_entries( result, pre, DiagnosticAction::Rebuilt );
368 pre = result.entry_count;
369 allocator::verify_free_tree( base, hdr, result ); // Phase 4: Rebuilt
370 mark_entries( result, pre, DiagnosticAction::Rebuilt );
371 // Repair phase: apply all fixes.
372 hdr->owns_memory = false;
373 hdr->prev_total_size = 0;
374 allocator::repair_linked_list( base, hdr );
375 allocator::recompute_counters( base, hdr );
376 allocator::rebuild_free_tree( base, hdr );
377 _initialized = true;
378 // Phase 5: forest registry diagnostics.
379 {
380 VerifyResult forest_verify;
381 verify_forest_registry_unlocked( forest_verify );
382 for ( std::size_t i = 0; i < forest_verify.entry_count; ++i )
383 {
384 const auto& e = forest_verify.entries[i];
385 result.add( e.type, DiagnosticAction::Repaired, e.block_index, e.expected, e.actual );
386 }
387 }
388 if ( !validate_or_bootstrap_forest_registry_unlocked() )
389 {
390 for ( std::size_t i = 0; i < result.entry_count; ++i )
391 {
392 if ( result.entries[i].type == ViolationType::ForestRegistryMissing ||
393 result.entries[i].type == ViolationType::ForestDomainMissing ||
394 result.entries[i].type == ViolationType::ForestDomainFlagsMissing )
395 result.entries[i].action = DiagnosticAction::Aborted;
396 }
397 _initialized = false;
398 return false;
399 }
400 if ( !validate_bootstrap_invariants_unlocked() )
401 {
402 _initialized = false;
403 return false;
404 }
405 _last_error = PmmError::Ok;
406 logging_policy::on_load();
407 return true;
408 }

547 template <typename T>
548 static pptr<T> reallocate_typed( pptr<T> p, std::size_t old_count, std::size_t new_count ) noexcept
549 {
550 static_assert( std::is_trivially_copyable_v<T>,
551 "reallocate_typed<T>: T must be trivially copyable for safe memcpy reallocation." );
552 if ( new_count == 0 )
553 {
554 _last_error = PmmError::InvalidSize;
555 return pptr<T>();
556 }
557 if ( p.is_null() )
558 return allocate_typed<T>( new_count );
559 if ( sizeof( T ) > 0 && new_count > ( std::numeric_limits<std::size_t>::max )() / sizeof( T ) )
560 {
561 _last_error = PmmError::Overflow;
562 return pptr<T>();
563 }
564 std::size_t new_user_size = sizeof( T ) * new_count;
565 typename thread_policy::unique_lock_type lock( _mutex );
566 if ( !_initialized )
567 {
568 _last_error = PmmError::NotInitialized;
569 return pptr<T>();
570 }
571 std::uint8_t* base = _backend.base_ptr();
572 detail::ManagerHeader<address_traits>* hdr = get_header( base );
573 // blk_idx = pptr.offset - floor(sizeof(Block) / granule)
574 static constexpr index_type kBlkHdrFloorGran =
575 static_cast<index_type>( sizeof( Block<address_traits> ) / address_traits::granule_size );
576 index_type blk_idx = static_cast<index_type>( p.offset() - kBlkHdrFloorGran );
577 void* blk_raw = detail::block_at<address_traits>( base, blk_idx );
578 index_type old_data_gran = BlockStateBase<address_traits>::get_weight( blk_raw );
579 index_type new_data_gran = detail::bytes_to_granules_t<address_traits>( new_user_size );
580 if ( new_data_gran == 0 )
581 new_data_gran = 1;
582 if ( new_data_gran == old_data_gran )
583 {
584 _last_error = PmmError::Ok;
585 return p;
586 }
587 // Skip in-place paths when resolve() overlaps block header (SmallAT).
588 static constexpr bool kBlockAligned = ( sizeof( Block<address_traits> ) % address_traits::granule_size == 0 );
589
590 if constexpr ( kBlockAligned )
591 {
592 if ( new_data_gran < old_data_gran )
593 {
594 allocator::realloc_shrink( base, hdr, blk_idx, blk_raw, old_data_gran, new_data_gran );
595 _last_error = PmmError::Ok;
596 return p;
597 }
598 if ( new_data_gran > old_data_gran )
599 {
600 if ( allocator::realloc_grow( base, hdr, blk_idx, blk_raw, old_data_gran, new_data_gran ) )
601 {
602 _last_error = PmmError::Ok;
603 return p;
604 }
605 }
606 }
607 // Fallback: allocate new + memmove + free old (under same lock).
608 static constexpr index_type kBlkHdrFloorGranFb =
609 static_cast<index_type>( sizeof( Block<address_traits> ) / address_traits::granule_size );
610 index_type new_data_gran_alloc = detail::bytes_to_granules_t<address_traits>( new_user_size );
611 if ( new_data_gran_alloc == 0 )
612 new_data_gran_alloc = 1;
613 if ( new_data_gran_alloc > std::numeric_limits<index_type>::max() - kBlockHdrGranules )
614 {
615 _last_error = PmmError::Overflow;
616 return pptr<T>();
617 }
618 index_type needed = kBlockHdrGranules + new_data_gran_alloc;
619 index_type new_idx = free_block_tree::find_best_fit( base, hdr, needed );
620 if ( new_idx == address_traits::no_block )
621 {
622 if ( !do_expand( new_user_size ) )
623 {
624 _last_error = PmmError::OutOfMemory;
625 logging_policy::on_allocation_failure( new_user_size, PmmError::OutOfMemory );
626 return pptr<T>();
627 }
628 base = _backend.base_ptr();
629 hdr = get_header( base );
630 new_idx = free_block_tree::find_best_fit( base, hdr, needed );
631 if ( new_idx == address_traits::no_block )
632 {
633 _last_error = PmmError::OutOfMemory;
634 logging_policy::on_allocation_failure( new_user_size, PmmError::OutOfMemory );
635 return pptr<T>();
636 }
637 }
638 void* new_raw = allocator::allocate_from_block( base, hdr, new_idx, new_user_size );
639 if ( new_raw == nullptr )
640 {
641 _last_error = PmmError::OutOfMemory;
642 return pptr<T>();
643 }
644 pptr<T> new_p = make_pptr_from_raw<T>( new_raw );
645 void* new_dst = base + static_cast<std::size_t>( new_p.offset() ) * address_traits::granule_size;
646 void* old_src = base + static_cast<std::size_t>( p.offset() ) * address_traits::granule_size;
647 std::size_t copy_sz = ( new_count < old_count ? new_count : old_count ) * sizeof( T );
648 std::memmove( new_dst, old_src, copy_sz );
649 // Free old block
650 index_type old_blk_idx = static_cast<index_type>( p.offset() - kBlkHdrFloorGranFb );
651 void* old_blk_raw = detail::block_at<address_traits>( base, old_blk_idx );
652 index_type freed_w = BlockStateBase<address_traits>::get_weight( old_blk_raw );
653 if ( BlockStateBase<address_traits>::get_node_type( old_blk_raw ) != pmm::kNodeReadOnly )
654 {
655 auto* old_alloc = AllocatedBlock<address_traits>::cast_from_raw( old_blk_raw );
656 if ( old_alloc != nullptr )
657 {
658 old_alloc->mark_as_free();
659 hdr->alloc_count--;
660 hdr->free_count++;
661 if ( hdr->used_size >= freed_w )
662 hdr->used_size -= freed_w;
663 allocator::coalesce( base, hdr, old_blk_idx );
664 }
665 }
666 _last_error = PmmError::Ok;
667 return new_p;
668 }

1138 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
1139 {
1140 typename thread_policy::shared_lock_type lock( _mutex );
1141 if ( !_initialized )
1142 return false;
1143 const std::uint8_t* base = _backend.base_ptr();
1144 using BlockState = BlockStateBase<address_traits>;
1145 const detail::ManagerHeader<address_traits>* hdr = get_header_c( base );
1146 index_type idx = hdr->first_block_offset;
1147 // Use address_traits::granule_size for correct byte offset computations.
1148 static constexpr std::size_t kGranSz = address_traits::granule_size;
1149 while ( idx != address_traits::no_block )
1150 {
1151 if ( static_cast<std::size_t>( idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
1152 break;
1153 const void* blk_raw = base + static_cast<std::size_t>( idx ) * kGranSz;
1154 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
1155 index_type total_gran = detail::block_total_granules( base, hdr, blk );
1156 auto w = BlockState::get_weight( blk_raw );
1157 bool is_used = ( w > 0 );
1158 std::size_t hdr_bytes = sizeof( Block<address_traits> );
1159 std::size_t data_bytes = is_used ? static_cast<std::size_t>( w ) * kGranSz : 0;
1160
1161 BlockView view;
1162 view.index = idx;
1163 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( idx ) * kGranSz );
1164 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
1165 view.header_size = hdr_bytes;
1166 view.user_size = data_bytes;
1167 view.alignment = kGranSz;
1168 view.used = is_used;
1169 callback( view );
1170 idx = BlockState::get_next_offset( blk_raw );
1171 }
1172 return true;
1173 }
