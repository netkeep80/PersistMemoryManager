126 static bool create( std::size_t initial_size ) noexcept
127 {
128 typename thread_policy::unique_lock_type lock( _mutex );
129 if ( initial_size < detail::kMinMemorySize )
130 return false;
131 // Issue #146: use address_traits::granule_size instead of hardcoded kGranuleSize.
132 static constexpr std::size_t kGranSzCreate = address_traits::granule_size;
133 std::size_t aligned = ( ( initial_size + kGranSzCreate - 1 ) / kGranSzCreate ) * kGranSzCreate;
134 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
135 {
136 // Либо буфера нет, либо он меньше требуемого — расширяем
137 std::size_t additional =
138 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
139 if ( !_backend.expand( additional ) )
140 return false;
141 }
142 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
143 return false;
144 return init_layout( _backend.base_ptr(), _backend.total_size() );
145 }

165 static bool load() noexcept
166 {
167 typename thread_policy::unique_lock_type lock( _mutex );
168 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
169 return false;
170 std::uint8_t* base = _backend.base_ptr();
171 detail::ManagerHeader* hdr = get_header( base );
172 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )
173 return false;
174 // Issue #146: compare stored granule size against address_traits::granule_size.
175 if ( hdr->granule_size != static_cast<std::uint16_t>( address_traits::granule_size ) )
176 return false;
177 hdr->owns_memory = hdr->prev_owns_memory = false;
178 hdr->prev_total_size = 0;
179 hdr->prev_base_ptr = nullptr;
180 allocator::repair_linked_list( base, hdr );
181 allocator::recompute_counters( base, hdr );
182 allocator::rebuild_free_tree( base, hdr );
183 _initialized = true;
184 return true;
185 }

214 static void* allocate( std::size_t user_size ) noexcept
215 {
216 typename thread_policy::unique_lock_type lock( _mutex );
217 if ( !_initialized || user_size == 0 )
218 return nullptr;
219
220 std::uint8_t* base = _backend.base_ptr();
221 detail::ManagerHeader* hdr = get_header( base );
222 // Issue #146: use AddressTraits-specific granule size for required granule computation.
223 std::uint32_t data_gran = detail::bytes_to_granules_t<address_traits>( user_size );
224 if ( data_gran == 0 )
225 data_gran = 1;
226 std::uint32_t needed = kBlockHdrGranules + data_gran;
227 std::uint32_t idx = free_block_tree::find_best_fit( base, hdr, needed );
228
229 if ( idx != detail::kNoBlock )
230 return allocator::allocate_from_block( base, hdr, idx, user_size );
231
232 // Попытка расширить (если бэкенд поддерживает)
233 if ( !do_expand( user_size ) )
234 return nullptr;
235
236 base = _backend.base_ptr();
237 hdr = get_header( base );
238 idx = free_block_tree::find_best_fit( base, hdr, needed );
239 if ( idx != detail::kNoBlock )
240 return allocator::allocate_from_block( base, hdr, idx, user_size );
241 return nullptr;
242 }

249 static void deallocate( void* ptr ) noexcept
250 {
251 typename thread_policy::unique_lock_type lock( _mutex );
252 if ( !_initialized || ptr == nullptr )
253 return;
254 std::uint8_t* base = _backend.base_ptr();
255 detail::ManagerHeader* hdr = get_header( base );
256 pmm::Block<address_traits>* blk =
257 detail::header_from_ptr_t<address_traits>( base, ptr, static_cast<std::size_t>( hdr->total_size ) );
258 if ( blk == nullptr )
259 return;
260 std::uint32_t freed = BlockStateBase<address_traits>::get_weight( blk );
261 if ( freed == 0 )
262 return;
263
264 // Issue #126: Permanently locked blocks cannot be freed.
265 if ( BlockStateBase<address_traits>::get_node_type( blk ) == pmm::kNodeReadOnly )
266 return;
267
268 index_type blk_idx = detail::block_idx_t<address_traits>( base, blk );
269
270 AllocatedBlock<address_traits>* alloc = AllocatedBlock<address_traits>::cast_from_raw( blk );
271 alloc->mark_as_free();
272
273 hdr->alloc_count--;
274 hdr->free_count++;
275 if ( hdr->used_size >= freed )
276 hdr->used_size -= freed;
277 allocator::coalesce( base, hdr, blk_idx );
278 }

707 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
708 {
709 typename thread_policy::shared_lock_type lock( _mutex );
710 if ( !_initialized )
711 return false;
712 const std::uint8_t* base = _backend.base_ptr();
713 using BlockState = BlockStateBase<address_traits>;
714 const detail::ManagerHeader* hdr = get_header_c( base );
715 std::uint32_t idx = hdr->first_block_offset;
716 // Issue #146: use address_traits::granule_size for correct byte offset computations.
717 static constexpr std::size_t kGranSz = address_traits::granule_size;
718 while ( idx != detail::kNoBlock )
719 {
720 if ( static_cast<std::size_t>( idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
721 break;
722 const void* blk_raw = base + static_cast<std::size_t>( idx ) * kGranSz;
723 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
724 std::uint32_t total_gran = detail::block_total_granules( base, hdr, blk );
725 auto w = BlockState::get_weight( blk_raw );
726 bool is_used = ( w > 0 );
727 std::size_t hdr_bytes = sizeof( Block<address_traits> );
728 std::size_t data_bytes = is_used ? static_cast<std::size_t>( w ) * kGranSz : 0;
729
730 BlockView view;
731 view.index = idx;
732 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( idx ) * kGranSz );
733 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
734 view.header_size = hdr_bytes;
735 view.user_size = data_bytes;
736 view.alignment = kGranSz;
737 view.used = is_used;
738 callback( view );
739 idx = detail::to_u32_idx<address_traits>( BlockState::get_next_offset( blk_raw ) );
740 }
741 return true;
742 }

787 template <typename Callback>
788 static void for_each_free_block_inorder( const std::uint8_t* base, const detail::ManagerHeader* hdr,
789 std::uint32_t node_idx, int depth, Callback&& callback ) noexcept
790 {
791 using BlockState = BlockStateBase<address_traits>;
792 // Issue #146: use address_traits::granule_size for correct byte offset computations.
793 static constexpr std::size_t kGranSz = address_traits::granule_size;
794 if ( node_idx == detail::kNoBlock )
795 return;
796 if ( static_cast<std::size_t>( node_idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
797 return;
798 const void* blk_raw = base + static_cast<std::size_t>( node_idx ) * kGranSz;
799 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
800
801 auto left_off_idx = BlockState::get_left_offset( blk_raw );
802 auto right_off_idx = BlockState::get_right_offset( blk_raw );
803 auto parent_off_idx = BlockState::get_parent_offset( blk_raw );
804 std::uint32_t left_off = detail::to_u32_idx<address_traits>( left_off_idx );
805 std::uint32_t right_off = detail::to_u32_idx<address_traits>( right_off_idx );
806 std::uint32_t parent_off = detail::to_u32_idx<address_traits>( parent_off_idx );
807
808 // Visit left subtree first (smaller blocks)
809 for_each_free_block_inorder( base, hdr, left_off, depth + 1, callback );
810
811 // Visit current node
812 std::uint32_t total_gran = detail::block_total_granules( base, hdr, blk );
813 FreeBlockView view;
814 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( node_idx ) * kGranSz );
815 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
816 view.free_size = static_cast<std::size_t>( total_gran - kBlockHdrGranules ) * kGranSz;
817 view.left_offset = ( left_off != detail::kNoBlock )
818 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( left_off ) * kGranSz )
819 : -1;
820 view.right_offset = ( right_off != detail::kNoBlock )
821 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( right_off ) * kGranSz )
822 : -1;
823 view.parent_offset = ( parent_off != detail::kNoBlock )
824 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( parent_off ) * kGranSz )
825 : -1;
826 view.avl_height = BlockState::get_avl_height( blk_raw );
827 view.avl_depth = depth;
828 callback( view );
829
830 // Visit right subtree (larger blocks)
831 for_each_free_block_inorder( base, hdr, right_off, depth + 1, callback );
832 }

853 static constexpr std::uint32_t kFreeBlkIdxLayout = kBlockHdrGranules + kMgrHdrGranules;

855 static detail::ManagerHeader* get_header( std::uint8_t* base ) noexcept
856 {
857 // Place ManagerHeader at a granule-aligned offset after Block_0 (Issue #146).
858 return reinterpret_cast<detail::ManagerHeader*>( base + kBlockHdrByteSize );
859 }
860
861 static const detail::ManagerHeader* get_header_c( const std::uint8_t* base ) noexcept
862 {
863 return reinterpret_cast<const detail::ManagerHeader*>( base + kBlockHdrByteSize );
864 }
865
866 static bool init_layout( std::uint8_t* base, std::size_t size ) noexcept
867 {
868 using BlockState = BlockStateBase<address_traits>;
869 static constexpr std::uint32_t kHdrBlkIdx = 0;
870 static constexpr std::uint32_t kFreeBlkIdx = kFreeBlkIdxLayout;
871 static constexpr std::size_t kGranSz = address_traits::granule_size;
872
873 // Minimum size check: Block_0 + ManagerHeader + Block_1 + at least 1 data granule
874 static constexpr std::size_t kMinBlockDataSize = kGranSz; // 1 data granule
875 if ( static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz + sizeof( Block<address_traits> ) + kMinBlockDataSize >
876 size )
877 return false;
878
879 // Инициализация блока-заголовка (Block_0) через state machine утилиты
880 void* hdr_blk = base;
881 std::memset( hdr_blk, 0, kBlockHdrByteSize ); // zero entire aligned region (including padding)
882 BlockState::init_fields( hdr_blk,
883 /*prev*/ address_traits::no_block,
884 /*next*/ static_cast<index_type>( kFreeBlkIdx ),
885 /*avl_height*/ 0,
886 /*weight*/ static_cast<index_type>( kMgrHdrGranules ),
887 /*root_offset*/ static_cast<index_type>( kHdrBlkIdx ) );
888
889 detail::ManagerHeader* hdr = get_header( base );
890 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
891 hdr->magic = kMagic;
892 hdr->total_size = size;
893 hdr->first_block_offset = kHdrBlkIdx;
894 hdr->last_block_offset = detail::kNoBlock;
895 hdr->free_tree_root = detail::kNoBlock;
896 hdr->granule_size = static_cast<std::uint16_t>( kGranSz );
897
898 // Инициализация первого свободного блока через state machine утилиты
899 void* blk = base + static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz;
900 std::memset( blk, 0, sizeof( Block<address_traits> ) );
901 BlockState::init_fields( blk,
902 /*prev*/ static_cast<index_type>( kHdrBlkIdx ),
903 /*next*/ address_traits::no_block,
904 /*avl_height*/ 1,
905 /*weight*/ 0,
906 /*root_offset*/ 0 );
907
908 hdr->last_block_offset = kFreeBlkIdx;
909 hdr->free_tree_root = kFreeBlkIdx;
910 hdr->block_count = 2;
911 hdr->free_count = 1;
912 hdr->alloc_count = 1;
913 hdr->used_size = kFreeBlkIdx + kBlockHdrGranules;
914
915 _initialized = true;
916 return true;
917 }
918
919 static bool do_expand( std::size_t user_size ) noexcept
920 {
921 using BlockState = BlockStateBase<address_traits>;
922 if ( !_initialized )
923 return false;
924 std::uint8_t* base = _backend.base_ptr();
925 detail::ManagerHeader* hdr = get_header( base );
926 std::size_t old_size = hdr->total_size;
927
928 // Issue #146: use AddressTraitsT-specific granule size for all computations.
929 static constexpr std::size_t kGranSz = address_traits::granule_size;
930 std::uint32_t data_gran_need = detail::bytes_to_granules_t<address_traits>( user_size );
931 if ( data_gran_need == 0 )
932 data_gran_need = 1;
933 // min_need = block_header + data + another block_header (for the new expand block)
934 std::size_t min_need =
935 static_cast<std::size_t>( kBlockHdrGranules + data_gran_need + kBlockHdrGranules ) * kGranSz;
936 std::size_t growth = old_size / 4;
937 if ( growth < min_need )
938 growth = min_need;
939
940 if ( !_backend.expand( growth ) )
941 return false;
942
943 std::uint8_t* new_base = _backend.base_ptr();
944 std::size_t new_size = _backend.total_size();
945 if ( new_base == nullptr || new_size <= old_size )
946 return false;
947
948 hdr = get_header( new_base );
949
950 // Issue #146: compute extra_idx using address_traits::granule_size.
951 std::uint32_t extra_idx = detail::byte_off_to_idx_t<address_traits>( old_size );
952 std::size_t extra_size = new_size - old_size;
953
954 void* last_blk_raw =
955 ( hdr->last_block_offset != detail::kNoBlock )
956 ? static_cast<void*>( new_base + static_cast<std::size_t>( hdr->last_block_offset ) * kGranSz )
957 : nullptr;
958
959 if ( last_blk_raw != nullptr && BlockState::get_weight( last_blk_raw ) == 0 )
960 {
961 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
962 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
963 free_block_tree::remove( new_base, hdr, loff );
964 hdr->total_size = new_size;
965 free_block_tree::insert( new_base, hdr, loff );
966 }
967 else
968 {
969 // Issue #146: minimum extra size check uses address_traits granule size.
970 if ( extra_size < sizeof( Block<address_traits> ) + kGranSz )
971 return false;
972 void* nb_blk = new_base + static_cast<std::size_t>( extra_idx ) * kGranSz;
973 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
974 if ( last_blk_raw != nullptr )
975 {
976 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
977 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
978 BlockState::init_fields( nb_blk,
979 /*prev*/ loff,
980 /*next*/ address_traits::no_block,
981 /*avl_height*/ 1,
982 /*weight*/ 0,
983 /*root_offset*/ 0 );
984 BlockState::set_next_offset_of( last_blk_raw, static_cast<index_type>( extra_idx ) );
985 }
986 else
987 {
988 BlockState::init_fields( nb_blk,
989 /*prev*/ address_traits::no_block,
990 /*next*/ address_traits::no_block,
991 /*avl_height*/ 1,
992 /*weight*/ 0,
993 /*root_offset*/ 0 );
994 hdr->first_block_offset = extra_idx;
995 }
996 hdr->last_block_offset = extra_idx;
997 hdr->block_count++;
998 hdr->free_count++;
999 hdr->total_size = new_size;
1000 free_block_tree::insert( new_base, hdr, extra_idx );
1001 }
1002 return true;
1003 }
