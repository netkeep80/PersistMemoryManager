126 static bool create( std::size_t initial_size ) noexcept
127 {
128 typename thread_policy::unique_lock_type lock( _mutex );
129 if ( initial_size < detail::kMinMemorySize )
130 return false;
131 std::size_t aligned = ( ( initial_size + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
132 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
133 {
134 // Либо буфера нет, либо он меньше требуемого — расширяем
135 std::size_t additional =
136 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
137 if ( !_backend.expand( additional ) )
138 return false;
139 }
140 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
141 return false;
142 return init_layout( _backend.base_ptr(), _backend.total_size() );
143 }

163 static bool load() noexcept
164 {
165 typename thread_policy::unique_lock_type lock( _mutex );
166 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
167 return false;
168 std::uint8_t* base = _backend.base_ptr();
169 detail::ManagerHeader* hdr = get_header( base );
170 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )
171 return false;
172 if ( hdr->granule_size != static_cast<std::uint16_t>( kGranuleSize ) )
173 return false;
174 hdr->owns_memory = hdr->prev_owns_memory = false;
175 hdr->prev_total_size = 0;
176 hdr->prev_base_ptr = nullptr;
177 allocator::repair_linked_list( base, hdr );
178 allocator::recompute_counters( base, hdr );
179 allocator::rebuild_free_tree( base, hdr );
180 _initialized = true;
181 return true;
182 }

211 static void* allocate( std::size_t user_size ) noexcept
212 {
213 typename thread_policy::unique_lock_type lock( _mutex );
214 if ( !_initialized || user_size == 0 )
215 return nullptr;
216
217 std::uint8_t* base = _backend.base_ptr();
218 detail::ManagerHeader* hdr = get_header( base );
219 std::uint32_t needed = detail::required_block_granules( user_size );
220 std::uint32_t idx = free_block_tree::find_best_fit( base, hdr, needed );
221
222 if ( idx != detail::kNoBlock )
223 return allocator::allocate_from_block( base, hdr, idx, user_size );
224
225 // Попытка расширить (если бэкенд поддерживает)
226 if ( !do_expand( user_size ) )
227 return nullptr;
228
229 base = _backend.base_ptr();
230 hdr = get_header( base );
231 idx = free_block_tree::find_best_fit( base, hdr, needed );
232 if ( idx != detail::kNoBlock )
233 return allocator::allocate_from_block( base, hdr, idx, user_size );
234 return nullptr;
235 }

240 static void deallocate( void* ptr ) noexcept
241 {
242 typename thread_policy::unique_lock_type lock( _mutex );
243 if ( !_initialized || ptr == nullptr )
244 return;
245 std::uint8_t* base = _backend.base_ptr();
246 detail::ManagerHeader* hdr = get_header( base );
247 pmm::Block<address_traits>* blk =
248 detail::header_from_ptr( base, ptr, static_cast<std::size_t>( hdr->total_size ) );
249 if ( blk == nullptr || blk->weight == 0 )
250 return;
251
252 std::uint32_t freed = blk->weight;
253 std::uint32_t blk_idx = detail::block_idx( base, blk );
254
255 AllocatedBlock<address_traits>* alloc = AllocatedBlock<address_traits>::cast_from_raw( blk );
256 alloc->mark_as_free();
257
258 hdr->alloc_count--;
259 hdr->free_count++;
260 if ( hdr->used_size >= freed )
261 hdr->used_size -= freed;
262 allocator::coalesce( base, hdr, blk_idx );
263 }

402 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
403 {
404 typename thread_policy::shared_lock_type lock( _mutex );
405 if ( !_initialized )
406 return false;
407 const std::uint8_t* base = _backend.base_ptr();
408 const detail::ManagerHeader* hdr = get_header_c( base );
409 std::uint32_t idx = hdr->first_block_offset;
410 while ( idx != detail::kNoBlock )
411 {
412 if ( detail::idx_to_byte_off( idx ) + sizeof( Block<address_traits> ) > hdr->total_size )
413 break;
414 const Block<address_traits>* blk =
415 reinterpret_cast<const Block<address_traits>*>( base + detail::idx_to_byte_off( idx ) );
416 std::uint32_t total_gran = detail::block_total_granules( base, hdr, blk );
417 bool is_used = ( blk->weight > 0 );
418 std::size_t hdr_bytes = sizeof( Block<address_traits> );
419 std::size_t data_bytes = is_used ? detail::granules_to_bytes( blk->weight ) : 0;
420
421 BlockView view;
422 view.index = idx;
423 view.offset = static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( idx ) );
424 view.total_size = detail::granules_to_bytes( total_gran );
425 view.header_size = hdr_bytes;
426 view.user_size = data_bytes;
427 view.alignment = kGranuleSize;
428 view.used = is_used;
429 callback( view );
430 idx = blk->next_offset;
431 }
432 return true;
433 }

478 template <typename Callback>
479 static void for_each_free_block_inorder( const std::uint8_t* base, const detail::ManagerHeader* hdr,
480 std::uint32_t node_idx, int depth, Callback&& callback ) noexcept
481 {
482 if ( node_idx == detail::kNoBlock )
483 return;
484 if ( detail::idx_to_byte_off( node_idx ) + sizeof( Block<address_traits> ) > hdr->total_size )
485 return;
486 const Block<address_traits>* blk =
487 reinterpret_cast<const Block<address_traits>*>( base + detail::idx_to_byte_off( node_idx ) );
488
489 // Visit left subtree first (smaller blocks)
490 for_each_free_block_inorder( base, hdr, blk->left_offset, depth + 1, callback );
491
492 // Visit current node
493 std::uint32_t total_gran = detail::block_total_granules( base, hdr, blk );
494 FreeBlockView view;
495 view.offset = static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( node_idx ) );
496 view.total_size = detail::granules_to_bytes( total_gran );
497 view.free_size = detail::granules_to_bytes( total_gran - detail::kBlockHeaderGranules );
498 view.left_offset = ( blk->left_offset != detail::kNoBlock )
499 ? static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( blk->left_offset ) )
500 : -1;
501 view.right_offset = ( blk->right_offset != detail::kNoBlock )
502 ? static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( blk->right_offset ) )
503 : -1;
504 view.parent_offset = ( blk->parent_offset != detail::kNoBlock )
505 ? static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( blk->parent_offset ) )
506 : -1;
507 view.avl_height = blk->avl_height;
508 view.avl_depth = depth;
509 callback( view );
510
511 // Visit right subtree (larger blocks)
512 for_each_free_block_inorder( base, hdr, blk->right_offset, depth + 1, callback );
513 }

525 static bool init_layout( std::uint8_t* base, std::size_t size ) noexcept
526 {
527 static constexpr std::uint32_t kHdrBlkIdx = 0;
528 static constexpr std::uint32_t kFreeBlkIdx = detail::kBlockHeaderGranules + detail::kManagerHeaderGranules;
529
530 if ( detail::idx_to_byte_off( kFreeBlkIdx ) + sizeof( Block<address_traits> ) + detail::kMinBlockSize > size )
531 return false;
532
533 Block<address_traits>* hdr_blk = reinterpret_cast<Block<address_traits>*>( base );
534 std::memset( hdr_blk, 0, sizeof( Block<address_traits> ) );
535 hdr_blk->weight = detail::kManagerHeaderGranules;
536 hdr_blk->prev_offset = detail::kNoBlock;
537 hdr_blk->next_offset = kFreeBlkIdx;
538 hdr_blk->left_offset = detail::kNoBlock;
539 hdr_blk->right_offset = detail::kNoBlock;
540 hdr_blk->parent_offset = detail::kNoBlock;
541 hdr_blk->avl_height = 0;
542 hdr_blk->root_offset = kHdrBlkIdx;
543
544 detail::ManagerHeader* hdr = get_header( base );
545 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
546 hdr->magic = kMagic;
547 hdr->total_size = size;
548 hdr->first_block_offset = kHdrBlkIdx;
549 hdr->last_block_offset = detail::kNoBlock;
550 hdr->free_tree_root = detail::kNoBlock;
551 hdr->granule_size = static_cast<std::uint16_t>( kGranuleSize );
552
553 Block<address_traits>* blk =
554 reinterpret_cast<Block<address_traits>*>( base + detail::idx_to_byte_off( kFreeBlkIdx ) );
555 std::memset( blk, 0, sizeof( Block<address_traits> ) );
556 blk->prev_offset = kHdrBlkIdx;
557 blk->next_offset = detail::kNoBlock;
558 blk->left_offset = detail::kNoBlock;
559 blk->right_offset = detail::kNoBlock;
560 blk->parent_offset = detail::kNoBlock;
561 blk->avl_height = 1;
562 blk->weight = 0;
563 blk->root_offset = 0;
564
565 hdr->last_block_offset = kFreeBlkIdx;
566 hdr->free_tree_root = kFreeBlkIdx;
567 hdr->block_count = 2;
568 hdr->free_count = 1;
569 hdr->alloc_count = 1;
570 hdr->used_size = kFreeBlkIdx + detail::kBlockHeaderGranules;
571
572 _initialized = true;
573 return true;
574 }

576 static bool do_expand( std::size_t user_size ) noexcept
577 {
578 if ( !_initialized )
579 return false;
580 std::uint8_t* base = _backend.base_ptr();
581 detail::ManagerHeader* hdr = get_header( base );
582 std::size_t old_size = hdr->total_size;
583
584 std::size_t min_need =
585 detail::granules_to_bytes( detail::required_block_granules( user_size ) + detail::kBlockHeaderGranules );
586 std::size_t growth = old_size / 4;
587 if ( growth < min_need )
588 growth = min_need;
589
590 if ( !_backend.expand( growth ) )
591 return false;
592
593 std::uint8_t* new_base = _backend.base_ptr();
594 std::size_t new_size = _backend.total_size();
595 if ( new_base == nullptr || new_size <= old_size )
596 return false;
597
598 hdr = get_header( new_base );
599
600 std::uint32_t extra_idx = detail::byte_off_to_idx( old_size );
601 std::size_t extra_size = new_size - old_size;
602
603 Block<address_traits>* last_blk = ( hdr->last_block_offset != detail::kNoBlock )
604 ? reinterpret_cast<Block<address_traits>*>(
605 new_base + detail::idx_to_byte_off( hdr->last_block_offset ) )
606 : nullptr;
607
608 if ( last_blk != nullptr && last_blk->weight == 0 )
609 {
610 std::uint32_t loff = detail::block_idx( new_base, last_blk );
611 free_block_tree::remove( new_base, hdr, loff );
612 hdr->total_size = new_size;
613 free_block_tree::insert( new_base, hdr, loff );
614 }
615 else
616 {
617 if ( extra_size < sizeof( Block<address_traits> ) + detail::kMinBlockSize )
618 return false;
619 Block<address_traits>* nb_blk =
620 reinterpret_cast<Block<address_traits>*>( new_base + detail::idx_to_byte_off( extra_idx ) );
621 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
622 nb_blk->left_offset = detail::kNoBlock;
623 nb_blk->right_offset = detail::kNoBlock;
624 nb_blk->parent_offset = detail::kNoBlock;
625 nb_blk->avl_height = 1;
626 nb_blk->weight = 0;
627 nb_blk->root_offset = 0;
628 if ( last_blk != nullptr )
629 {
630 std::uint32_t loff = detail::block_idx( new_base, last_blk );
631 nb_blk->prev_offset = loff;
632 nb_blk->next_offset = detail::kNoBlock;
633 last_blk->next_offset = extra_idx;
634 }
635 else
636 {
637 nb_blk->prev_offset = detail::kNoBlock;
638 nb_blk->next_offset = detail::kNoBlock;
639 hdr->first_block_offset = extra_idx;
640 }
641 hdr->last_block_offset = extra_idx;
642 hdr->block_count++;
643 hdr->free_count++;
644 hdr->total_size = new_size;
645 free_block_tree::insert( new_base, hdr, extra_idx );
646 }
647 return true;
648 }
