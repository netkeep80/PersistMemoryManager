169 static bool create( std::size_t initial_size ) noexcept
170 {
171 typename thread_policy::unique_lock_type lock( _mutex );
172 if ( initial_size < detail::kMinMemorySize )
173 return false;
174 // Issue #146: use address_traits::granule_size instead of hardcoded kGranuleSize.
175 // Issue #172: guard against overflow when initial_size is close to size_t max.
176 static constexpr std::size_t kGranSzCreate = address_traits::granule_size;
177 if ( initial_size > std::numeric_limits<std::size_t>::max() - ( kGranSzCreate - 1 ) )
178 return false; // overflow in alignment computation
179 std::size_t aligned = ( ( initial_size + kGranSzCreate - 1 ) / kGranSzCreate ) * kGranSzCreate;
180 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
181 {
182 // Либо буфера нет, либо он меньше требуемого — расширяем
183 std::size_t additional =
184 ( _backend.total_size() < aligned ) ? ( aligned - _backend.total_size() ) : aligned;
185 if ( !_backend.expand( additional ) )
186 return false;
187 }
188 if ( _backend.base_ptr() == nullptr || _backend.total_size() < aligned )
189 return false;
190 return init_layout( _backend.base_ptr(), _backend.total_size() );
191 }

211 static bool load() noexcept
212 {
213 typename thread_policy::unique_lock_type lock( _mutex );
214 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
215 return false;
216 std::uint8_t* base = _backend.base_ptr();
217 detail::ManagerHeader<address_traits>* hdr = get_header( base );
218 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )
219 return false;

221 if ( hdr->granule_size != static_cast<std::uint16_t>( address_traits::granule_size ) )
222 return false;

224 hdr->prev_total_size = 0;
225 allocator::repair_linked_list( base, hdr );
226 allocator::recompute_counters( base, hdr );
227 allocator::rebuild_free_tree( base, hdr );
228 _initialized = true;
229 return true;

260 static void* allocate( std::size_t user_size ) noexcept
261 {
262 typename thread_policy::unique_lock_type lock( _mutex );
263 if ( !_initialized || user_size == 0 )
264 return nullptr;
265
266 std::uint8_t* base = _backend.base_ptr();
267 detail::ManagerHeader<address_traits>* hdr = get_header( base );
268 // Issue #146: use AddressTraits-specific granule size for required granule computation.
269 index_type data_gran = detail::bytes_to_granules_t<address_traits>( user_size );
270 if ( data_gran == 0 )
271 data_gran = 1;
272 index_type needed = kBlockHdrGranules + data_gran;
273 index_type idx = free_block_tree::find_best_fit( base, hdr, needed );
274
275 if ( idx != address_traits::no_block )
276 return allocator::allocate_from_block( base, hdr, idx, user_size );
277
278 // Попытка расширить (если бэкенд поддерживает)
279 if ( !do_expand( user_size ) )
280 return nullptr;
281
282 base = _backend.base_ptr();
283 hdr = get_header( base );
284 idx = free_block_tree::find_best_fit( base, hdr, needed );
285 if ( idx != address_traits::no_block )
286 return allocator::allocate_from_block( base, hdr, idx, user_size );
287 return nullptr;

295 static void deallocate( void* ptr ) noexcept
296 {
297 typename thread_policy::unique_lock_type lock( _mutex );
298 if ( !_initialized || ptr == nullptr )
299 return;
300 pmm::Block<address_traits>* blk = find_block_from_user_ptr( ptr );
301 if ( blk == nullptr )
302 return;
303 index_type freed = BlockStateBase<address_traits>::get_weight( blk );
304 if ( freed == 0 )
305 return;
306
307 // Issue #126: Permanently locked blocks cannot be freed.
308 if ( BlockStateBase<address_traits>::get_node_type( blk ) == pmm::kNodeReadOnly )
309 return;
310
311 std::uint8_t* base = _backend.base_ptr();
312 detail::ManagerHeader<address_traits>* hdr = get_header( base );
313 index_type blk_idx = detail::block_idx_t<address_traits>( base, blk );
314
315 AllocatedBlock<address_traits>* alloc = AllocatedBlock<address_traits>::cast_from_raw( blk );
316 alloc->mark_as_free();
317
318 hdr->alloc_count--;
319 hdr->free_count++;
320 if ( hdr->used_size >= freed )
321 hdr->used_size -= freed;
322 allocator::coalesce( base, hdr, blk_idx );
323 }

804 template <typename Callback> static bool for_each_block( Callback&& callback ) noexcept
805 {
806 typename thread_policy::shared_lock_type lock( _mutex );
807 if ( !_initialized )
808 return false;
809 const std::uint8_t* base = _backend.base_ptr();
810 using BlockState = BlockStateBase<address_traits>;
811 const detail::ManagerHeader<address_traits>* hdr = get_header_c( base );
812 index_type idx = hdr->first_block_offset;
813 // Issue #146: use address_traits::granule_size for correct byte offset computations.
814 static constexpr std::size_t kGranSz = address_traits::granule_size;
815 while ( idx != address_traits::no_block )
816 {
817 if ( static_cast<std::size_t>( idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
818 break;
819 const void* blk_raw = base + static_cast<std::size_t>( idx ) * kGranSz;
820 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
821 index_type total_gran = detail::block_total_granules( base, hdr, blk );
822 auto w = BlockState::get_weight( blk_raw );
823 bool is_used = ( w > 0 );
824 std::size_t hdr_bytes = sizeof( Block<address_traits> );
825 std::size_t data_bytes = is_used ? static_cast<std::size_t>( w ) * kGranSz : 0;
826
827 BlockView view;
828 view.index = idx;
829 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( idx ) * kGranSz );
830 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
831 view.header_size = hdr_bytes;
832 view.user_size = data_bytes;
833 view.alignment = kGranSz;
834 view.used = is_used;
835 callback( view );
836 idx = BlockState::get_next_offset( blk_raw );
837 }
838 return true;
839 }

920 // ─── Issue #179: blk_raw helpers ──────────────────────────────────────────

922 // Formula: base + offset * granule_size - sizeof(Block<AT>) points to the

943 template <typename Callback>
944 static void for_each_free_block_inorder( const std::uint8_t* base, const detail::ManagerHeader<address_traits>* hdr,
945 index_type node_idx, int depth, Callback&& callback ) noexcept
946 {
947 using BlockState = BlockStateBase<address_traits>;
948 // Issue #146: use address_traits::granule_size for correct byte offset computations.
949 static constexpr std::size_t kGranSz = address_traits::granule_size;
950 if ( node_idx == address_traits::no_block )
951 return;
952 if ( static_cast<std::size_t>( node_idx ) * kGranSz + sizeof( Block<address_traits> ) > hdr->total_size )
953 return;
954 const void* blk_raw = base + static_cast<std::size_t>( node_idx ) * kGranSz;
955 const Block<address_traits>* blk = reinterpret_cast<const Block<address_traits>*>( blk_raw );
956
957 index_type left_off = BlockState::get_left_offset( blk_raw );
958 index_type right_off = BlockState::get_right_offset( blk_raw );
959 index_type parent_off = BlockState::get_parent_offset( blk_raw );
960
961 // Visit left subtree first (smaller blocks)
962 for_each_free_block_inorder( base, hdr, left_off, depth + 1, callback );
963
964 // Visit current node
965 index_type total_gran = detail::block_total_granules( base, hdr, blk );
966 FreeBlockView view;
967 view.offset = static_cast<std::ptrdiff_t>( static_cast<std::size_t>( node_idx ) * kGranSz );
968 view.total_size = static_cast<std::size_t>( total_gran ) * kGranSz;
969 view.free_size = static_cast<std::size_t>( total_gran - kBlockHdrGranules ) * kGranSz;
970 view.left_offset = ( left_off != address_traits::no_block )
971 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( left_off ) * kGranSz )
972 : -1;
973 view.right_offset = ( right_off != address_traits::no_block )
974 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( right_off ) * kGranSz )
975 : -1;
976 view.parent_offset = ( parent_off != address_traits::no_block )
977 ? static_cast<std::ptrdiff_t>( static_cast<std::size_t>( parent_off ) * kGranSz )
978 : -1;
979 view.avl_height = BlockState::get_avl_height( blk_raw );
980 view.avl_depth = depth;
981 callback( view );
982
983 // Visit right subtree (larger blocks)
984 for_each_free_block_inorder( base, hdr, right_off, depth + 1, callback );
985 }

987 // ─── Address-traits-specific layout constants (Issue #146) ──────────────────
988 // These compute the correct granule indices based on the actual address_traits
989 // granule size, rather than using the hardcoded DefaultAddressTraits constants.
990
993 static constexpr std::size_t kBlockHdrByteSize =
994 ( ( sizeof( Block<address_traits> ) + address_traits::granule_size - 1 ) / address_traits::granule_size ) *
995 address_traits::granule_size;
996
998 static constexpr index_type kBlockHdrGranules =
999 static_cast<index_type>( kBlockHdrByteSize / address_traits::granule_size );
1000
1003 static constexpr index_type kMgrHdrGranules = detail::kManagerHeaderGranules_t<address_traits>;
1004
1006 static constexpr index_type kFreeBlkIdxLayout = kBlockHdrGranules + kMgrHdrGranules;

1008 static detail::ManagerHeader<address_traits>* get_header( std::uint8_t* base ) noexcept
1009 {
1010 // Place ManagerHeader at a granule-aligned offset after Block_0 (Issue #146).
1011 return reinterpret_cast<detail::ManagerHeader<address_traits>*>( base + kBlockHdrByteSize );
1012 }

1019 static bool init_layout( std::uint8_t* base, std::size_t size ) noexcept
1020 {
1021 using BlockState = BlockStateBase<address_traits>;
1022 static constexpr index_type kHdrBlkIdx = 0;
1023 static constexpr index_type kFreeBlkIdx = kFreeBlkIdxLayout;
1024 static constexpr std::size_t kGranSz = address_traits::granule_size;
1025
1026 // Minimum size check: Block_0 + ManagerHeader + Block_1 + at least 1 data granule
1027 static constexpr std::size_t kMinBlockDataSize = kGranSz; // 1 data granule
1028 if ( static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz + sizeof( Block<address_traits> ) + kMinBlockDataSize >
1029 size )
1030 return false;
1031
1032 // Инициализация блока-заголовка (Block_0) через state machine утилиты
1033 void* hdr_blk = base;
1034 std::memset( hdr_blk, 0, kBlockHdrByteSize ); // zero entire aligned region (including padding)
1035 BlockState::init_fields( hdr_blk,
1036 /*prev*/ address_traits::no_block,
1037 /*next*/ kFreeBlkIdx,
1038 /*avl_height*/ 0,
1039 /*weight*/ kMgrHdrGranules,
1040 /*root_offset*/ kHdrBlkIdx );
1041
1042 detail::ManagerHeader<address_traits>* hdr = get_header( base );
1043 std::memset( hdr, 0, sizeof( detail::ManagerHeader<address_traits> ) );
1044 hdr->magic = kMagic;
1045 hdr->total_size = size;
1046 hdr->first_block_offset = kHdrBlkIdx;
1047 hdr->last_block_offset = address_traits::no_block;
1048 hdr->free_tree_root = address_traits::no_block;
1049 hdr->granule_size = static_cast<std::uint16_t>( kGranSz );
1050
1051 // Инициализация первого свободного блока через state machine утилиты
1052 void* blk = base + static_cast<std::size_t>( kFreeBlkIdx ) * kGranSz;
1053 std::memset( blk, 0, sizeof( Block<address_traits> ) );
1054 BlockState::init_fields( blk,
1055 /*prev*/ kHdrBlkIdx,
1056 /*next*/ address_traits::no_block,
1057 /*avl_height*/ 1,
1058 /*weight*/ 0,
1059 /*root_offset*/ 0 );
1060
1061 hdr->last_block_offset = kFreeBlkIdx;
1062 hdr->free_tree_root = kFreeBlkIdx;
1063 hdr->block_count = 2;
1064 hdr->free_count = 1;
1065 hdr->alloc_count = 1;
1066 hdr->used_size = kFreeBlkIdx + kBlockHdrGranules;
1067
1068 _initialized = true;
1069 return true;
1070 }
1071
1072 static bool do_expand( std::size_t user_size ) noexcept
1073 {
1074 using BlockState = BlockStateBase<address_traits>;
1075 if ( !_initialized )
1076 return false;
1077 std::uint8_t* base = _backend.base_ptr();
1078 detail::ManagerHeader<address_traits>* hdr = get_header( base );
1079 std::size_t old_size = hdr->total_size;
1080
1081 // Issue #146: use AddressTraitsT-specific granule size for all computations.
1082 static constexpr std::size_t kGranSz = address_traits::granule_size;
1083 index_type data_gran_need = detail::bytes_to_granules_t<address_traits>( user_size );
1084 if ( data_gran_need == 0 )
1085 data_gran_need = 1;
1086 // min_need = block_header + data + another block_header (for the new expand block)
1087 std::size_t min_need =
1088 static_cast<std::size_t>( kBlockHdrGranules + data_gran_need + kBlockHdrGranules ) * kGranSz;
1089 std::size_t growth = old_size / 4;
1090 if ( growth < min_need )
1091 growth = min_need;
1092
1093 if ( !_backend.expand( growth ) )
1094 return false;
1095
1096 std::uint8_t* new_base = _backend.base_ptr();
1097 std::size_t new_size = _backend.total_size();
1098 if ( new_base == nullptr || new_size <= old_size )
1099 return false;
1100
1101 hdr = get_header( new_base );
1102
1103 // Issue #146: compute extra_idx using address_traits::granule_size.
1104 index_type extra_idx = detail::byte_off_to_idx_t<address_traits>( old_size );
1105 std::size_t extra_size = new_size - old_size;
1106
1107 void* last_blk_raw =
1108 ( hdr->last_block_offset != address_traits::no_block )
1109 ? static_cast<void*>( new_base + static_cast<std::size_t>( hdr->last_block_offset ) * kGranSz )
1110 : nullptr;
1111
1112 if ( last_blk_raw != nullptr && BlockState::get_weight( last_blk_raw ) == 0 )
1113 {
1114 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1115 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1116 free_block_tree::remove( new_base, hdr, loff );
1117 hdr->total_size = new_size;
1118 free_block_tree::insert( new_base, hdr, loff );
1119 }
1120 else
1121 {
1122 // Issue #146: minimum extra size check uses address_traits granule size.
1123 if ( extra_size < sizeof( Block<address_traits> ) + kGranSz )
1124 return false;
1125 void* nb_blk = new_base + static_cast<std::size_t>( extra_idx ) * kGranSz;
1126 std::memset( nb_blk, 0, sizeof( Block<address_traits> ) );
1127 if ( last_blk_raw != nullptr )
1128 {
1129 Block<address_traits>* last_blk = reinterpret_cast<Block<address_traits>*>( last_blk_raw );
1130 index_type loff = detail::block_idx_t<address_traits>( new_base, last_blk );
1131 BlockState::init_fields( nb_blk,
1132 /*prev*/ loff,
1133 /*next*/ address_traits::no_block,
1134 /*avl_height*/ 1,
1135 /*weight*/ 0,
1136 /*root_offset*/ 0 );
1137 BlockState::set_next_offset_of( last_blk_raw, static_cast<index_type>( extra_idx ) );
1138 }
1139 else
1140 {
1141 BlockState::init_fields( nb_blk,
1142 /*prev*/ address_traits::no_block,
1143 /*next*/ address_traits::no_block,
1144 /*avl_height*/ 1,
1145 /*weight*/ 0,
1146 /*root_offset*/ 0 );
1147 hdr->first_block_offset = extra_idx;
1148 }
1149 hdr->last_block_offset = extra_idx;
1150 hdr->block_count++;
1151 hdr->free_count++;
1152 hdr->total_size = new_size;
1153 free_block_tree::insert( new_base, hdr, extra_idx );
1154 }
1155 return true;
1156 }
