219 static psview_pptr _intern( const char* s ) noexcept
220 {
221 if ( s == nullptr )
222 s = "";
223
224 // Ищем в AVL-дереве.
225 psview_pptr found = _avl_find( s );
226 if ( !found.is_null() )
227 return found;
228
229 // Не найдено — создаём новый объект pstringview.
230 auto len = static_cast<std::uint32_t>( std::strlen( s ) );
231
232 // Выделяем один блок для pstringview + строковых данных.
233 // Размер = offsetof(pstringview, str) + len + 1 (null-terminator).
234 // Используем offsetof для корректного вычисления без учёта str[1].
235 std::size_t alloc_size = offsetof( pstringview, str ) + static_cast<std::size_t>( len ) + 1;
236
237 // Выделяем память через allocate() напрямую, т.к. размер переменный.
238 void* raw = ManagerT::allocate( alloc_size );
239 if ( raw == nullptr )
240 return psview_pptr();
241
242 // Создаём pptr вручную из raw указателя.
243 std::uint8_t* base = ManagerT::backend().base_ptr();
244 psview_pptr new_node( detail::ptr_to_granule_idx<typename ManagerT::address_traits>( base, raw ) );
245
246 pstringview* obj = static_cast<pstringview*>( raw );
247 obj->length = len;
248 // Копируем строку включая null-terminator.
249 std::memcpy( obj->str, s, static_cast<std::size_t>( len ) + 1 );
250
251 // Инициализируем AVL-поля нового узла.
252 detail::avl_init_node( new_node );
253
254 // Блокируем блок pstringview навечно.
255 ManagerT::lock_block_permanent( obj );
256
257 // Вставляем в AVL-дерево.
258 _avl_insert( new_node );
259
260 return new_node;
261 }

267 {

269 if ( domain == nullptr )

272 domain->root_offset,

276 return ( obj != nullptr ) ? std::strcmp( s, obj->c_str() ) : 0;
277 },

282 static void _avl_insert( psview_pptr new_node ) noexcept
283 {
284 auto* domain = ManagerT::symbol_domain_record_unlocked();
285 if ( domain == nullptr )
286 return;
287 pstringview* new_obj = ManagerT::template resolve<pstringview>( new_node );
288 const char* new_str = ( new_obj != nullptr ) ? new_obj->c_str() : "";
289 detail::avl_insert(
290 new_node, domain->root_offset,
291 [&]( psview_pptr cur ) -> bool
292 {
293 pstringview* obj = ManagerT::template resolve<pstringview>( cur );
294 return ( obj != nullptr ) && ( std::strcmp( new_str, obj->c_str() ) < 0 );
295 },
