196 static psview_pptr _intern( const char* s ) noexcept
197 {
198 if ( s == nullptr )
199 s = "";
200
201 // Ищем в AVL-дереве.
202 psview_pptr found = _avl_find( s );
203 if ( !found.is_null() )
204 return found;
205
206 // Не найдено — создаём новый объект pstringview.
207 auto len = static_cast<std::uint32_t>( std::strlen( s ) );
208
209 // Создаём char[] в ПАП и блокируем навечно.
210 index_type new_chars = _create_chars( s, len );
211 if ( new_chars == static_cast<index_type>( 0 ) && len > 0 )
212 return psview_pptr();
213
214 // Создаём объект pstringview в ПАП.
215 psview_pptr new_node = ManagerT::template allocate_typed<pstringview>();
216 if ( new_node.is_null() )
217 return psview_pptr();
218
219 pstringview* obj = ManagerT::template resolve<pstringview>( new_node );
220 if ( obj == nullptr )
221 return psview_pptr();
222 obj->chars_idx = new_chars;
223 obj->length = len;
224
225 // Инициализируем AVL-поля нового узла (пустые ссылки, высота 1).
226 auto& tn = new_node.tree_node();
227 tn.set_left( static_cast<index_type>( 0 ) );
228 tn.set_right( static_cast<index_type>( 0 ) );
229 tn.set_parent( static_cast<index_type>( 0 ) );
230 tn.set_height( static_cast<std::int16_t>( 1 ) );
231
232 // Блокируем блок pstringview навечно (Issue #151, Issue #126).
233 ManagerT::lock_block_permanent( obj );
234
235 // Вставляем в AVL-дерево.
236 _avl_insert( new_node );
237
238 return new_node;
239 }

244 static index_type _create_chars( const char* s, std::uint32_t len ) noexcept
245 {
246 if ( len == 0 )
247 {
248 // Пустая строка: выделяем один байт для нулевого терминатора.
249 char_pptr arr = ManagerT::template allocate_typed<char>( 1 );
250 if ( arr.is_null() )
251 return static_cast<index_type>( 0 );
252 char* dst = ManagerT::template resolve<char>( arr );
253 if ( dst != nullptr )
254 dst[0] = '\0';
255 if ( dst != nullptr )
256 ManagerT::lock_block_permanent( dst );
257 return arr.offset();
258 }
259
260 char_pptr arr = ManagerT::template allocate_typed<char>( static_cast<std::size_t>( len + 1 ) );
261 if ( arr.is_null() )
262 return static_cast<index_type>( 0 );
263 char* dst = ManagerT::template resolve<char>( arr );
264 if ( dst != nullptr )
265 std::memcpy( dst, s, static_cast<std::size_t>( len + 1 ) );
266 if ( dst != nullptr )
267 ManagerT::lock_block_permanent( dst );
268 return arr.offset();
269 }

388 static void _rebalance_up( psview_pptr p ) noexcept
389 {
390 while ( !p.is_null() )
391 {
392 _update_height( p );
393 std::int16_t bf = _balance_factor( p );
394 if ( bf > 1 )
395 {
396 // Левое поддерево перевешивает.
397 psview_pptr left( p.get_tree_left().offset() );
398 if ( _balance_factor( left ) < 0 )
399 {
400 // LR-случай: сначала левый поворот левого потомка.
401 _rotate_left( left );
402 }
403 p = _rotate_right( p );
404 }
405 else if ( bf < -1 )
406 {
407 // Правое поддерево перевешивает.
408 psview_pptr right( p.get_tree_right().offset() );
409 if ( _balance_factor( right ) > 0 )
410 {
411 // RL-случай: сначала правый поворот правого потомка.
412 _rotate_right( right );
413 }
414 p = _rotate_left( p );
415 }
416 // Переходим к родителю.
417 p = psview_pptr( p.get_tree_parent().offset() );
418 }

422 static psview_pptr _avl_find( const char* s ) noexcept
423 {
424 psview_pptr cur( _root_idx );
425 while ( !cur.is_null() )
426 {
427 pstringview* obj = ManagerT::template resolve<pstringview>( cur );
428 if ( obj == nullptr )
429 break;
430 int cmp = std::strcmp( s, obj->c_str() );
431 if ( cmp == 0 )
432 return cur;
433 else if ( cmp < 0 )
434 cur = psview_pptr( cur.get_tree_left().offset() );
435 else
436 cur = psview_pptr( cur.get_tree_right().offset() );
437 }
438 return psview_pptr(); // null
439 }

442 static void _avl_insert( psview_pptr new_node ) noexcept
443 {
444 if ( new_node.is_null() )
445 return;
446
447 pstringview* new_obj = ManagerT::template resolve<pstringview>( new_node );
448 if ( new_obj == nullptr )
449 return;
450 const char* new_str = new_obj->c_str();
451
452 if ( _root_idx == static_cast<index_type>( 0 ) )
453 {
454 // Дерево пустое — новый узел становится корнем.
455 new_node.set_tree_left( psview_pptr() );
456 new_node.set_tree_right( psview_pptr() );
457 new_node.set_tree_parent( psview_pptr() );
458 new_node.set_tree_height( static_cast<std::int16_t>( 1 ) );
459 _root_idx = new_node.offset();
460 return;
461 }
462
463 // Ищем место для вставки.
464 psview_pptr cur( _root_idx );
465 psview_pptr parent;
466 bool go_left = false;
467
468 while ( !cur.is_null() )
469 {
470 pstringview* obj = ManagerT::template resolve<pstringview>( cur );
471 if ( obj == nullptr )
472 break;
473 parent = cur;
474 int cmp = std::strcmp( new_str, obj->c_str() );
475 go_left = ( cmp < 0 );
476 if ( go_left )
477 cur = psview_pptr( cur.get_tree_left().offset() );
478 else
479 cur = psview_pptr( cur.get_tree_right().offset() );
480 }
481
482 // Устанавливаем родителя нового узла.
483 new_node.set_tree_parent( parent );
484
485 // Прикрепляем новый узел к родителю.
486 if ( go_left )
487 parent.set_tree_left( new_node );
488 else
489 parent.set_tree_right( new_node );
490
491 // Ребалансировка вверх от родителя.
492 _rebalance_up( parent );
493 }
