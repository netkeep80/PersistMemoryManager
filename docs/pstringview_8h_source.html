206 static psview_pptr _intern( const char* s ) noexcept
207 {
208 if ( s == nullptr )
209 s = "";
210
211 // Ищем в AVL-дереве.
212 psview_pptr found = _avl_find( s );
213 if ( !found.is_null() )
214 return found;
215
216 // Не найдено — создаём новый объект pstringview.
217 auto len = static_cast<std::uint32_t>( std::strlen( s ) );
218
219 // Issue #184: выделяем один блок для pstringview + строковых данных.
220 // Размер = offsetof(pstringview, str) + len + 1 (null-terminator).
221 // Используем offsetof для корректного вычисления без учёта str[1].
222 std::size_t alloc_size = offsetof( pstringview, str ) + static_cast<std::size_t>( len ) + 1;
223
224 // Выделяем память через allocate() напрямую, т.к. размер переменный.
225 void* raw = ManagerT::allocate( alloc_size );
226 if ( raw == nullptr )
227 return psview_pptr();
228
229 // Создаём pptr вручную из raw указателя.
230 std::uint8_t* base = ManagerT::backend().base_ptr();
231 std::size_t byte_off = static_cast<std::uint8_t*>( raw ) - base;
232 psview_pptr new_node( static_cast<index_type>( byte_off / ManagerT::address_traits::granule_size ) );
233
234 pstringview* obj = static_cast<pstringview*>( raw );
235 obj->length = len;
236 // Копируем строку включая null-terminator.
237 std::memcpy( obj->str, s, static_cast<std::size_t>( len ) + 1 );
238
239 // Инициализируем AVL-поля нового узла (пустые ссылки, высота 1).
240 auto& tn = new_node.tree_node();
241 tn.set_left( static_cast<index_type>( 0 ) );
242 tn.set_right( static_cast<index_type>( 0 ) );
243 tn.set_parent( static_cast<index_type>( 0 ) );
244 tn.set_height( static_cast<std::int16_t>( 1 ) );
245
246 // Блокируем блок pstringview навечно (Issue #151, Issue #126).
247 ManagerT::lock_block_permanent( obj );
248
249 // Вставляем в AVL-дерево.
250 _avl_insert( new_node );
251
252 return new_node;
253 }

258 static psview_pptr _avl_find( const char* s ) noexcept
259 {

261 _root_idx,

265 return ( obj != nullptr ) ? std::strcmp( s, obj->c_str() ) : 0;
266 },

268 }

271 static void _avl_insert( psview_pptr new_node ) noexcept
272 {
273 pstringview* new_obj = ManagerT::template resolve<pstringview>( new_node );
274 const char* new_str = ( new_obj != nullptr ) ? new_obj->c_str() : "";
275 detail::avl_insert(
276 new_node, _root_idx,
277 [&]( psview_pptr cur ) -> bool
278 {
279 pstringview* obj = ManagerT::template resolve<pstringview>( cur );
280 return ( obj != nullptr ) && ( std::strcmp( new_str, obj->c_str() ) < 0 );
281 },
282 []( psview_pptr p ) -> pstringview* { return ManagerT::template resolve<pstringview>( p ); } );
