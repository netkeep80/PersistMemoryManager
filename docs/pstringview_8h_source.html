199 static psview_pptr _intern( const char* s ) noexcept
200 {
201 if ( s == nullptr )
202 s = "";
203
204 // Ищем в AVL-дереве.
205 psview_pptr found = _avl_find( s );
206 if ( !found.is_null() )
207 return found;
208
209 // Не найдено — создаём новый объект pstringview.
210 auto len = static_cast<std::uint32_t>( std::strlen( s ) );
211
212 // Создаём char[] в ПАП и блокируем навечно.
213 index_type new_chars = _create_chars( s, len );
214 if ( new_chars == static_cast<index_type>( 0 ) && len > 0 )
215 return psview_pptr();
216
217 // Создаём объект pstringview в ПАП.
218 psview_pptr new_node = ManagerT::template allocate_typed<pstringview>();
219 if ( new_node.is_null() )
220 return psview_pptr();
221
222 pstringview* obj = ManagerT::template resolve<pstringview>( new_node );
223 if ( obj == nullptr )
224 return psview_pptr();
225 obj->chars_idx = new_chars;
226 obj->length = len;
227
228 // Инициализируем AVL-поля нового узла (пустые ссылки, высота 1).
229 auto& tn = new_node.tree_node();
230 tn.set_left( static_cast<index_type>( 0 ) );
231 tn.set_right( static_cast<index_type>( 0 ) );
232 tn.set_parent( static_cast<index_type>( 0 ) );
233 tn.set_height( static_cast<std::int16_t>( 1 ) );
234
235 // Блокируем блок pstringview навечно (Issue #151, Issue #126).
236 ManagerT::lock_block_permanent( obj );
237
238 // Вставляем в AVL-дерево.
239 _avl_insert( new_node );
240
241 return new_node;
242 }

247 static index_type _create_chars( const char* s, std::uint32_t len ) noexcept
248 {
249 if ( len == 0 )
250 {
251 // Пустая строка: выделяем один байт для нулевого терминатора.
252 char_pptr arr = ManagerT::template allocate_typed<char>( 1 );
253 if ( arr.is_null() )
254 return static_cast<index_type>( 0 );
255 char* dst = ManagerT::template resolve<char>( arr );
256 if ( dst != nullptr )
257 dst[0] = '\0';
258 if ( dst != nullptr )
259 ManagerT::lock_block_permanent( dst );
260 return arr.offset();
261 }
262
263 char_pptr arr = ManagerT::template allocate_typed<char>( static_cast<std::size_t>( len + 1 ) );
264 if ( arr.is_null() )
265 return static_cast<index_type>( 0 );
266 char* dst = ManagerT::template resolve<char>( arr );
267 if ( dst != nullptr )
268 std::memcpy( dst, s, static_cast<std::size_t>( len + 1 ) );
269 if ( dst != nullptr )
270 ManagerT::lock_block_permanent( dst );
271 return arr.offset();
272 }

290 static void _avl_insert( psview_pptr new_node ) noexcept
291 {
292 pstringview* new_obj = ManagerT::template resolve<pstringview>( new_node );
293 const char* new_str = ( new_obj != nullptr ) ? new_obj->c_str() : "";
294 detail::avl_insert(
295 new_node, _root_idx,
296 [&]( psview_pptr cur ) -> bool
297 {
298 pstringview* obj = ManagerT::template resolve<pstringview>( cur );
299 return ( obj != nullptr ) && ( std::strcmp( new_str, obj->c_str() ) < 0 );
300 },
301 []( psview_pptr p ) -> pstringview* { return ManagerT::template resolve<pstringview>( p ); } );
