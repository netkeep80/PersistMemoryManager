145 static constexpr index_type bytes_to_granules( std::size_t bytes ) noexcept
146 {
147 if ( bytes == 0 )
148 return static_cast<index_type>( 0 );
149 // Overflow-safe ceiling division: (bytes + granule_size - 1) / granule_size
150 if ( bytes > std::numeric_limits<std::size_t>::max() - ( granule_size - 1 ) )
151 return static_cast<index_type>( 0 ); // overflow
152 std::size_t granules = ( bytes + granule_size - 1 ) / granule_size;
153 if ( granules > static_cast<std::size_t>( std::numeric_limits<IndexT>::max() ) )
154 return static_cast<index_type>( 0 ); // overflow for IndexT
155 return static_cast<index_type>( granules );
156 }
157
161 static constexpr std::size_t granules_to_bytes( index_type granules ) noexcept
162 {
163 return static_cast<std::size_t>( granules ) * granule_size;
164 }
165
169 static constexpr std::size_t idx_to_byte_off( index_type idx ) noexcept
170 {
171 return static_cast<std::size_t>( idx ) * granule_size;
172 }
173
177 static index_type byte_off_to_idx( std::size_t byte_off ) noexcept
178 {
179 assert( byte_off % granule_size == 0 );
180 assert( byte_off / granule_size <= static_cast<std::size_t>( std::numeric_limits<IndexT>::max() ) );
181 return static_cast<index_type>( byte_off / granule_size );
182 }

1637 return false;

1640 auto next_off = BlockState::get_next_offset( blk );
1641 std::uint32_t total_gran = ( next_off != kNoBlock )
1642 ? ( next_off - idx )
1643 : ( byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - idx );
1644 if ( BlockState::get_weight( blk ) >= total_gran )
1645 return false;

1648 return false;
1649 if ( next_off != kNoBlock && next_off <= idx )
1650 return false;
1651 if ( BlockState::get_avl_height( blk ) >= 32 )
1652 return false;
1653 auto left_off = BlockState::get_left_offset( blk );
1654 auto right_off = BlockState::get_right_offset( blk );

1656 const bool l = ( left_off != kNoBlock );
1657 const bool r = ( right_off != kNoBlock );
1658 const bool p = ( parent_off != kNoBlock );
1659 if ( ( l || r || p ) && ( ( l && r && left_off == right_off ) || ( l && p && left_off == parent_off ) ||
1660 ( r && p && right_off == parent_off ) ) )
1661 return false;
1662 return true;
1663}

1787 static void insert( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
1788 {
1789 void* blk = blk_at( base, blk_idx );
1790 BlockState::set_left_offset_of( blk, detail::kNoBlock );
1791 BlockState::set_right_offset_of( blk, detail::kNoBlock );
1792 BlockState::set_parent_offset_of( blk, detail::kNoBlock );
1793 BlockState::set_avl_height_of( blk, 1 );
1794 if ( hdr->free_tree_root == detail::kNoBlock )
1795 {
1796 hdr->free_tree_root = blk_idx;
1797 return;
1798 }
1799 // Issue #59: cache total_gran once; compute blk size in granules before the traversal loop
1800 std::uint32_t total_gran = detail::byte_off_to_idx( hdr->total_size );
1801 index_type blk_next = BlockState::get_next_offset( blk );
1802 std::uint32_t blk_gran = ( blk_next != detail::kNoBlock ) ? ( blk_next - blk_idx ) : ( total_gran - blk_idx );
1803 std::uint32_t cur = hdr->free_tree_root, parent = detail::kNoBlock;
1804 bool go_left = false;
1805 while ( cur != detail::kNoBlock )
1806 {
1807 parent = cur;
1808 const void* n = blk_at( base, cur );
1809 index_type n_next = BlockState::get_next_offset( n );
1810 std::uint32_t n_gran = ( n_next != detail::kNoBlock ) ? ( n_next - cur ) : ( total_gran - cur );
1811 bool smaller = ( blk_gran < n_gran ) || ( blk_gran == n_gran && blk_idx < cur );
1812 go_left = smaller;
1813 cur = smaller ? BlockState::get_left_offset( n ) : BlockState::get_right_offset( n );
1814 }
1815 BlockState::set_parent_offset_of( blk, parent );
1816 if ( go_left )
1817 BlockState::set_left_offset_of( blk_at( base, parent ), blk_idx );
1818 else
1819 BlockState::set_right_offset_of( blk_at( base, parent ), blk_idx );
1820 rebalance_up( base, hdr, parent );
1821 }

1824 static void remove( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
1825 {
1826 void* blk = blk_at( base, blk_idx );
1827 std::uint32_t parent = BlockState::get_parent_offset( blk );
1828 std::uint32_t left = BlockState::get_left_offset( blk );
1829 std::uint32_t right = BlockState::get_right_offset( blk );
1830 std::uint32_t rebal = detail::kNoBlock;
1831
1832 if ( left == detail::kNoBlock && right == detail::kNoBlock )
1833 {
1834 set_child( base, hdr, parent, blk_idx, detail::kNoBlock );
1835 rebal = parent;
1836 }
1837 else if ( left == detail::kNoBlock || right == detail::kNoBlock )
1838 {
1839 std::uint32_t child = ( left != detail::kNoBlock ) ? left : right;
1840 BlockState::set_parent_offset_of( blk_at( base, child ), parent );
1841 set_child( base, hdr, parent, blk_idx, child );
1842 rebal = parent;
1843 }
1844 else
1845 {
1846 std::uint32_t succ_idx = min_node( base, right );
1847 void* succ = blk_at( base, succ_idx );
1848 std::uint32_t succ_parent = BlockState::get_parent_offset( succ );
1849 std::uint32_t succ_right = BlockState::get_right_offset( succ );
1850
1851 if ( succ_parent != blk_idx )
1852 {
1853 set_child( base, hdr, succ_parent, succ_idx, succ_right );
1854 if ( succ_right != detail::kNoBlock )
1855 BlockState::set_parent_offset_of( blk_at( base, succ_right ), succ_parent );
1856 BlockState::set_right_offset_of( succ, right );
1857 BlockState::set_parent_offset_of( blk_at( base, right ), succ_idx );
1858 rebal = succ_parent;
1859 }
1860 else
1861 {
1862 rebal = succ_idx;
1863 }
1864 BlockState::set_left_offset_of( succ, left );
1865 BlockState::set_parent_offset_of( blk_at( base, left ), succ_idx );
1866 BlockState::set_parent_offset_of( succ, parent );
1867 set_child( base, hdr, parent, blk_idx, succ_idx );
1868 update_height( base, succ_idx );
1869 }
1870 BlockState::set_left_offset_of( blk, detail::kNoBlock );
1871 BlockState::set_right_offset_of( blk, detail::kNoBlock );
1872 BlockState::set_parent_offset_of( blk, detail::kNoBlock );
1873 BlockState::set_avl_height_of( blk, 0 );
1874 rebalance_up( base, hdr, rebal );
1875 }

1881 std::uint32_t total_gran = detail::byte_off_to_idx( hdr->total_size );
1882 std::uint32_t cur = hdr->free_tree_root, result = detail::kNoBlock;
1883 while ( cur != detail::kNoBlock )
1884 {
1885 const void* node = blk_at( base, cur );
1886 index_type node_next = BlockState::get_next_offset( node );
1887 std::uint32_t cur_gran = ( node_next != detail::kNoBlock ) ? ( node_next - cur ) : ( total_gran - cur );
1888 if ( cur_gran >= needed_granules )
1889 {
1890 result = cur;
1891 cur = BlockState::get_left_offset( node );
1892 }
1893 else
1894 {
1895 cur = BlockState::get_right_offset( node );
1896 }
1897 }
1898 return result;

3632 view.alignment = kGranuleSize;
