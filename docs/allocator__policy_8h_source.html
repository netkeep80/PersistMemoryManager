108 static void* allocate_from_block( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx,
109 std::size_t user_size )
110 {
111 // State: FreeBlock → FreeBlockRemovedAVL [remove_from_avl]
112 FreeBlockTreeT::remove( base, hdr, blk_idx );
113 FreeBlock<AddressTraitsT>* fb = FreeBlock<AddressTraitsT>::cast_from_raw( blk_at( base, blk_idx ) );
114 FreeBlockRemovedAVL<AddressTraitsT>* removed = fb->remove_from_avl();
115
116 std::uint32_t blk_total_gran = detail::block_total_granules( base, hdr, blk_at( base, blk_idx ) );
117 std::uint32_t data_gran = detail::bytes_to_granules( user_size );
118 std::uint32_t needed_gran = detail::kBlockHeaderGranules + data_gran;
119 std::uint32_t min_rem_gran = detail::kBlockHeaderGranules + 1;
120 bool can_split = ( blk_total_gran >= needed_gran + min_rem_gran );
121
122 if ( can_split )
123 {
124 // State: FreeBlockRemovedAVL → SplittingBlock [begin_splitting]
125 SplittingBlock<AddressTraitsT>* splitting = removed->begin_splitting();
126
127 std::uint32_t new_idx = blk_idx + needed_gran;
128 void* new_blk_ptr = blk_at( base, new_idx );
129
130 // Capture old_next before initialize_new_block modifies splitting->next_offset()
131 index_type curr_next = splitting->next_offset();
132 BlockT* old_next = ( curr_next != detail::kNoBlock ) ? blk_at( base, curr_next ) : nullptr;
133
134 // SplittingBlock::initialize_new_block — инициализировать новый (remainder) блок
135 splitting->initialize_new_block( new_blk_ptr, new_idx, blk_idx );
136
137 // SplittingBlock::link_new_block — обновить связный список
138 splitting->link_new_block( old_next, new_idx );
139 if ( old_next == nullptr )
140 hdr->last_block_offset = new_idx;
141
142 hdr->block_count++;
143 hdr->free_count++;
144 hdr->used_size += detail::kBlockHeaderGranules;
145 FreeBlockTreeT::insert( base, hdr, new_idx );
146
147 // State: SplittingBlock → AllocatedBlock [finalize_split]
148 AllocatedBlock<AddressTraitsT>* alloc = splitting->finalize_split( data_gran, blk_idx );
149 (void)alloc; // allocated block pointer obtained via state machine
150 }
151 else
152 {
153 // State: FreeBlockRemovedAVL → AllocatedBlock [mark_as_allocated]
154 AllocatedBlock<AddressTraitsT>* alloc = removed->mark_as_allocated( data_gran, blk_idx );
155 (void)alloc; // allocated block pointer obtained via state machine
156 }
157
158 hdr->alloc_count++;
159 hdr->free_count--;
160 hdr->used_size += data_gran;
161
162 // Return user data pointer (after block header)
163 return reinterpret_cast<std::uint8_t*>( blk_at( base, blk_idx ) ) + sizeof( BlockT );
164 }

184 static void coalesce( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
185 {
186 // State: FreeBlockNotInAVL → CoalescingBlock [begin_coalescing]
187 FreeBlockNotInAVL<AddressTraitsT>* not_avl =
188 FreeBlockNotInAVL<AddressTraitsT>::cast_from_raw( blk_at( base, blk_idx ) );
189 CoalescingBlock<AddressTraitsT>* coalescing = not_avl->begin_coalescing();
190
191 std::uint32_t b_idx = blk_idx;
192
193 // Слияние с правым соседом
194 // State: CoalescingBlock::coalesce_with_next
195 index_type curr_next = coalescing->next_offset();
196 if ( curr_next != detail::kNoBlock )
197 {
198 const BlockStateBase<AddressTraitsT>* nxt_state =
199 reinterpret_cast<const BlockStateBase<AddressTraitsT>*>( blk_at( base, curr_next ) );
200 if ( nxt_state->weight() == 0 ) // free block
201 {
202 std::uint32_t nxt_idx = curr_next;
203 index_type nxt_next = nxt_state->next_offset();
204 BlockT* nxt_nxt_blk = ( nxt_next != detail::kNoBlock ) ? blk_at( base, nxt_next ) : nullptr;
205
206 FreeBlockTreeT::remove( base, hdr, nxt_idx );
207
208 // CoalescingBlock::coalesce_with_next (also updates nxt_nxt_blk->prev_offset internally)
209 coalescing->coalesce_with_next( blk_at( base, nxt_idx ), nxt_nxt_blk, b_idx );
210
211 if ( nxt_nxt_blk == nullptr )
212 hdr->last_block_offset = b_idx;
213
214 hdr->block_count--;
215 hdr->free_count--;
216 if ( hdr->used_size >= detail::kBlockHeaderGranules )
217 hdr->used_size -= detail::kBlockHeaderGranules;
218 }
219 }
220
221 // Слияние с левым соседом
222 // State: CoalescingBlock::coalesce_with_prev → результат CoalescingBlock(prv)
223 index_type curr_prev = coalescing->prev_offset();
224 if ( curr_prev != detail::kNoBlock )
225 {
226 const BlockStateBase<AddressTraitsT>* prv_state =
227 reinterpret_cast<const BlockStateBase<AddressTraitsT>*>( blk_at( base, curr_prev ) );
228 if ( prv_state->weight() == 0 ) // free block
229 {
230 std::uint32_t prv_idx = curr_prev;
231 index_type blk_next = coalescing->next_offset();
232 BlockT* next_blk = ( blk_next != detail::kNoBlock ) ? blk_at( base, blk_next ) : nullptr;
233
234 FreeBlockTreeT::remove( base, hdr, prv_idx );
235
236 // CoalescingBlock::coalesce_with_prev — current block (blk) is absorbed into prv
237 CoalescingBlock<AddressTraitsT>* result_coalescing =
238 coalescing->coalesce_with_prev( blk_at( base, prv_idx ), next_blk, prv_idx );
239
240 if ( next_blk == nullptr )
241 hdr->last_block_offset = prv_idx;
242
243 hdr->block_count--;
244 hdr->free_count--;
245 if ( hdr->used_size >= detail::kBlockHeaderGranules )
246 hdr->used_size -= detail::kBlockHeaderGranules;
247
248 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
249 FreeBlock<AddressTraitsT>* fb = result_coalescing->finalize_coalesce();
250 (void)fb;
251 FreeBlockTreeT::insert( base, hdr, prv_idx );
252 return;
253 }
254 }
255
256 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
257 FreeBlock<AddressTraitsT>* fb = coalescing->finalize_coalesce();
258 (void)fb;
259 FreeBlockTreeT::insert( base, hdr, b_idx );
260 }

274 static void rebuild_free_tree( std::uint8_t* base, detail::ManagerHeader* hdr )
275 {
276 hdr->free_tree_root = detail::kNoBlock;
277 hdr->last_block_offset = detail::kNoBlock;
278 std::uint32_t idx = hdr->first_block_offset;
279 while ( idx != detail::kNoBlock )
280 {
281 void* blk_ptr = blk_at( base, idx );
282
283 // Reset AVL fields via state machine utility (Issue #114)
284 reset_block_avl_fields<AddressTraitsT>( blk_ptr );
285
286 // Issue #106: recover_block_state — исправить некорректные переходные состояния
287 recover_block_state<AddressTraitsT>( blk_ptr, idx );
288
289 if ( read_block_weight<AddressTraitsT>( blk_ptr ) == 0 ) // free block
290 FreeBlockTreeT::insert( base, hdr, idx );
291 index_type next_idx = read_block_next_offset<AddressTraitsT>( blk_ptr );
292 if ( next_idx == detail::kNoBlock )
293 hdr->last_block_offset = idx;
294 idx = next_idx;
295 }
296 }

307 static void repair_linked_list( std::uint8_t* base, detail::ManagerHeader* hdr )
308 {
309 std::uint32_t idx = hdr->first_block_offset;
310 index_type prev = detail::kNoBlock;
311 while ( idx != detail::kNoBlock )
312 {
313 if ( detail::idx_to_byte_off( idx ) + sizeof( BlockT ) > hdr->total_size )
314 break;
315 void* blk_ptr = blk_at( base, idx );
316 repair_block_prev_offset<AddressTraitsT>( blk_ptr, prev ); // Issue #114
317 prev = idx;
318 index_type next_offset = read_block_next_offset<AddressTraitsT>( blk_ptr );
319 idx = next_offset;
320 }

333 static void recompute_counters( std::uint8_t* base, detail::ManagerHeader* hdr )
334 {
335 std::uint32_t block_count = 0, free_count = 0, alloc_count = 0;
336 std::uint32_t used_gran = 0;
337 std::uint32_t idx = hdr->first_block_offset;
338 while ( idx != detail::kNoBlock )
339 {
340 if ( detail::idx_to_byte_off( idx ) + sizeof( BlockT ) > hdr->total_size )
341 break;
342 const void* blk_ptr = blk_at( base, idx );
343 block_count++;
344 used_gran += detail::kBlockHeaderGranules;
345 index_type w = read_block_weight<AddressTraitsT>( blk_ptr ); // Issue #114
346 if ( w > 0 ) // allocated block
347 {
348 alloc_count++;
349 used_gran += w;
350 }
351 else
352 {
353 free_count++;
354 }
355 idx = read_block_next_offset<AddressTraitsT>( blk_ptr ); // Issue #114
356 }
357 hdr->block_count = block_count;
358 hdr->free_count = free_count;
359 hdr->alloc_count = alloc_count;
360 hdr->used_size = used_gran;
361 }

367 assert( idx != detail::kNoBlock );
