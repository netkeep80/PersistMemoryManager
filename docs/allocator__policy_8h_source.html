111 static void* allocate_from_block( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr,
112 index_type blk_idx, std::size_t user_size )
113 {
114 // State: FreeBlock → FreeBlockRemovedAVL [remove_from_avl]
115 FreeBlockTreeT::remove( base, hdr, blk_idx );
116 FreeBlock<AddressTraitsT>* fb =
117 FreeBlock<AddressTraitsT>::cast_from_raw( detail::block_at<AddressTraitsT>( base, blk_idx ) );
118 FreeBlockRemovedAVL<AddressTraitsT>* removed = fb->remove_from_avl();
119
120 // Use AddressTraitsT-specific granule computations.
121 static constexpr index_type kBlkHdrGran =
122 detail::kBlockHeaderGranules_t<AddressTraitsT>;
123
124 index_type blk_total_gran =
125 detail::block_total_granules( base, hdr, detail::block_at<AddressTraitsT>( base, blk_idx ) );
126 index_type data_gran = detail::bytes_to_granules_t<AddressTraitsT>( user_size );
127
128 // Overflow protection for needed_gran computation.
129 if ( data_gran > std::numeric_limits<index_type>::max() - kBlkHdrGran )
130 return nullptr; // overflow: request too large
131
132 index_type needed_gran = kBlkHdrGran + data_gran;
133 index_type min_rem_gran = kBlkHdrGran + 1;
134
135 // Overflow protection for split check (needed_gran + min_rem_gran).
136 bool can_split = false;
137 if ( needed_gran <= std::numeric_limits<index_type>::max() - min_rem_gran )
138 can_split = ( blk_total_gran >= needed_gran + min_rem_gran );
139
140 if ( can_split )
141 {
142 // State: FreeBlockRemovedAVL → SplittingBlock [begin_splitting]
143 SplittingBlock<AddressTraitsT>* splitting = removed->begin_splitting();
144
145 index_type new_idx = blk_idx + needed_gran;
146 void* new_blk_ptr = detail::block_at<AddressTraitsT>( base, new_idx );
147
148 // Capture old_next before initialize_new_block modifies splitting->next_offset()
149 // Compare against AddressTraitsT::no_block (correct sentinel for index_type).
150 index_type curr_next = splitting->next_offset();
151 BlockT* old_next = ( curr_next != AddressTraitsT::no_block )
152 ? detail::block_at<AddressTraitsT>( base, curr_next )
153 : nullptr;
154
155 // SplittingBlock::initialize_new_block — инициализировать новый (remainder) блок
156 splitting->initialize_new_block( new_blk_ptr, new_idx, blk_idx );
157
158 // SplittingBlock::link_new_block — обновить связный список
159 splitting->link_new_block( old_next, new_idx );
160 if ( old_next == nullptr )
161 hdr->last_block_offset = new_idx;
162
163 hdr->block_count++;
164 hdr->free_count++;
165 hdr->used_size += kBlkHdrGran;
166 FreeBlockTreeT::insert( base, hdr, new_idx );
167
168 // State: SplittingBlock → AllocatedBlock [finalize_split]
169 AllocatedBlock<AddressTraitsT>* alloc = splitting->finalize_split( data_gran, blk_idx );
170 (void)alloc; // allocated block pointer obtained via state machine
171 }
172 else
173 {
174 // State: FreeBlockRemovedAVL → AllocatedBlock [mark_as_allocated]
175 AllocatedBlock<AddressTraitsT>* alloc = removed->mark_as_allocated( data_gran, blk_idx );
176 (void)alloc; // allocated block pointer obtained via state machine
177 }
178
179 hdr->alloc_count++;
180 hdr->free_count--;
181 hdr->used_size += data_gran;
182
183 // Return user data pointer (after block header) via canonical detail::user_ptr
184 return detail::user_ptr<AddressTraitsT>( detail::block_at<AddressTraitsT>( base, blk_idx ) );
185 }

205 static void coalesce( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type blk_idx )
206 {
207 // State: FreeBlockNotInAVL → CoalescingBlock [begin_coalescing]
208 FreeBlockNotInAVL<AddressTraitsT>* not_avl =
209 FreeBlockNotInAVL<AddressTraitsT>::cast_from_raw( detail::block_at<AddressTraitsT>( base, blk_idx ) );
210 CoalescingBlock<AddressTraitsT>* coalescing = not_avl->begin_coalescing();
211
212 // Use AddressTraitsT-specific block header granule count.
213 static constexpr index_type kBlkHdrGran = detail::kBlockHeaderGranules_t<AddressTraitsT>;
214
215 index_type b_idx = blk_idx;
216
217 // Слияние с правым соседом
218 // State: CoalescingBlock::coalesce_with_next
219 // Compare against AddressTraitsT::no_block for correct sentinel check.
220 index_type curr_next = coalescing->next_offset();
221 if ( curr_next != AddressTraitsT::no_block )
222 {
223 const BlockStateBase<AddressTraitsT>* nxt_state = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>(
224 detail::block_at<AddressTraitsT>( base, curr_next ) );
225 if ( nxt_state->weight() == 0 ) // free block
226 {
227 index_type nxt_idx = curr_next;
228 index_type nxt_next = nxt_state->next_offset();
229 BlockT* nxt_nxt_blk = ( nxt_next != AddressTraitsT::no_block )
230 ? detail::block_at<AddressTraitsT>( base, nxt_next )
231 : nullptr;
232
233 FreeBlockTreeT::remove( base, hdr, nxt_idx );
234
235 // CoalescingBlock::coalesce_with_next (also updates nxt_nxt_blk->prev_offset internally)
236 coalescing->coalesce_with_next( detail::block_at<AddressTraitsT>( base, nxt_idx ), nxt_nxt_blk, b_idx );
237
238 if ( nxt_nxt_blk == nullptr )
239 hdr->last_block_offset = b_idx;
240
241 hdr->block_count--;
242 hdr->free_count--;
243 if ( hdr->used_size >= kBlkHdrGran )
244 hdr->used_size -= kBlkHdrGran;
245 }
246 }
247
248 // Слияние с левым соседом
249 // State: CoalescingBlock::coalesce_with_prev → результат CoalescingBlock(prv)
250 index_type curr_prev = coalescing->prev_offset();
251 if ( curr_prev != AddressTraitsT::no_block )
252 {
253 const BlockStateBase<AddressTraitsT>* prv_state = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>(
254 detail::block_at<AddressTraitsT>( base, curr_prev ) );
255 if ( prv_state->weight() == 0 ) // free block
256 {
257 index_type prv_idx = curr_prev;
258 index_type blk_next = coalescing->next_offset();
259 BlockT* next_blk = ( blk_next != AddressTraitsT::no_block )
260 ? detail::block_at<AddressTraitsT>( base, blk_next )
261 : nullptr;
262
263 FreeBlockTreeT::remove( base, hdr, prv_idx );
264
265 // CoalescingBlock::coalesce_with_prev — current block (blk) is absorbed into prv
266 CoalescingBlock<AddressTraitsT>* result_coalescing = coalescing->coalesce_with_prev(
267 detail::block_at<AddressTraitsT>( base, prv_idx ), next_blk, prv_idx );
268
269 if ( next_blk == nullptr )
270 hdr->last_block_offset = prv_idx;
271
272 hdr->block_count--;
273 hdr->free_count--;
274 if ( hdr->used_size >= kBlkHdrGran )
275 hdr->used_size -= kBlkHdrGran;
276
277 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
278 FreeBlock<AddressTraitsT>* fb = result_coalescing->finalize_coalesce();
279 (void)fb;
280 FreeBlockTreeT::insert( base, hdr, prv_idx );
281 return;
282 }
283 }
284
285 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
286 FreeBlock<AddressTraitsT>* fb = coalescing->finalize_coalesce();
287 (void)fb;
288 FreeBlockTreeT::insert( base, hdr, b_idx );
289 }

303 static void rebuild_free_tree( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr )
304 {
305 hdr->free_tree_root = AddressTraitsT::no_block;
306 hdr->last_block_offset = AddressTraitsT::no_block;
307 index_type idx = hdr->first_block_offset;
308 while ( idx != AddressTraitsT::no_block )
309 {
310 void* blk_ptr = detail::block_at<AddressTraitsT>( base, idx );
311
312 // Repair: fix incorrect transitional states (weight/root_offset mismatch)
313 BlockState::recover_state( blk_ptr, idx );
314
315 if ( BlockState::get_weight( blk_ptr ) == 0 ) // free block
316 {
317 // Reset AVL fields only for free blocks — allocated blocks may use
318 // TreeNode AVL fields for user-level trees (symbol tree, pmap, etc.)
319 // so resetting them would corrupt those data structures.
320 BlockState::reset_avl_fields_of( blk_ptr );
321 FreeBlockTreeT::insert( base, hdr, idx );
322 }
323 // Use AddressTraitsT::no_block for correct sentinel check.
324 index_type next_idx = BlockState::get_next_offset( blk_ptr );
325 if ( next_idx == AddressTraitsT::no_block )
326 hdr->last_block_offset = idx;
327 idx = next_idx;
328 }
329 }

340 static void repair_linked_list( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr )
341 {
342 index_type idx = hdr->first_block_offset;
343 index_type prev = AddressTraitsT::no_block;
344 while ( idx != AddressTraitsT::no_block )
345 {
346 // Use AddressTraitsT::granule_size for correct size check.
347 if ( static_cast<std::size_t>( idx ) * AddressTraitsT::granule_size + sizeof( BlockT ) > hdr->total_size )
348 break;
349 void* blk_ptr = detail::block_at<AddressTraitsT>( base, idx );
350 BlockState::repair_prev_offset( blk_ptr, prev );
351 prev = idx;
352 index_type next_offset = BlockState::get_next_offset( blk_ptr );
353 idx = next_offset;
354 }

367 static void recompute_counters( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr )
368 {
369 // Use AddressTraitsT-specific block header granule count.
370 static constexpr index_type kBlkHdrGran = detail::kBlockHeaderGranules_t<AddressTraitsT>;
371
372 index_type block_count = 0, free_count = 0, alloc_count = 0;
373 index_type used_gran = 0;
374 index_type idx = hdr->first_block_offset;
375 while ( idx != AddressTraitsT::no_block )
376 {
377 // Use AddressTraitsT::granule_size for correct size check.
378 if ( static_cast<std::size_t>( idx ) * AddressTraitsT::granule_size + sizeof( BlockT ) > hdr->total_size )
379 break;
380 const void* blk_ptr = detail::block_at<AddressTraitsT>( base, idx );
381 block_count++;
382 used_gran += kBlkHdrGran;
383 index_type w = BlockState::get_weight( blk_ptr );
384 if ( w > 0 ) // allocated block
385 {
386 alloc_count++;
387 used_gran += w;
388 }
389 else
390 {
391 free_count++;
392 }
393 idx = BlockState::get_next_offset( blk_ptr );
394 }
395 hdr->block_count = block_count;
396 hdr->free_count = free_count;
397 hdr->alloc_count = alloc_count;
398 hdr->used_size = used_gran;
399 }

415 {

516 static void verify_free_tree( const std::uint8_t* base, const detail::ManagerHeader<AddressTraitsT>* hdr,
517 VerifyResult& result ) noexcept
518 {
519 // Count free blocks by walking the linked list.
520 index_type free_count = 0;
521 index_type idx = hdr->first_block_offset;
522 while ( idx != AddressTraitsT::no_block )
523 {
524 if ( static_cast<std::size_t>( idx ) * AddressTraitsT::granule_size + sizeof( BlockT ) > hdr->total_size )
525 break;
526 const void* blk_ptr = detail::block_at<AddressTraitsT>( base, idx );
527 if ( BlockState::get_weight( blk_ptr ) == 0 )
528 free_count++;
529 idx = BlockState::get_next_offset( blk_ptr );
530 }
531 // If free blocks exist but the tree root is no_block (or vice versa),
532 // the free tree is stale and needs rebuild.
533 bool root_present = ( hdr->free_tree_root != AddressTraitsT::no_block );
534 if ( ( free_count > 0 && !root_present ) || ( free_count == 0 && root_present ) )
535 {
536 result.add( ViolationType::FreeTreeStale, DiagnosticAction::NoAction, 0,
537 static_cast<std::uint64_t>( free_count ), static_cast<std::uint64_t>( hdr->free_tree_root ) );
538 }

544 static void realloc_shrink( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type blk_idx,
545 void* blk_raw, index_type old_data_gran, index_type new_data_gran ) noexcept
546 {
547 static constexpr index_type kBlkHdrGran = detail::kBlockHeaderGranules_t<AddressTraitsT>;
548 index_type remainder = old_data_gran - new_data_gran;
549 if ( remainder >= kBlkHdrGran + 1 )
550 {
551 index_type new_free_idx = blk_idx + kBlkHdrGran + new_data_gran;
552 void* new_free_blk = detail::block_at<AddressTraitsT>( base, new_free_idx );
553 index_type old_next = BlockState::get_next_offset( blk_raw );
554 auto* old_next_blk =
555 ( old_next != AddressTraitsT::no_block ) ? detail::block_at<AddressTraitsT>( base, old_next ) : nullptr;
556 std::memset( new_free_blk, 0, sizeof( BlockT ) );
557 BlockState::init_fields( new_free_blk, blk_idx, old_next, 1, 0, 0 );
558 BlockState::set_next_offset_of( blk_raw, new_free_idx );
559 if ( old_next_blk != nullptr )
560 BlockState::set_prev_offset_of( old_next_blk, new_free_idx );
561 else
562 hdr->last_block_offset = new_free_idx;
563 BlockState::set_weight_of( blk_raw, new_data_gran );
564 hdr->block_count++;
565 hdr->free_count++;
566 hdr->used_size += kBlkHdrGran;
567 hdr->used_size -= ( old_data_gran - new_data_gran );
568 coalesce( base, hdr, new_free_idx );
569 }
570 else
571 {
572 BlockState::set_weight_of( blk_raw, new_data_gran );
573 hdr->used_size -= ( old_data_gran - new_data_gran );
574 }
575 }

578 static bool realloc_grow( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type blk_idx,
579 void* blk_raw, index_type old_data_gran, index_type new_data_gran ) noexcept
580 {
581 static constexpr index_type kBlkHdrGran = detail::kBlockHeaderGranules_t<AddressTraitsT>;
582 index_type next_idx = BlockState::get_next_offset( blk_raw );
583 if ( next_idx == AddressTraitsT::no_block )
584 return false;
585 void* next_blk = detail::block_at<AddressTraitsT>( base, next_idx );
586 if ( BlockState::get_weight( next_blk ) != 0 )
587 return false;
588 index_type next_total =
589 detail::block_total_granules( base, hdr, detail::block_at<AddressTraitsT>( base, next_idx ) );
590 index_type available = old_data_gran + next_total;
591 if ( available < new_data_gran )
592 return false;
593 FreeBlockTreeT::remove( base, hdr, next_idx );
594 index_type next_next = BlockState::get_next_offset( next_blk );
595 BlockState::set_next_offset_of( blk_raw, next_next );
596 if ( next_next != AddressTraitsT::no_block )
597 BlockState::set_prev_offset_of( detail::block_at<AddressTraitsT>( base, next_next ), blk_idx );
598 else
599 hdr->last_block_offset = blk_idx;
600 std::memset( next_blk, 0, sizeof( BlockT ) );
601 hdr->block_count--;
602 hdr->free_count--;
603 if ( hdr->used_size >= kBlkHdrGran )
604 hdr->used_size -= kBlkHdrGran;
605 index_type rem = available - new_data_gran;
606 if ( rem >= kBlkHdrGran + 1 )
607 {
608 index_type rem_idx = blk_idx + kBlkHdrGran + new_data_gran;
609 void* rem_blk = detail::block_at<AddressTraitsT>( base, rem_idx );
610 index_type blk_new_next = BlockState::get_next_offset( blk_raw );
611 std::memset( rem_blk, 0, sizeof( BlockT ) );
612 BlockState::init_fields( rem_blk, blk_idx, blk_new_next, 1, 0, 0 );
613 BlockState::set_next_offset_of( blk_raw, rem_idx );
614 if ( blk_new_next != AddressTraitsT::no_block )
615 BlockState::set_prev_offset_of( detail::block_at<AddressTraitsT>( base, blk_new_next ), rem_idx );
616 else
617 hdr->last_block_offset = rem_idx;
618 hdr->block_count++;
619 hdr->free_count++;
620 hdr->used_size += kBlkHdrGran;
621 FreeBlockTreeT::insert( base, hdr, rem_idx );
622 }
623 BlockState::set_weight_of( blk_raw, new_data_gran );
624 hdr->used_size += ( new_data_gran - old_data_gran );
625 return true;
626 }
