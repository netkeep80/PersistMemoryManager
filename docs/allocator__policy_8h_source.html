108 static void* allocate_from_block( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx,
109 std::size_t user_size )
110 {
111 // State: FreeBlock → FreeBlockRemovedAVL [remove_from_avl]
112 FreeBlockTreeT::remove( base, hdr, blk_idx );
113 FreeBlock<AddressTraitsT>* fb =
114 FreeBlock<AddressTraitsT>::cast_from_raw( detail::block_at<AddressTraitsT>( base, blk_idx ) );
115 FreeBlockRemovedAVL<AddressTraitsT>* removed = fb->remove_from_avl();
116
117 // Issue #146: use AddressTraitsT-specific granule computations.
118 static constexpr std::uint32_t kBlkHdrGran =
119 detail::kBlockHeaderGranules_t<AddressTraitsT>;
120
121 std::uint32_t blk_total_gran =
122 detail::block_total_granules( base, hdr, detail::block_at<AddressTraitsT>( base, blk_idx ) );
123 std::uint32_t data_gran = detail::bytes_to_granules_t<AddressTraitsT>( user_size );
124 std::uint32_t needed_gran = kBlkHdrGran + data_gran;
125 std::uint32_t min_rem_gran = kBlkHdrGran + 1;
126 bool can_split = ( blk_total_gran >= needed_gran + min_rem_gran );
127
128 if ( can_split )
129 {
130 // State: FreeBlockRemovedAVL → SplittingBlock [begin_splitting]
131 SplittingBlock<AddressTraitsT>* splitting = removed->begin_splitting();
132
133 std::uint32_t new_idx = blk_idx + needed_gran;
134 void* new_blk_ptr = detail::block_at<AddressTraitsT>( base, new_idx );
135
136 // Capture old_next before initialize_new_block modifies splitting->next_offset()
137 // Issue #146: compare against AddressTraitsT::no_block (correct sentinel for index_type).
138 index_type curr_next = splitting->next_offset();
139 BlockT* old_next = ( curr_next != AddressTraitsT::no_block )
140 ? detail::block_at<AddressTraitsT>( base, curr_next )
141 : nullptr;
142
143 // SplittingBlock::initialize_new_block — инициализировать новый (remainder) блок
144 splitting->initialize_new_block( new_blk_ptr, new_idx, blk_idx );
145
146 // SplittingBlock::link_new_block — обновить связный список
147 splitting->link_new_block( old_next, new_idx );
148 if ( old_next == nullptr )
149 hdr->last_block_offset = new_idx;
150
151 hdr->block_count++;
152 hdr->free_count++;
153 hdr->used_size += kBlkHdrGran;
154 FreeBlockTreeT::insert( base, hdr, new_idx );
155
156 // State: SplittingBlock → AllocatedBlock [finalize_split]
157 AllocatedBlock<AddressTraitsT>* alloc = splitting->finalize_split( data_gran, blk_idx );
158 (void)alloc; // allocated block pointer obtained via state machine
159 }
160 else
161 {
162 // State: FreeBlockRemovedAVL → AllocatedBlock [mark_as_allocated]
163 AllocatedBlock<AddressTraitsT>* alloc = removed->mark_as_allocated( data_gran, blk_idx );
164 (void)alloc; // allocated block pointer obtained via state machine
165 }
166
167 hdr->alloc_count++;
168 hdr->free_count--;
169 hdr->used_size += data_gran;
170
171 // Return user data pointer (after block header) via canonical detail::user_ptr (Issue #141)
172 return detail::user_ptr<AddressTraitsT>( detail::block_at<AddressTraitsT>( base, blk_idx ) );
173 }

193 static void coalesce( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
194 {
195 // State: FreeBlockNotInAVL → CoalescingBlock [begin_coalescing]
196 FreeBlockNotInAVL<AddressTraitsT>* not_avl =
197 FreeBlockNotInAVL<AddressTraitsT>::cast_from_raw( detail::block_at<AddressTraitsT>( base, blk_idx ) );
198 CoalescingBlock<AddressTraitsT>* coalescing = not_avl->begin_coalescing();
199
200 // Issue #146: use AddressTraitsT-specific block header granule count.
201 static constexpr std::uint32_t kBlkHdrGran = detail::kBlockHeaderGranules_t<AddressTraitsT>;
202
203 std::uint32_t b_idx = blk_idx;
204
205 // Слияние с правым соседом
206 // State: CoalescingBlock::coalesce_with_next
207 // Issue #146: compare against AddressTraitsT::no_block for correct sentinel check.
208 index_type curr_next = coalescing->next_offset();
209 if ( curr_next != AddressTraitsT::no_block )
210 {
211 const BlockStateBase<AddressTraitsT>* nxt_state = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>(
212 detail::block_at<AddressTraitsT>( base, curr_next ) );
213 if ( nxt_state->weight() == 0 ) // free block
214 {
215 std::uint32_t nxt_idx = static_cast<std::uint32_t>( curr_next );
216 index_type nxt_next = nxt_state->next_offset();
217 BlockT* nxt_nxt_blk = ( nxt_next != AddressTraitsT::no_block )
218 ? detail::block_at<AddressTraitsT>( base, nxt_next )
219 : nullptr;
220
221 FreeBlockTreeT::remove( base, hdr, nxt_idx );
222
223 // CoalescingBlock::coalesce_with_next (also updates nxt_nxt_blk->prev_offset internally)
224 coalescing->coalesce_with_next( detail::block_at<AddressTraitsT>( base, nxt_idx ), nxt_nxt_blk, b_idx );
225
226 if ( nxt_nxt_blk == nullptr )
227 hdr->last_block_offset = b_idx;
228
229 hdr->block_count--;
230 hdr->free_count--;
231 if ( hdr->used_size >= kBlkHdrGran )
232 hdr->used_size -= kBlkHdrGran;
233 }
234 }
235
236 // Слияние с левым соседом
237 // State: CoalescingBlock::coalesce_with_prev → результат CoalescingBlock(prv)
238 index_type curr_prev = coalescing->prev_offset();
239 if ( curr_prev != AddressTraitsT::no_block )
240 {
241 const BlockStateBase<AddressTraitsT>* prv_state = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>(
242 detail::block_at<AddressTraitsT>( base, curr_prev ) );
243 if ( prv_state->weight() == 0 ) // free block
244 {
245 std::uint32_t prv_idx = static_cast<std::uint32_t>( curr_prev );
246 index_type blk_next = coalescing->next_offset();
247 BlockT* next_blk = ( blk_next != AddressTraitsT::no_block )
248 ? detail::block_at<AddressTraitsT>( base, blk_next )
249 : nullptr;
250
251 FreeBlockTreeT::remove( base, hdr, prv_idx );
252
253 // CoalescingBlock::coalesce_with_prev — current block (blk) is absorbed into prv
254 CoalescingBlock<AddressTraitsT>* result_coalescing = coalescing->coalesce_with_prev(
255 detail::block_at<AddressTraitsT>( base, prv_idx ), next_blk, prv_idx );
256
257 if ( next_blk == nullptr )
258 hdr->last_block_offset = prv_idx;
259
260 hdr->block_count--;
261 hdr->free_count--;
262 if ( hdr->used_size >= kBlkHdrGran )
263 hdr->used_size -= kBlkHdrGran;
264
265 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
266 FreeBlock<AddressTraitsT>* fb = result_coalescing->finalize_coalesce();
267 (void)fb;
268 FreeBlockTreeT::insert( base, hdr, prv_idx );
269 return;
270 }
271 }
272
273 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
274 FreeBlock<AddressTraitsT>* fb = coalescing->finalize_coalesce();
275 (void)fb;
276 FreeBlockTreeT::insert( base, hdr, b_idx );
277 }

291 static void rebuild_free_tree( std::uint8_t* base, detail::ManagerHeader* hdr )
292 {
293 hdr->free_tree_root = detail::kNoBlock;
294 hdr->last_block_offset = detail::kNoBlock;
295 std::uint32_t idx = hdr->first_block_offset;
296 while ( idx != detail::kNoBlock )
297 {
298 void* blk_ptr = detail::block_at<AddressTraitsT>( base, idx );
299
300 // Reset AVL fields via state machine utility (Issue #114)
301 reset_block_avl_fields<AddressTraitsT>( blk_ptr );
302
303 // Issue #106: recover_block_state — исправить некорректные переходные состояния
304 recover_block_state<AddressTraitsT>( blk_ptr, static_cast<index_type>( idx ) );
305
306 if ( read_block_weight<AddressTraitsT>( blk_ptr ) == 0 ) // free block
307 FreeBlockTreeT::insert( base, hdr, idx );
308 // Issue #146: use AddressTraitsT::no_block for correct sentinel check.
309 index_type next_idx = read_block_next_offset<AddressTraitsT>( blk_ptr );
310 if ( next_idx == AddressTraitsT::no_block )
311 hdr->last_block_offset = idx;
312 idx = detail::to_u32_idx<AddressTraitsT>( next_idx );
313 }
314 }

325 static void repair_linked_list( std::uint8_t* base, detail::ManagerHeader* hdr )
326 {
327 std::uint32_t idx = hdr->first_block_offset;
328 // Issue #146: use AddressTraitsT::no_block for correct sentinel value.
329 index_type prev = AddressTraitsT::no_block;
330 while ( idx != detail::kNoBlock )
331 {
332 // Issue #146: use AddressTraitsT::granule_size for correct size check.
333 if ( static_cast<std::size_t>( idx ) * AddressTraitsT::granule_size + sizeof( BlockT ) > hdr->total_size )
334 break;
335 void* blk_ptr = detail::block_at<AddressTraitsT>( base, idx );
336 repair_block_prev_offset<AddressTraitsT>( blk_ptr, prev ); // Issue #114
337 prev = static_cast<index_type>( idx );
338 index_type next_offset = read_block_next_offset<AddressTraitsT>( blk_ptr );
339 // Issue #146: use sentinel-aware translation to convert index_type back to uint32_t.
340 idx = detail::to_u32_idx<AddressTraitsT>( next_offset );
341 }

354 static void recompute_counters( std::uint8_t* base, detail::ManagerHeader* hdr )
355 {
356 // Issue #146: use AddressTraitsT-specific block header granule count.
357 static constexpr std::uint32_t kBlkHdrGran = detail::kBlockHeaderGranules_t<AddressTraitsT>;
358
359 std::uint32_t block_count = 0, free_count = 0, alloc_count = 0;
360 std::uint32_t used_gran = 0;
361 std::uint32_t idx = hdr->first_block_offset;
362 while ( idx != detail::kNoBlock )
363 {
364 // Issue #146: use AddressTraitsT::granule_size for correct size check.
365 if ( static_cast<std::size_t>( idx ) * AddressTraitsT::granule_size + sizeof( BlockT ) > hdr->total_size )
366 break;
367 const void* blk_ptr = detail::block_at<AddressTraitsT>( base, idx );
368 block_count++;
369 used_gran += kBlkHdrGran;
370 index_type w = read_block_weight<AddressTraitsT>( blk_ptr ); // Issue #114
371 if ( w > 0 ) // allocated block
372 {
373 alloc_count++;
374 used_gran += static_cast<std::uint32_t>( w );
375 }
376 else
377 {
378 free_count++;
379 }
380 // Issue #146: use sentinel-aware translation to convert index_type back to uint32_t.
381 idx = detail::to_u32_idx<AddressTraitsT>( read_block_next_offset<AddressTraitsT>( blk_ptr ) );
382 }
383 hdr->block_count = block_count;
384 hdr->free_count = free_count;
385 hdr->alloc_count = alloc_count;
386 hdr->used_size = used_gran;
387 }
