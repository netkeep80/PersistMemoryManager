100 static void* allocate_from_block( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx,
101 std::size_t user_size )
102 {
103 // State: FreeBlock → FreeBlockRemovedAVL [remove_from_avl]
104 FreeBlockTreeT::remove( base, hdr, blk_idx );
105 FreeBlock<AddressTraitsT>* fb = FreeBlock<AddressTraitsT>::cast_from_raw( blk_at( base, blk_idx ) );
106 FreeBlockRemovedAVL<AddressTraitsT>* removed = fb->remove_from_avl();
107
108 std::uint32_t blk_total_gran = detail::block_total_granules( base, hdr, blk_at( base, blk_idx ) );
109 std::uint32_t data_gran = detail::bytes_to_granules( user_size );
110 std::uint32_t needed_gran = detail::kBlockHeaderGranules + data_gran;
111 std::uint32_t min_rem_gran = detail::kBlockHeaderGranules + 1;
112 bool can_split = ( blk_total_gran >= needed_gran + min_rem_gran );
113
114 if ( can_split )
115 {
116 // State: FreeBlockRemovedAVL → SplittingBlock [begin_splitting]
117 SplittingBlock<AddressTraitsT>* splitting = removed->begin_splitting();
118
119 std::uint32_t new_idx = blk_idx + needed_gran;
120 void* new_blk_ptr = blk_at( base, new_idx );
121
122 // SplittingBlock::initialize_new_block — инициализировать новый (remainder) блок
123 splitting->initialize_new_block( new_blk_ptr, new_idx, blk_idx );
124
125 // SplittingBlock::link_new_block — обновить связный список
126 BlockT* old_next_blk =
127 ( splitting->next_offset() != detail::kNoBlock ) ? blk_at( base, splitting->next_offset() ) : nullptr;
128 if ( old_next_blk == nullptr && blk_at( base, blk_idx )->next_offset != detail::kNoBlock )
129 old_next_blk = blk_at( base, blk_at( base, blk_idx )->next_offset );
130
131 // Determine old_next before linking
132 std::uint32_t curr_next = blk_at( base, blk_idx )->next_offset;
133 BlockT* old_next = ( curr_next != detail::kNoBlock ) ? blk_at( base, curr_next ) : nullptr;
134
135 // Initialize new block in place
136 std::memset( new_blk_ptr, 0, sizeof( BlockT ) );
137 BlockT* new_blk = blk_at( base, new_idx );
138 new_blk->prev_offset = blk_idx;
139 new_blk->next_offset = curr_next;
140 new_blk->left_offset = detail::kNoBlock;
141 new_blk->right_offset = detail::kNoBlock;
142 new_blk->parent_offset = detail::kNoBlock;
143 new_blk->avl_height = 1;
144 new_blk->weight = 0;
145 new_blk->root_offset = 0;
146
147 // Link new block into linked list
148 if ( old_next != nullptr )
149 old_next->prev_offset = new_idx;
150 else
151 hdr->last_block_offset = new_idx;
152 blk_at( base, blk_idx )->next_offset = new_idx;
153
154 hdr->block_count++;
155 hdr->free_count++;
156 hdr->used_size += detail::kBlockHeaderGranules;
157 FreeBlockTreeT::insert( base, hdr, new_idx );
158
159 // State: SplittingBlock → AllocatedBlock [finalize_split]
160 // Use direct field assignment on the block (block_state finalize_split does the same)
161 BlockT* blk = blk_at( base, blk_idx );
162 blk->weight = data_gran;
163 blk->root_offset = blk_idx;
164 blk->left_offset = detail::kNoBlock;
165 blk->right_offset = detail::kNoBlock;
166 blk->parent_offset = detail::kNoBlock;
167 blk->avl_height = 0;
168 }
169 else
170 {
171 // State: FreeBlockRemovedAVL → AllocatedBlock [mark_as_allocated]
172 AllocatedBlock<AddressTraitsT>* alloc = removed->mark_as_allocated( data_gran, blk_idx );
173 (void)alloc; // allocated block pointer obtained via state machine
174 }
175
176 hdr->alloc_count++;
177 hdr->free_count--;
178 hdr->used_size += data_gran;
179
180 // Return user data pointer (after block header)
181 return reinterpret_cast<std::uint8_t*>( blk_at( base, blk_idx ) ) + sizeof( BlockT );
182 }

202 static void coalesce( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
203 {
204 // State: FreeBlockNotInAVL → CoalescingBlock [begin_coalescing]
205 FreeBlockNotInAVL<AddressTraitsT>* not_avl =
206 FreeBlockNotInAVL<AddressTraitsT>::cast_from_raw( blk_at( base, blk_idx ) );
207 CoalescingBlock<AddressTraitsT>* coalescing = not_avl->begin_coalescing();
208
209 std::uint32_t b_idx = blk_idx;
210
211 // Слияние с правым соседом
212 // State: CoalescingBlock::coalesce_with_next
213 std::uint32_t curr_next = blk_at( base, b_idx )->next_offset;
214 if ( curr_next != detail::kNoBlock )
215 {
216 BlockT* nxt = blk_at( base, curr_next );
217 if ( nxt->weight == 0 ) // free block
218 {
219 std::uint32_t nxt_idx = curr_next;
220 std::uint32_t nxt_next = nxt->next_offset;
221 BlockT* nxt_nxt_blk = ( nxt_next != detail::kNoBlock ) ? blk_at( base, nxt_next ) : nullptr;
222
223 FreeBlockTreeT::remove( base, hdr, nxt_idx );
224
225 // CoalescingBlock::coalesce_with_next
226 coalescing->coalesce_with_next( nxt, nxt_nxt_blk, b_idx );
227
228 if ( nxt_nxt_blk == nullptr )
229 hdr->last_block_offset = b_idx;
230 else
231 nxt_nxt_blk->prev_offset = b_idx;
232
233 hdr->block_count--;
234 hdr->free_count--;
235 if ( hdr->used_size >= detail::kBlockHeaderGranules )
236 hdr->used_size -= detail::kBlockHeaderGranules;
237 }
238 }
239
240 // Слияние с левым соседом
241 // State: CoalescingBlock::coalesce_with_prev → результат CoalescingBlock(prv)
242 std::uint32_t curr_prev = blk_at( base, b_idx )->prev_offset;
243 if ( curr_prev != detail::kNoBlock )
244 {
245 BlockT* prv = blk_at( base, curr_prev );
246 if ( prv->weight == 0 ) // free block
247 {
248 std::uint32_t prv_idx = curr_prev;
249 std::uint32_t blk_next = blk_at( base, b_idx )->next_offset;
250 BlockT* next_blk = ( blk_next != detail::kNoBlock ) ? blk_at( base, blk_next ) : nullptr;
251
252 FreeBlockTreeT::remove( base, hdr, prv_idx );
253
254 // CoalescingBlock::coalesce_with_prev — current block (blk) is absorbed into prv
255 CoalescingBlock<AddressTraitsT>* result_coalescing =
256 coalescing->coalesce_with_prev( prv, next_blk, prv_idx );
257
258 if ( next_blk == nullptr )
259 hdr->last_block_offset = prv_idx;
260
261 hdr->block_count--;
262 hdr->free_count--;
263 if ( hdr->used_size >= detail::kBlockHeaderGranules )
264 hdr->used_size -= detail::kBlockHeaderGranules;
265
266 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
267 FreeBlock<AddressTraitsT>* fb = result_coalescing->finalize_coalesce();
268 (void)fb;
269 FreeBlockTreeT::insert( base, hdr, prv_idx );
270 return;
271 }
272 }
273
274 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
275 FreeBlock<AddressTraitsT>* fb = coalescing->finalize_coalesce();
276 (void)fb;
277 FreeBlockTreeT::insert( base, hdr, b_idx );
278 }

292 static void rebuild_free_tree( std::uint8_t* base, detail::ManagerHeader* hdr )
293 {
294 hdr->free_tree_root = detail::kNoBlock;
295 hdr->last_block_offset = detail::kNoBlock;
296 std::uint32_t idx = hdr->first_block_offset;
297 while ( idx != detail::kNoBlock )
298 {
299 BlockT* blk = blk_at( base, idx );
300 blk->left_offset = detail::kNoBlock;
301 blk->right_offset = detail::kNoBlock;
302 blk->parent_offset = detail::kNoBlock;
303 blk->avl_height = 0;
304
305 // Issue #106: recover_block_state — исправить некорректные переходные состояния
306 recover_block_state<AddressTraitsT>( blk, idx );
307
308 if ( blk->weight == 0 ) // free block (Block<A> layout: weight field)
309 FreeBlockTreeT::insert( base, hdr, idx );
310 if ( blk->next_offset == detail::kNoBlock )
311 hdr->last_block_offset = idx;
312 idx = blk->next_offset;
313 }
314 }

325 static void repair_linked_list( std::uint8_t* base, detail::ManagerHeader* hdr )
326 {
327 std::uint32_t idx = hdr->first_block_offset;
328 std::uint32_t prev = detail::kNoBlock;
329 while ( idx != detail::kNoBlock )
330 {
331 if ( detail::idx_to_byte_off( idx ) + sizeof( BlockT ) > hdr->total_size )
332 break;
333 BlockT* blk = blk_at( base, idx );
334 blk->prev_offset = prev;
335 prev = idx;
336 idx = blk->next_offset;
337 }

350 static void recompute_counters( std::uint8_t* base, detail::ManagerHeader* hdr )
351 {
352 std::uint32_t block_count = 0, free_count = 0, alloc_count = 0;
353 std::uint32_t used_gran = 0;
354 std::uint32_t idx = hdr->first_block_offset;
355 while ( idx != detail::kNoBlock )
356 {
357 if ( detail::idx_to_byte_off( idx ) + sizeof( BlockT ) > hdr->total_size )
358 break;
359 BlockT* blk = blk_at( base, idx );
360 block_count++;
361 used_gran += detail::kBlockHeaderGranules;
362 if ( blk->weight > 0 ) // allocated block (Block<A> layout: weight > 0)
363 {
364 alloc_count++;
365 used_gran += blk->weight;
366 }
367 else
368 {
369 free_count++;
370 }
371 idx = blk->next_offset;
372 }
373 hdr->block_count = block_count;
374 hdr->free_count = free_count;
375 hdr->alloc_count = alloc_count;
376 hdr->used_size = used_gran;
377 }

382 {
383 assert( idx != detail::kNoBlock );
384 return reinterpret_cast<BlockT*>( base + detail::idx_to_byte_off( idx ) );
