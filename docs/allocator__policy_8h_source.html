114 static void* allocate_from_block( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr,
115 index_type blk_idx, std::size_t user_size )
116 {
117 // State: FreeBlock → FreeBlockRemovedAVL [remove_from_avl]
118 FreeBlockTreeT::remove( base, hdr, blk_idx );
119 FreeBlock<AddressTraitsT>* fb =
120 FreeBlock<AddressTraitsT>::cast_from_raw( detail::block_at<AddressTraitsT>( base, blk_idx ) );
121 FreeBlockRemovedAVL<AddressTraitsT>* removed = fb->remove_from_avl();
122
123 // Issue #146: use AddressTraitsT-specific granule computations.
124 static constexpr index_type kBlkHdrGran =
125 detail::kBlockHeaderGranules_t<AddressTraitsT>;
126
127 index_type blk_total_gran =
128 detail::block_total_granules( base, hdr, detail::block_at<AddressTraitsT>( base, blk_idx ) );
129 index_type data_gran = detail::bytes_to_granules_t<AddressTraitsT>( user_size );
130
131 // Issue #43 Phase 1.3: Overflow protection for needed_gran computation.
132 if ( data_gran > std::numeric_limits<index_type>::max() - kBlkHdrGran )
133 return nullptr; // overflow: request too large
134
135 index_type needed_gran = kBlkHdrGran + data_gran;
136 index_type min_rem_gran = kBlkHdrGran + 1;
137
138 // Issue #43 Phase 1.3: Overflow protection for split check (needed_gran + min_rem_gran).
139 bool can_split = false;
140 if ( needed_gran <= std::numeric_limits<index_type>::max() - min_rem_gran )
141 can_split = ( blk_total_gran >= needed_gran + min_rem_gran );
142
143 if ( can_split )
144 {
145 // State: FreeBlockRemovedAVL → SplittingBlock [begin_splitting]
146 SplittingBlock<AddressTraitsT>* splitting = removed->begin_splitting();
147
148 index_type new_idx = blk_idx + needed_gran;
149 void* new_blk_ptr = detail::block_at<AddressTraitsT>( base, new_idx );
150
151 // Capture old_next before initialize_new_block modifies splitting->next_offset()
152 // Issue #146: compare against AddressTraitsT::no_block (correct sentinel for index_type).
153 index_type curr_next = splitting->next_offset();
154 BlockT* old_next = ( curr_next != AddressTraitsT::no_block )
155 ? detail::block_at<AddressTraitsT>( base, curr_next )
156 : nullptr;
157
158 // SplittingBlock::initialize_new_block — инициализировать новый (remainder) блок
159 splitting->initialize_new_block( new_blk_ptr, new_idx, blk_idx );
160
161 // SplittingBlock::link_new_block — обновить связный список
162 splitting->link_new_block( old_next, new_idx );
163 if ( old_next == nullptr )
164 hdr->last_block_offset = new_idx;
165
166 hdr->block_count++;
167 hdr->free_count++;
168 hdr->used_size += kBlkHdrGran;
169 FreeBlockTreeT::insert( base, hdr, new_idx );
170
171 // State: SplittingBlock → AllocatedBlock [finalize_split]
172 AllocatedBlock<AddressTraitsT>* alloc = splitting->finalize_split( data_gran, blk_idx );
173 (void)alloc; // allocated block pointer obtained via state machine
174 }
175 else
176 {
177 // State: FreeBlockRemovedAVL → AllocatedBlock [mark_as_allocated]
178 AllocatedBlock<AddressTraitsT>* alloc = removed->mark_as_allocated( data_gran, blk_idx );
179 (void)alloc; // allocated block pointer obtained via state machine
180 }
181
182 hdr->alloc_count++;
183 hdr->free_count--;
184 hdr->used_size += data_gran;
185
186 // Return user data pointer (after block header) via canonical detail::user_ptr (Issue #141)
187 return detail::user_ptr<AddressTraitsT>( detail::block_at<AddressTraitsT>( base, blk_idx ) );
188 }

208 static void coalesce( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type blk_idx )
209 {
210 // State: FreeBlockNotInAVL → CoalescingBlock [begin_coalescing]
211 FreeBlockNotInAVL<AddressTraitsT>* not_avl =
212 FreeBlockNotInAVL<AddressTraitsT>::cast_from_raw( detail::block_at<AddressTraitsT>( base, blk_idx ) );
213 CoalescingBlock<AddressTraitsT>* coalescing = not_avl->begin_coalescing();
214
215 // Issue #146: use AddressTraitsT-specific block header granule count.
216 static constexpr index_type kBlkHdrGran = detail::kBlockHeaderGranules_t<AddressTraitsT>;
217
218 index_type b_idx = blk_idx;
219
220 // Слияние с правым соседом
221 // State: CoalescingBlock::coalesce_with_next
222 // Issue #146: compare against AddressTraitsT::no_block for correct sentinel check.
223 index_type curr_next = coalescing->next_offset();
224 if ( curr_next != AddressTraitsT::no_block )
225 {
226 const BlockStateBase<AddressTraitsT>* nxt_state = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>(
227 detail::block_at<AddressTraitsT>( base, curr_next ) );
228 if ( nxt_state->weight() == 0 ) // free block
229 {
230 index_type nxt_idx = curr_next;
231 index_type nxt_next = nxt_state->next_offset();
232 BlockT* nxt_nxt_blk = ( nxt_next != AddressTraitsT::no_block )
233 ? detail::block_at<AddressTraitsT>( base, nxt_next )
234 : nullptr;
235
236 FreeBlockTreeT::remove( base, hdr, nxt_idx );
237
238 // CoalescingBlock::coalesce_with_next (also updates nxt_nxt_blk->prev_offset internally)
239 coalescing->coalesce_with_next( detail::block_at<AddressTraitsT>( base, nxt_idx ), nxt_nxt_blk, b_idx );
240
241 if ( nxt_nxt_blk == nullptr )
242 hdr->last_block_offset = b_idx;
243
244 hdr->block_count--;
245 hdr->free_count--;
246 if ( hdr->used_size >= kBlkHdrGran )
247 hdr->used_size -= kBlkHdrGran;
248 }
249 }
250
251 // Слияние с левым соседом
252 // State: CoalescingBlock::coalesce_with_prev → результат CoalescingBlock(prv)
253 index_type curr_prev = coalescing->prev_offset();
254 if ( curr_prev != AddressTraitsT::no_block )
255 {
256 const BlockStateBase<AddressTraitsT>* prv_state = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>(
257 detail::block_at<AddressTraitsT>( base, curr_prev ) );
258 if ( prv_state->weight() == 0 ) // free block
259 {
260 index_type prv_idx = curr_prev;
261 index_type blk_next = coalescing->next_offset();
262 BlockT* next_blk = ( blk_next != AddressTraitsT::no_block )
263 ? detail::block_at<AddressTraitsT>( base, blk_next )
264 : nullptr;
265
266 FreeBlockTreeT::remove( base, hdr, prv_idx );
267
268 // CoalescingBlock::coalesce_with_prev — current block (blk) is absorbed into prv
269 CoalescingBlock<AddressTraitsT>* result_coalescing = coalescing->coalesce_with_prev(
270 detail::block_at<AddressTraitsT>( base, prv_idx ), next_blk, prv_idx );
271
272 if ( next_blk == nullptr )
273 hdr->last_block_offset = prv_idx;
274
275 hdr->block_count--;
276 hdr->free_count--;
277 if ( hdr->used_size >= kBlkHdrGran )
278 hdr->used_size -= kBlkHdrGran;
279
280 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
281 FreeBlock<AddressTraitsT>* fb = result_coalescing->finalize_coalesce();
282 (void)fb;
283 FreeBlockTreeT::insert( base, hdr, prv_idx );
284 return;
285 }
286 }
287
288 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
289 FreeBlock<AddressTraitsT>* fb = coalescing->finalize_coalesce();
290 (void)fb;
291 FreeBlockTreeT::insert( base, hdr, b_idx );
292 }

306 static void rebuild_free_tree( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr )
307 {
308 hdr->free_tree_root = AddressTraitsT::no_block;
309 hdr->last_block_offset = AddressTraitsT::no_block;
310 index_type idx = hdr->first_block_offset;
311 while ( idx != AddressTraitsT::no_block )
312 {
313 void* blk_ptr = detail::block_at<AddressTraitsT>( base, idx );
314
315 // Reset AVL fields via BlockStateBase (Issue #114, #168)
316 BlockState::reset_avl_fields_of( blk_ptr );
317
318 // Issue #106: recover state — fix incorrect transitional states
319 BlockState::recover_state( blk_ptr, idx );
320
321 if ( BlockState::get_weight( blk_ptr ) == 0 ) // free block
322 FreeBlockTreeT::insert( base, hdr, idx );
323 // Issue #146: use AddressTraitsT::no_block for correct sentinel check.
324 index_type next_idx = BlockState::get_next_offset( blk_ptr );
325 if ( next_idx == AddressTraitsT::no_block )
326 hdr->last_block_offset = idx;
327 idx = next_idx;
328 }
329 }

340 static void repair_linked_list( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr )
341 {
342 index_type idx = hdr->first_block_offset;
343 index_type prev = AddressTraitsT::no_block;
344 while ( idx != AddressTraitsT::no_block )
345 {
346 // Issue #146: use AddressTraitsT::granule_size for correct size check.
347 if ( static_cast<std::size_t>( idx ) * AddressTraitsT::granule_size + sizeof( BlockT ) > hdr->total_size )
348 break;
349 void* blk_ptr = detail::block_at<AddressTraitsT>( base, idx );
350 BlockState::repair_prev_offset( blk_ptr, prev ); // Issue #114, #168
351 prev = idx;
352 index_type next_offset = BlockState::get_next_offset( blk_ptr );
353 idx = next_offset;
354 }

367 static void recompute_counters( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr )
368 {
369 // Issue #146: use AddressTraitsT-specific block header granule count.
370 static constexpr index_type kBlkHdrGran = detail::kBlockHeaderGranules_t<AddressTraitsT>;
371
372 index_type block_count = 0, free_count = 0, alloc_count = 0;
373 index_type used_gran = 0;
374 index_type idx = hdr->first_block_offset;
375 while ( idx != AddressTraitsT::no_block )
376 {
377 // Issue #146: use AddressTraitsT::granule_size for correct size check.
378 if ( static_cast<std::size_t>( idx ) * AddressTraitsT::granule_size + sizeof( BlockT ) > hdr->total_size )
379 break;
380 const void* blk_ptr = detail::block_at<AddressTraitsT>( base, idx );
381 block_count++;
382 used_gran += kBlkHdrGran;
383 index_type w = BlockState::get_weight( blk_ptr ); // Issue #114, #168
384 if ( w > 0 ) // allocated block
385 {
386 alloc_count++;
387 used_gran += w;
388 }
389 else
390 {
391 free_count++;
392 }
393 idx = BlockState::get_next_offset( blk_ptr );
394 }
395 hdr->block_count = block_count;
396 hdr->free_count = free_count;
397 hdr->alloc_count = alloc_count;
398 hdr->used_size = used_gran;
