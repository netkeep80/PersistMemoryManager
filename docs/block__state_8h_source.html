163 static void recover_state( void* raw_blk, index_type own_idx ) noexcept
164 {
165 auto* blk = reinterpret_cast<BlockStateBase*>( raw_blk );
166 // Если weight > 0, но root_offset неверен — исправляем
167 if ( blk->weight() > 0 && blk->root_offset() != own_idx )
168 blk->set_root_offset( own_idx );
169 // Если weight == 0, но root_offset != 0 — исправляем
170 if ( blk->weight() == 0 && blk->root_offset() != 0 )
171 blk->set_root_offset( 0 );
172 }

868template <typename AddressTraitsT>
869int detect_block_state( const void* raw_blk, typename AddressTraitsT::index_type own_idx ) noexcept
870{
871 const auto* base = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>( raw_blk );
872 if ( base->is_free() )
873 return 0; // FreeBlock (или переходное — требует проверки AVL)
874 if ( base->is_allocated( own_idx ) )
875 return 1; // AllocatedBlock
876 return -1; // Неопределённое состояние (ошибка или переходное)
