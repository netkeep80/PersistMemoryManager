538template <typename AddressTraitsT>
539int detect_block_state( const void* raw_blk, typename AddressTraitsT::index_type own_idx ) noexcept
540{
541 const auto* base = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>( raw_blk );
542 if ( base->is_free() )
543 return 0; // FreeBlock (или переходное — требует проверки AVL)
544 if ( base->is_allocated( own_idx ) )
545 return 1; // AllocatedBlock
546 return -1; // Неопределённое состояние (ошибка или переходное)
547}

559template <typename AddressTraitsT>
560void recover_block_state( void* raw_blk, typename AddressTraitsT::index_type own_idx ) noexcept
561{
562 auto* blk = reinterpret_cast<Block<AddressTraitsT>*>( raw_blk );
563
564 // Если weight > 0, но root_offset неверен — исправляем
565 if ( blk->weight > 0 && blk->root_offset != own_idx )
566 {
567 blk->root_offset = own_idx;
568 }
569
570 // Если weight == 0, но root_offset != 0 — исправляем
571 if ( blk->weight == 0 && blk->root_offset != 0 )
572 {
573 blk->root_offset = 0;
574 }
575}
