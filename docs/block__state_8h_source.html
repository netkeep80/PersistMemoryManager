841template <typename AddressTraitsT>
842int detect_block_state( const void* raw_blk, typename AddressTraitsT::index_type own_idx ) noexcept
843{
844 const auto* base = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>( raw_blk );
845 if ( base->is_free() )
