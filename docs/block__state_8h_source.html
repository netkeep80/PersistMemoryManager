729 void coalesce_with_next( void* next_blk, void* next_next_blk, index_type own_idx ) noexcept
730 {
731 auto* nxt = reinterpret_cast<CoalescingBlock<AddressTraitsT>*>( next_blk );
732

742 std::memset( next_blk, 0, sizeof( Block<AddressTraitsT> ) );

758 prv->set_next_offset( Base::next_offset() );

796template <typename AddressTraitsT>
797int detect_block_state( const void* raw_blk, typename AddressTraitsT::index_type own_idx ) noexcept
798{
799 const auto* base = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>( raw_blk );
800 if ( base->is_free() )
801 return 0; // FreeBlock (или переходное — требует проверки AVL)
802 if ( base->is_allocated( own_idx ) )
803 return 1; // AllocatedBlock
804 return -1; // Неопределённое состояние (ошибка или переходное)
805}
