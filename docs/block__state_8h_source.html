148 static void recover_state( void* raw_blk, index_type own_idx ) noexcept
149 {

151 // Если weight > 0, но root_offset неверен — исправляем

702 void coalesce_with_next( void* next_blk, void* next_next_blk, index_type own_idx ) noexcept
703 {

769template <typename AddressTraitsT>
770int detect_block_state( const void* raw_blk, typename AddressTraitsT::index_type own_idx ) noexcept
771{
772 const auto* base = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>( raw_blk );
773 if ( base->is_free() )
774 return 0; // FreeBlock (или переходное — требует проверки AVL)
775 if ( base->is_allocated( own_idx ) )
776 return 1; // AllocatedBlock
777 return -1; // Неопределённое состояние (ошибка или переходное)
778}
