785 prv->set_next_offset( Base::next_offset() );

823template <typename AddressTraitsT>
824int detect_block_state( const void* raw_blk, typename AddressTraitsT::index_type own_idx ) noexcept
825{
826 const auto* base = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>( raw_blk );
827 if ( base->is_free() )
828 return 0; // FreeBlock (или переходное — требует проверки AVL)
829 if ( base->is_allocated( own_idx ) )
830 return 1; // AllocatedBlock
