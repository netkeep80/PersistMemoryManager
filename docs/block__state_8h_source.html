163 static void recover_state( void* raw_blk, index_type own_idx ) noexcept
164 {
165 auto* blk = reinterpret_cast<BlockStateBase*>( raw_blk );
166 // Если weight > 0, но root_offset неверен — исправляем
167 if ( blk->weight() > 0 && blk->root_offset() != own_idx )
168 blk->set_root_offset( own_idx );
169 // Если weight == 0, но root_offset != 0 — исправляем
170 if ( blk->weight() == 0 && blk->root_offset() != 0 )
171 blk->set_root_offset( 0 );
172 }

184 static void verify_state( const void* raw_blk, index_type own_idx, VerifyResult& result ) noexcept
185 {
186 const auto* blk = reinterpret_cast<const BlockStateBase*>( raw_blk );
187 if ( blk->weight() > 0 && blk->root_offset() != own_idx )
188 {
189 result.add( ViolationType::BlockStateInconsistent, DiagnosticAction::NoAction,
190 static_cast<std::uint64_t>( own_idx ), static_cast<std::uint64_t>( own_idx ),

811template <typename AddressTraitsT>
812int detect_block_state( const void* raw_blk, typename AddressTraitsT::index_type own_idx ) noexcept
813{
814 const auto* base = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>( raw_blk );
815 if ( base->is_free() )
816 return 0; // FreeBlock (или переходное — требует проверки AVL)
817 if ( base->is_allocated( own_idx ) )
818 return 1; // AllocatedBlock
819 return -1; // Неопределённое состояние (ошибка или переходное)
820}
