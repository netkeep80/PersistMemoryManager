398 struct _WeightUpdateFn
399 {
400 void operator()( node_pptr p ) const noexcept
401 {
402 if ( p.is_null() )
403 return;
404 // Update height via shared helper.
405 detail::avl_update_height( p );
406 // Update weight (subtree size) = 1 + left_weight + right_weight.
407 auto& tn = p.tree_node();
408
409 auto left_idx = tn.get_left();
410 auto right_idx = tn.get_right();
411 node_pptr left_p = ( left_idx != no_block ) ? node_pptr( left_idx ) : node_pptr();
412 node_pptr right_p = ( right_idx != no_block ) ? node_pptr( right_idx ) : node_pptr();
413 index_type lw = _subtree_size( left_p );
414 index_type rw = _subtree_size( right_p );
415 tn.set_weight( static_cast<index_type>( 1 + lw + rw ) );
416 }

438 static node_pptr _avl_find_by_index( node_pptr p, std::size_t index ) noexcept
439 {
440 while ( !p.is_null() )
441 {
442 auto& tn = p.tree_node();
443 auto left_idx = tn.get_left();
444
445 index_type left_size = ( left_idx != no_block ) ? node_pptr( left_idx ).tree_node().get_weight()
446 : static_cast<index_type>( 0 );
447
448 if ( index < static_cast<std::size_t>( left_size ) )
449 {
450 // Ищем в левом поддереве.
451 p = node_pptr( left_idx );
452 }
453 else if ( index == static_cast<std::size_t>( left_size ) )
454 {
455 // Нашли!
456 return p;
457 }
458 else
459 {
460 // Ищем в правом поддереве с скорректированным индексом.
461 index -= static_cast<std::size_t>( left_size ) + 1;
462 auto right_idx = tn.get_right();
463 if ( right_idx == no_block )
464 return node_pptr();
465 p = node_pptr( right_idx );
466 }
467 }
468 return node_pptr();
469 }
