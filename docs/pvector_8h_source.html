338 while ( true )
339 {
340 auto parent_idx = cur.tree_node().get_parent();
341 if ( parent_idx == no_block )
342 {
343 // Достигли корня снизу справа — конец обхода.
344 _current_idx = static_cast<index_type>( 0 );
345 return *this;
346 }
347 node_pptr parent( parent_idx );
348 auto parent_left = parent.tree_node().get_left();
349 if ( parent_left == cur.offset() )

352 _current_idx = parent_idx;

406 static void _update_node( node_pptr p ) noexcept
407 {
408 if ( p.is_null() )
409 return;
410 auto& tn = p.tree_node();
411
412 auto left_idx = tn.get_left();
413 auto right_idx = tn.get_right();
414
415 node_pptr left_p = ( left_idx != no_block ) ? node_pptr( left_idx ) : node_pptr();
416 node_pptr right_p = ( right_idx != no_block ) ? node_pptr( right_idx ) : node_pptr();
417
418 std::int16_t lh = _height( left_p );
419 std::int16_t rh = _height( right_p );
420 tn.set_height( static_cast<std::int16_t>( 1 + ( lh > rh ? lh : rh ) ) );
421
422 index_type lw = _subtree_size( left_p );
423 index_type rw = _subtree_size( right_p );
424 tn.set_weight( static_cast<index_type>( 1 + lw + rw ) );

523 void _rebalance_up( node_pptr p ) noexcept
524 {
525 while ( !p.is_null() )
526 {
527 _update_node( p );
528
529 auto& tn = p.tree_node();
530
531 auto left_idx = tn.get_left();
532 auto right_idx = tn.get_right();
533
534 node_pptr left_p = ( left_idx != no_block ) ? node_pptr( left_idx ) : node_pptr();
535 node_pptr right_p = ( right_idx != no_block ) ? node_pptr( right_idx ) : node_pptr();
536
537 std::int16_t bf = static_cast<std::int16_t>( _height( left_p ) - _height( right_p ) );
538
539 if ( bf > 1 )
540 {
541 // Левый перевес
542 auto ll_idx = left_p.tree_node().get_left();
543 auto lr_idx = left_p.tree_node().get_right();
544 auto ll_h = ( ll_idx != no_block ) ? _height( node_pptr( ll_idx ) ) : std::int16_t( 0 );
545 auto lr_h = ( lr_idx != no_block ) ? _height( node_pptr( lr_idx ) ) : std::int16_t( 0 );
546 if ( lr_h > ll_h )
547 _rotate_left( left_p );
548 _rotate_right( p );
549 // После поворота p переместился вниз; его новый родитель — это тот, кто сейчас на его месте.
550 // Продолжаем снизу вверх от нового положения p.
551 auto p_par = p.tree_node().get_parent();
552 p = ( p_par != no_block ) ? node_pptr( p_par ) : node_pptr();
553 }
554 else if ( bf < -1 )
555 {
556 // Правый перевес
557 auto rl_idx = right_p.tree_node().get_left();
558 auto rr_idx = right_p.tree_node().get_right();
559 auto rl_h = ( rl_idx != no_block ) ? _height( node_pptr( rl_idx ) ) : std::int16_t( 0 );
560 auto rr_h = ( rr_idx != no_block ) ? _height( node_pptr( rr_idx ) ) : std::int16_t( 0 );
561 if ( rl_h > rr_h )
562 _rotate_right( right_p );
563 _rotate_left( p );
564 auto p_par = p.tree_node().get_parent();
565 p = ( p_par != no_block ) ? node_pptr( p_par ) : node_pptr();
566 }
567 else
568 {
569 auto p_par = tn.get_parent();
570 p = ( p_par != no_block ) ? node_pptr( p_par ) : node_pptr();
571 }
572 }
573 }

576 void _avl_insert_rightmost( node_pptr new_node ) noexcept
577 {
578 if ( _root_idx == static_cast<index_type>( 0 ) )
579 {
580 _root_idx = new_node.offset();
581 return;
582 }
583
584 // Идём всегда вправо до листа.
585 node_pptr cur( _root_idx );
586 while ( true )
587 {
588 auto right_idx = cur.tree_node().get_right();
589 if ( right_idx == no_block )
590 break;
591 cur = node_pptr( right_idx );
592 }
593
594 // Вставляем new_node как правый потомок cur.
595 cur.tree_node().set_right( new_node.offset() );
596 new_node.tree_node().set_parent( cur.offset() );
597
598 // Перебалансируем снизу вверх.
599 _rebalance_up( cur );

611 static node_pptr _avl_find_by_index( node_pptr p, std::size_t index ) noexcept
612 {
613 while ( !p.is_null() )
614 {
615 auto& tn = p.tree_node();
616 auto left_idx = tn.get_left();
617
618 index_type left_size = ( left_idx != no_block ) ? node_pptr( left_idx ).tree_node().get_weight()
619 : static_cast<index_type>( 0 );
620
621 if ( index < static_cast<std::size_t>( left_size ) )
622 {
623 // Ищем в левом поддереве.
624 p = node_pptr( left_idx );
625 }
626 else if ( index == static_cast<std::size_t>( left_size ) )
627 {
628 // Нашли!
629 return p;
630 }
631 else
632 {
633 // Ищем в правом поддереве с скорректированным индексом.
634 index -= static_cast<std::size_t>( left_size ) + 1;
635 auto right_idx = tn.get_right();
636 if ( right_idx == no_block )
637 return node_pptr();
638 p = node_pptr( right_idx );
639 }
640 }
641 return node_pptr();
642 }

650 void _avl_remove( node_pptr target ) noexcept
651 {
652 auto& tn = target.tree_node();
653 auto left_idx = tn.get_left();
654 auto right_idx = tn.get_right();
655 auto par_idx = tn.get_parent();
656
657 node_pptr par_p = ( par_idx != no_block ) ? node_pptr( par_idx ) : node_pptr();
658
659 if ( left_idx == no_block && right_idx == no_block )
660 {
661 // Листовой узел — просто удаляем.
662 _set_child( par_p, target, node_pptr() );
663 if ( !par_p.is_null() )
664 _rebalance_up( par_p );
665 }
666 else if ( left_idx == no_block )
667 {
668 // Только правый потомок.
669 node_pptr right_p( right_idx );
670 right_p.tree_node().set_parent( par_idx );
671 _set_child( par_p, target, right_p );
672 if ( !par_p.is_null() )
673 _rebalance_up( par_p );
674 else
675 _update_node( right_p );
676 }
677 else if ( right_idx == no_block )
678 {
679 // Только левый потомок.
680 node_pptr left_p( left_idx );
681 left_p.tree_node().set_parent( par_idx );
682 _set_child( par_p, target, left_p );
683 if ( !par_p.is_null() )
684 _rebalance_up( par_p );
685 else
686 _update_node( left_p );
687 }
688 else
689 {
690 // Два потомка — ищем in-order successor (крайний левый в правом поддереве).
691 node_pptr successor( right_idx );
692 while ( true )
693 {
694 auto sl = successor.tree_node().get_left();
695 if ( sl == no_block )
696 break;
697 successor = node_pptr( sl );
698 }
699
700 // Запоминаем родителя successor'а до перестановки.
701 auto succ_par_idx = successor.tree_node().get_parent();
702 auto succ_rgt_idx = successor.tree_node().get_right();
703 auto& succ_tn = successor.tree_node();
704
705 node_pptr succ_par_p = ( succ_par_idx != static_cast<index_type>( target.offset() ) )
706 ? node_pptr( succ_par_idx )
707 : node_pptr();
708
709 // Отсоединяем successor от его текущего места.
710 if ( succ_par_idx == target.offset() )
711 {
712 // Successor — прямой правый потомок target.
713 // ничего не делаем — reconnect ниже
714 }
715 else
716 {
717 // Правый потомок successor'а становится левым потомком его родителя.
718 node_pptr succ_parent( succ_par_idx );
719 if ( succ_rgt_idx != no_block )
720 {
721 node_pptr succ_rgt( succ_rgt_idx );
722 succ_rgt.tree_node().set_parent( succ_par_idx );
723 succ_parent.tree_node().set_left( succ_rgt_idx );
724 }
725 else
726 {
727 succ_parent.tree_node().set_left( no_block );
728 }
729 }
730
731 // Ставим successor на место target.
732 succ_tn.set_left( left_idx );
733 node_pptr( left_idx ).tree_node().set_parent( successor.offset() );
734
735 if ( succ_par_idx == target.offset() )
736 {
737 // Правый потомок successor'а остаётся как есть.
738 succ_tn.set_right( succ_rgt_idx );
739 if ( succ_rgt_idx != no_block )
740 node_pptr( succ_rgt_idx ).tree_node().set_parent( successor.offset() );
741 }
742 else
743 {
744 succ_tn.set_right( right_idx );
745 node_pptr( right_idx ).tree_node().set_parent( successor.offset() );
746 }
747
748 succ_tn.set_parent( par_idx );
749 _set_child( par_p, target, successor );
750
751 // Перебалансируем от нижней точки изменений.
752 node_pptr rebalance_start = ( succ_par_idx == target.offset() ) ? successor : node_pptr( succ_par_idx );
753 _rebalance_up( rebalance_start );
754 }
