319 struct _WeightUpdateFn
320 {
321 void operator()( node_pptr p ) const noexcept
322 {
323 if ( p.is_null() )
324 return;
325 // Update height via shared helper.
326 detail::avl_update_height( p );
327 // Update weight (subtree size) = 1 + left_weight + right_weight.
328 auto& tn = p.tree_node();
329
330 auto left_idx = tn.get_left();
331 auto right_idx = tn.get_right();
332 node_pptr left_p = ( left_idx != no_block ) ? node_pptr( left_idx ) : node_pptr();
333 node_pptr right_p = ( right_idx != no_block ) ? node_pptr( right_idx ) : node_pptr();
334 index_type lw = _subtree_size( left_p );
335 index_type rw = _subtree_size( right_p );
336 tn.set_weight( static_cast<index_type>( 1 + lw + rw ) );
337 }
338 };

341 void _avl_insert_rightmost( node_pptr new_node ) noexcept
342 {
343 // Use shared avl_insert with "always go right" comparator (Issue #188).
344 detail::avl_insert(
345 new_node, _root_idx, []( node_pptr ) -> bool { return false; }, // never go left — always rightmost
346 []( node_pptr p ) -> node_type* { return manager_type::template resolve<node_type>( p ); },

359 static node_pptr _avl_find_by_index( node_pptr p, std::size_t index ) noexcept
360 {
361 while ( !p.is_null() )
362 {
363 auto& tn = p.tree_node();
364 auto left_idx = tn.get_left();
365
366 index_type left_size = ( left_idx != no_block ) ? node_pptr( left_idx ).tree_node().get_weight()
367 : static_cast<index_type>( 0 );
368
369 if ( index < static_cast<std::size_t>( left_size ) )
370 {
371 // Ищем в левом поддереве.
372 p = node_pptr( left_idx );
373 }
374 else if ( index == static_cast<std::size_t>( left_size ) )
375 {
376 // Нашли!
377 return p;
378 }
379 else
380 {
381 // Ищем в правом поддереве с скорректированным индексом.
382 index -= static_cast<std::size_t>( left_size ) + 1;
383 auto right_idx = tn.get_right();
384 if ( right_idx == no_block )
385 return node_pptr();
386 p = node_pptr( right_idx );
387 }
388 }
389 return node_pptr();
390 }
