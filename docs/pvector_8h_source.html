346 return *this;

348 node_pptr cur( _current_idx );

350 // Если есть правый потомок — идём в его крайний левый узел.
351 auto right_idx = cur.tree_node().get_right();

353 {
354 node_pptr right( right_idx );

356 {
357 auto left_idx = right.tree_node().get_left();
358 if ( left_idx == no_block )
359 break;
360 right = node_pptr( left_idx );
361 }
362 _current_idx = right.offset();

364 }
365
366 // Иначе — идём вверх, пока не окажемся левым потомком.
367 while ( true )
368 {
369 auto parent_idx = cur.tree_node().get_parent();
370 if ( parent_idx == no_block )
371 {
372 // Достигли корня снизу справа — конец обхода.
373 _current_idx = static_cast<index_type>( 0 );
374 return *this;
375 }
376 node_pptr parent( parent_idx );
377 auto parent_left = parent.tree_node().get_left();
378 if ( parent_left == cur.offset() )
379 {
380 // cur — левый потомок: parent — следующий.
381 _current_idx = parent_idx;
382 return *this;

425 struct _WeightUpdateFn
426 {
427 void operator()( node_pptr p ) const noexcept
428 {
429 if ( p.is_null() )
430 return;
431 // Update height via shared helper.
432 detail::avl_update_height( p );
433 // Update weight (subtree size) = 1 + left_weight + right_weight.
434 auto& tn = p.tree_node();
435
436 auto left_idx = tn.get_left();
437 auto right_idx = tn.get_right();
438 node_pptr left_p = ( left_idx != no_block ) ? node_pptr( left_idx ) : node_pptr();
439 node_pptr right_p = ( right_idx != no_block ) ? node_pptr( right_idx ) : node_pptr();
440 index_type lw = _subtree_size( left_p );
441 index_type rw = _subtree_size( right_p );
442 tn.set_weight( static_cast<index_type>( 1 + lw + rw ) );
443 }
444 };

465 static node_pptr _avl_find_by_index( node_pptr p, std::size_t index ) noexcept
466 {
467 while ( !p.is_null() )
468 {
469 auto& tn = p.tree_node();
470 auto left_idx = tn.get_left();
471
472 index_type left_size = ( left_idx != no_block ) ? node_pptr( left_idx ).tree_node().get_weight()
473 : static_cast<index_type>( 0 );
474
475 if ( index < static_cast<std::size_t>( left_size ) )
476 {
477 // Ищем в левом поддереве.
478 p = node_pptr( left_idx );
479 }
480 else if ( index == static_cast<std::size_t>( left_size ) )
481 {
482 // Нашли!
483 return p;
484 }
485 else
486 {
487 // Ищем в правом поддереве с скорректированным индексом.
488 index -= static_cast<std::size_t>( left_size ) + 1;
489 auto right_idx = tn.get_right();
490 if ( right_idx == no_block )
491 return node_pptr();
492 p = node_pptr( right_idx );
493 }
494 }
495 return node_pptr();
496 }
