109 static void insert( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type blk_idx )
110 {
111 void* blk = detail::block_at<AddressTraitsT>( base, blk_idx );
112 BlockState::set_left_offset_of( blk, AddressTraitsT::no_block );
113 BlockState::set_right_offset_of( blk, AddressTraitsT::no_block );
114 BlockState::set_parent_offset_of( blk, AddressTraitsT::no_block );
115 BlockState::set_avl_height_of( blk, 1 );
116 if ( hdr->free_tree_root == AddressTraitsT::no_block )
117 {
118 hdr->free_tree_root = blk_idx;
119 return;
120 }
121 // Issue #59: cache total_gran once; compute blk size in granules before the traversal loop
122 // Issue #146: use AddressTraitsT::granule_size for correct byte→granule conversion.
123 index_type total_gran = detail::byte_off_to_idx_t<AddressTraitsT>( hdr->total_size );
124 index_type blk_next = BlockState::get_next_offset( blk );
125 index_type blk_gran =
126 ( blk_next != AddressTraitsT::no_block ) ? ( blk_next - blk_idx ) : ( total_gran - blk_idx );
127 index_type cur = hdr->free_tree_root, parent = AddressTraitsT::no_block;
128 bool go_left = false;
129 while ( cur != AddressTraitsT::no_block )
130 {
131 parent = cur;
132 const void* n = detail::block_at<AddressTraitsT>( base, cur );
133 index_type n_next = BlockState::get_next_offset( n );
134 index_type n_gran = ( n_next != AddressTraitsT::no_block ) ? ( n_next - cur ) : ( total_gran - cur );
135 bool smaller = ( blk_gran < n_gran ) || ( blk_gran == n_gran && blk_idx < cur );
136 go_left = smaller;
137 cur = smaller ? BlockState::get_left_offset( n ) : BlockState::get_right_offset( n );
138 }
139 BlockState::set_parent_offset_of( blk, parent );
140 if ( go_left )
141 BlockState::set_left_offset_of( detail::block_at<AddressTraitsT>( base, parent ), blk_idx );
142 else
143 BlockState::set_right_offset_of( detail::block_at<AddressTraitsT>( base, parent ), blk_idx );
144 rebalance_up( base, hdr, parent );
145 }

148 static void remove( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type blk_idx )
149 {
150 void* blk = detail::block_at<AddressTraitsT>( base, blk_idx );
151 index_type parent = BlockState::get_parent_offset( blk );
152 index_type left = BlockState::get_left_offset( blk );
153 index_type right = BlockState::get_right_offset( blk );
154 index_type rebal = AddressTraitsT::no_block;
155
156 if ( left == AddressTraitsT::no_block && right == AddressTraitsT::no_block )
157 {
158 set_child( base, hdr, parent, blk_idx, AddressTraitsT::no_block );
159 rebal = parent;
160 }
161 else if ( left == AddressTraitsT::no_block || right == AddressTraitsT::no_block )
162 {
163 index_type child = ( left != AddressTraitsT::no_block ) ? left : right;
164 BlockState::set_parent_offset_of( detail::block_at<AddressTraitsT>( base, child ), parent );
165 set_child( base, hdr, parent, blk_idx, child );
166 rebal = parent;
167 }
168 else
169 {
170 index_type succ_idx = min_node( base, right );
171 void* succ = detail::block_at<AddressTraitsT>( base, succ_idx );
172 index_type succ_parent = BlockState::get_parent_offset( succ );
173 index_type succ_right = BlockState::get_right_offset( succ );
174
175 if ( succ_parent != blk_idx )
176 {
177 set_child( base, hdr, succ_parent, succ_idx, succ_right );
178 if ( succ_right != AddressTraitsT::no_block )
179 BlockState::set_parent_offset_of( detail::block_at<AddressTraitsT>( base, succ_right ),
180 succ_parent );
181 BlockState::set_right_offset_of( succ, right );
182 BlockState::set_parent_offset_of( detail::block_at<AddressTraitsT>( base, right ), succ_idx );
183 rebal = succ_parent;
184 }
185 else
186 {
187 rebal = succ_idx;
188 }
189 BlockState::set_left_offset_of( succ, left );
190 BlockState::set_parent_offset_of( detail::block_at<AddressTraitsT>( base, left ), succ_idx );
191 BlockState::set_parent_offset_of( succ, parent );
192 set_child( base, hdr, parent, blk_idx, succ_idx );
193 update_height( base, succ_idx );
194 }
195 BlockState::set_left_offset_of( blk, AddressTraitsT::no_block );
196 BlockState::set_right_offset_of( blk, AddressTraitsT::no_block );
197 BlockState::set_parent_offset_of( blk, AddressTraitsT::no_block );
198 BlockState::set_avl_height_of( blk, 0 );
199 rebalance_up( base, hdr, rebal );
200 }

203 static index_type find_best_fit( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr,
204 index_type needed_granules )
205 {
206 // Issue #59: cache total_gran once to avoid repeated hdr->total_size reads in the hot path
207 // Issue #146: use AddressTraitsT::granule_size for correct byte→granule conversion.
208 index_type total_gran = detail::byte_off_to_idx_t<AddressTraitsT>( hdr->total_size );
209 index_type cur = hdr->free_tree_root, result = AddressTraitsT::no_block;
210 while ( cur != AddressTraitsT::no_block )
211 {
212 const void* node = detail::block_at<AddressTraitsT>( base, cur );
213 index_type node_next = BlockState::get_next_offset( node );
214 // Issue #146: compare against AddressTraitsT::no_block (not detail::kNoBlock)
215 // to correctly handle SmallAddressTraits (uint16_t) sentinel 0xFFFF.
216 index_type cur_gran =
217 ( node_next != AddressTraitsT::no_block ) ? ( node_next - cur ) : ( total_gran - cur );
218 if ( cur_gran >= needed_granules )
219 {
220 result = cur;
221 cur = BlockState::get_left_offset( node );
222 }
223 else
224 {
225 cur = BlockState::get_right_offset( node );
226 }
227 }
228 return result;
229 }

239 static void update_height( std::uint8_t* base, index_type node_idx )
240 {
241 void* node = detail::block_at<AddressTraitsT>( base, node_idx );
242 std::int32_t h = 1 + ( std::max )( height( base, BlockState::get_left_offset( node ) ),
243 height( base, BlockState::get_right_offset( node ) ) );
244 assert( h <= std::numeric_limits<std::int16_t>::max() ); // tree height must fit in int16_t
245 BlockState::set_avl_height_of( node, static_cast<std::int16_t>( h ) );
246 }

256 static void set_child( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type parent,
257 index_type old_child, index_type new_child )
258 {
259 if ( parent == AddressTraitsT::no_block )

261 hdr->free_tree_root = new_child;
262 return;

264 void* p = detail::block_at<AddressTraitsT>( base, parent );
265 if ( BlockState::get_left_offset( p ) == old_child )
266 BlockState::set_left_offset_of( p, new_child );
267 else
268 BlockState::set_right_offset_of( p, new_child );

271 static index_type rotate_right( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type y_idx )
272 {
273 void* y = detail::block_at<AddressTraitsT>( base, y_idx );
274 index_type x_idx = BlockState::get_left_offset( y );
275 void* x = detail::block_at<AddressTraitsT>( base, x_idx );
276 index_type t2 = BlockState::get_right_offset( x );
277 index_type y_parent = BlockState::get_parent_offset( y );
278
279 BlockState::set_right_offset_of( x, y_idx );
280 BlockState::set_left_offset_of( y, t2 );
281 // x takes y's old parent; y's new parent is x.
282 BlockState::set_parent_offset_of( x, y_parent );
283 BlockState::set_parent_offset_of( y, x_idx );
284 if ( t2 != AddressTraitsT::no_block )
285 BlockState::set_parent_offset_of( detail::block_at<AddressTraitsT>( base, t2 ), y_idx );
286 set_child( base, hdr, y_parent, y_idx, x_idx );
287 update_height( base, y_idx );
288 update_height( base, x_idx );
289 return x_idx;
290 }

292 static index_type rotate_left( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type x_idx )
293 {
294 void* x = detail::block_at<AddressTraitsT>( base, x_idx );
295 index_type y_idx = BlockState::get_right_offset( x );
296 void* y = detail::block_at<AddressTraitsT>( base, y_idx );
297 index_type t2 = BlockState::get_left_offset( y );
298 index_type x_parent = BlockState::get_parent_offset( x );
299
300 BlockState::set_left_offset_of( y, x_idx );
301 BlockState::set_right_offset_of( x, t2 );
302 // y takes x's old parent; x's new parent is y.
303 BlockState::set_parent_offset_of( y, x_parent );
304 BlockState::set_parent_offset_of( x, y_idx );
305 if ( t2 != AddressTraitsT::no_block )
306 BlockState::set_parent_offset_of( detail::block_at<AddressTraitsT>( base, t2 ), x_idx );
307 set_child( base, hdr, x_parent, x_idx, y_idx );
308 update_height( base, x_idx );
309 update_height( base, y_idx );
310 return y_idx;
311 }

313 static void rebalance_up( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type node_idx )
314 {
315 index_type cur = node_idx;
316 while ( cur != AddressTraitsT::no_block )
317 {
318 update_height( base, cur );
319 std::int32_t bf = balance_factor( base, cur );
320 if ( bf > 1 )
321 {
322 void* node = detail::block_at<AddressTraitsT>( base, cur );
323 index_type left_idx = BlockState::get_left_offset( node );
324 if ( balance_factor( base, left_idx ) < 0 )
325 rotate_left( base, hdr, left_idx );
326 cur = rotate_right( base, hdr, cur );
327 }
328 else if ( bf < -1 )
329 {
330 void* node = detail::block_at<AddressTraitsT>( base, cur );
331 index_type right_idx = BlockState::get_right_offset( node );
332 if ( balance_factor( base, right_idx ) > 0 )
333 rotate_right( base, hdr, right_idx );
334 cur = rotate_left( base, hdr, cur );
335 }
336 cur = BlockState::get_parent_offset( detail::block_at<AddressTraitsT>( base, cur ) );
337 }
338 }

340 static index_type min_node( std::uint8_t* base, index_type node_idx )
341 {
342 while ( node_idx != AddressTraitsT::no_block )
343 {
344 index_type left = BlockState::get_left_offset( detail::block_at<AddressTraitsT>( base, node_idx ) );
345 if ( left == AddressTraitsT::no_block )
346 break;
347 node_idx = left;
348 }
349 return node_idx;
350 }
