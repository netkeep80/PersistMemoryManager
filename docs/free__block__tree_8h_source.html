102 static void insert( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type blk_idx )
103 {
104 void* blk = detail::block_at<AddressTraitsT>( base, blk_idx );
105 BlockState::set_left_offset_of( blk, AddressTraitsT::no_block );
106 BlockState::set_right_offset_of( blk, AddressTraitsT::no_block );
107 BlockState::set_parent_offset_of( blk, AddressTraitsT::no_block );
108 BlockState::set_avl_height_of( blk, 1 );
109 if ( hdr->free_tree_root == AddressTraitsT::no_block )
110 {
111 hdr->free_tree_root = blk_idx;
112 return;
113 }
114 // Derive sort key from linear PAP geometry: block_size = next_offset - block_index.
115 // Use AddressTraitsT::granule_size for correct byte→granule conversion.
116 index_type total_gran = detail::byte_off_to_idx_t<AddressTraitsT>( hdr->total_size );
117 index_type blk_next = BlockState::get_next_offset( blk );
118 index_type blk_gran =
119 ( blk_next != AddressTraitsT::no_block ) ? ( blk_next - blk_idx ) : ( total_gran - blk_idx );
120 index_type cur = hdr->free_tree_root, parent = AddressTraitsT::no_block;
121 bool go_left = false;
122 while ( cur != AddressTraitsT::no_block )
123 {
124 parent = cur;
125 const void* n = detail::block_at<AddressTraitsT>( base, cur );
126 index_type n_next = BlockState::get_next_offset( n );
127 index_type n_gran = ( n_next != AddressTraitsT::no_block ) ? ( n_next - cur ) : ( total_gran - cur );
128 // Forest-policy ordering: (block_size, block_index) — smaller size goes left.
129 bool smaller = ( blk_gran < n_gran ) || ( blk_gran == n_gran && blk_idx < cur );
130 go_left = smaller;
131 cur = smaller ? BlockState::get_left_offset( n ) : BlockState::get_right_offset( n );
132 }
133 BlockState::set_parent_offset_of( blk, parent );
134 if ( go_left )
135 BlockState::set_left_offset_of( detail::block_at<AddressTraitsT>( base, parent ), blk_idx );
136 else
137 BlockState::set_right_offset_of( detail::block_at<AddressTraitsT>( base, parent ), blk_idx );
138 // Delegate rebalancing to shared AVL function via BlockPPtr adapter.
139 detail::avl_rebalance_up( BPPtr( base, parent ), hdr->free_tree_root );
140 }

143 static void remove( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type blk_idx )
144 {
145 void* blk = detail::block_at<AddressTraitsT>( base, blk_idx );
146 index_type parent = BlockState::get_parent_offset( blk );
147 index_type left = BlockState::get_left_offset( blk );
148 index_type right = BlockState::get_right_offset( blk );
149 index_type rebal = AddressTraitsT::no_block;
150
151 if ( left == AddressTraitsT::no_block && right == AddressTraitsT::no_block )
152 {
153 set_child( base, hdr, parent, blk_idx, AddressTraitsT::no_block );
154 rebal = parent;
155 }
156 else if ( left == AddressTraitsT::no_block || right == AddressTraitsT::no_block )
157 {
158 index_type child = ( left != AddressTraitsT::no_block ) ? left : right;
159 BlockState::set_parent_offset_of( detail::block_at<AddressTraitsT>( base, child ), parent );
160 set_child( base, hdr, parent, blk_idx, child );
161 rebal = parent;
162 }
163 else
164 {
165 // Delegate min_node to shared AVL function via BlockPPtr.
166 BPPtr succ = detail::avl_min_node( BPPtr( base, right ) );
167 index_type succ_idx = succ.offset();
168 void* succ_raw = detail::block_at<AddressTraitsT>( base, succ_idx );
169 index_type succ_parent = BlockState::get_parent_offset( succ_raw );
170 index_type succ_right = BlockState::get_right_offset( succ_raw );
171
172 if ( succ_parent != blk_idx )
173 {
174 set_child( base, hdr, succ_parent, succ_idx, succ_right );
175 if ( succ_right != AddressTraitsT::no_block )
176 BlockState::set_parent_offset_of( detail::block_at<AddressTraitsT>( base, succ_right ),
177 succ_parent );
178 BlockState::set_right_offset_of( succ_raw, right );
179 BlockState::set_parent_offset_of( detail::block_at<AddressTraitsT>( base, right ), succ_idx );
180 rebal = succ_parent;
181 }
182 else
183 {
184 rebal = succ_idx;
185 }
186 BlockState::set_left_offset_of( succ_raw, left );
187 BlockState::set_parent_offset_of( detail::block_at<AddressTraitsT>( base, left ), succ_idx );
188 BlockState::set_parent_offset_of( succ_raw, parent );
189 set_child( base, hdr, parent, blk_idx, succ_idx );
190 // Delegate height update to shared AVL function via BlockPPtr.
191 detail::avl_update_height( BPPtr( base, succ_idx ) );
192 }
193 BlockState::set_left_offset_of( blk, AddressTraitsT::no_block );
194 BlockState::set_right_offset_of( blk, AddressTraitsT::no_block );
195 BlockState::set_parent_offset_of( blk, AddressTraitsT::no_block );
196 BlockState::set_avl_height_of( blk, 0 );
197 // Delegate rebalancing to shared AVL function via BlockPPtr adapter.
198 detail::avl_rebalance_up( BPPtr( base, rebal ), hdr->free_tree_root );
199 }

204 static index_type find_best_fit( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr,
205 index_type needed_granules )
206 {
207 // Cache total_gran once to avoid repeated hdr->total_size reads in the hot path
208 // Use AddressTraitsT::granule_size for correct byte→granule conversion.
209 index_type total_gran = detail::byte_off_to_idx_t<AddressTraitsT>( hdr->total_size );
210 index_type cur = hdr->free_tree_root, result = AddressTraitsT::no_block;
211 while ( cur != AddressTraitsT::no_block )
212 {
213 const void* node = detail::block_at<AddressTraitsT>( base, cur );
214 index_type node_next = BlockState::get_next_offset( node );
215 // Compare against AddressTraitsT::no_block (not detail::kNoBlock)
216 // to correctly handle SmallAddressTraits (uint16_t) sentinel 0xFFFF.
217 index_type cur_gran =
218 ( node_next != AddressTraitsT::no_block ) ? ( node_next - cur ) : ( total_gran - cur );
219 if ( cur_gran >= needed_granules )
220 {
221 result = cur;
222 cur = BlockState::get_left_offset( node );
223 }
224 else
225 {
226 cur = BlockState::get_right_offset( node );
227 }
228 }
229 return result;
230 }

234 static void set_child( std::uint8_t* base, detail::ManagerHeader<AddressTraitsT>* hdr, index_type parent,
235 index_type old_child, index_type new_child )
236 {
237 if ( parent == AddressTraitsT::no_block )
238 {
239 hdr->free_tree_root = new_child;
240 return;
241 }
242 void* p = detail::block_at<AddressTraitsT>( base, parent );
243 if ( BlockState::get_left_offset( p ) == old_child )
244 BlockState::set_left_offset_of( p, new_child );
245 else
246 BlockState::set_right_offset_of( p, new_child );
247 }
