134 static void insert( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
135 {
136 BlockT* blk = blk_at( base, blk_idx );
137 blk->left_offset = detail::kNoBlock;
138 blk->right_offset = detail::kNoBlock;
139 blk->parent_offset = detail::kNoBlock;
140 blk->avl_height = 1;
141 if ( hdr->free_tree_root == detail::kNoBlock )
142 {
143 hdr->free_tree_root = blk_idx;
144 return;
145 }
146 // Issue #59: cache total_gran once; compute blk size in granules before the traversal loop
147 std::uint32_t total_gran = detail::byte_off_to_idx( hdr->total_size );
148 std::uint32_t blk_gran =
149 ( blk->next_offset != detail::kNoBlock ) ? ( blk->next_offset - blk_idx ) : ( total_gran - blk_idx );
150 std::uint32_t cur = hdr->free_tree_root, parent = detail::kNoBlock;
151 bool go_left = false;
152 while ( cur != detail::kNoBlock )
153 {
154 parent = cur;
155 BlockT* n = blk_at( base, cur );
156 std::uint32_t n_gran =
157 ( n->next_offset != detail::kNoBlock ) ? ( n->next_offset - cur ) : ( total_gran - cur );
158 bool smaller = ( blk_gran < n_gran ) || ( blk_gran == n_gran && blk_idx < cur );
159 go_left = smaller;
160 cur = smaller ? n->left_offset : n->right_offset;
161 }
162 blk->parent_offset = parent;
163 if ( go_left )
164 blk_at( base, parent )->left_offset = blk_idx;
165 else
166 blk_at( base, parent )->right_offset = blk_idx;
167 rebalance_up( base, hdr, parent );
168 }

171 static void remove( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
172 {
173 BlockT* blk = blk_at( base, blk_idx );
174 std::uint32_t parent = blk->parent_offset;
175 std::uint32_t left = blk->left_offset;
176 std::uint32_t right = blk->right_offset;
177 std::uint32_t rebal = detail::kNoBlock;
178
179 if ( left == detail::kNoBlock && right == detail::kNoBlock )
180 {
181 set_child( base, hdr, parent, blk_idx, detail::kNoBlock );
182 rebal = parent;
183 }
184 else if ( left == detail::kNoBlock || right == detail::kNoBlock )
185 {
186 std::uint32_t child = ( left != detail::kNoBlock ) ? left : right;
187 blk_at( base, child )->parent_offset = parent;
188 set_child( base, hdr, parent, blk_idx, child );
189 rebal = parent;
190 }
191 else
192 {
193 std::uint32_t succ_idx = min_node( base, right );
194 BlockT* succ = blk_at( base, succ_idx );
195 std::uint32_t succ_parent = succ->parent_offset;
196 std::uint32_t succ_right = succ->right_offset;
197
198 if ( succ_parent != blk_idx )
199 {
200 set_child( base, hdr, succ_parent, succ_idx, succ_right );
201 if ( succ_right != detail::kNoBlock )
202 blk_at( base, succ_right )->parent_offset = succ_parent;
203 succ->right_offset = right;
204 blk_at( base, right )->parent_offset = succ_idx;
205 rebal = succ_parent;
206 }
207 else
208 {
209 rebal = succ_idx;
210 }
211 succ->left_offset = left;
212 blk_at( base, left )->parent_offset = succ_idx;
213 succ->parent_offset = parent;
214 set_child( base, hdr, parent, blk_idx, succ_idx );
215 update_height( base, succ );
216 }
217 blk->left_offset = detail::kNoBlock;
218 blk->right_offset = detail::kNoBlock;
219 blk->parent_offset = detail::kNoBlock;
220 blk->avl_height = 0;
221 rebalance_up( base, hdr, rebal );
222 }

225 static std::uint32_t find_best_fit( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t needed_granules )
226 {
227 // Issue #59: cache total_gran once to avoid repeated hdr->total_size reads in the hot path
228 std::uint32_t total_gran = detail::byte_off_to_idx( hdr->total_size );
229 std::uint32_t cur = hdr->free_tree_root, result = detail::kNoBlock;
230 while ( cur != detail::kNoBlock )
231 {
232 const BlockT* node = blk_at( base, cur );
233 std::uint32_t cur_gran =
234 ( node->next_offset != detail::kNoBlock ) ? ( node->next_offset - cur ) : ( total_gran - cur );
235 if ( cur_gran >= needed_granules )
236 {
237 result = cur;
238 cur = node->left_offset;
239 }
240 else
241 {
242 cur = node->right_offset;
243 }
244 }
245 return result;
246 }

254 static void update_height( std::uint8_t* base, BlockT* node )
255 {
256 std::int32_t h = 1 + ( std::max )( height( base, node->left_offset ), height( base, node->right_offset ) );
257 assert( h <= std::numeric_limits<std::int16_t>::max() ); // tree height must fit in int16_t
258 node->avl_height = static_cast<std::int16_t>( h );
259 }

267 static void set_child( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t parent,
268 std::uint32_t old_child, std::uint32_t new_child )
269 {
270 if ( parent == detail::kNoBlock )
271 {
272 hdr->free_tree_root = new_child;
273 return;
274 }
275 BlockT* p = blk_at( base, parent );
276 if ( p->left_offset == old_child )

282 static std::uint32_t rotate_right( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t y_idx )
283 {
284 BlockT* y = blk_at( base, y_idx );
285 std::uint32_t x_idx = y->left_offset;
286 BlockT* x = blk_at( base, x_idx );
287 std::uint32_t t2 = x->right_offset;
288
289 x->right_offset = y_idx;
290 y->left_offset = t2;
291 x->parent_offset = y->parent_offset;
292 y->parent_offset = x_idx;
293 if ( t2 != detail::kNoBlock )
294 blk_at( base, t2 )->parent_offset = y_idx;
295 set_child( base, hdr, x->parent_offset, y_idx, x_idx );
296 update_height( base, y );
297 update_height( base, x );
298 return x_idx;
299 }

301 static std::uint32_t rotate_left( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t x_idx )
302 {
303 BlockT* x = blk_at( base, x_idx );
304 std::uint32_t y_idx = x->right_offset;
305 BlockT* y = blk_at( base, y_idx );
306 std::uint32_t t2 = y->left_offset;
307
308 y->left_offset = x_idx;
309 x->right_offset = t2;
310 y->parent_offset = x->parent_offset;
311 x->parent_offset = y_idx;
312 if ( t2 != detail::kNoBlock )
313 blk_at( base, t2 )->parent_offset = x_idx;
314 set_child( base, hdr, y->parent_offset, x_idx, y_idx );
315 update_height( base, x );
316 update_height( base, y );
317 return y_idx;
318 }

320 static void rebalance_up( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t node_idx )
321 {
322 std::uint32_t cur = node_idx;
323 while ( cur != detail::kNoBlock )
324 {
325 BlockT* node = blk_at( base, cur );
326 update_height( base, node );
327 std::int32_t bf = balance_factor( base, node );
328 if ( bf > 1 )
329 {
330 if ( balance_factor( base, blk_at( base, node->left_offset ) ) < 0 )
331 rotate_left( base, hdr, node->left_offset );
332 cur = rotate_right( base, hdr, cur );
333 }
334 else if ( bf < -1 )
335 {
336 if ( balance_factor( base, blk_at( base, node->right_offset ) ) > 0 )
337 rotate_right( base, hdr, node->right_offset );
338 cur = rotate_left( base, hdr, cur );
339 }
340 cur = blk_at( base, cur )->parent_offset;
341 }
342 }

344 static std::uint32_t min_node( std::uint8_t* base, std::uint32_t node_idx )
345 {
346 while ( node_idx != detail::kNoBlock )
347 {
348 std::uint32_t left = blk_at( base, node_idx )->left_offset;
349 if ( left == detail::kNoBlock )
350 break;
351 node_idx = left;
352 }
353 return node_idx;
354 }
