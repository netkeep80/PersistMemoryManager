143template <typename PPtr, typename IndexType> static void avl_rebalance_up( PPtr p, IndexType& root_idx ) noexcept
144{
145 while ( !p.is_null() )
146 {
147 avl_update_height( p );
148 std::int16_t bf = avl_balance_factor( p );
149 if ( bf > 1 )
150 {
151 PPtr left( p.get_tree_left().offset() );
152 if ( avl_balance_factor( left ) < 0 )
153 avl_rotate_left( left, root_idx );
154 p = avl_rotate_right( p, root_idx );
155 }
156 else if ( bf < -1 )
157 {
158 PPtr right( p.get_tree_right().offset() );
159 if ( avl_balance_factor( right ) > 0 )
160 avl_rotate_right( right, root_idx );
161 p = avl_rotate_left( p, root_idx );
162 }
163 p = PPtr( p.get_tree_parent().offset() );
164 }
165}

180template <typename PPtr, typename IndexType, typename CompareThreeWayFn, typename ResolveFn>
181static PPtr avl_find( IndexType root_idx, CompareThreeWayFn&& compare_three_way, ResolveFn&& resolve ) noexcept
182{
183 PPtr cur( root_idx );
184 while ( !cur.is_null() )
185 {
186 if ( resolve( cur ) == nullptr )
187 break;
188 int cmp = compare_three_way( cur );
189 if ( cmp == 0 )
190 return cur;
191 else if ( cmp < 0 )
192 cur = PPtr( cur.get_tree_left().offset() );
193 else
194 cur = PPtr( cur.get_tree_right().offset() );
195 }
196 return PPtr(); // not found
197}

205template <typename PPtr, typename IndexType, typename GoLeftFn, typename ResolveFn>
206static void avl_insert( PPtr new_node, IndexType& root_idx, GoLeftFn&& go_left, ResolveFn&& resolve ) noexcept
207{
208 if ( new_node.is_null() )
209 return;
210 if ( resolve( new_node ) == nullptr )
211 return;
212
213 if ( root_idx == static_cast<IndexType>( 0 ) )
214 {
215 new_node.set_tree_left( PPtr() );
216 new_node.set_tree_right( PPtr() );
217 new_node.set_tree_parent( PPtr() );
218 new_node.set_tree_height( static_cast<std::int16_t>( 1 ) );
219 root_idx = new_node.offset();
220 return;
221 }
222
223 PPtr cur( root_idx );
224 PPtr parent;
225 bool left = false;
226
227 while ( !cur.is_null() )
228 {
229 if ( resolve( cur ) == nullptr )
230 break;
231 parent = cur;
232 left = go_left( cur );
233 if ( left )
234 cur = PPtr( cur.get_tree_left().offset() );
235 else
236 cur = PPtr( cur.get_tree_right().offset() );
237 }
238
239 new_node.set_tree_parent( parent );
240 if ( left )
241 parent.set_tree_left( new_node );
242 else
243 parent.set_tree_right( new_node );
244
245 avl_rebalance_up( parent, root_idx );
246}
