193template <typename PPtr, typename IndexType> static void avl_rebalance_up( PPtr p, IndexType& root_idx ) noexcept
194{
195 while ( !p.is_null() )
196 {
197 avl_update_height( p );
198 std::int16_t bf = avl_balance_factor( p );
199 if ( bf > 1 )
200 {
201 PPtr left = pptr_get_left( p );
202 if ( avl_balance_factor( left ) < 0 )
203 avl_rotate_left( left, root_idx );
204 p = avl_rotate_right( p, root_idx );
205 }
206 else if ( bf < -1 )
207 {
208 PPtr right = pptr_get_right( p );
209 if ( avl_balance_factor( right ) > 0 )
210 avl_rotate_right( right, root_idx );
211 p = avl_rotate_left( p, root_idx );
212 }
213 p = pptr_get_parent( p );
214 }
215}

230template <typename PPtr, typename IndexType, typename CompareThreeWayFn, typename ResolveFn>
231static PPtr avl_find( IndexType root_idx, CompareThreeWayFn&& compare_three_way, ResolveFn&& resolve ) noexcept
232{
233 PPtr cur( root_idx );
234 while ( !cur.is_null() )
235 {
236 if ( resolve( cur ) == nullptr )
237 break;
238 int cmp = compare_three_way( cur );
239 if ( cmp == 0 )
240 return cur;
241 else if ( cmp < 0 )
242 cur = pptr_get_left( cur );

255template <typename PPtr, typename IndexType, typename GoLeftFn, typename ResolveFn>
256static void avl_insert( PPtr new_node, IndexType& root_idx, GoLeftFn&& go_left, ResolveFn&& resolve ) noexcept
257{
258 if ( new_node.is_null() )
259 return;
260 if ( resolve( new_node ) == nullptr )
261 return;
262
263 if ( root_idx == static_cast<IndexType>( 0 ) )
264 {
265 pptr_set_left( new_node, PPtr() );
266 pptr_set_right( new_node, PPtr() );
267 pptr_set_parent( new_node, PPtr() );
268 new_node.tree_node().set_height( static_cast<std::int16_t>( 1 ) );
269 root_idx = new_node.offset();
270 return;
271 }
272
273 PPtr cur( root_idx );
274 PPtr parent;
275 bool left = false;
276
277 while ( !cur.is_null() )
278 {
279 if ( resolve( cur ) == nullptr )
280 break;
281 parent = cur;
282 left = go_left( cur );
283 if ( left )
284 cur = pptr_get_left( cur );
285 else
286 cur = pptr_get_right( cur );
287 }
288
289 pptr_set_parent( new_node, parent );
290 if ( left )
291 pptr_set_left( parent, new_node );
292 else
293 pptr_set_right( parent, new_node );
294
295 avl_rebalance_up( parent, root_idx );
296}
