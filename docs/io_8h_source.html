96template <typename MgrT> inline bool save_manager( const char* filename )
97{
98 using address_traits = typename MgrT::address_traits;
99
100 if ( filename == nullptr || !MgrT::is_initialized() )
101 return false;
102 std::uint8_t* data = MgrT::backend().base_ptr();
103 std::size_t total = MgrT::backend().total_size();
104 if ( data == nullptr || total == 0 )
105 return false;
106
107 // Compute and store CRC32 in the manager header.
108 // The header is located after Block_0 (sizeof(Block<AT>) bytes from base).
109 constexpr std::size_t kHdrOffset = sizeof( pmm::Block<address_traits> );
110 auto* hdr = reinterpret_cast<detail::ManagerHeader<address_traits>*>( data + kHdrOffset );
111 hdr->crc32 = 0; // zero the field before computing CRC
112 hdr->crc32 = detail::compute_image_crc32<address_traits>( data, total );
113
114 // Atomic save — write to temp file, then rename.
115 std::string tmp_path = std::string( filename ) + ".tmp";
116
117 std::FILE* f = std::fopen( tmp_path.c_str(), "wb" );
118 if ( f == nullptr )
119 return false;
120 std::size_t written = std::fwrite( data, 1, total, f );
121 if ( std::fflush( f ) != 0 )
122 {
123 std::fclose( f );
124 std::remove( tmp_path.c_str() );
125 return false;
126 }
127 std::fclose( f );
128
129 if ( written != total )
130 {
131 std::remove( tmp_path.c_str() );
132 return false;
133 }
134
135 // Atomic rename: tmp → final
136 if ( !detail::atomic_rename( tmp_path.c_str(), filename ) )
137 {
138 std::remove( tmp_path.c_str() );
139 return false;
140 }
141 return true;
142}

164template <typename MgrT> inline bool load_manager_from_file( const char* filename, VerifyResult& result )
165{
166 using address_traits = typename MgrT::address_traits;
167
168 if ( filename == nullptr )
169 return false;
170
171 std::uint8_t* buf = MgrT::backend().base_ptr();
172 std::size_t size = MgrT::backend().total_size();
173 if ( buf == nullptr || size < detail::kMinMemorySize )
174 return false;
175
176 std::FILE* f = std::fopen( filename, "rb" );
177 if ( f == nullptr )
178 return false;
179
180 if ( std::fseek( f, 0, SEEK_END ) != 0 )
181 {
182 std::fclose( f );
183 return false;
184 }
185 long file_size_long = std::ftell( f );
186 if ( file_size_long <= 0 )
187 {
188 std::fclose( f );
189 return false;
190 }
191 std::rewind( f );
192
193 std::size_t file_size = static_cast<std::size_t>( file_size_long );
194 if ( file_size > size )
195 {
196 std::fclose( f );
197 return false;
198 }
199
200 std::size_t read_bytes = std::fread( buf, 1, file_size, f );
201 std::fclose( f );
202
203 if ( read_bytes != file_size )
204 return false;
205
206 // Verify CRC32 before calling load().
207 constexpr std::size_t kHdrOffset = sizeof( pmm::Block<address_traits> );
208 if ( file_size >= kHdrOffset + sizeof( detail::ManagerHeader<address_traits> ) )
209 {
210 auto* hdr = reinterpret_cast<detail::ManagerHeader<address_traits>*>( buf + kHdrOffset );
211 std::uint32_t stored_crc = hdr->crc32;
212 std::uint32_t computed_crc = detail::compute_image_crc32<address_traits>( buf, file_size );
213 if ( stored_crc != computed_crc )
214 {
215 MgrT::set_last_error( PmmError::CrcMismatch );
216 MgrT::logging_policy::on_corruption_detected( PmmError::CrcMismatch );
217 return false;
218 }
219 }
220
221 return MgrT::load( result );
222}
