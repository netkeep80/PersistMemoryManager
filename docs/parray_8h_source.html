357 bool ensure_capacity( std::uint32_t required ) noexcept
358 {
359 if ( required <= _capacity )
360 return true;
361
362 // New capacity: double current or required, whichever is larger.
363 // Minimum 4 elements to avoid frequent reallocations for small arrays.
364 std::uint32_t new_cap = _capacity * 2;
365 if ( new_cap < required )
366 new_cap = required;
367 if ( new_cap < 4 )
368 new_cap = 4;
369
370 // Check for overflow in allocation size.
371 std::size_t alloc_size = static_cast<std::size_t>( new_cap ) * sizeof( T );
372 if ( sizeof( T ) > 0 && alloc_size / sizeof( T ) != static_cast<std::size_t>( new_cap ) )
373 return false; // overflow
374
375 void* new_raw = ManagerT::allocate( alloc_size );
376 if ( new_raw == nullptr )
377 return false;
378
379 // Compute new index (Issue #188: shared ptr_to_granule_idx).
380 std::uint8_t* base = ManagerT::backend().base_ptr();
381 index_type new_dat_idx = detail::ptr_to_granule_idx<typename ManagerT::address_traits>( base, new_raw );
382
383 // Copy old data.
384 if ( _size > 0 && _data_idx != static_cast<index_type>( 0 ) )
385 {
386 T* old_data = resolve_data();
387 if ( old_data != nullptr )
388 std::memcpy( new_raw, old_data, static_cast<std::size_t>( _size ) * sizeof( T ) );
389 }
390
391 // Free old block (Issue #188: shared resolve_granule_ptr).
392 if ( _data_idx != static_cast<index_type>( 0 ) )
393 ManagerT::deallocate( detail::resolve_granule_ptr<typename ManagerT::address_traits>( base, _data_idx ) );
394
395 _data_idx = new_dat_idx;
396 _capacity = new_cap;
397 return true;
