406 bool ensure_capacity( std::uint32_t required ) noexcept
407 {
408 if ( required <= _capacity )
409 return true;
410
411 // New capacity: double current or required, whichever is larger.
412 // Minimum 4 elements to avoid frequent reallocations for small arrays.
413 std::uint32_t new_cap = _capacity * 2;
414 if ( new_cap < required )
415 new_cap = required;
416 if ( new_cap < 4 )
417 new_cap = 4;
418
419 // Check for overflow in allocation size.
420 std::size_t alloc_size = static_cast<std::size_t>( new_cap ) * sizeof( T );
421 if ( sizeof( T ) > 0 && alloc_size / sizeof( T ) != static_cast<std::size_t>( new_cap ) )
422 return false; // overflow
423
424 void* new_raw = ManagerT::allocate( alloc_size );
425 if ( new_raw == nullptr )
426 return false;
427
428 // Compute new index.
429 std::uint8_t* base = ManagerT::backend().base_ptr();
430 index_type new_dat_idx = detail::ptr_to_granule_idx<typename ManagerT::address_traits>( base, new_raw );
431
432 // Copy old data.
433 if ( _size > 0 && _data_idx != static_cast<index_type>( 0 ) )
434 {
435 T* old_data = resolve_data();
436 if ( old_data != nullptr )
437 std::memcpy( new_raw, old_data, static_cast<std::size_t>( _size ) * sizeof( T ) );
438 }
439
440 // Free old block.
441 if ( _data_idx != static_cast<index_type>( 0 ) )
442 ManagerT::deallocate( detail::resolve_granule_ptr<typename ManagerT::address_traits>( base, _data_idx ) );
443
444 _data_idx = new_dat_idx;
445 _capacity = new_cap;
446 return true;
447 }
