245inline bool is_valid_block( const std::uint8_t* base, const ManagerHeader* hdr, std::uint32_t idx )
246{
247 if ( idx == kNoBlock )
248 return false;
249 if ( idx_to_byte_off( idx ) + sizeof( BlockHeader ) > hdr->total_size )
250 return false;
251
252 const BlockHeader* blk = reinterpret_cast<const BlockHeader*>( base + idx_to_byte_off( idx ) );
253 std::uint32_t total_gran = ( blk->next_offset != kNoBlock )
254 ? ( blk->next_offset - idx )
255 : ( byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - idx );
256 if ( blk->size >= total_gran )
257 return false;
258 if ( blk->prev_offset != kNoBlock && blk->prev_offset >= idx )
259 return false;
260 if ( blk->next_offset != kNoBlock && blk->next_offset <= idx )
261 return false;
262 if ( blk->avl_height >= 32 )
263 return false;
264 const bool l = ( blk->left_offset != kNoBlock );
265 const bool r = ( blk->right_offset != kNoBlock );
266 const bool p = ( blk->parent_offset != kNoBlock );
267 if ( ( l || r || p ) && ( ( l && r && blk->left_offset == blk->right_offset ) ||
268 ( l && p && blk->left_offset == blk->parent_offset ) ||
269 ( r && p && blk->right_offset == blk->parent_offset ) ) )
270 return false;
271 return true;
272}

387inline void avl_rebalance_up( std::uint8_t* base, ManagerHeader* hdr, std::uint32_t node_idx )
388{
389 std::uint32_t cur = node_idx;
390 while ( cur != kNoBlock )
391 {
392 BlockHeader* node = block_at( base, cur );
393 avl_update_height( base, node );
394 std::int32_t bf = avl_bf( base, node );
395 if ( bf > 1 )
396 {
397 if ( avl_bf( base, block_at( base, node->left_offset ) ) < 0 )
398 avl_rotate_left( base, hdr, node->left_offset );
399 cur = avl_rotate_right( base, hdr, cur );
400 }
401 else if ( bf < -1 )
402 {
403 if ( avl_bf( base, block_at( base, node->right_offset ) ) > 0 )
404 avl_rotate_right( base, hdr, node->right_offset );
405 cur = avl_rotate_left( base, hdr, cur );

407 cur = block_at( base, cur )->parent_offset;

411inline void avl_insert( std::uint8_t* base, ManagerHeader* hdr, std::uint32_t blk_idx )
412{
413 BlockHeader* blk = block_at( base, blk_idx );
414 blk->left_offset = kNoBlock;
415 blk->right_offset = kNoBlock;
416 blk->parent_offset = kNoBlock;
417 blk->avl_height = 1;
418 if ( hdr->free_tree_root == kNoBlock )
419 {
420 hdr->free_tree_root = blk_idx;
421 return;
422 }
423 // Issue #59: cache total_gran once; compute blk size in granules before the traversal loop
424 std::uint32_t total_gran = byte_off_to_idx( hdr->total_size );
425 std::uint32_t blk_gran =
426 ( blk->next_offset != kNoBlock ) ? ( blk->next_offset - blk_idx ) : ( total_gran - blk_idx );
427 std::uint32_t cur = hdr->free_tree_root, parent = kNoBlock;
428 bool go_left = false;
429 while ( cur != kNoBlock )
430 {
431 parent = cur;
432 BlockHeader* n = block_at( base, cur );
433 std::uint32_t n_gran = ( n->next_offset != kNoBlock ) ? ( n->next_offset - cur ) : ( total_gran - cur );
434 bool smaller = ( blk_gran < n_gran ) || ( blk_gran == n_gran && blk_idx < cur );
435 go_left = smaller;
436 cur = smaller ? n->left_offset : n->right_offset;
437 }
438 blk->parent_offset = parent;
439 if ( go_left )
440 block_at( base, parent )->left_offset = blk_idx;
441 else
442 block_at( base, parent )->right_offset = blk_idx;
443 avl_rebalance_up( base, hdr, parent );
444}

458inline void avl_remove( std::uint8_t* base, ManagerHeader* hdr, std::uint32_t blk_idx )
459{
460 BlockHeader* blk = block_at( base, blk_idx );
461 std::uint32_t parent = blk->parent_offset;
462 std::uint32_t left = blk->left_offset;
463 std::uint32_t right = blk->right_offset;
464 std::uint32_t rebal = kNoBlock;
465
466 if ( left == kNoBlock && right == kNoBlock )
467 {
468 avl_set_child( base, hdr, parent, blk_idx, kNoBlock );
469 rebal = parent;
470 }
471 else if ( left == kNoBlock || right == kNoBlock )
472 {
473 std::uint32_t child = ( left != kNoBlock ) ? left : right;
474 block_at( base, child )->parent_offset = parent;
475 avl_set_child( base, hdr, parent, blk_idx, child );
476 rebal = parent;
477 }
478 else
479 {
480 std::uint32_t succ_idx = avl_min_node( base, right );
481 BlockHeader* succ = block_at( base, succ_idx );
482 std::uint32_t succ_parent = succ->parent_offset;
483 std::uint32_t succ_right = succ->right_offset;
484
485 if ( succ_parent != blk_idx )
486 {
487 avl_set_child( base, hdr, succ_parent, succ_idx, succ_right );
488 if ( succ_right != kNoBlock )
489 block_at( base, succ_right )->parent_offset = succ_parent;
490 succ->right_offset = right;
491 block_at( base, right )->parent_offset = succ_idx;
492 rebal = succ_parent;
493 }
494 else
495 {
496 rebal = succ_idx;
497 }
498 succ->left_offset = left;
499 block_at( base, left )->parent_offset = succ_idx;
500 succ->parent_offset = parent;
501 avl_set_child( base, hdr, parent, blk_idx, succ_idx );
502 avl_update_height( base, succ );
503 }
504 blk->left_offset = kNoBlock;
505 blk->right_offset = kNoBlock;
506 blk->parent_offset = kNoBlock;
507 blk->avl_height = 0;
508 avl_rebalance_up( base, hdr, rebal );
509}

595 static bool create( void* memory, std::size_t size )
596 {
597 std::unique_lock<std::shared_mutex> lock( s_mutex );
598 if ( memory == nullptr || size < kMinMemorySize )
599 return false;
600 // Issue #59: max 64 GB (2^32 * kGranuleSize)
601 if ( size > static_cast<std::uint64_t>( std::numeric_limits<std::uint32_t>::max() ) * kGranuleSize )
602 return false;
603 // Size must be a multiple of kGranuleSize
604 if ( size % kGranuleSize != 0 )
605 size -= ( size % kGranuleSize );
606
607 std::uint8_t* base = static_cast<std::uint8_t*>( memory );
608 // Issue #75: PAP homogenization — BlockHeader_0 at granule 0 holds ManagerHeader as user data.
609 // Layout: [BlockHeader_0(0)][ManagerHeader(2)][BlockHeader_free(6)][free_space...]
610 static constexpr std::uint32_t kHdrBlkIdx = 0; // BlockHeader_0 granule index
611 static constexpr std::uint32_t kFreeBlkIdx = // first free block
612 detail::kBlockHeaderGranules + detail::kManagerHeaderGranules;
613
614 if ( detail::idx_to_byte_off( kFreeBlkIdx ) + sizeof( detail::BlockHeader ) + kMinBlockSize > size )
615 return false;
616
617 // Set up BlockHeader_0 — the allocated block that holds ManagerHeader
618 detail::BlockHeader* hdr_blk = detail::block_at( base, kHdrBlkIdx );
619 std::memset( hdr_blk, 0, sizeof( detail::BlockHeader ) );
620 hdr_blk->size = detail::kManagerHeaderGranules; // holds ManagerHeader (4 gran)
621 hdr_blk->prev_offset = detail::kNoBlock;
622 hdr_blk->next_offset = kFreeBlkIdx;
623 hdr_blk->left_offset = detail::kNoBlock;
624 hdr_blk->right_offset = detail::kNoBlock;
625 hdr_blk->parent_offset = detail::kNoBlock;
626 hdr_blk->avl_height = 0;
627 hdr_blk->root_offset = kHdrBlkIdx; // own_idx==0: root of PAP meta-tree (Issue #75)
628
629 // Initialise ManagerHeader inside BlockHeader_0's user data
630 detail::ManagerHeader* hdr = reinterpret_cast<detail::ManagerHeader*>( base + sizeof( detail::BlockHeader ) );
631 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
632 hdr->magic = kMagic;
633 hdr->total_size = size;
634 hdr->first_block_offset = kHdrBlkIdx;
635 hdr->last_block_offset = detail::kNoBlock; // filled below
636 hdr->free_tree_root = detail::kNoBlock;
637 hdr->owns_memory = false;
638 hdr->prev_owns_memory = false;
639
640 // Set up first free block at kFreeBlkIdx
641 detail::BlockHeader* blk = detail::block_at( base, kFreeBlkIdx );
642 std::memset( blk, 0, sizeof( detail::BlockHeader ) );
643 blk->size = 0; // free block
644 blk->prev_offset = kHdrBlkIdx;
645 blk->next_offset = detail::kNoBlock;
646 blk->left_offset = detail::kNoBlock;
647 blk->right_offset = detail::kNoBlock;
648 blk->parent_offset = detail::kNoBlock;
649 blk->avl_height = 1;
650 blk->root_offset = 0; // free-blocks tree (Issue #75)
651
652 hdr->last_block_offset = kFreeBlkIdx;
653 hdr->free_tree_root = kFreeBlkIdx;
654 hdr->block_count = 2; // BlockHeader_0 + free block
655 hdr->free_count = 1;
656 hdr->alloc_count = 1; // BlockHeader_0 is allocated (holds ManagerHeader)
657 // used_size: BlockHeader_0 header (2) + ManagerHeader data (4) + free BlockHeader (2)
658 hdr->used_size = kFreeBlkIdx + detail::kBlockHeaderGranules;
659
660 s_instance = reinterpret_cast<PersistMemoryManager*>( base );
661 return true;

899 return false;
900 return ( block_count == hdr->block_count && free_count == hdr->free_count && alloc_count == hdr->alloc_count );

1013 bool expand( std::size_t user_size )
1014 {
1015 detail::ManagerHeader* hdr = header();
1016 std::size_t old_size = hdr->total_size;
1017 std::uint32_t needed = detail::required_block_granules( user_size );
1018
1019 std::size_t growth = ( old_size / kGrowDenominator ) * kGrowNumerator; // 5/4 growth
1020 std::size_t new_size = ( ( growth + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
1021 if ( new_size <= old_size )
1022 new_size = old_size + kGranuleSize; // at minimum grow by one granule
1023 if ( new_size < old_size + detail::granules_to_bytes( needed ) )
1024 {
1025 std::size_t extra = detail::granules_to_bytes( needed + detail::kBlockHeaderGranules );
1026 if ( extra > std::numeric_limits<std::size_t>::max() - old_size - ( kGranuleSize - 1 ) )
1027 return false;
1028 new_size = ( ( old_size + extra + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
1029 }
1030
1031 if ( new_size > static_cast<std::uint64_t>( std::numeric_limits<std::uint32_t>::max() ) * kGranuleSize )
1032 return false;
1033
1034 void* new_memory = std::malloc( new_size );
1035 if ( new_memory == nullptr )
1036 return false;
1037
1038 std::memcpy( new_memory, base_ptr(), old_size );
1039 std::uint8_t* nb = static_cast<std::uint8_t*>( new_memory );
1040 // Issue #75: ManagerHeader at nb + sizeof(BlockHeader) (inside BlockHeader_0)
1041 detail::ManagerHeader* nh = reinterpret_cast<detail::ManagerHeader*>( nb + sizeof( detail::BlockHeader ) );
1042 nh->owns_memory = true;
1043 std::uint32_t extra_idx = detail::byte_off_to_idx( old_size );
1044 std::size_t extra_size = new_size - old_size;
1045 detail::BlockHeader* last_blk =
1046 ( nh->last_block_offset != detail::kNoBlock ) ? detail::block_at( nb, nh->last_block_offset ) : nullptr;
1047
1048 if ( last_blk != nullptr && last_blk->size == 0 )
1049 {
1050 std::uint32_t loff = detail::block_idx( nb, last_blk );
1051 detail::avl_remove( nb, nh, loff );
1052 nh->total_size = new_size;
1053 detail::avl_insert( nb, nh, loff );
1054 }
1055 else
1056 {
1057 if ( extra_size < sizeof( detail::BlockHeader ) + kMinBlockSize )
1058 {
1059 std::free( new_memory );
1060 return false;
1061 }
1062 detail::BlockHeader* nb_blk = detail::block_at( nb, extra_idx );
1063 std::memset( nb_blk, 0, sizeof( detail::BlockHeader ) );
1064 nb_blk->size = 0; // free block
1065 nb_blk->left_offset = detail::kNoBlock;
1066 nb_blk->right_offset = detail::kNoBlock;
1067 nb_blk->parent_offset = detail::kNoBlock;
1068 nb_blk->avl_height = 1;
1069 nb_blk->root_offset = 0; // free-blocks tree (Issue #75)
1070 if ( last_blk != nullptr )
1071 {
1072 std::uint32_t loff = detail::block_idx( nb, last_blk );
1073 nb_blk->prev_offset = loff;
1074 nb_blk->next_offset = detail::kNoBlock;
1075 last_blk->next_offset = extra_idx;
1076 }
1077 else
1078 {
1079 nb_blk->prev_offset = detail::kNoBlock;
1080 nb_blk->next_offset = detail::kNoBlock;
1081 nh->first_block_offset = extra_idx;
1082 }
1083 nh->last_block_offset = extra_idx;
1084 nh->block_count++;
1085 nh->free_count++;
1086 nh->total_size = new_size;
1087 detail::avl_insert( nb, nh, extra_idx );
1088 }
1089 nh->prev_base_ptr = base_ptr();
1090 nh->prev_total_size = old_size;
1091 nh->prev_owns_memory = hdr->owns_memory;
1092 s_instance = reinterpret_cast<PersistMemoryManager*>( new_memory );
1093 return true;
1094 }

1247 {

1271 new_blk->root_offset = 0; // free-blocks tree (Issue #75)
1272 if ( blk->next_offset != detail::kNoBlock )
1273 detail::block_at( base, blk->next_offset )->prev_offset = new_idx;
1274 else
1275 hdr->last_block_offset = new_idx;
1276 blk->next_offset = new_idx;
1277 hdr->block_count++;
1278 hdr->free_count++;
1279 hdr->used_size += detail::kBlockHeaderGranules;
1280 detail::avl_insert( base, hdr, new_idx );
1281 }
1282
1283 blk->size = data_gran; // store in granules (Issue #75: was used_size)
1284 blk->root_offset = blk_idx; // allocated block is root of its own tree (Issue #75)
1285 blk->left_offset = detail::kNoBlock;
1286 blk->right_offset = detail::kNoBlock;
1287 blk->parent_offset = detail::kNoBlock;
1288 blk->avl_height = 0;

1290 hdr->free_count--;
1291 hdr->used_size += data_gran;

1293 }

1297 {

1299 return true;
1300 if ( detail::idx_to_byte_off( node_idx ) >= hdr->total_size )
1301 return false;
1302 if ( !detail::is_valid_block( base, hdr, node_idx ) )
1303 return false;

1305 if ( node->size != 0 )
1306 return false;
1307 count++;
1308 if ( !validate_avl( base, hdr, node->left_offset, count ) )
1309 return false;
1310 if ( !validate_avl( base, hdr, node->right_offset, count ) )
1311 return false;
1312 if ( node->left_offset != detail::kNoBlock )
1313 {
1314 const detail::BlockHeader* lc = detail::block_at( const_cast<std::uint8_t*>( base ), node->left_offset );
1315 if ( lc->parent_offset != node_idx )
1316 return false;
1317 }
1318 if ( node->right_offset != detail::kNoBlock )
1319 {
1320 const detail::BlockHeader* rc = detail::block_at( const_cast<std::uint8_t*>( base ), node->right_offset );
1321 if ( rc->parent_offset != node_idx )
1322 return false;
1323 }
1324 return true;
