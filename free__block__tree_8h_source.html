109 static void insert( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
110 {
111 detail::BlockHeader* blk = detail::block_at( base, blk_idx );
112 blk->left_offset = detail::kNoBlock;
113 blk->right_offset = detail::kNoBlock;
114 blk->parent_offset = detail::kNoBlock;
115 blk->avl_height = 1;
116 if ( hdr->free_tree_root == detail::kNoBlock )
117 {
118 hdr->free_tree_root = blk_idx;
119 return;
120 }
121 // Issue #59: cache total_gran once; compute blk size in granules before the traversal loop
122 std::uint32_t total_gran = detail::byte_off_to_idx( hdr->total_size );
123 std::uint32_t blk_gran =
124 ( blk->next_offset != detail::kNoBlock ) ? ( blk->next_offset - blk_idx ) : ( total_gran - blk_idx );
125 std::uint32_t cur = hdr->free_tree_root, parent = detail::kNoBlock;
126 bool go_left = false;
127 while ( cur != detail::kNoBlock )
128 {
129 parent = cur;
130 detail::BlockHeader* n = detail::block_at( base, cur );
131 std::uint32_t n_gran =
132 ( n->next_offset != detail::kNoBlock ) ? ( n->next_offset - cur ) : ( total_gran - cur );
133 bool smaller = ( blk_gran < n_gran ) || ( blk_gran == n_gran && blk_idx < cur );
134 go_left = smaller;
135 cur = smaller ? n->left_offset : n->right_offset;
136 }
137 blk->parent_offset = parent;
138 if ( go_left )
139 detail::block_at( base, parent )->left_offset = blk_idx;
140 else
141 detail::block_at( base, parent )->right_offset = blk_idx;
142 rebalance_up( base, hdr, parent );
143 }

146 static void remove( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
147 {
148 detail::BlockHeader* blk = detail::block_at( base, blk_idx );
149 std::uint32_t parent = blk->parent_offset;
150 std::uint32_t left = blk->left_offset;
151 std::uint32_t right = blk->right_offset;
152 std::uint32_t rebal = detail::kNoBlock;
153
154 if ( left == detail::kNoBlock && right == detail::kNoBlock )
155 {
156 set_child( base, hdr, parent, blk_idx, detail::kNoBlock );
157 rebal = parent;
158 }
159 else if ( left == detail::kNoBlock || right == detail::kNoBlock )
160 {
161 std::uint32_t child = ( left != detail::kNoBlock ) ? left : right;
162 detail::block_at( base, child )->parent_offset = parent;
163 set_child( base, hdr, parent, blk_idx, child );
164 rebal = parent;
165 }
166 else
167 {
168 std::uint32_t succ_idx = min_node( base, right );
169 detail::BlockHeader* succ = detail::block_at( base, succ_idx );
170 std::uint32_t succ_parent = succ->parent_offset;
171 std::uint32_t succ_right = succ->right_offset;
172
173 if ( succ_parent != blk_idx )
174 {
175 set_child( base, hdr, succ_parent, succ_idx, succ_right );
176 if ( succ_right != detail::kNoBlock )
177 detail::block_at( base, succ_right )->parent_offset = succ_parent;
178 succ->right_offset = right;
179 detail::block_at( base, right )->parent_offset = succ_idx;
180 rebal = succ_parent;
181 }
182 else
183 {
184 rebal = succ_idx;
185 }
186 succ->left_offset = left;
187 detail::block_at( base, left )->parent_offset = succ_idx;
188 succ->parent_offset = parent;
189 set_child( base, hdr, parent, blk_idx, succ_idx );
190 update_height( base, succ );
191 }
192 blk->left_offset = detail::kNoBlock;
193 blk->right_offset = detail::kNoBlock;
194 blk->parent_offset = detail::kNoBlock;
195 blk->avl_height = 0;
196 rebalance_up( base, hdr, rebal );
197 }

200 static std::uint32_t find_best_fit( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t needed_granules )
201 {
202 // Issue #59: cache total_gran once to avoid repeated hdr->total_size reads in the hot path
203 std::uint32_t total_gran = detail::byte_off_to_idx( hdr->total_size );
204 std::uint32_t cur = hdr->free_tree_root, result = detail::kNoBlock;
205 while ( cur != detail::kNoBlock )
206 {
207 detail::BlockHeader* node = detail::block_at( base, cur );
208 std::uint32_t cur_gran =
209 ( node->next_offset != detail::kNoBlock ) ? ( node->next_offset - cur ) : ( total_gran - cur );
210 if ( cur_gran >= needed_granules )
211 {
212 result = cur;
213 cur = node->left_offset;
214 }
215 else
216 {
217 cur = node->right_offset;
218 }
219 }
220 return result;
221 }

229 static void update_height( std::uint8_t* base, detail::BlockHeader* node )
230 {
231 std::int32_t h = 1 + std::max( height( base, node->left_offset ), height( base, node->right_offset ) );

257 static std::uint32_t rotate_right( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t y_idx )
258 {
259 detail::BlockHeader* y = detail::block_at( base, y_idx );
260 std::uint32_t x_idx = y->left_offset;
261 detail::BlockHeader* x = detail::block_at( base, x_idx );
262 std::uint32_t t2 = x->right_offset;
263
264 x->right_offset = y_idx;
265 y->left_offset = t2;
266 x->parent_offset = y->parent_offset;
267 y->parent_offset = x_idx;
268 if ( t2 != detail::kNoBlock )
269 detail::block_at( base, t2 )->parent_offset = y_idx;
270 set_child( base, hdr, x->parent_offset, y_idx, x_idx );
271 update_height( base, y );
272 update_height( base, x );
273 return x_idx;
274 }

277 {

281 std::uint32_t t2 = y->left_offset;

283 y->left_offset = x_idx;

285 y->parent_offset = x->parent_offset;
286 x->parent_offset = y_idx;
287 if ( t2 != detail::kNoBlock )
288 detail::block_at( base, t2 )->parent_offset = x_idx;
289 set_child( base, hdr, y->parent_offset, x_idx, y_idx );

295 static void rebalance_up( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t node_idx )
296 {
297 std::uint32_t cur = node_idx;
298 while ( cur != detail::kNoBlock )
299 {
300 detail::BlockHeader* node = detail::block_at( base, cur );
301 update_height( base, node );
302 std::int32_t bf = balance_factor( base, node );
303 if ( bf > 1 )
304 {
305 if ( balance_factor( base, detail::block_at( base, node->left_offset ) ) < 0 )
306 rotate_left( base, hdr, node->left_offset );
307 cur = rotate_right( base, hdr, cur );
308 }
309 else if ( bf < -1 )
310 {
311 if ( balance_factor( base, detail::block_at( base, node->right_offset ) ) > 0 )
312 rotate_right( base, hdr, node->right_offset );
313 cur = rotate_left( base, hdr, cur );
314 }
315 cur = detail::block_at( base, cur )->parent_offset;
316 }
317 }

319 static std::uint32_t min_node( std::uint8_t* base, std::uint32_t node_idx )
320 {
321 while ( node_idx != detail::kNoBlock )
322 {
323 std::uint32_t left = detail::block_at( base, node_idx )->left_offset;
324 if ( left == detail::kNoBlock )
325 break;
326 node_idx = left;
327 }
328 return node_idx;
329 }
