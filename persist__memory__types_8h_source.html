193inline std::uint32_t bytes_to_granules( std::size_t bytes )
194{
195 if ( bytes > std::numeric_limits<std::size_t>::max() - ( kGranuleSize - 1 ) )
196 return 0;
197 std::size_t granules = ( bytes + kGranuleSize - 1 ) / kGranuleSize;
198 if ( granules > std::numeric_limits<std::uint32_t>::max() )
199 return 0;
200 return static_cast<std::uint32_t>( granules );
201}

253inline std::uint32_t block_total_granules( const std::uint8_t* base, const ManagerHeader* hdr, const BlockHeader* blk )
254{
255 std::uint32_t this_idx = block_idx( base, blk );
256 if ( blk->next_offset != kNoBlock )
257 return blk->next_offset - this_idx;
258 return byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - this_idx;
259}

263inline bool is_valid_block( const std::uint8_t* base, const ManagerHeader* hdr, std::uint32_t idx )
264{
265 if ( idx == kNoBlock )
266 return false;
267 if ( idx_to_byte_off( idx ) + sizeof( BlockHeader ) > hdr->total_size )
268 return false;
269
270 const BlockHeader* blk = reinterpret_cast<const BlockHeader*>( base + idx_to_byte_off( idx ) );
271 std::uint32_t total_gran = ( blk->next_offset != kNoBlock )
272 ? ( blk->next_offset - idx )
273 : ( byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - idx );
274 if ( blk->size >= total_gran )
275 return false;
276 if ( blk->prev_offset != kNoBlock && blk->prev_offset >= idx )
277 return false;
278 if ( blk->next_offset != kNoBlock && blk->next_offset <= idx )
279 return false;
280 if ( blk->avl_height >= 32 )
281 return false;
282 const bool l = ( blk->left_offset != kNoBlock );
283 const bool r = ( blk->right_offset != kNoBlock );
284 const bool p = ( blk->parent_offset != kNoBlock );
285 if ( ( l || r || p ) && ( ( l && r && blk->left_offset == blk->right_offset ) ||
286 ( l && p && blk->left_offset == blk->parent_offset ) ||
287 ( r && p && blk->right_offset == blk->parent_offset ) ) )
288 return false;
289 return true;
290}

299inline BlockHeader* header_from_ptr( std::uint8_t* base, void* ptr, std::size_t total_size )
300{
301 if ( ptr == nullptr )
302 return nullptr;
303 std::uint8_t* raw_ptr = reinterpret_cast<std::uint8_t*>( ptr );
304 // First user data starts after BlockHeader_0 + ManagerHeader + BlockHeader_1 (Issue #75)
305 std::uint8_t* min_addr = base + sizeof( BlockHeader ) + sizeof( ManagerHeader ) + sizeof( BlockHeader );
306 if ( raw_ptr < min_addr )
307 return nullptr;
308 if ( raw_ptr > base + total_size )
309 return nullptr;
310 std::uint8_t* cand_addr = raw_ptr - sizeof( BlockHeader );
311 if ( ( reinterpret_cast<std::size_t>( cand_addr ) - reinterpret_cast<std::size_t>( base ) ) % kGranuleSize != 0 )
312 return nullptr;
313 std::uint32_t cand_idx = static_cast<std::uint32_t>( ( cand_addr - base ) / kGranuleSize );
314 // Issue #83: ManagerHeader is at base + sizeof(BlockHeader), not base.
315 const ManagerHeader* hdr_const = reinterpret_cast<const ManagerHeader*>( base + sizeof( BlockHeader ) );
316 if ( !is_valid_block( base, hdr_const, cand_idx ) )
317 return nullptr;
318 BlockHeader* cand = reinterpret_cast<BlockHeader*>( cand_addr );
319 if ( cand->size == 0 )
320 return nullptr;
321 return cand;
322}

325inline std::uint32_t required_block_granules( std::size_t user_bytes )
