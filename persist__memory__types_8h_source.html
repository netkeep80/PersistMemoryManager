152inline std::uint32_t bytes_to_granules( std::size_t bytes )
153{
154 if ( bytes > std::numeric_limits<std::size_t>::max() - ( kGranuleSize - 1 ) )
155 return 0;
156 std::size_t granules = ( bytes + kGranuleSize - 1 ) / kGranuleSize;
157 if ( granules > std::numeric_limits<std::uint32_t>::max() )
158 return 0;
159 return static_cast<std::uint32_t>( granules );
160}

205inline std::uint32_t block_total_granules( const std::uint8_t* base, const ManagerHeader* hdr, const BlockHeader* blk )
206{
207 std::uint32_t this_idx = block_idx( base, blk );
208 if ( blk->next_offset != kNoBlock )
209 return blk->next_offset - this_idx;
210 return byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - this_idx;
211}

215inline bool is_valid_block( const std::uint8_t* base, const ManagerHeader* hdr, std::uint32_t idx )
216{
217 if ( idx == kNoBlock )
218 return false;
219 if ( idx_to_byte_off( idx ) + sizeof( BlockHeader ) > hdr->total_size )
220 return false;
221
222 const BlockHeader* blk = reinterpret_cast<const BlockHeader*>( base + idx_to_byte_off( idx ) );
223 std::uint32_t total_gran = ( blk->next_offset != kNoBlock )
224 ? ( blk->next_offset - idx )
225 : ( byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - idx );
226 if ( blk->size >= total_gran )
227 return false;
228 if ( blk->prev_offset != kNoBlock && blk->prev_offset >= idx )
229 return false;
230 if ( blk->next_offset != kNoBlock && blk->next_offset <= idx )
231 return false;
232 if ( blk->avl_height >= 32 )
233 return false;
234 const bool l = ( blk->left_offset != kNoBlock );
235 const bool r = ( blk->right_offset != kNoBlock );
236 const bool p = ( blk->parent_offset != kNoBlock );
237 if ( ( l || r || p ) && ( ( l && r && blk->left_offset == blk->right_offset ) ||
238 ( l && p && blk->left_offset == blk->parent_offset ) ||
239 ( r && p && blk->right_offset == blk->parent_offset ) ) )
240 return false;
241 return true;
242}

251inline BlockHeader* header_from_ptr( std::uint8_t* base, void* ptr, std::size_t total_size )
252{
253 if ( ptr == nullptr )
254 return nullptr;
255 std::uint8_t* raw_ptr = reinterpret_cast<std::uint8_t*>( ptr );
256 // First user data starts after BlockHeader_0 + ManagerHeader + BlockHeader_1 (Issue #75)
257 std::uint8_t* min_addr = base + sizeof( BlockHeader ) + sizeof( ManagerHeader ) + sizeof( BlockHeader );
258 if ( raw_ptr < min_addr )
259 return nullptr;
260 if ( raw_ptr > base + total_size )
261 return nullptr;
262 std::uint8_t* cand_addr = raw_ptr - sizeof( BlockHeader );
263 if ( ( reinterpret_cast<std::size_t>( cand_addr ) - reinterpret_cast<std::size_t>( base ) ) % kGranuleSize != 0 )
264 return nullptr;
265 std::uint32_t cand_idx = static_cast<std::uint32_t>( ( cand_addr - base ) / kGranuleSize );
266 const ManagerHeader* hdr_const = reinterpret_cast<const ManagerHeader*>( base );
267 if ( !is_valid_block( base, hdr_const, cand_idx ) )
268 return nullptr;
269 BlockHeader* cand = reinterpret_cast<BlockHeader*>( cand_addr );
270 if ( cand->size == 0 )
271 return nullptr;
272 return cand;
273}

276inline std::uint32_t required_block_granules( std::size_t user_bytes )
277{
