153inline std::uint32_t bytes_to_granules( std::size_t bytes )
154{
155 if ( bytes > std::numeric_limits<std::size_t>::max() - ( kGranuleSize - 1 ) )
156 return 0;
157 std::size_t granules = ( bytes + kGranuleSize - 1 ) / kGranuleSize;
158 if ( granules > std::numeric_limits<std::uint32_t>::max() )
159 return 0;
160 return static_cast<std::uint32_t>( granules );
161}

213inline std::uint32_t block_total_granules( const std::uint8_t* base, const ManagerHeader* hdr, const BlockHeader* blk )
214{
215 std::uint32_t this_idx = block_idx( base, blk );
216 if ( blk->next_offset != kNoBlock )
217 return blk->next_offset - this_idx;
218 return byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - this_idx;
219}

223inline bool is_valid_block( const std::uint8_t* base, const ManagerHeader* hdr, std::uint32_t idx )
224{
225 if ( idx == kNoBlock )
226 return false;
227 if ( idx_to_byte_off( idx ) + sizeof( BlockHeader ) > hdr->total_size )
228 return false;
229
230 const BlockHeader* blk = reinterpret_cast<const BlockHeader*>( base + idx_to_byte_off( idx ) );
231 std::uint32_t total_gran = ( blk->next_offset != kNoBlock )
232 ? ( blk->next_offset - idx )
233 : ( byte_off_to_idx( static_cast<std::size_t>( hdr->total_size ) ) - idx );
234 if ( blk->size >= total_gran )
235 return false;
236 if ( blk->prev_offset != kNoBlock && blk->prev_offset >= idx )
237 return false;
238 if ( blk->next_offset != kNoBlock && blk->next_offset <= idx )
239 return false;
240 if ( blk->avl_height >= 32 )
241 return false;
242 const bool l = ( blk->left_offset != kNoBlock );
243 const bool r = ( blk->right_offset != kNoBlock );
244 const bool p = ( blk->parent_offset != kNoBlock );
245 if ( ( l || r || p ) && ( ( l && r && blk->left_offset == blk->right_offset ) ||
246 ( l && p && blk->left_offset == blk->parent_offset ) ||
247 ( r && p && blk->right_offset == blk->parent_offset ) ) )
248 return false;
249 return true;
250}

259inline BlockHeader* header_from_ptr( std::uint8_t* base, void* ptr, std::size_t total_size )
260{
261 if ( ptr == nullptr )
262 return nullptr;
263 std::uint8_t* raw_ptr = reinterpret_cast<std::uint8_t*>( ptr );
264 // First user data starts after BlockHeader_0 + ManagerHeader + BlockHeader_1 (Issue #75)
265 std::uint8_t* min_addr = base + sizeof( BlockHeader ) + sizeof( ManagerHeader ) + sizeof( BlockHeader );
266 if ( raw_ptr < min_addr )
267 return nullptr;
268 if ( raw_ptr > base + total_size )
269 return nullptr;
270 std::uint8_t* cand_addr = raw_ptr - sizeof( BlockHeader );
271 if ( ( reinterpret_cast<std::size_t>( cand_addr ) - reinterpret_cast<std::size_t>( base ) ) % kGranuleSize != 0 )
272 return nullptr;
273 std::uint32_t cand_idx = static_cast<std::uint32_t>( ( cand_addr - base ) / kGranuleSize );
274 // Issue #83: ManagerHeader is at base + sizeof(BlockHeader), not base.
275 const ManagerHeader* hdr_const = reinterpret_cast<const ManagerHeader*>( base + sizeof( BlockHeader ) );
276 if ( !is_valid_block( base, hdr_const, cand_idx ) )
277 return nullptr;
278 BlockHeader* cand = reinterpret_cast<BlockHeader*>( cand_addr );
279 if ( cand->size == 0 )
280 return nullptr;
281 return cand;
282}

285inline std::uint32_t required_block_granules( std::size_t user_bytes )
286{
