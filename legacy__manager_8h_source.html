90template <typename Base> class StatsMixin : public Base
91{
92 public:
93 static MemoryStats get_stats()
94 {
95 using Derived = typename Base::derived_type;
96 MemoryStats stats{};
97 Derived* mgr = Derived::s_instance;
98 if ( mgr == nullptr )
99 return stats;
100 const std::uint8_t* base = mgr->const_base_ptr();
101 const detail::ManagerHeader* hdr = mgr->header();
102 stats.total_blocks = hdr->block_count;
103 stats.free_blocks = hdr->free_count;
104 stats.allocated_blocks = hdr->alloc_count;
105 bool first_free = true;
106 std::uint32_t idx = hdr->first_block_offset;
107 while ( idx != detail::kNoBlock )
108 {
109 const detail::BlockHeader* blk = detail::block_at( base, idx );
110 if ( blk->size == 0 )
111 {
112 std::uint32_t gran = ( blk->next_offset != detail::kNoBlock )
113 ? ( blk->next_offset - idx )
114 : ( detail::byte_off_to_idx( hdr->total_size ) - idx );
115 std::size_t blk_size = detail::granules_to_bytes( gran );
116 if ( first_free )
117 {
118 stats.largest_free = blk_size;
119 stats.smallest_free = blk_size;
120 first_free = false;
121 }
122 else
123 {
124 stats.largest_free = std::max( stats.largest_free, blk_size );
125 stats.smallest_free = std::min( stats.smallest_free, blk_size );
126 }
127 stats.total_fragmentation += blk_size;
128 }
129 idx = blk->next_offset;
130 }
131 return stats;
132 }
133};

138template <typename Base> class ValidationMixin : public Base
139{
140 public:
141 static bool validate()
142 {
143 using Derived = typename Base::derived_type;
144 using LockPolicy = typename Derived::config_type::lock_policy;
145 Derived* mgr = Derived::s_instance;
146 if ( mgr == nullptr )
147 return false;
148 typename LockPolicy::shared_lock_type lock( Derived::s_mutex );
149 const std::uint8_t* base = mgr->const_base_ptr();
150 const detail::ManagerHeader* hdr = mgr->header();
151
152 if ( hdr->magic != kMagic )
153 return false;
154 std::size_t block_count = 0, free_count = 0, alloc_count = 0;
155 std::uint32_t idx = hdr->first_block_offset;
156 while ( idx != detail::kNoBlock )
157 {
158 if ( !detail::is_valid_block( base, hdr, idx ) )
159 return false;
160 const detail::BlockHeader* blk = detail::block_at( base, idx );
161 block_count++;
162 if ( blk->size > 0 )
163 alloc_count++;
164 else
165 free_count++;
166 if ( blk->next_offset != detail::kNoBlock )
167 {
168 const detail::BlockHeader* nxt = detail::block_at( base, blk->next_offset );
169 if ( nxt->prev_offset != idx )
170 return false;
171 }
172 idx = blk->next_offset;
173 }
174 std::size_t tree_free = 0;
175 if ( !validate_avl( base, hdr, hdr->free_tree_root, tree_free ) )
176 return false;
177 if ( tree_free != free_count )
178 return false;
179 return ( block_count == hdr->block_count && free_count == hdr->free_count && alloc_count == hdr->alloc_count );
180 }
181
182 private:
183 static bool validate_avl( const std::uint8_t* base, const detail::ManagerHeader* hdr, std::uint32_t node_idx,
184 std::size_t& count )
185 {
186 if ( node_idx == detail::kNoBlock )
187 return true;
188 if ( detail::idx_to_byte_off( node_idx ) >= hdr->total_size )
189 return false;
190 if ( !detail::is_valid_block( base, hdr, node_idx ) )
191 return false;
192 const detail::BlockHeader* node = detail::block_at( base, node_idx );
193 if ( node->size != 0 )
194 return false;
195 count++;
196 if ( !validate_avl( base, hdr, node->left_offset, count ) )
197 return false;
198 if ( !validate_avl( base, hdr, node->right_offset, count ) )
199 return false;
200 if ( node->left_offset != detail::kNoBlock )
201 {
202 const detail::BlockHeader* lc = detail::block_at( base, node->left_offset );
203 if ( lc->parent_offset != node_idx )
204 return false;
205 }
206 if ( node->right_offset != detail::kNoBlock )
207 {
208 const detail::BlockHeader* rc = detail::block_at( base, node->right_offset );
209 if ( rc->parent_offset != node_idx )
210 return false;
211 }
212 return true;
213 }

235 static bool create( void* memory, std::size_t size )
236 {
237 typename LockPolicy::unique_lock_type lock( s_mutex );
238 if ( memory == nullptr || size < detail::kMinMemorySize )
239 return false;
240 if ( size > static_cast<std::uint64_t>( std::numeric_limits<std::uint32_t>::max() ) * kGranuleSize )
241 return false;
242 if ( size % kGranuleSize != 0 )
243 size -= ( size % kGranuleSize );
244
245 std::uint8_t* base = static_cast<std::uint8_t*>( memory );
246 // Issue #75: BlockHeader_0 at granule 0 holds ManagerHeader as user data.
247 static constexpr std::uint32_t kHdrBlkIdx = 0;
248 static constexpr std::uint32_t kFreeBlkIdx = detail::kBlockHeaderGranules + detail::kManagerHeaderGranules;
249
250 if ( detail::idx_to_byte_off( kFreeBlkIdx ) + sizeof( detail::BlockHeader ) + detail::kMinBlockSize > size )
251 return false;
252
253 detail::BlockHeader* hdr_blk = detail::block_at( base, kHdrBlkIdx );
254 std::memset( hdr_blk, 0, sizeof( detail::BlockHeader ) );
255 hdr_blk->size = detail::kManagerHeaderGranules;
256 hdr_blk->prev_offset = detail::kNoBlock;
257 hdr_blk->next_offset = kFreeBlkIdx;
258 hdr_blk->left_offset = detail::kNoBlock;
259 hdr_blk->right_offset = detail::kNoBlock;
260 hdr_blk->parent_offset = detail::kNoBlock;
261 // hdr_blk->avl_height = 0; — already zero from memset
262 hdr_blk->root_offset = kHdrBlkIdx;
263
264 detail::ManagerHeader* hdr = reinterpret_cast<detail::ManagerHeader*>( base + sizeof( detail::BlockHeader ) );
265 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
266 hdr->magic = kMagic;
267 hdr->total_size = size;
268 hdr->first_block_offset = kHdrBlkIdx;
269 hdr->last_block_offset = detail::kNoBlock;
270 hdr->free_tree_root = detail::kNoBlock;
271 // hdr->owns_memory = false; — already false from memset
272 // hdr->prev_owns_memory = false; — already false from memset
273 hdr->granule_size = static_cast<std::uint16_t>( kGranuleSize );
274
275 detail::BlockHeader* blk = detail::block_at( base, kFreeBlkIdx );
276 std::memset( blk, 0, sizeof( detail::BlockHeader ) );
277 // blk->size = 0; — already zero from memset
278 blk->prev_offset = kHdrBlkIdx;
279 blk->next_offset = detail::kNoBlock;
280 blk->left_offset = detail::kNoBlock;
281 blk->right_offset = detail::kNoBlock;
282 blk->parent_offset = detail::kNoBlock;
283 blk->avl_height = 1;
284 // blk->root_offset = 0; — already zero from memset
285
286 hdr->last_block_offset = kFreeBlkIdx;
287 hdr->free_tree_root = kFreeBlkIdx;
288 hdr->block_count = 2;
289 hdr->free_count = 1;
290 hdr->alloc_count = 1;
291 hdr->used_size = kFreeBlkIdx + detail::kBlockHeaderGranules;
292
293 s_instance = reinterpret_cast<PersistMemoryManager*>( base );
294 return true;
295 }

298 static bool load( void* memory, std::size_t size )
299 {
300 typename LockPolicy::unique_lock_type lock( s_mutex );
301 if ( memory == nullptr || size < detail::kMinMemorySize )
302 return false;
303 std::uint8_t* base = static_cast<std::uint8_t*>( memory );
304 auto* hdr = reinterpret_cast<detail::ManagerHeader*>( base + sizeof( detail::BlockHeader ) );
305 if ( hdr->magic != kMagic || hdr->total_size != size )
306 return false;
307 if ( hdr->granule_size != static_cast<std::uint16_t>( kGranuleSize ) )
308 return false;
309 hdr->owns_memory = hdr->prev_owns_memory = false;
310 hdr->prev_total_size = 0;
311 hdr->prev_base_ptr = nullptr;
312 auto* mgr = reinterpret_cast<PersistMemoryManager*>( base );
313 mgr->repair_linked_list();
314 mgr->recompute_counters();
315 mgr->rebuild_free_tree();
316 s_instance = mgr;
317 return true;
318 }

321 static void destroy()
322 {
323 typename LockPolicy::unique_lock_type lock( s_mutex );
324 if ( s_instance == nullptr )
325 return;
326 detail::ManagerHeader* hdr = s_instance->header();
327 hdr->magic = 0;
328 bool owns = hdr->owns_memory;
329 void* buf = s_instance->base_ptr();
330 void* prev = hdr->prev_base_ptr;
331 bool prev_owns = hdr->prev_owns_memory;
332 s_instance = nullptr;
333 while ( prev != nullptr )
334 {
335 auto* ph = reinterpret_cast<detail::ManagerHeader*>( static_cast<std::uint8_t*>( prev ) +
336 sizeof( detail::BlockHeader ) );
337 void* next_prev = ph->prev_base_ptr;
338 bool next_owns = ph->prev_owns_memory;
339 if ( prev_owns )
340 std::free( prev );
341 prev = next_prev;
342 prev_owns = next_owns;
343 }
344 if ( owns )
345 std::free( buf );
346 }

350 template <class T> static pptr<T> allocate_typed()
351 {

361 template <class T> static pptr<T> allocate_typed( std::size_t count )
362 {
363 if ( count == 0 )
364 return pptr<T>();
365 if ( sizeof( T ) > 0 && count > std::numeric_limits<std::size_t>::max() / sizeof( T ) )
366 return pptr<T>();
367 void* raw = s_instance ? s_instance->allocate_raw( sizeof( T ) * count ) : nullptr;
368 if ( raw == nullptr )
369 return pptr<T>();
370 std::uint8_t* base = s_instance->base_ptr();
371 std::size_t byte_off = static_cast<std::uint8_t*>( raw ) - base;
372 assert( byte_off % kGranuleSize == 0 );

385 template <class T> static pptr<T> reallocate_typed( pptr<T> p, std::size_t count )
386 {
387 if ( p.is_null() )
388 return allocate_typed<T>( count );
389 if ( count == 0 )
390 {
391 deallocate_typed( p );
392 return pptr<T>();
393 }
394 if ( s_instance == nullptr )
395 return pptr<T>();
396
397 std::uint8_t* base = s_instance->base_ptr();
398 void* old_raw = base + detail::idx_to_byte_off( p.offset() );
399 detail::BlockHeader* blk =
400 detail::header_from_ptr( base, old_raw, static_cast<std::size_t>( s_instance->header()->total_size ) );
401 if ( blk == nullptr || blk->size == 0 )
402 return pptr<T>();
403
404 if ( count > 0 && sizeof( T ) > std::numeric_limits<std::size_t>::max() / count )
405 return pptr<T>();
406 std::uint32_t new_granules = detail::bytes_to_granules( sizeof( T ) * count );
407 if ( new_granules <= blk->size )
408 return p;
409
410 std::size_t old_bytes = detail::granules_to_bytes( blk->size );
411 std::uint32_t old_offset = p.offset();
412 pptr<T> new_p = allocate_typed<T>( count );
413 if ( new_p.is_null() )
414 return pptr<T>();
415
416 // Re-derive after possible auto-expand (Issue #67).
417 std::uint8_t* cur_base = s_instance->base_ptr();
418 void* new_raw = cur_base + detail::idx_to_byte_off( new_p.offset() );
419 void* cur_old = cur_base + detail::idx_to_byte_off( old_offset );
420 std::memcpy( new_raw, cur_old, old_bytes );
421 deallocate_typed( p );
422 return new_p;
423 }

520 {

523 return nullptr;

542 if ( new_idx != detail::kNoBlock )

547 void deallocate_raw( void* ptr )
548 {
549 typename LockPolicy::unique_lock_type lock( s_mutex );
550 if ( ptr == nullptr )
551 return;
552 ptr = translate_ptr( ptr );
553 std::uint8_t* base = this->base_ptr();
554 detail::ManagerHeader* hdr = this->header();
555 detail::BlockHeader* blk = detail::header_from_ptr( base, ptr, static_cast<std::size_t>( hdr->total_size ) );

560 blk->size = 0;

565 hdr->used_size -= freed;
566 coalesce( blk );

569 bool expand( std::size_t user_size )
570 {
571 detail::ManagerHeader* hdr = this->header();
572 std::size_t old_size = hdr->total_size;
573 std::uint32_t needed = detail::required_block_granules( user_size );
574
575 std::size_t growth = ( old_size / Config::grow_denominator ) * Config::grow_numerator;
576 std::size_t new_size = ( ( growth + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
577 if ( new_size <= old_size )
578 new_size = old_size + kGranuleSize;
579 if ( new_size < old_size + detail::granules_to_bytes( needed ) )
580 {
581 std::size_t extra = detail::granules_to_bytes( needed + detail::kBlockHeaderGranules );
582 if ( extra > std::numeric_limits<std::size_t>::max() - old_size - ( kGranuleSize - 1 ) )
583 return false;
584 new_size = ( ( old_size + extra + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
585 }
586
587 if ( new_size > static_cast<std::uint64_t>( std::numeric_limits<std::uint32_t>::max() ) * kGranuleSize )
588 return false;
589
590 void* new_memory = std::malloc( new_size );
591 if ( new_memory == nullptr )
592 return false;
593
594 std::memcpy( new_memory, this->base_ptr(), old_size );
595 std::uint8_t* nb = static_cast<std::uint8_t*>( new_memory );
596 detail::ManagerHeader* nh = reinterpret_cast<detail::ManagerHeader*>( nb + sizeof( detail::BlockHeader ) );
597 nh->owns_memory = true;
598 std::uint32_t extra_idx = detail::byte_off_to_idx( old_size );
599 std::size_t extra_size = new_size - old_size;
600 detail::BlockHeader* last_blk =
601 ( nh->last_block_offset != detail::kNoBlock ) ? detail::block_at( nb, nh->last_block_offset ) : nullptr;
602
603 if ( last_blk != nullptr && last_blk->size == 0 )
604 {
605 std::uint32_t loff = detail::block_idx( nb, last_blk );
606 PersistentAvlTree::remove( nb, nh, loff );
607 nh->total_size = new_size;
608 PersistentAvlTree::insert( nb, nh, loff );
609 }
610 else
611 {
612 if ( extra_size < sizeof( detail::BlockHeader ) + detail::kMinBlockSize )
613 {
614 std::free( new_memory );
615 return false;
616 }
617 detail::BlockHeader* nb_blk = detail::block_at( nb, extra_idx );
618 std::memset( nb_blk, 0, sizeof( detail::BlockHeader ) );
619 // nb_blk->size = 0; — already zero from memset
620 nb_blk->left_offset = detail::kNoBlock;
621 nb_blk->right_offset = detail::kNoBlock;
622 nb_blk->parent_offset = detail::kNoBlock;
623 nb_blk->avl_height = 1;
624 // nb_blk->root_offset = 0; — already zero from memset
625 if ( last_blk != nullptr )
626 {
627 std::uint32_t loff = detail::block_idx( nb, last_blk );
628 nb_blk->prev_offset = loff;
629 nb_blk->next_offset = detail::kNoBlock;
630 last_blk->next_offset = extra_idx;
631 }
632 else
633 {
634 nb_blk->prev_offset = detail::kNoBlock;
635 nb_blk->next_offset = detail::kNoBlock;
636 nh->first_block_offset = extra_idx;
637 }
638 nh->last_block_offset = extra_idx;
639 nh->block_count++;
640 nh->free_count++;
641 nh->total_size = new_size;
642 PersistentAvlTree::insert( nb, nh, extra_idx );
643 }
644 nh->prev_base_ptr = this->base_ptr();
645 nh->prev_total_size = old_size;
646 nh->prev_owns_memory = hdr->owns_memory;
647 s_instance = reinterpret_cast<PersistMemoryManager*>( new_memory );
648 return true;
649 }

651 void rebuild_free_tree()
652 {
653 std::uint8_t* base = this->base_ptr();
654 detail::ManagerHeader* hdr = this->header();
655 hdr->free_tree_root = detail::kNoBlock;
656 hdr->last_block_offset = detail::kNoBlock;
657 std::uint32_t idx = hdr->first_block_offset;

690 void recompute_counters()
691 {
692 std::uint8_t* base = this->base_ptr();
693 detail::ManagerHeader* hdr = this->header();
694 std::uint32_t block_count = 0, free_count = 0, alloc_count = 0;
695 std::uint32_t used_gran = 0;
696 std::uint32_t idx = hdr->first_block_offset;
697 while ( idx != detail::kNoBlock )
698 {
699 if ( detail::idx_to_byte_off( idx ) + sizeof( detail::BlockHeader ) > hdr->total_size )
700 break;
701 detail::BlockHeader* blk = detail::block_at( base, idx );
702 block_count++;
703 used_gran += detail::kBlockHeaderGranules;
704 if ( blk->size > 0 )
705 {
706 alloc_count++;
707 used_gran += blk->size;
708 }
709 else
710 {
711 free_count++;
712 }
713 idx = blk->next_offset;
714 }
715 hdr->block_count = block_count;
716 hdr->free_count = free_count;
717 hdr->alloc_count = alloc_count;
718 hdr->used_size = used_gran;

721 void* translate_ptr( void* ptr ) const noexcept
722 {
723 std::uint8_t* base = const_cast<std::uint8_t*>( this->const_base_ptr() );
724 const detail::ManagerHeader* hdr = this->header();
725 std::uint8_t* raw = static_cast<std::uint8_t*>( ptr );
726 void* prev_buf = hdr->prev_base_ptr;
727 std::size_t prev_sz = hdr->prev_total_size;
728 while ( prev_buf != nullptr && prev_sz > 0 )
729 {
730 auto* prev = static_cast<std::uint8_t*>( prev_buf );
731 if ( raw >= prev && raw < prev + prev_sz )
732 return base + ( raw - prev );
733 auto* ph = reinterpret_cast<detail::ManagerHeader*>( prev + sizeof( detail::BlockHeader ) );
734 prev_buf = ph->prev_base_ptr;
735 prev_sz = ph->prev_total_size;
736 }
737 return ptr;
738 }

740 void coalesce( detail::BlockHeader* blk )
741 {
742 std::uint8_t* base = this->base_ptr();
743 detail::ManagerHeader* hdr = this->header();
744 std::uint32_t b_idx = detail::block_idx( base, blk );
745
746 if ( blk->next_offset != detail::kNoBlock )
747 {
748 detail::BlockHeader* nxt = detail::block_at( base, blk->next_offset );
749 if ( nxt->size == 0 )
750 {
751 std::uint32_t nxt_idx = blk->next_offset;
752 PersistentAvlTree::remove( base, hdr, nxt_idx );
753 blk->next_offset = nxt->next_offset;
754 if ( nxt->next_offset != detail::kNoBlock )
755 detail::block_at( base, nxt->next_offset )->prev_offset = b_idx;
756 else
757 hdr->last_block_offset = b_idx;
758 std::memset( nxt, 0, sizeof( detail::BlockHeader ) );
759 hdr->block_count--;
760 hdr->free_count--;
761 if ( hdr->used_size >= detail::kBlockHeaderGranules )
762 hdr->used_size -= detail::kBlockHeaderGranules;
763 }
764 }
765
766 if ( blk->prev_offset != detail::kNoBlock )
767 {
768 detail::BlockHeader* prv = detail::block_at( base, blk->prev_offset );
769 if ( prv->size == 0 )
770 {
771 std::uint32_t prv_idx = blk->prev_offset;
772 PersistentAvlTree::remove( base, hdr, prv_idx );
773 prv->next_offset = blk->next_offset;
774 if ( blk->next_offset != detail::kNoBlock )
775 detail::block_at( base, blk->next_offset )->prev_offset = prv_idx;
776 else
777 hdr->last_block_offset = prv_idx;
778 std::memset( blk, 0, sizeof( detail::BlockHeader ) );
779 hdr->block_count--;
780 hdr->free_count--;
781 if ( hdr->used_size >= detail::kBlockHeaderGranules )
782 hdr->used_size -= detail::kBlockHeaderGranules;
783 PersistentAvlTree::insert( base, hdr, prv_idx );
784 return;
785 }
786 }
787
788 PersistentAvlTree::insert( base, hdr, b_idx );
789 }

791 void* allocate_from_block( detail::BlockHeader* blk, std::size_t user_size )
792 {
793 std::uint8_t* base = this->base_ptr();
794 detail::ManagerHeader* hdr = this->header();
795 std::uint32_t blk_idx = detail::block_idx( base, blk );
796 PersistentAvlTree::remove( base, hdr, blk_idx );
797
798 std::uint32_t blk_total_gran = detail::block_total_granules( base, hdr, blk );
799 std::uint32_t data_gran = detail::bytes_to_granules( user_size );
800 std::uint32_t needed_gran = detail::kBlockHeaderGranules + data_gran;
801 std::uint32_t min_rem_gran = detail::kBlockHeaderGranules + 1;
802 bool can_split = ( blk_total_gran >= needed_gran + min_rem_gran );
803
804 if ( can_split )
805 {
806 std::uint32_t new_idx = blk_idx + needed_gran;
807 detail::BlockHeader* new_blk = detail::block_at( base, new_idx );
808 std::memset( new_blk, 0, sizeof( detail::BlockHeader ) );
809 // new_blk->size = 0; — already zero from memset
810 new_blk->prev_offset = blk_idx;
811 new_blk->next_offset = blk->next_offset;
812 new_blk->left_offset = detail::kNoBlock;
813 new_blk->right_offset = detail::kNoBlock;
814 new_blk->parent_offset = detail::kNoBlock;
815 new_blk->avl_height = 1;
816 // new_blk->root_offset = 0; — already zero from memset
817 if ( blk->next_offset != detail::kNoBlock )
818 detail::block_at( base, blk->next_offset )->prev_offset = new_idx;
819 else
820 hdr->last_block_offset = new_idx;
821 blk->next_offset = new_idx;
822 hdr->block_count++;
823 hdr->free_count++;
824 hdr->used_size += detail::kBlockHeaderGranules;
825 PersistentAvlTree::insert( base, hdr, new_idx );
826 }
827
828 blk->size = data_gran;
829 blk->root_offset = blk_idx;
830 blk->left_offset = detail::kNoBlock;
831 blk->right_offset = detail::kNoBlock;
832 blk->parent_offset = detail::kNoBlock;
833 blk->avl_height = 0;
834 hdr->alloc_count++;
835 hdr->free_count--;
836 hdr->used_size += data_gran;
837 return detail::user_ptr( blk );
838 }

905template <typename Callback> inline void for_each_block( Callback&& cb )
906{
907 PersistMemoryManager<>* mgr = PersistMemoryManager<>::instance();
908 if ( mgr == nullptr )
909 return;
910 using LockPolicy = config::DefaultConfig::lock_policy;
911 typename LockPolicy::shared_lock_type lock( PersistMemoryManager<>::s_mutex );
912 const std::uint8_t* base = mgr->const_base_ptr();
913 const detail::ManagerHeader* hdr = mgr->header();
914 std::uint32_t idx = hdr->first_block_offset;
915 std::size_t index = 0;
916 while ( idx != detail::kNoBlock )
917 {
918 if ( detail::idx_to_byte_off( idx ) >= hdr->total_size )
919 break;
920 const detail::BlockHeader* blk = detail::block_at( base, idx );
921 std::uint32_t gran = ( blk->next_offset != detail::kNoBlock )
922 ? ( blk->next_offset - idx )
923 : ( detail::byte_off_to_idx( hdr->total_size ) - idx );
924 BlockView view;
925 view.index = index;
926 view.offset = static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( idx ) );
927 view.total_size = detail::granules_to_bytes( gran );
928 view.header_size = sizeof( detail::BlockHeader );
929 view.user_size = detail::granules_to_bytes( blk->size );
930 view.alignment = kGranuleSize;
931 view.used = ( blk->size > 0 );
932 cb( view );
933 ++index;
934 idx = blk->next_offset;
935 }
936}
