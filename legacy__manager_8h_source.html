127template <typename Base> class StatsMixin : public Base
128{
129 public:
130 static MemoryStats get_stats()
131 {
132 using Derived = typename Base::derived_type;
133 MemoryStats stats{};
134 Derived* mgr = Derived::s_instance;
135 if ( mgr == nullptr )
136 return stats;
137 const std::uint8_t* base = mgr->const_base_ptr();
138 const detail::ManagerHeader* hdr = mgr->header();
139 stats.total_blocks = hdr->block_count;
140 stats.free_blocks = hdr->free_count;
141 stats.allocated_blocks = hdr->alloc_count;
142 bool first_free = true;
143 std::uint32_t idx = hdr->first_block_offset;
144 while ( idx != detail::kNoBlock )
145 {
146 const detail::BlockHeader* blk = detail::block_at( base, idx );
147 if ( blk->size == 0 )
148 {
149 std::uint32_t gran = ( blk->next_offset != detail::kNoBlock )
150 ? ( blk->next_offset - idx )
151 : ( detail::byte_off_to_idx( hdr->total_size ) - idx );
152 std::size_t blk_size = detail::granules_to_bytes( gran );
153 if ( first_free )
154 {
155 stats.largest_free = blk_size;
156 stats.smallest_free = blk_size;
157 first_free = false;
158 }
159 else
160 {
161 stats.largest_free = std::max( stats.largest_free, blk_size );
162 stats.smallest_free = std::min( stats.smallest_free, blk_size );
163 }
164 stats.total_fragmentation += blk_size;
165 }
166 idx = blk->next_offset;
167 }
168 return stats;
169 }

175template <typename Base> class ValidationMixin : public Base
176{
177 public:
178 static bool validate()
179 {
180 using Derived = typename Base::derived_type;
181 using LockPolicy = typename Derived::config_type::lock_policy;
182 Derived* mgr = Derived::s_instance;
183 if ( mgr == nullptr )
184 return false;
185 typename LockPolicy::shared_lock_type lock( Derived::s_mutex );
186 const std::uint8_t* base = mgr->const_base_ptr();
187 const detail::ManagerHeader* hdr = mgr->header();
188
189 if ( hdr->magic != kMagic )
190 return false;
191 std::size_t block_count = 0, free_count = 0, alloc_count = 0;
192 std::uint32_t idx = hdr->first_block_offset;
193 while ( idx != detail::kNoBlock )
194 {
195 if ( !detail::is_valid_block( base, hdr, idx ) )
196 return false;
197 const detail::BlockHeader* blk = detail::block_at( base, idx );
198 block_count++;
199 if ( blk->size > 0 )
200 alloc_count++;
201 else
202 free_count++;
203 if ( blk->next_offset != detail::kNoBlock )
204 {
205 const detail::BlockHeader* nxt = detail::block_at( base, blk->next_offset );
206 if ( nxt->prev_offset != idx )
207 return false;
208 }
209 idx = blk->next_offset;
210 }
211 std::size_t tree_free = 0;
212 if ( !validate_avl( base, hdr, hdr->free_tree_root, tree_free ) )
213 return false;
214 if ( tree_free != free_count )
215 return false;
216 return ( block_count == hdr->block_count && free_count == hdr->free_count && alloc_count == hdr->alloc_count );
217 }
218
219 private:
220 static bool validate_avl( const std::uint8_t* base, const detail::ManagerHeader* hdr, std::uint32_t node_idx,
221 std::size_t& count )
222 {
223 if ( node_idx == detail::kNoBlock )
224 return true;
225 if ( detail::idx_to_byte_off( node_idx ) >= hdr->total_size )
226 return false;
227 if ( !detail::is_valid_block( base, hdr, node_idx ) )
228 return false;
229 const detail::BlockHeader* node = detail::block_at( base, node_idx );
230 if ( node->size != 0 )
231 return false;
232 count++;
233 if ( !validate_avl( base, hdr, node->left_offset, count ) )
234 return false;
235 if ( !validate_avl( base, hdr, node->right_offset, count ) )
236 return false;
237 if ( node->left_offset != detail::kNoBlock )
238 {
239 const detail::BlockHeader* lc = detail::block_at( base, node->left_offset );
240 if ( lc->parent_offset != node_idx )
241 return false;
242 }
243 if ( node->right_offset != detail::kNoBlock )
244 {
245 const detail::BlockHeader* rc = detail::block_at( base, node->right_offset );
246 if ( rc->parent_offset != node_idx )
247 return false;
248 }
249 return true;
250 }

272 static bool create( void* memory, std::size_t size )
273 {
274 typename LockPolicy::unique_lock_type lock( s_mutex );
275 if ( memory == nullptr || size < detail::kMinMemorySize )
276 return false;
277 if ( size > static_cast<std::uint64_t>( std::numeric_limits<std::uint32_t>::max() ) * kGranuleSize )
278 return false;
279 if ( size % kGranuleSize != 0 )
280 size -= ( size % kGranuleSize );
281
282 std::uint8_t* base = static_cast<std::uint8_t*>( memory );
283 // Issue #75: BlockHeader_0 at granule 0 holds ManagerHeader as user data.
284 static constexpr std::uint32_t kHdrBlkIdx = 0;
285 static constexpr std::uint32_t kFreeBlkIdx = detail::kBlockHeaderGranules + detail::kManagerHeaderGranules;
286
287 if ( detail::idx_to_byte_off( kFreeBlkIdx ) + sizeof( detail::BlockHeader ) + detail::kMinBlockSize > size )
288 return false;
289
290 detail::BlockHeader* hdr_blk = detail::block_at( base, kHdrBlkIdx );
291 std::memset( hdr_blk, 0, sizeof( detail::BlockHeader ) );
292 hdr_blk->size = detail::kManagerHeaderGranules;
293 hdr_blk->prev_offset = detail::kNoBlock;
294 hdr_blk->next_offset = kFreeBlkIdx;
295 hdr_blk->left_offset = detail::kNoBlock;
296 hdr_blk->right_offset = detail::kNoBlock;
297 hdr_blk->parent_offset = detail::kNoBlock;
298 // hdr_blk->avl_height = 0; — already zero from memset
299 hdr_blk->root_offset = kHdrBlkIdx;
300
301 detail::ManagerHeader* hdr = reinterpret_cast<detail::ManagerHeader*>( base + sizeof( detail::BlockHeader ) );
302 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
303 hdr->magic = kMagic;
304 hdr->total_size = size;
305 hdr->first_block_offset = kHdrBlkIdx;
306 hdr->last_block_offset = detail::kNoBlock;
307 hdr->free_tree_root = detail::kNoBlock;
308 // hdr->owns_memory = false; — already false from memset
309 // hdr->prev_owns_memory = false; — already false from memset
310 hdr->granule_size = static_cast<std::uint16_t>( kGranuleSize );
311
312 detail::BlockHeader* blk = detail::block_at( base, kFreeBlkIdx );
313 std::memset( blk, 0, sizeof( detail::BlockHeader ) );
314 // blk->size = 0; — already zero from memset
315 blk->prev_offset = kHdrBlkIdx;
316 blk->next_offset = detail::kNoBlock;
317 blk->left_offset = detail::kNoBlock;
318 blk->right_offset = detail::kNoBlock;
319 blk->parent_offset = detail::kNoBlock;
320 blk->avl_height = 1;
321 // blk->root_offset = 0; — already zero from memset
322
323 hdr->last_block_offset = kFreeBlkIdx;
324 hdr->free_tree_root = kFreeBlkIdx;
325 hdr->block_count = 2;
326 hdr->free_count = 1;
327 hdr->alloc_count = 1;
328 hdr->used_size = kFreeBlkIdx + detail::kBlockHeaderGranules;
329
330 s_instance = reinterpret_cast<PersistMemoryManager*>( base );
331 return true;
332 }

335 static bool load( void* memory, std::size_t size )
336 {
337 typename LockPolicy::unique_lock_type lock( s_mutex );
338 if ( memory == nullptr || size < detail::kMinMemorySize )
339 return false;
340 std::uint8_t* base = static_cast<std::uint8_t*>( memory );
341 auto* hdr = reinterpret_cast<detail::ManagerHeader*>( base + sizeof( detail::BlockHeader ) );
342 if ( hdr->magic != kMagic || hdr->total_size != size )
343 return false;
344 if ( hdr->granule_size != static_cast<std::uint16_t>( kGranuleSize ) )
345 return false;
346 hdr->owns_memory = hdr->prev_owns_memory = false;
347 hdr->prev_total_size = 0;
348 hdr->prev_base_ptr = nullptr;
349 auto* mgr = reinterpret_cast<PersistMemoryManager*>( base );
350 mgr->repair_linked_list();
351 mgr->recompute_counters();
352 mgr->rebuild_free_tree();
353 s_instance = mgr;
354 return true;
355 }

358 static void destroy()
359 {
360 typename LockPolicy::unique_lock_type lock( s_mutex );
361 if ( s_instance == nullptr )
362 return;
363 detail::ManagerHeader* hdr = s_instance->header();
364 hdr->magic = 0;
365 bool owns = hdr->owns_memory;
366 void* buf = s_instance->base_ptr();
367 void* prev = hdr->prev_base_ptr;
368 bool prev_owns = hdr->prev_owns_memory;
369 s_instance = nullptr;
370 while ( prev != nullptr )
371 {
372 auto* ph = reinterpret_cast<detail::ManagerHeader*>( static_cast<std::uint8_t*>( prev ) +
373 sizeof( detail::BlockHeader ) );
374 void* next_prev = ph->prev_base_ptr;
375 bool next_owns = ph->prev_owns_memory;
376 if ( prev_owns )
377 std::free( prev );
378 prev = next_prev;
379 prev_owns = next_owns;
380 }
381 if ( owns )
382 std::free( buf );
383 }

422 template <class T> static pptr<T> reallocate_typed( pptr<T> p, std::size_t count )
423 {
424 if ( p.is_null() )
425 return allocate_typed<T>( count );
426 if ( count == 0 )
427 {
428 deallocate_typed( p );
429 return pptr<T>();
430 }
431 if ( s_instance == nullptr )
432 return pptr<T>();
433
434 std::uint8_t* base = s_instance->base_ptr();
435 void* old_raw = base + detail::idx_to_byte_off( p.offset() );
436 detail::BlockHeader* blk =
437 detail::header_from_ptr( base, old_raw, static_cast<std::size_t>( s_instance->header()->total_size ) );
438 if ( blk == nullptr || blk->size == 0 )
439 return pptr<T>();
440
441 if ( count > 0 && sizeof( T ) > std::numeric_limits<std::size_t>::max() / count )
442 return pptr<T>();
443 std::uint32_t new_granules = detail::bytes_to_granules( sizeof( T ) * count );
444 if ( new_granules <= blk->size )
445 return p;
446
447 std::size_t old_bytes = detail::granules_to_bytes( blk->size );
448 std::uint32_t old_offset = p.offset();
449 pptr<T> new_p = allocate_typed<T>( count );
450 if ( new_p.is_null() )
451 return pptr<T>();
452
453 // Re-derive after possible auto-expand (Issue #67).
454 std::uint8_t* cur_base = s_instance->base_ptr();
455 void* new_raw = cur_base + detail::idx_to_byte_off( new_p.offset() );
456 void* cur_old = cur_base + detail::idx_to_byte_off( old_offset );
457 std::memcpy( new_raw, cur_old, old_bytes );
458 deallocate_typed( p );
459 return new_p;
460 }

557 {

560 return nullptr;

584 void deallocate_raw( void* ptr )
585 {
586 typename LockPolicy::unique_lock_type lock( s_mutex );
587 if ( ptr == nullptr )
588 return;
589 ptr = translate_ptr( ptr );
590 std::uint8_t* base = this->base_ptr();
591 detail::ManagerHeader* hdr = this->header();
592 detail::BlockHeader* blk = detail::header_from_ptr( base, ptr, static_cast<std::size_t>( hdr->total_size ) );
593 if ( blk == nullptr || blk->size == 0 )
594 return;
595

597 blk->size = 0;
598 blk->root_offset = 0;
599 hdr->alloc_count--;
600 hdr->free_count++;
601 if ( hdr->used_size >= freed )
602 hdr->used_size -= freed;
603 coalesce( blk );

606 bool expand( std::size_t user_size )
607 {
608 detail::ManagerHeader* hdr = this->header();
609 std::size_t old_size = hdr->total_size;
610 std::uint32_t needed = detail::required_block_granules( user_size );
611
612 std::size_t growth = ( old_size / Config::grow_denominator ) * Config::grow_numerator;
613 std::size_t new_size = ( ( growth + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
614 if ( new_size <= old_size )
615 new_size = old_size + kGranuleSize;
616 if ( new_size < old_size + detail::granules_to_bytes( needed ) )
617 {
618 std::size_t extra = detail::granules_to_bytes( needed + detail::kBlockHeaderGranules );
619 if ( extra > std::numeric_limits<std::size_t>::max() - old_size - ( kGranuleSize - 1 ) )
620 return false;
621 new_size = ( ( old_size + extra + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
622 }
623
624 if ( new_size > static_cast<std::uint64_t>( std::numeric_limits<std::uint32_t>::max() ) * kGranuleSize )
625 return false;
626
627 void* new_memory = std::malloc( new_size );
628 if ( new_memory == nullptr )
629 return false;
630
631 std::memcpy( new_memory, this->base_ptr(), old_size );
632 std::uint8_t* nb = static_cast<std::uint8_t*>( new_memory );
633 detail::ManagerHeader* nh = reinterpret_cast<detail::ManagerHeader*>( nb + sizeof( detail::BlockHeader ) );
634 nh->owns_memory = true;
635 std::uint32_t extra_idx = detail::byte_off_to_idx( old_size );
636 std::size_t extra_size = new_size - old_size;
637 detail::BlockHeader* last_blk =
638 ( nh->last_block_offset != detail::kNoBlock ) ? detail::block_at( nb, nh->last_block_offset ) : nullptr;
639
640 if ( last_blk != nullptr && last_blk->size == 0 )
641 {
642 std::uint32_t loff = detail::block_idx( nb, last_blk );
643 PersistentAvlTree::remove( nb, nh, loff );
644 nh->total_size = new_size;
645 PersistentAvlTree::insert( nb, nh, loff );
646 }
647 else
648 {
649 if ( extra_size < sizeof( detail::BlockHeader ) + detail::kMinBlockSize )
650 {
651 std::free( new_memory );
652 return false;
653 }

656 // nb_blk->size = 0; — already zero from memset

659 nb_blk->parent_offset = detail::kNoBlock;
660 nb_blk->avl_height = 1;
661 // nb_blk->root_offset = 0; — already zero from memset
662 if ( last_blk != nullptr )
663 {
664 std::uint32_t loff = detail::block_idx( nb, last_blk );
665 nb_blk->prev_offset = loff;
666 nb_blk->next_offset = detail::kNoBlock;
667 last_blk->next_offset = extra_idx;
668 }
669 else
670 {
671 nb_blk->prev_offset = detail::kNoBlock;

676 nh->block_count++;
677 nh->free_count++;
678 nh->total_size = new_size;
679 PersistentAvlTree::insert( nb, nh, extra_idx );
680 }
681 nh->prev_base_ptr = this->base_ptr();
682 nh->prev_total_size = old_size;
683 nh->prev_owns_memory = hdr->owns_memory;
684 s_instance = reinterpret_cast<PersistMemoryManager*>( new_memory );

728 {
729 std::uint8_t* base = this->base_ptr();
730 detail::ManagerHeader* hdr = this->header();
731 std::uint32_t block_count = 0, free_count = 0, alloc_count = 0;
732 std::uint32_t used_gran = 0;
733 std::uint32_t idx = hdr->first_block_offset;
734 while ( idx != detail::kNoBlock )
735 {
736 if ( detail::idx_to_byte_off( idx ) + sizeof( detail::BlockHeader ) > hdr->total_size )
737 break;
738 detail::BlockHeader* blk = detail::block_at( base, idx );
739 block_count++;

741 if ( blk->size > 0 )
742 {

744 used_gran += blk->size;

746 else
747 {
748 free_count++;
749 }
750 idx = blk->next_offset;
751 }

754 hdr->alloc_count = alloc_count;

758 void* translate_ptr( void* ptr ) const noexcept
759 {
760 std::uint8_t* base = const_cast<std::uint8_t*>( this->const_base_ptr() );
761 const detail::ManagerHeader* hdr = this->header();
762 std::uint8_t* raw = static_cast<std::uint8_t*>( ptr );
763 void* prev_buf = hdr->prev_base_ptr;
764 std::size_t prev_sz = hdr->prev_total_size;
765 while ( prev_buf != nullptr && prev_sz > 0 )
766 {
767 auto* prev = static_cast<std::uint8_t*>( prev_buf );
768 if ( raw >= prev && raw < prev + prev_sz )
769 return base + ( raw - prev );
770 auto* ph = reinterpret_cast<detail::ManagerHeader*>( prev + sizeof( detail::BlockHeader ) );
771 prev_buf = ph->prev_base_ptr;
772 prev_sz = ph->prev_total_size;
773 }
774 return ptr;
775 }

777 void coalesce( detail::BlockHeader* blk )
778 {
779 std::uint8_t* base = this->base_ptr();
780 detail::ManagerHeader* hdr = this->header();
781 std::uint32_t b_idx = detail::block_idx( base, blk );
782
783 if ( blk->next_offset != detail::kNoBlock )
784 {
785 detail::BlockHeader* nxt = detail::block_at( base, blk->next_offset );
786 if ( nxt->size == 0 )
787 {
788 std::uint32_t nxt_idx = blk->next_offset;
789 PersistentAvlTree::remove( base, hdr, nxt_idx );
790 blk->next_offset = nxt->next_offset;
791 if ( nxt->next_offset != detail::kNoBlock )
792 detail::block_at( base, nxt->next_offset )->prev_offset = b_idx;
793 else
794 hdr->last_block_offset = b_idx;
795 std::memset( nxt, 0, sizeof( detail::BlockHeader ) );
796 hdr->block_count--;
797 hdr->free_count--;
798 if ( hdr->used_size >= detail::kBlockHeaderGranules )
799 hdr->used_size -= detail::kBlockHeaderGranules;
800 }
801 }
802
803 if ( blk->prev_offset != detail::kNoBlock )
804 {
805 detail::BlockHeader* prv = detail::block_at( base, blk->prev_offset );
806 if ( prv->size == 0 )
807 {
808 std::uint32_t prv_idx = blk->prev_offset;
809 PersistentAvlTree::remove( base, hdr, prv_idx );
810 prv->next_offset = blk->next_offset;
811 if ( blk->next_offset != detail::kNoBlock )
812 detail::block_at( base, blk->next_offset )->prev_offset = prv_idx;
813 else
814 hdr->last_block_offset = prv_idx;
815 std::memset( blk, 0, sizeof( detail::BlockHeader ) );
816 hdr->block_count--;
817 hdr->free_count--;
818 if ( hdr->used_size >= detail::kBlockHeaderGranules )
819 hdr->used_size -= detail::kBlockHeaderGranules;
820 PersistentAvlTree::insert( base, hdr, prv_idx );
821 return;
822 }
823 }
824
825 PersistentAvlTree::insert( base, hdr, b_idx );
826 }

828 void* allocate_from_block( detail::BlockHeader* blk, std::size_t user_size )
829 {
830 std::uint8_t* base = this->base_ptr();
831 detail::ManagerHeader* hdr = this->header();
832 std::uint32_t blk_idx = detail::block_idx( base, blk );
833 PersistentAvlTree::remove( base, hdr, blk_idx );
834
835 std::uint32_t blk_total_gran = detail::block_total_granules( base, hdr, blk );
836 std::uint32_t data_gran = detail::bytes_to_granules( user_size );
837 std::uint32_t needed_gran = detail::kBlockHeaderGranules + data_gran;
838 std::uint32_t min_rem_gran = detail::kBlockHeaderGranules + 1;
839 bool can_split = ( blk_total_gran >= needed_gran + min_rem_gran );
840
841 if ( can_split )
842 {
843 std::uint32_t new_idx = blk_idx + needed_gran;
844 detail::BlockHeader* new_blk = detail::block_at( base, new_idx );
845 std::memset( new_blk, 0, sizeof( detail::BlockHeader ) );
846 // new_blk->size = 0; — already zero from memset
847 new_blk->prev_offset = blk_idx;
848 new_blk->next_offset = blk->next_offset;
849 new_blk->left_offset = detail::kNoBlock;
850 new_blk->right_offset = detail::kNoBlock;
851 new_blk->parent_offset = detail::kNoBlock;
852 new_blk->avl_height = 1;
853 // new_blk->root_offset = 0; — already zero from memset
854 if ( blk->next_offset != detail::kNoBlock )
855 detail::block_at( base, blk->next_offset )->prev_offset = new_idx;
856 else
857 hdr->last_block_offset = new_idx;
858 blk->next_offset = new_idx;
859 hdr->block_count++;
860 hdr->free_count++;
861 hdr->used_size += detail::kBlockHeaderGranules;
862 PersistentAvlTree::insert( base, hdr, new_idx );
863 }
864
865 blk->size = data_gran;
866 blk->root_offset = blk_idx;
867 blk->left_offset = detail::kNoBlock;
868 blk->right_offset = detail::kNoBlock;
869 blk->parent_offset = detail::kNoBlock;
870 blk->avl_height = 0;
871 hdr->alloc_count++;
872 hdr->free_count--;
873 hdr->used_size += data_gran;
874 return detail::user_ptr( blk );
875 }

942template <typename Callback> inline void for_each_block( Callback&& cb )
943{
944 PersistMemoryManager<>* mgr = PersistMemoryManager<>::instance();
945 if ( mgr == nullptr )
946 return;
947 using LockPolicy = config::DefaultConfig::lock_policy;
948 typename LockPolicy::shared_lock_type lock( PersistMemoryManager<>::s_mutex );
949 const std::uint8_t* base = mgr->const_base_ptr();
950 const detail::ManagerHeader* hdr = mgr->header();
951 std::uint32_t idx = hdr->first_block_offset;
952 std::size_t index = 0;
953 while ( idx != detail::kNoBlock )
954 {
955 if ( detail::idx_to_byte_off( idx ) >= hdr->total_size )
956 break;
957 const detail::BlockHeader* blk = detail::block_at( base, idx );
958 std::uint32_t gran = ( blk->next_offset != detail::kNoBlock )
959 ? ( blk->next_offset - idx )
960 : ( detail::byte_off_to_idx( hdr->total_size ) - idx );
961 BlockView view;
962 view.index = index;
963 view.offset = static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( idx ) );
964 view.total_size = detail::granules_to_bytes( gran );
965 view.header_size = sizeof( detail::BlockHeader );
966 view.user_size = detail::granules_to_bytes( blk->size );
967 view.alignment = kGranuleSize;
968 view.used = ( blk->size > 0 );
969 cb( view );
970 ++index;
971 idx = blk->next_offset;
972 }
973}

976{

978 if ( mgr == nullptr )

986 for ( std::uint32_t idx = hdr->first_block_offset; idx != detail::kNoBlock; )
