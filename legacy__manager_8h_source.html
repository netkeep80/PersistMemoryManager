91
92 constexpr bool operator==( const pptr<T>& other ) const noexcept { return _idx == other._idx; }

133template <typename Base> class StatsMixin : public Base
134{
135 public:
136 static MemoryStats get_stats()
137 {
138 using Derived = typename Base::derived_type;
139 MemoryStats stats{};
140 Derived* mgr = Derived::s_instance;
141 if ( mgr == nullptr )
142 return stats;
143 const std::uint8_t* base = mgr->const_base_ptr();
144 const detail::ManagerHeader* hdr = mgr->header();
145 stats.total_blocks = hdr->block_count;
146 stats.free_blocks = hdr->free_count;
147 stats.allocated_blocks = hdr->alloc_count;
148 bool first_free = true;
149 std::uint32_t idx = hdr->first_block_offset;
150 while ( idx != detail::kNoBlock )
151 {
152 const detail::BlockHeader* blk = detail::block_at( base, idx );
153 if ( blk->size == 0 )
154 {
155 std::uint32_t gran = ( blk->next_offset != detail::kNoBlock )
156 ? ( blk->next_offset - idx )
157 : ( detail::byte_off_to_idx( hdr->total_size ) - idx );
158 std::size_t blk_size = detail::granules_to_bytes( gran );
159 if ( first_free )
160 {
161 stats.largest_free = blk_size;
162 stats.smallest_free = blk_size;
163 first_free = false;
164 }
165 else
166 {
167 stats.largest_free = std::max( stats.largest_free, blk_size );
168 stats.smallest_free = std::min( stats.smallest_free, blk_size );
169 }
170 stats.total_fragmentation += blk_size;
171 }
172 idx = blk->next_offset;
173 }
174 return stats;
175 }

181template <typename Base> class ValidationMixin : public Base
182{
183 public:
184 static bool validate()
185 {
186 using Derived = typename Base::derived_type;
187 using LockPolicy = typename Derived::config_type::lock_policy;
188 Derived* mgr = Derived::s_instance;
189 if ( mgr == nullptr )
190 return false;
191 typename LockPolicy::shared_lock_type lock( Derived::s_mutex );
192 const std::uint8_t* base = mgr->const_base_ptr();
193 const detail::ManagerHeader* hdr = mgr->header();
194
195 if ( hdr->magic != kMagic )
196 return false;
197 std::size_t block_count = 0, free_count = 0, alloc_count = 0;
198 std::uint32_t idx = hdr->first_block_offset;
199 while ( idx != detail::kNoBlock )
200 {
201 if ( !detail::is_valid_block( base, hdr, idx ) )
202 return false;
203 const detail::BlockHeader* blk = detail::block_at( base, idx );
204 block_count++;
205 if ( blk->size > 0 )
206 alloc_count++;
207 else
208 free_count++;
209 if ( blk->next_offset != detail::kNoBlock )
210 {
211 const detail::BlockHeader* nxt = detail::block_at( base, blk->next_offset );
212 if ( nxt->prev_offset != idx )
213 return false;
214 }
215 idx = blk->next_offset;
216 }
217 std::size_t tree_free = 0;
218 if ( !validate_avl( base, hdr, hdr->free_tree_root, tree_free ) )
219 return false;
220 if ( tree_free != free_count )
221 return false;
222 return ( block_count == hdr->block_count && free_count == hdr->free_count && alloc_count == hdr->alloc_count );
223 }
224
225 private:
226 static bool validate_avl( const std::uint8_t* base, const detail::ManagerHeader* hdr, std::uint32_t node_idx,
227 std::size_t& count )
228 {
229 if ( node_idx == detail::kNoBlock )
230 return true;
231 if ( detail::idx_to_byte_off( node_idx ) >= hdr->total_size )
232 return false;
233 if ( !detail::is_valid_block( base, hdr, node_idx ) )
234 return false;
235 const detail::BlockHeader* node = detail::block_at( base, node_idx );
236 if ( node->size != 0 )
237 return false;
238 count++;
239 if ( !validate_avl( base, hdr, node->left_offset, count ) )
240 return false;
241 if ( !validate_avl( base, hdr, node->right_offset, count ) )
242 return false;
243 if ( node->left_offset != detail::kNoBlock )
244 {
245 const detail::BlockHeader* lc = detail::block_at( base, node->left_offset );
246 if ( lc->parent_offset != node_idx )
247 return false;
248 }
249 if ( node->right_offset != detail::kNoBlock )
250 {
251 const detail::BlockHeader* rc = detail::block_at( base, node->right_offset );
252 if ( rc->parent_offset != node_idx )
253 return false;
254 }
255 return true;
256 }

278 static bool create( void* memory, std::size_t size )
279 {
280 typename LockPolicy::unique_lock_type lock( s_mutex );
281 if ( memory == nullptr || size < detail::kMinMemorySize )
282 return false;
283 if ( size > static_cast<std::uint64_t>( std::numeric_limits<std::uint32_t>::max() ) * kGranuleSize )
284 return false;
285 if ( size % kGranuleSize != 0 )
286 size -= ( size % kGranuleSize );
287
288 std::uint8_t* base = static_cast<std::uint8_t*>( memory );
289 // Issue #75: BlockHeader_0 at granule 0 holds ManagerHeader as user data.
290 static constexpr std::uint32_t kHdrBlkIdx = 0;
291 static constexpr std::uint32_t kFreeBlkIdx = detail::kBlockHeaderGranules + detail::kManagerHeaderGranules;
292
293 if ( detail::idx_to_byte_off( kFreeBlkIdx ) + sizeof( detail::BlockHeader ) + detail::kMinBlockSize > size )
294 return false;
295
296 detail::BlockHeader* hdr_blk = detail::block_at( base, kHdrBlkIdx );
297 std::memset( hdr_blk, 0, sizeof( detail::BlockHeader ) );
298 hdr_blk->size = detail::kManagerHeaderGranules;
299 hdr_blk->prev_offset = detail::kNoBlock;
300 hdr_blk->next_offset = kFreeBlkIdx;
301 hdr_blk->left_offset = detail::kNoBlock;
302 hdr_blk->right_offset = detail::kNoBlock;
303 hdr_blk->parent_offset = detail::kNoBlock;
304 // hdr_blk->avl_height = 0; — already zero from memset
305 hdr_blk->root_offset = kHdrBlkIdx;
306
307 detail::ManagerHeader* hdr = reinterpret_cast<detail::ManagerHeader*>( base + sizeof( detail::BlockHeader ) );
308 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
309 hdr->magic = kMagic;
310 hdr->total_size = size;
311 hdr->first_block_offset = kHdrBlkIdx;
312 hdr->last_block_offset = detail::kNoBlock;
313 hdr->free_tree_root = detail::kNoBlock;
314 // hdr->owns_memory = false; — already false from memset
315 // hdr->prev_owns_memory = false; — already false from memset
316 hdr->granule_size = static_cast<std::uint16_t>( kGranuleSize );
317
318 detail::BlockHeader* blk = detail::block_at( base, kFreeBlkIdx );
319 std::memset( blk, 0, sizeof( detail::BlockHeader ) );
320 // blk->size = 0; — already zero from memset
321 blk->prev_offset = kHdrBlkIdx;
322 blk->next_offset = detail::kNoBlock;
323 blk->left_offset = detail::kNoBlock;
324 blk->right_offset = detail::kNoBlock;
325 blk->parent_offset = detail::kNoBlock;
326 blk->avl_height = 1;
327 // blk->root_offset = 0; — already zero from memset
328
329 hdr->last_block_offset = kFreeBlkIdx;
330 hdr->free_tree_root = kFreeBlkIdx;
331 hdr->block_count = 2;
332 hdr->free_count = 1;
333 hdr->alloc_count = 1;
334 hdr->used_size = kFreeBlkIdx + detail::kBlockHeaderGranules;
335
336 s_instance = reinterpret_cast<PersistMemoryManager*>( base );
337 return true;
338 }

341 static bool load( void* memory, std::size_t size )
342 {
343 typename LockPolicy::unique_lock_type lock( s_mutex );
344 if ( memory == nullptr || size < detail::kMinMemorySize )
345 return false;
346 std::uint8_t* base = static_cast<std::uint8_t*>( memory );
347 auto* hdr = reinterpret_cast<detail::ManagerHeader*>( base + sizeof( detail::BlockHeader ) );
348 if ( hdr->magic != kMagic || hdr->total_size != size )
349 return false;
350 if ( hdr->granule_size != static_cast<std::uint16_t>( kGranuleSize ) )
351 return false;
352 hdr->owns_memory = hdr->prev_owns_memory = false;
353 hdr->prev_total_size = 0;
354 hdr->prev_base_ptr = nullptr;
355 auto* mgr = reinterpret_cast<PersistMemoryManager*>( base );
356 mgr->repair_linked_list();
357 mgr->recompute_counters();
358 mgr->rebuild_free_tree();
359 s_instance = mgr;
360 return true;
361 }

364 static void destroy()
365 {
366 typename LockPolicy::unique_lock_type lock( s_mutex );
367 if ( s_instance == nullptr )
368 return;
369 detail::ManagerHeader* hdr = s_instance->header();
370 hdr->magic = 0;
371 bool owns = hdr->owns_memory;
372 void* buf = s_instance->base_ptr();
373 void* prev = hdr->prev_base_ptr;
374 bool prev_owns = hdr->prev_owns_memory;
375 s_instance = nullptr;
376 while ( prev != nullptr )
377 {
378 auto* ph = reinterpret_cast<detail::ManagerHeader*>( static_cast<std::uint8_t*>( prev ) +
379 sizeof( detail::BlockHeader ) );
380 void* next_prev = ph->prev_base_ptr;
381 bool next_owns = ph->prev_owns_memory;
382 if ( prev_owns )
383 std::free( prev );
384 prev = next_prev;
385 prev_owns = next_owns;
386 }
387 if ( owns )
388 std::free( buf );
389 }

428 template <class T> static pptr<T> reallocate_typed( pptr<T> p, std::size_t count )
429 {
430 if ( p.is_null() )
431 return allocate_typed<T>( count );
432 if ( count == 0 )
433 {
434 deallocate_typed( p );
435 return pptr<T>();
436 }
437 if ( s_instance == nullptr )
438 return pptr<T>();
439
440 std::uint8_t* base = s_instance->base_ptr();
441 void* old_raw = base + detail::idx_to_byte_off( p.offset() );
442 detail::BlockHeader* blk =
443 detail::header_from_ptr( base, old_raw, static_cast<std::size_t>( s_instance->header()->total_size ) );
444 if ( blk == nullptr || blk->size == 0 )
445 return pptr<T>();
446
447 if ( count > 0 && sizeof( T ) > std::numeric_limits<std::size_t>::max() / count )
448 return pptr<T>();
449 std::uint32_t new_granules = detail::bytes_to_granules( sizeof( T ) * count );
450 if ( new_granules <= blk->size )
451 return p;
452
453 std::size_t old_bytes = detail::granules_to_bytes( blk->size );
454 std::uint32_t old_offset = p.offset();
455 pptr<T> new_p = allocate_typed<T>( count );
456 if ( new_p.is_null() )
457 return pptr<T>();
458
459 // Re-derive after possible auto-expand (Issue #67).
460 std::uint8_t* cur_base = s_instance->base_ptr();
461 void* new_raw = cur_base + detail::idx_to_byte_off( new_p.offset() );
462 void* cur_old = cur_base + detail::idx_to_byte_off( old_offset );
463 std::memcpy( new_raw, cur_old, old_bytes );
464 deallocate_typed( p );
465 return new_p;
466 }

563 {

566 return nullptr;

584 std::uint32_t new_idx = PersistentAvlTree::find_best_fit( nb, nh, needed );
585 if ( new_idx != detail::kNoBlock )

590 void deallocate_raw( void* ptr )
591 {
592 typename LockPolicy::unique_lock_type lock( s_mutex );
593 if ( ptr == nullptr )
594 return;
595 ptr = translate_ptr( ptr );
596 std::uint8_t* base = this->base_ptr();
597 detail::ManagerHeader* hdr = this->header();
598 detail::BlockHeader* blk = detail::header_from_ptr( base, ptr, static_cast<std::size_t>( hdr->total_size ) );
599 if ( blk == nullptr || blk->size == 0 )
600 return;
601
602 std::uint32_t freed = blk->size;
603 blk->size = 0;
604 blk->root_offset = 0;
605 hdr->alloc_count--;
606 hdr->free_count++;
607 if ( hdr->used_size >= freed )
608 hdr->used_size -= freed;
609 coalesce( blk );
610 }

612 bool expand( std::size_t user_size )
613 {
614 detail::ManagerHeader* hdr = this->header();
615 std::size_t old_size = hdr->total_size;
616 std::uint32_t needed = detail::required_block_granules( user_size );
617
618 std::size_t growth = ( old_size / Config::grow_denominator ) * Config::grow_numerator;
619 std::size_t new_size = ( ( growth + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
620 if ( new_size <= old_size )
621 new_size = old_size + kGranuleSize;
622 if ( new_size < old_size + detail::granules_to_bytes( needed ) )
623 {
624 std::size_t extra = detail::granules_to_bytes( needed + detail::kBlockHeaderGranules );
625 if ( extra > std::numeric_limits<std::size_t>::max() - old_size - ( kGranuleSize - 1 ) )
626 return false;
627 new_size = ( ( old_size + extra + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
628 }
629
630 if ( new_size > static_cast<std::uint64_t>( std::numeric_limits<std::uint32_t>::max() ) * kGranuleSize )
631 return false;
632
633 void* new_memory = std::malloc( new_size );
634 if ( new_memory == nullptr )
635 return false;
636
637 std::memcpy( new_memory, this->base_ptr(), old_size );
638 std::uint8_t* nb = static_cast<std::uint8_t*>( new_memory );
639 detail::ManagerHeader* nh = reinterpret_cast<detail::ManagerHeader*>( nb + sizeof( detail::BlockHeader ) );
640 nh->owns_memory = true;
641 std::uint32_t extra_idx = detail::byte_off_to_idx( old_size );
642 std::size_t extra_size = new_size - old_size;
643 detail::BlockHeader* last_blk =
644 ( nh->last_block_offset != detail::kNoBlock ) ? detail::block_at( nb, nh->last_block_offset ) : nullptr;
645
646 if ( last_blk != nullptr && last_blk->size == 0 )
647 {
648 std::uint32_t loff = detail::block_idx( nb, last_blk );
649 PersistentAvlTree::remove( nb, nh, loff );
650 nh->total_size = new_size;
651 PersistentAvlTree::insert( nb, nh, loff );
652 }
653 else
654 {
655 if ( extra_size < sizeof( detail::BlockHeader ) + detail::kMinBlockSize )
656 {
657 std::free( new_memory );
658 return false;
659 }

665 nb_blk->parent_offset = detail::kNoBlock;
666 nb_blk->avl_height = 1;
667 // nb_blk->root_offset = 0; — already zero from memset
668 if ( last_blk != nullptr )

670 std::uint32_t loff = detail::block_idx( nb, last_blk );
671 nb_blk->prev_offset = loff;
672 nb_blk->next_offset = detail::kNoBlock;
673 last_blk->next_offset = extra_idx;
674 }
675 else
676 {
677 nb_blk->prev_offset = detail::kNoBlock;
678 nb_blk->next_offset = detail::kNoBlock;

682 nh->block_count++;
683 nh->free_count++;
684 nh->total_size = new_size;
685 PersistentAvlTree::insert( nb, nh, extra_idx );
686 }
687 nh->prev_base_ptr = this->base_ptr();
688 nh->prev_total_size = old_size;
689 nh->prev_owns_memory = hdr->owns_memory;
690 s_instance = reinterpret_cast<PersistMemoryManager*>( new_memory );

764 void* translate_ptr( void* ptr ) const noexcept
765 {
766 std::uint8_t* base = const_cast<std::uint8_t*>( this->const_base_ptr() );
767 const detail::ManagerHeader* hdr = this->header();
768 std::uint8_t* raw = static_cast<std::uint8_t*>( ptr );
769 void* prev_buf = hdr->prev_base_ptr;
770 std::size_t prev_sz = hdr->prev_total_size;
771 while ( prev_buf != nullptr && prev_sz > 0 )
772 {
773 auto* prev = static_cast<std::uint8_t*>( prev_buf );
774 if ( raw >= prev && raw < prev + prev_sz )
775 return base + ( raw - prev );
776 auto* ph = reinterpret_cast<detail::ManagerHeader*>( prev + sizeof( detail::BlockHeader ) );
777 prev_buf = ph->prev_base_ptr;
778 prev_sz = ph->prev_total_size;
779 }
780 return ptr;

783 void coalesce( detail::BlockHeader* blk )
784 {
785 std::uint8_t* base = this->base_ptr();
786 detail::ManagerHeader* hdr = this->header();
787 std::uint32_t b_idx = detail::block_idx( base, blk );
788
789 if ( blk->next_offset != detail::kNoBlock )
790 {
791 detail::BlockHeader* nxt = detail::block_at( base, blk->next_offset );
792 if ( nxt->size == 0 )
793 {
794 std::uint32_t nxt_idx = blk->next_offset;
795 PersistentAvlTree::remove( base, hdr, nxt_idx );
796 blk->next_offset = nxt->next_offset;
797 if ( nxt->next_offset != detail::kNoBlock )
798 detail::block_at( base, nxt->next_offset )->prev_offset = b_idx;
799 else
800 hdr->last_block_offset = b_idx;
801 std::memset( nxt, 0, sizeof( detail::BlockHeader ) );
802 hdr->block_count--;
803 hdr->free_count--;
804 if ( hdr->used_size >= detail::kBlockHeaderGranules )
805 hdr->used_size -= detail::kBlockHeaderGranules;
806 }
807 }
808
809 if ( blk->prev_offset != detail::kNoBlock )
810 {
811 detail::BlockHeader* prv = detail::block_at( base, blk->prev_offset );
812 if ( prv->size == 0 )
813 {
814 std::uint32_t prv_idx = blk->prev_offset;
815 PersistentAvlTree::remove( base, hdr, prv_idx );
816 prv->next_offset = blk->next_offset;
817 if ( blk->next_offset != detail::kNoBlock )
818 detail::block_at( base, blk->next_offset )->prev_offset = prv_idx;
819 else
820 hdr->last_block_offset = prv_idx;
821 std::memset( blk, 0, sizeof( detail::BlockHeader ) );
822 hdr->block_count--;
823 hdr->free_count--;
824 if ( hdr->used_size >= detail::kBlockHeaderGranules )
825 hdr->used_size -= detail::kBlockHeaderGranules;
826 PersistentAvlTree::insert( base, hdr, prv_idx );
827 return;
828 }
829 }
830
831 PersistentAvlTree::insert( base, hdr, b_idx );
832 }

834 void* allocate_from_block( detail::BlockHeader* blk, std::size_t user_size )
835 {
836 std::uint8_t* base = this->base_ptr();
837 detail::ManagerHeader* hdr = this->header();
838 std::uint32_t blk_idx = detail::block_idx( base, blk );
839 PersistentAvlTree::remove( base, hdr, blk_idx );
840
841 std::uint32_t blk_total_gran = detail::block_total_granules( base, hdr, blk );
842 std::uint32_t data_gran = detail::bytes_to_granules( user_size );
843 std::uint32_t needed_gran = detail::kBlockHeaderGranules + data_gran;
844 std::uint32_t min_rem_gran = detail::kBlockHeaderGranules + 1;
845 bool can_split = ( blk_total_gran >= needed_gran + min_rem_gran );
846
847 if ( can_split )
848 {
849 std::uint32_t new_idx = blk_idx + needed_gran;
850 detail::BlockHeader* new_blk = detail::block_at( base, new_idx );
851 std::memset( new_blk, 0, sizeof( detail::BlockHeader ) );
852 // new_blk->size = 0; — already zero from memset
853 new_blk->prev_offset = blk_idx;
854 new_blk->next_offset = blk->next_offset;
855 new_blk->left_offset = detail::kNoBlock;
856 new_blk->right_offset = detail::kNoBlock;
857 new_blk->parent_offset = detail::kNoBlock;
858 new_blk->avl_height = 1;
859 // new_blk->root_offset = 0; — already zero from memset
860 if ( blk->next_offset != detail::kNoBlock )
861 detail::block_at( base, blk->next_offset )->prev_offset = new_idx;
862 else
863 hdr->last_block_offset = new_idx;
864 blk->next_offset = new_idx;
865 hdr->block_count++;
866 hdr->free_count++;
867 hdr->used_size += detail::kBlockHeaderGranules;
868 PersistentAvlTree::insert( base, hdr, new_idx );
869 }
870
871 blk->size = data_gran;
872 blk->root_offset = blk_idx;
873 blk->left_offset = detail::kNoBlock;
874 blk->right_offset = detail::kNoBlock;
875 blk->parent_offset = detail::kNoBlock;
876 blk->avl_height = 0;
877 hdr->alloc_count++;
878 hdr->free_count--;
879 hdr->used_size += data_gran;
880 return detail::user_ptr( blk );
881 }

948template <typename Callback> inline void for_each_block( Callback&& cb )
949{
950 PersistMemoryManager<>* mgr = PersistMemoryManager<>::instance();
951 if ( mgr == nullptr )
952 return;
953 using LockPolicy = config::DefaultConfig::lock_policy;
954 typename LockPolicy::shared_lock_type lock( PersistMemoryManager<>::s_mutex );
955 const std::uint8_t* base = mgr->const_base_ptr();
956 const detail::ManagerHeader* hdr = mgr->header();
957 std::uint32_t idx = hdr->first_block_offset;
958 std::size_t index = 0;
959 while ( idx != detail::kNoBlock )
960 {
961 if ( detail::idx_to_byte_off( idx ) >= hdr->total_size )
962 break;
963 const detail::BlockHeader* blk = detail::block_at( base, idx );
964 std::uint32_t gran = ( blk->next_offset != detail::kNoBlock )
965 ? ( blk->next_offset - idx )
966 : ( detail::byte_off_to_idx( hdr->total_size ) - idx );
967 BlockView view;
968 view.index = index;
969 view.offset = static_cast<std::ptrdiff_t>( detail::idx_to_byte_off( idx ) );
970 view.total_size = detail::granules_to_bytes( gran );
971 view.header_size = sizeof( detail::BlockHeader );
972 view.user_size = detail::granules_to_bytes( blk->size );
973 view.alignment = kGranuleSize;
974 view.used = ( blk->size > 0 );
975 cb( view );
976 ++index;
977 idx = blk->next_offset;
978 }

982{

992 for ( std::uint32_t idx = hdr->first_block_offset; idx != detail::kNoBlock; )

1014 }

1016 }
