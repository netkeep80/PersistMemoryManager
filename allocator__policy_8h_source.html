97 static void* allocate_from_block( std::uint8_t* base, detail::ManagerHeader* hdr, detail::BlockHeader* blk,
98 std::size_t user_size )
99 {
100 std::uint32_t blk_idx = detail::block_idx( base, blk );
101 // State: FreeBlock → FreeBlockRemovedAVL
102 FreeBlockTreeT::remove( base, hdr, blk_idx );
103
104 std::uint32_t blk_total_gran = detail::block_total_granules( base, hdr, blk );
105 std::uint32_t data_gran = detail::bytes_to_granules( user_size );
106 std::uint32_t needed_gran = detail::kBlockHeaderGranules + data_gran;
107 std::uint32_t min_rem_gran = detail::kBlockHeaderGranules + 1;
108 bool can_split = ( blk_total_gran >= needed_gran + min_rem_gran );
109
110 if ( can_split )
111 {
112 // State: FreeBlockRemovedAVL → SplittingBlock [begin_splitting]
113 std::uint32_t new_idx = blk_idx + needed_gran;
114 detail::BlockHeader* new_blk = detail::block_at( base, new_idx );
115 // SplittingBlock::initialize_new_block
116 std::memset( new_blk, 0, sizeof( detail::BlockHeader ) );
117 new_blk->prev_offset = blk_idx;
118 new_blk->next_offset = blk->next_offset;
119 new_blk->left_offset = detail::kNoBlock;
120 new_blk->right_offset = detail::kNoBlock;
121 new_blk->parent_offset = detail::kNoBlock;
122 new_blk->avl_height = 1;
123 // SplittingBlock::link_new_block
124 if ( blk->next_offset != detail::kNoBlock )
125 detail::block_at( base, blk->next_offset )->prev_offset = new_idx;
126 else
127 hdr->last_block_offset = new_idx;
128 blk->next_offset = new_idx;
129 hdr->block_count++;
130 hdr->free_count++;
131 hdr->used_size += detail::kBlockHeaderGranules;
132 FreeBlockTreeT::insert( base, hdr, new_idx );
133 }
134
135 // State: FreeBlockRemovedAVL/SplittingBlock → AllocatedBlock
136 // [mark_as_allocated / finalize_split: set size=data_gran, root_offset=blk_idx]
137 blk->size = data_gran;
138 blk->root_offset = blk_idx;
139 blk->left_offset = detail::kNoBlock;
140 blk->right_offset = detail::kNoBlock;
141 blk->parent_offset = detail::kNoBlock;
142 blk->avl_height = 0;
143 hdr->alloc_count++;
144 hdr->free_count--;
145 hdr->used_size += data_gran;
146 return detail::user_ptr( blk );
147 }

167 static void coalesce( std::uint8_t* base, detail::ManagerHeader* hdr, detail::BlockHeader* blk )
168 {
169 // State: FreeBlockNotInAVL → CoalescingBlock [begin_coalescing]
170 std::uint32_t b_idx = detail::block_idx( base, blk );
171
172 // Слияние с правым соседом
173 // State: CoalescingBlock::coalesce_with_next
174 if ( blk->next_offset != detail::kNoBlock )
175 {
176 detail::BlockHeader* nxt = detail::block_at( base, blk->next_offset );
177 if ( nxt->size == 0 )
178 {
179 std::uint32_t nxt_idx = blk->next_offset;
180 FreeBlockTreeT::remove( base, hdr, nxt_idx );
181 blk->next_offset = nxt->next_offset;
182 if ( nxt->next_offset != detail::kNoBlock )
183 detail::block_at( base, nxt->next_offset )->prev_offset = b_idx;
184 else
185 hdr->last_block_offset = b_idx;
186 std::memset( nxt, 0, sizeof( detail::BlockHeader ) );
187 hdr->block_count--;
188 hdr->free_count--;
189 if ( hdr->used_size >= detail::kBlockHeaderGranules )
190 hdr->used_size -= detail::kBlockHeaderGranules;
191 }
192 }
193
194 // Слияние с левым соседом
195 // State: CoalescingBlock::coalesce_with_prev → результат CoalescingBlock(prv)
196 if ( blk->prev_offset != detail::kNoBlock )
197 {
198 detail::BlockHeader* prv = detail::block_at( base, blk->prev_offset );
199 if ( prv->size == 0 )
200 {
201 std::uint32_t prv_idx = blk->prev_offset;
202 FreeBlockTreeT::remove( base, hdr, prv_idx );
203 prv->next_offset = blk->next_offset;
204 if ( blk->next_offset != detail::kNoBlock )
205 detail::block_at( base, blk->next_offset )->prev_offset = prv_idx;
206 else
207 hdr->last_block_offset = prv_idx;
208 std::memset( blk, 0, sizeof( detail::BlockHeader ) );
209 hdr->block_count--;
210 hdr->free_count--;
211 if ( hdr->used_size >= detail::kBlockHeaderGranules )
212 hdr->used_size -= detail::kBlockHeaderGranules;
213 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
214 FreeBlockTreeT::insert( base, hdr, prv_idx );
215 return;
216 }
217 }
218
219 // State: CoalescingBlock::finalize_coalesce → FreeBlock; вставка в AVL
220 FreeBlockTreeT::insert( base, hdr, b_idx );
221 }

234 static void rebuild_free_tree( std::uint8_t* base, detail::ManagerHeader* hdr )
235 {
236 hdr->free_tree_root = detail::kNoBlock;
237 hdr->last_block_offset = detail::kNoBlock;
238 std::uint32_t idx = hdr->first_block_offset;
239 while ( idx != detail::kNoBlock )
240 {
241 detail::BlockHeader* blk = detail::block_at( base, idx );
242 blk->left_offset = detail::kNoBlock;
243 blk->right_offset = detail::kNoBlock;
244 blk->parent_offset = detail::kNoBlock;
245 blk->avl_height = 0;
246 if ( blk->size == 0 )
247 FreeBlockTreeT::insert( base, hdr, idx );
248 if ( blk->next_offset == detail::kNoBlock )
249 hdr->last_block_offset = idx;
250 idx = blk->next_offset;
251 }
252 }

263 static void repair_linked_list( std::uint8_t* base, detail::ManagerHeader* hdr )
264 {
265 std::uint32_t idx = hdr->first_block_offset;
266 std::uint32_t prev = detail::kNoBlock;

287 static void recompute_counters( std::uint8_t* base, detail::ManagerHeader* hdr )
288 {
289 std::uint32_t block_count = 0, free_count = 0, alloc_count = 0;
290 std::uint32_t used_gran = 0;
291 std::uint32_t idx = hdr->first_block_offset;
292 while ( idx != detail::kNoBlock )
293 {
294 if ( detail::idx_to_byte_off( idx ) + sizeof( detail::BlockHeader ) > hdr->total_size )
295 break;
296 detail::BlockHeader* blk = detail::block_at( base, idx );
297 block_count++;
298 used_gran += detail::kBlockHeaderGranules;
299 if ( blk->size > 0 )
300 {
301 alloc_count++;
302 used_gran += blk->size;
303 }
304 else
305 {
306 free_count++;
307 }
308 idx = blk->next_offset;
309 }
310 hdr->block_count = block_count;
311 hdr->free_count = free_count;
312 hdr->alloc_count = alloc_count;
313 hdr->used_size = used_gran;
314 }
