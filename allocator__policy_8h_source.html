73 static void* allocate_from_block( std::uint8_t* base, detail::ManagerHeader* hdr, detail::BlockHeader* blk,
74 std::size_t user_size )
75 {
76 std::uint32_t blk_idx = detail::block_idx( base, blk );
77 FreeBlockTreeT::remove( base, hdr, blk_idx );
78
79 std::uint32_t blk_total_gran = detail::block_total_granules( base, hdr, blk );
80 std::uint32_t data_gran = detail::bytes_to_granules( user_size );
81 std::uint32_t needed_gran = detail::kBlockHeaderGranules + data_gran;
82 std::uint32_t min_rem_gran = detail::kBlockHeaderGranules + 1;
83 bool can_split = ( blk_total_gran >= needed_gran + min_rem_gran );
84
85 if ( can_split )
86 {
87 std::uint32_t new_idx = blk_idx + needed_gran;
88 detail::BlockHeader* new_blk = detail::block_at( base, new_idx );
89 std::memset( new_blk, 0, sizeof( detail::BlockHeader ) );
90 new_blk->prev_offset = blk_idx;
91 new_blk->next_offset = blk->next_offset;
92 new_blk->left_offset = detail::kNoBlock;
93 new_blk->right_offset = detail::kNoBlock;
94 new_blk->parent_offset = detail::kNoBlock;
95 new_blk->avl_height = 1;
96 if ( blk->next_offset != detail::kNoBlock )
97 detail::block_at( base, blk->next_offset )->prev_offset = new_idx;
98 else
99 hdr->last_block_offset = new_idx;
100 blk->next_offset = new_idx;
101 hdr->block_count++;
102 hdr->free_count++;
103 hdr->used_size += detail::kBlockHeaderGranules;
104 FreeBlockTreeT::insert( base, hdr, new_idx );
105 }
106
107 blk->size = data_gran;
108 blk->root_offset = blk_idx;
109 blk->left_offset = detail::kNoBlock;
110 blk->right_offset = detail::kNoBlock;
111 blk->parent_offset = detail::kNoBlock;
112 blk->avl_height = 0;
113 hdr->alloc_count++;
114 hdr->free_count--;
115 hdr->used_size += data_gran;
116 return detail::user_ptr( blk );
117 }

131 static void coalesce( std::uint8_t* base, detail::ManagerHeader* hdr, detail::BlockHeader* blk )
132 {
133 std::uint32_t b_idx = detail::block_idx( base, blk );
134
135 // Слияние с правым соседом
136 if ( blk->next_offset != detail::kNoBlock )
137 {
138 detail::BlockHeader* nxt = detail::block_at( base, blk->next_offset );
139 if ( nxt->size == 0 )
140 {
141 std::uint32_t nxt_idx = blk->next_offset;
142 FreeBlockTreeT::remove( base, hdr, nxt_idx );
143 blk->next_offset = nxt->next_offset;
144 if ( nxt->next_offset != detail::kNoBlock )
145 detail::block_at( base, nxt->next_offset )->prev_offset = b_idx;
146 else
147 hdr->last_block_offset = b_idx;
148 std::memset( nxt, 0, sizeof( detail::BlockHeader ) );
149 hdr->block_count--;
150 hdr->free_count--;
151 if ( hdr->used_size >= detail::kBlockHeaderGranules )
152 hdr->used_size -= detail::kBlockHeaderGranules;
153 }
154 }
155
156 // Слияние с левым соседом
157 if ( blk->prev_offset != detail::kNoBlock )
158 {
159 detail::BlockHeader* prv = detail::block_at( base, blk->prev_offset );
160 if ( prv->size == 0 )
161 {
162 std::uint32_t prv_idx = blk->prev_offset;
163 FreeBlockTreeT::remove( base, hdr, prv_idx );
164 prv->next_offset = blk->next_offset;
165 if ( blk->next_offset != detail::kNoBlock )
166 detail::block_at( base, blk->next_offset )->prev_offset = prv_idx;
167 else
168 hdr->last_block_offset = prv_idx;
169 std::memset( blk, 0, sizeof( detail::BlockHeader ) );
170 hdr->block_count--;
171 hdr->free_count--;
172 if ( hdr->used_size >= detail::kBlockHeaderGranules )
173 hdr->used_size -= detail::kBlockHeaderGranules;
174 FreeBlockTreeT::insert( base, hdr, prv_idx );
175 return;
176 }
177 }
178
179 FreeBlockTreeT::insert( base, hdr, b_idx );
180 }

193 static void rebuild_free_tree( std::uint8_t* base, detail::ManagerHeader* hdr )
194 {
195 hdr->free_tree_root = detail::kNoBlock;
196 hdr->last_block_offset = detail::kNoBlock;
197 std::uint32_t idx = hdr->first_block_offset;
198 while ( idx != detail::kNoBlock )
199 {
200 detail::BlockHeader* blk = detail::block_at( base, idx );
201 blk->left_offset = detail::kNoBlock;
202 blk->right_offset = detail::kNoBlock;
203 blk->parent_offset = detail::kNoBlock;
204 blk->avl_height = 0;
205 if ( blk->size == 0 )
206 FreeBlockTreeT::insert( base, hdr, idx );
207 if ( blk->next_offset == detail::kNoBlock )
208 hdr->last_block_offset = idx;
209 idx = blk->next_offset;
210 }
211 }

222 static void repair_linked_list( std::uint8_t* base, detail::ManagerHeader* hdr )

246 static void recompute_counters( std::uint8_t* base, detail::ManagerHeader* hdr )
247 {
248 std::uint32_t block_count = 0, free_count = 0, alloc_count = 0;
249 std::uint32_t used_gran = 0;
250 std::uint32_t idx = hdr->first_block_offset;
251 while ( idx != detail::kNoBlock )
252 {
253 if ( detail::idx_to_byte_off( idx ) + sizeof( detail::BlockHeader ) > hdr->total_size )
254 break;
255 detail::BlockHeader* blk = detail::block_at( base, idx );
256 block_count++;
257 used_gran += detail::kBlockHeaderGranules;
258 if ( blk->size > 0 )
259 {
260 alloc_count++;
261 used_gran += blk->size;
262 }
263 else
264 {
265 free_count++;
266 }
267 idx = blk->next_offset;
268 }
269 hdr->block_count = block_count;
270 hdr->free_count = free_count;
271 hdr->alloc_count = alloc_count;
272 hdr->used_size = used_gran;
