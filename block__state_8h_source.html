532template <typename AddressTraitsT>
533int detect_block_state( const void* raw_blk, typename AddressTraitsT::index_type own_idx ) noexcept
534{
535 const auto* base = reinterpret_cast<const BlockStateBase<AddressTraitsT>*>( raw_blk );
536 if ( base->is_free() )
537 return 0; // FreeBlock (или переходное — требует проверки AVL)
538 if ( base->is_allocated( own_idx ) )
539 return 1; // AllocatedBlock
540 return -1; // Неопределённое состояние (ошибка или переходное)
541}

553template <typename AddressTraitsT>
554void recover_block_state( void* raw_blk, typename AddressTraitsT::index_type own_idx ) noexcept
555{
556 auto* blk = reinterpret_cast<Block<AddressTraitsT>*>( raw_blk );
557
558 // Если weight > 0, но root_offset неверен — исправляем
559 if ( blk->weight > 0 && blk->root_offset != own_idx )
560 {
561 blk->root_offset = own_idx;
562 }
563
564 // Если weight == 0, но root_offset != 0 — исправляем
565 if ( blk->weight == 0 && blk->root_offset != 0 )
566 {
567 blk->root_offset = 0;
568 }
569}
