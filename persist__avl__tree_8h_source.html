36 static void insert( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
37 {
38 detail::BlockHeader* blk = detail::block_at( base, blk_idx );
39 blk->left_offset = detail::kNoBlock;
40 blk->right_offset = detail::kNoBlock;
41 blk->parent_offset = detail::kNoBlock;
42 blk->avl_height = 1;
43 if ( hdr->free_tree_root == detail::kNoBlock )
44 {
45 hdr->free_tree_root = blk_idx;
46 return;
47 }
48 // Issue #59: cache total_gran once; compute blk size in granules before the traversal loop
49 std::uint32_t total_gran = detail::byte_off_to_idx( hdr->total_size );
50 std::uint32_t blk_gran =
51 ( blk->next_offset != detail::kNoBlock ) ? ( blk->next_offset - blk_idx ) : ( total_gran - blk_idx );
52 std::uint32_t cur = hdr->free_tree_root, parent = detail::kNoBlock;
53 bool go_left = false;
54 while ( cur != detail::kNoBlock )
55 {
56 parent = cur;
57 detail::BlockHeader* n = detail::block_at( base, cur );
58 std::uint32_t n_gran =
59 ( n->next_offset != detail::kNoBlock ) ? ( n->next_offset - cur ) : ( total_gran - cur );
60 bool smaller = ( blk_gran < n_gran ) || ( blk_gran == n_gran && blk_idx < cur );
61 go_left = smaller;
62 cur = smaller ? n->left_offset : n->right_offset;
63 }
64 blk->parent_offset = parent;
65 if ( go_left )
66 detail::block_at( base, parent )->left_offset = blk_idx;
67 else
68 detail::block_at( base, parent )->right_offset = blk_idx;
69 rebalance_up( base, hdr, parent );
70 }

73 static void remove( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t blk_idx )
74 {
75 detail::BlockHeader* blk = detail::block_at( base, blk_idx );
76 std::uint32_t parent = blk->parent_offset;
77 std::uint32_t left = blk->left_offset;
78 std::uint32_t right = blk->right_offset;
79 std::uint32_t rebal = detail::kNoBlock;
80
81 if ( left == detail::kNoBlock && right == detail::kNoBlock )
82 {
83 set_child( base, hdr, parent, blk_idx, detail::kNoBlock );
84 rebal = parent;
85 }
86 else if ( left == detail::kNoBlock || right == detail::kNoBlock )
87 {
88 std::uint32_t child = ( left != detail::kNoBlock ) ? left : right;
89 detail::block_at( base, child )->parent_offset = parent;
90 set_child( base, hdr, parent, blk_idx, child );
91 rebal = parent;
92 }
93 else
94 {
95 std::uint32_t succ_idx = min_node( base, right );
96 detail::BlockHeader* succ = detail::block_at( base, succ_idx );
97 std::uint32_t succ_parent = succ->parent_offset;
98 std::uint32_t succ_right = succ->right_offset;
99
100 if ( succ_parent != blk_idx )
101 {
102 set_child( base, hdr, succ_parent, succ_idx, succ_right );
103 if ( succ_right != detail::kNoBlock )
104 detail::block_at( base, succ_right )->parent_offset = succ_parent;
105 succ->right_offset = right;
106 detail::block_at( base, right )->parent_offset = succ_idx;
107 rebal = succ_parent;
108 }
109 else
110 {
111 rebal = succ_idx;
112 }
113 succ->left_offset = left;
114 detail::block_at( base, left )->parent_offset = succ_idx;
115 succ->parent_offset = parent;
116 set_child( base, hdr, parent, blk_idx, succ_idx );
117 update_height( base, succ );
118 }
119 blk->left_offset = detail::kNoBlock;
120 blk->right_offset = detail::kNoBlock;
121 blk->parent_offset = detail::kNoBlock;
122 blk->avl_height = 0;
123 rebalance_up( base, hdr, rebal );
124 }

127 static std::uint32_t find_best_fit( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t needed_granules )
128 {
129 // Issue #59: cache total_gran once to avoid repeated hdr->total_size reads in the hot path
130 std::uint32_t total_gran = detail::byte_off_to_idx( hdr->total_size );
131 std::uint32_t cur = hdr->free_tree_root, result = detail::kNoBlock;
132 while ( cur != detail::kNoBlock )
133 {
134 detail::BlockHeader* node = detail::block_at( base, cur );
135 std::uint32_t cur_gran =
136 ( node->next_offset != detail::kNoBlock ) ? ( node->next_offset - cur ) : ( total_gran - cur );
137 if ( cur_gran >= needed_granules )
138 {
139 result = cur;
140 cur = node->left_offset;
141 }
142 else
143 {
144 cur = node->right_offset;
145 }
146 }
147 return result;
148 }

222 static void rebalance_up( std::uint8_t* base, detail::ManagerHeader* hdr, std::uint32_t node_idx )
223 {
224 std::uint32_t cur = node_idx;
225 while ( cur != detail::kNoBlock )
226 {
227 detail::BlockHeader* node = detail::block_at( base, cur );
228 update_height( base, node );
229 std::int32_t bf = balance_factor( base, node );
230 if ( bf > 1 )
231 {
232 if ( balance_factor( base, detail::block_at( base, node->left_offset ) ) < 0 )
233 rotate_left( base, hdr, node->left_offset );
234 cur = rotate_right( base, hdr, cur );
235 }
236 else if ( bf < -1 )
237 {
238 if ( balance_factor( base, detail::block_at( base, node->right_offset ) ) > 0 )
239 rotate_right( base, hdr, node->right_offset );
240 cur = rotate_left( base, hdr, cur );
241 }
242 cur = detail::block_at( base, cur )->parent_offset;
243 }
244 }

246 static std::uint32_t min_node( std::uint8_t* base, std::uint32_t node_idx )
247 {
248 while ( node_idx != detail::kNoBlock )
249 {
250 std::uint32_t left = detail::block_at( base, node_idx )->left_offset;
251 if ( left == detail::kNoBlock )
252 break;
253 node_idx = left;
254 }
