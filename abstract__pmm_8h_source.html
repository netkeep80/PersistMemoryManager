131 bool create( std::size_t initial_size ) noexcept
132 {
133 typename ThreadPolicyT::unique_lock_type lock( _mutex );
134 if ( initial_size < detail::kMinMemorySize )
135 return false;
136 // Align to granule_size
137 std::size_t aligned = ( ( initial_size + kGranuleSize - 1 ) / kGranuleSize ) * kGranuleSize;
138 if ( _backend.base_ptr() == nullptr )

140 // Backend has no buffer yet — ask it to expand from zero

156 bool load() noexcept
157 {
158 typename ThreadPolicyT::unique_lock_type lock( _mutex );
159 if ( _backend.base_ptr() == nullptr || _backend.total_size() < detail::kMinMemorySize )
160 return false;
161 std::uint8_t* base = _backend.base_ptr();
162 detail::ManagerHeader* hdr = get_header( base );
163 if ( hdr->magic != kMagic || hdr->total_size != _backend.total_size() )
164 return false;
165 if ( hdr->granule_size != static_cast<std::uint16_t>( kGranuleSize ) )
166 return false;
167 hdr->owns_memory = hdr->prev_owns_memory = false;
168 hdr->prev_total_size = 0;
169 hdr->prev_base_ptr = nullptr;
170 allocator::repair_linked_list( base, hdr );
171 allocator::recompute_counters( base, hdr );
172 allocator::rebuild_free_tree( base, hdr );
173 _initialized = true;
174 return true;
175 }

199 void* allocate( std::size_t user_size ) noexcept
200 {
201 typename ThreadPolicyT::unique_lock_type lock( _mutex );
202 if ( !_initialized || user_size == 0 )
203 return nullptr;
204
205 std::uint8_t* base = _backend.base_ptr();
206 detail::ManagerHeader* hdr = get_header( base );
207 std::uint32_t needed = detail::required_block_granules( user_size );
208 std::uint32_t idx = FreeBlockTreeT::find_best_fit( base, hdr, needed );
209
210 if ( idx != detail::kNoBlock )
211 return allocator::allocate_from_block( base, hdr, detail::block_at( base, idx ), user_size );
212
213 // Попытка расширить (если бэкенд поддерживает)
214 if ( !do_expand( user_size ) )
215 return nullptr;
216
217 base = _backend.base_ptr();
218 hdr = get_header( base );
219 idx = FreeBlockTreeT::find_best_fit( base, hdr, needed );
220 if ( idx != detail::kNoBlock )
221 return allocator::allocate_from_block( base, hdr, detail::block_at( base, idx ), user_size );
222 return nullptr;
223 }

228 void deallocate( void* ptr ) noexcept
229 {
230 typename ThreadPolicyT::unique_lock_type lock( _mutex );
231 if ( !_initialized || ptr == nullptr )
232 return;
233 std::uint8_t* base = _backend.base_ptr();
234 detail::ManagerHeader* hdr = get_header( base );
235 detail::BlockHeader* blk = detail::header_from_ptr( base, ptr, static_cast<std::size_t>( hdr->total_size ) );
236 if ( blk == nullptr || blk->size == 0 )
237 return;
238
239 std::uint32_t freed = blk->size;
240 blk->size = 0;
241 blk->root_offset = 0;
242 hdr->alloc_count--;
243 hdr->free_count++;
244 if ( hdr->used_size >= freed )
245 hdr->used_size -= freed;
246 allocator::coalesce( base, hdr, blk );
247 }

432 bool init_layout( std::uint8_t* base, std::size_t size ) noexcept
433 {
434 static constexpr std::uint32_t kHdrBlkIdx = 0;
435 static constexpr std::uint32_t kFreeBlkIdx = detail::kBlockHeaderGranules + detail::kManagerHeaderGranules;
436
437 if ( detail::idx_to_byte_off( kFreeBlkIdx ) + sizeof( detail::BlockHeader ) + detail::kMinBlockSize > size )
438 return false;
439
440 detail::BlockHeader* hdr_blk = detail::block_at( base, kHdrBlkIdx );
441 std::memset( hdr_blk, 0, sizeof( detail::BlockHeader ) );
442 hdr_blk->size = detail::kManagerHeaderGranules;
443 hdr_blk->prev_offset = detail::kNoBlock;
444 hdr_blk->next_offset = kFreeBlkIdx;
445 hdr_blk->left_offset = detail::kNoBlock;
446 hdr_blk->right_offset = detail::kNoBlock;
447 hdr_blk->parent_offset = detail::kNoBlock;
448 hdr_blk->root_offset = kHdrBlkIdx;
449
450 detail::ManagerHeader* hdr = get_header( base );
451 std::memset( hdr, 0, sizeof( detail::ManagerHeader ) );
452 hdr->magic = kMagic;
453 hdr->total_size = size;
454 hdr->first_block_offset = kHdrBlkIdx;
455 hdr->last_block_offset = detail::kNoBlock;
456 hdr->free_tree_root = detail::kNoBlock;
457 hdr->granule_size = static_cast<std::uint16_t>( kGranuleSize );
458
459 detail::BlockHeader* blk = detail::block_at( base, kFreeBlkIdx );
460 std::memset( blk, 0, sizeof( detail::BlockHeader ) );
461 blk->prev_offset = kHdrBlkIdx;
462 blk->next_offset = detail::kNoBlock;
463 blk->left_offset = detail::kNoBlock;
464 blk->right_offset = detail::kNoBlock;
465 blk->parent_offset = detail::kNoBlock;
466 blk->avl_height = 1;
467
468 hdr->last_block_offset = kFreeBlkIdx;
469 hdr->free_tree_root = kFreeBlkIdx;
470 hdr->block_count = 2;
471 hdr->free_count = 1;
472 hdr->alloc_count = 1;
473 hdr->used_size = kFreeBlkIdx + detail::kBlockHeaderGranules;
474
475 _initialized = true;
476 return true;
477 }

479 bool do_expand( std::size_t user_size ) noexcept
480 {
481 if ( !_initialized )
482 return false;
483 std::uint8_t* base = _backend.base_ptr();
484 detail::ManagerHeader* hdr = get_header( base );
485 std::size_t old_size = hdr->total_size;
486
487 // Вычисляем минимально необходимый прирост
488 std::size_t min_need =
489 detail::granules_to_bytes( detail::required_block_granules( user_size ) + detail::kBlockHeaderGranules );
490 std::size_t growth = old_size / 4; // 25% рост
491 if ( growth < min_need )
492 growth = min_need;
493
494 if ( !_backend.expand( growth ) )
495 return false;
496
497 std::uint8_t* new_base = _backend.base_ptr();
498 std::size_t new_size = _backend.total_size();
499 if ( new_base == nullptr || new_size <= old_size )
500 return false;
501
502 // Обновляем total_size в заголовке и добавляем новый свободный блок
503 hdr = get_header( new_base );
504
505 std::uint32_t extra_idx = detail::byte_off_to_idx( old_size );
506 std::size_t extra_size = new_size - old_size;
507
508 detail::BlockHeader* last_blk = ( hdr->last_block_offset != detail::kNoBlock )
509 ? detail::block_at( new_base, hdr->last_block_offset )
510 : nullptr;
511
512 if ( last_blk != nullptr && last_blk->size == 0 )
513 {
514 // Расширяем последний свободный блок
515 std::uint32_t loff = detail::block_idx( new_base, last_blk );
516 FreeBlockTreeT::remove( new_base, hdr, loff );
517 hdr->total_size = new_size;
518 FreeBlockTreeT::insert( new_base, hdr, loff );
519 }
520 else
521 {
522 if ( extra_size < sizeof( detail::BlockHeader ) + detail::kMinBlockSize )
523 return false;
524 detail::BlockHeader* nb_blk = detail::block_at( new_base, extra_idx );
525 std::memset( nb_blk, 0, sizeof( detail::BlockHeader ) );
526 nb_blk->left_offset = detail::kNoBlock;
527 nb_blk->right_offset = detail::kNoBlock;
528 nb_blk->parent_offset = detail::kNoBlock;
529 nb_blk->avl_height = 1;
530 if ( last_blk != nullptr )
531 {
532 std::uint32_t loff = detail::block_idx( new_base, last_blk );
533 nb_blk->prev_offset = loff;
534 nb_blk->next_offset = detail::kNoBlock;
535 last_blk->next_offset = extra_idx;
536 }
537 else
538 {
539 nb_blk->prev_offset = detail::kNoBlock;
540 nb_blk->next_offset = detail::kNoBlock;
541 hdr->first_block_offset = extra_idx;
542 }
543 hdr->last_block_offset = extra_idx;
544 hdr->block_count++;
545 hdr->free_count++;
546 hdr->total_size = new_size;
547 FreeBlockTreeT::insert( new_base, hdr, extra_idx );
548 }
549 return true;
550 }
